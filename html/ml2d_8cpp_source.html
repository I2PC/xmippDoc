119 {
120 //#define DEBUG_JM
121 #ifdef DEBUG_JM
122  std::cerr<<">>>>>>>> entering checkConvergence"<<std::endl;
123 #endif
124 
125  if (iter == 0)
126  return false;
127 
128  bool converged = true;
129  double convv, avg2;
130  MultidimArray<double> Maux;
131 
132  Maux.resize(dim, dim);
133  Maux.setXmippOrigin();
134 
135  conv.clear();
136 
137  for (int refno = 0; refno < model.n_ref; refno++)
138  {
139  convv = -1;
140 
141  if (model.Iref[refno].weight() > 0.)
142  {
143  Maux = Iold[refno]() * Iold[refno]();
144  avg2 = Maux.computeAvg();
145  if (avg2 > 0.)
146  {
147  convv = 1. / avg2;

149  Maux = Maux * Maux;

151  if (convv > eps)

153  }
154  }

156  }

359 void ModelML2D::combineModel(const ModelML2D &model, int sign)
360 {
361  if (n_ref != model.n_ref)
362  REPORT_ERROR(ERR_VALUE_INCORRECT, "Can not add models with different 'n_ref'");
363 
364 #define COMBINE(var) var += sign * model.var
365  COMBINE(wsum_sigma_offset);
366  COMBINE(wsum_sigma_noise);
367  COMBINE(sumw_allrefs);
368  COMBINE(sumw_allrefs2);
369  COMBINE(sumfracweight);
370  COMBINE(LL);
371 
372  MultidimArray<double> tmp;
373  double sumweight = 0., w1 = 0., w2 = 0.;
374 
375  for (int refno = 0; refno < n_ref; refno++)
376  {
377  w1 = WsumMref[refno].weight();
378  w2 = sign * model.WsumMref[refno].weight();
379  tmp = model.WsumMref[refno]();
380  tmp *= sign;
381  WsumMref[refno]() += tmp;
382  sumweight = w1 + w2;
383 
384  if (sumweight > 0.)
385  {
386  Iref[refno].setWeight(sumweight);
387  WsumMref[refno].setWeight(sumweight);
388  COMBINE(sumw_mirror[refno]);
389  COMBINE(sumwsc[refno]);
390  }
391  else
392  {
393  Iref[refno]().initZeros();
394  WsumMref[refno]().initZeros();
395  Iref[refno].setWeight(0);
396  WsumMref[refno].setWeight(0.);
397  sumw_mirror[refno] = 0.;
398  sumwsc[refno] = 0.;
399  }
400  }

413 void ModelML2D::update()
414 {
415  if (sumw_allrefs < 0)
416  REPORT_ERROR(ERR_VALUE_INCORRECT, "updateFractions: sumw_allrefs should be greater than 0 ");
417 
418  //update sigma_noise
419  double sum = (do_student && do_student_sigma_trick) ? sumw_allrefs2
420  : sumw_allrefs;
421  double sigma_noise2 = wsum_sigma_noise / (sum * dim * dim);
422 
423  if (sigma_noise2 < 0.)
424  REPORT_ERROR(ERR_VALUE_INCORRECT, "sqrt of negative 'sigma_noise2'");
425  sigma_noise = sqrt(sigma_noise2);
426 
427  //update sigma_offset
428  if (wsum_sigma_offset < 0.)
429  REPORT_ERROR(ERR_VALUE_INCORRECT, "sqrt of negative 'wsum_sigma_offset'");
430  sigma_offset = sqrt(wsum_sigma_offset / (2. * sumw_allrefs));
431 
432  //update avePmax
433  avePmax = sumfracweight / sumw_allrefs;
434 
435  double weight, inv_weight;
436 
437 // std::cerr << "DEBUG_JM: sumw_allrefs: " << sumw_allrefs << std::endl;
438 
439  for (int refno = 0; refno < n_ref; ++refno)
440  {
441  weight = WsumMref[refno].weight();
442 
443 // std::cerr << " DEBUG_JM: refno: " << refno << std::endl;
444 
445  if (weight > 0.)
446  {
447  //update weights
448  Iref[refno].setWeight(weight);
449  inv_weight = 1 / weight; //just to speed-up
450  Iref[refno]() = WsumMref[refno]();
451  Iref[refno]() *= inv_weight;
452  //update fractions
453  alpha_k[refno] = weight / sumw_allrefs;
454  mirror_fraction[refno] = sumw_mirror[refno] * inv_weight;
455  scale[refno] = sumwsc[refno] * inv_weight;
456 // std::cerr << " DEBUG_JM: weight: " << weight << std::endl;
457 // std::cerr << " DEBUG_JM: sumw_mirror[refno]: " << sumw_mirror[refno] << std::endl;
458  }
459  else
460  {
461 // std::cerr << " DEBUG_JM: all zeros... " << std::endl;
462  //zero weights
463  Iref[refno].setWeight(0.);
464  Iref[refno]().initZeros(WsumMref[refno]());
465  alpha_k[refno] = 0.;
466  mirror_fraction[refno] = 0.;
467  scale[refno] = 0.;
468  }
469  }
470 }
