381 /* Get mask value ---------------------------------------------------------- */
382 double FourierFilter::maskValue(const Matrix1D<double> &w)
383 {
384  double absw = w.module();
385  double wx=fabs(XX(w));
386  double wy=fabs(YY(w));
387 
388  // Generate mask
389  switch (FilterBand)
390  {
391  case LOWPASS:
392  switch (FilterShape)
393  {
394  case RAISED_COSINE:
395  if (absw<w1)
396  return 1;
397  else if (absw<w1+raised_w)
398  return (1+cos(PI/raised_w*(absw-w1)))/2;
399  else
400  return 0;
401  break;
402  case GAUSSIAN:
403  return 1/sqrt(2*PI*w1)*exp(-0.5*absw*absw/(w1*w1));
404  break;
405  case REALGAUSSIAN:
406  return exp(-PI*PI*absw*absw*w1*w1);
407  break;
408  }
409  break;
410  case HIGHPASS:
411  switch (FilterShape)
412  {
413  case RAISED_COSINE:
414  if (absw>w1)
415  return 1;
416  else if (absw>w1-raised_w)
417  return (1+cos(PI/raised_w*(w1-absw)))/2;
418  else
419  return 0;
420  break;
421  }
422  break;
423  case BANDPASS:
424  switch (FilterShape)
425  {
426  case RAISED_COSINE:
427  if (absw>=w1 && absw<=w2)
428  return 1;
429  else if (absw>w1-raised_w && absw<w1)
430  return (1+cos(PI/raised_w*(w1-absw)))/2;
431  else if (absw<w2+raised_w && absw>w2)
432  return (1+cos(PI/raised_w*(w2-absw)))/2;
433  else
434  return 0;
435  break;
436  }
437  break;
438  case STOPBAND:
439  switch (FilterShape)
440  {
441  case RAISED_COSINE:
442  if (absw>=w1 && absw<=w2)
443  return 0;
444  else if (absw>w1-raised_w && absw<w1)
445  return 1-(1+cos(PI/raised_w*(w1-absw)))/2;
446  else if (absw<w2+raised_w && absw>w2)
447  return 1-(1+cos(PI/raised_w*(w2-absw)))/2;
448  else
449  return 1;
450  break;
451  }
452  break;
453  case STOPLOWBANDX:
454  switch (FilterShape)
455  {
456  case RAISED_COSINE:
457  if (wx<w1)
458  return 0;
459  else if (wx<w1+raised_w)
460  return (1+cos(PI/raised_w*(w1-wx)))/2;
461  else
462  return 1.0;
463  break;
464  }
465  break;
466  case STOPLOWBANDY:
467  switch (FilterShape)
468  {
469  case RAISED_COSINE:
470  if (wy<w1)
471  return 0;
472  else if (wy<w1+raised_w)
473  return (1+cos(PI/raised_w*(w1-wy)))/2;
474  else
475  return 1.0;
476  break;
477  }
478  break;
479  case CTF:
480  ctf.precomputeValues(XX(w)/ctf.Tm,YY(w)/ctf.Tm);
481  return ctf.getValueAt();
482  break;
483  case CTFPOS:
484  ctf.precomputeValues(XX(w)/ctf.Tm,YY(w)/ctf.Tm);
485  return fabs(ctf.getValueAt());
486  break;
487  case CTFINV:
488  {
489  ctf.precomputeValues(XX(w)/ctf.Tm,YY(w)/ctf.Tm);
490  double ctfval=ctf.getValueAt();
491  if (fabs(ctfval)<=minCTF)
492  return 0.0;
493  else
494  return 1.0/ctfval;
495  }
496  break;
497  case CTFPOSINV:
498  {
499  ctf.precomputeValues(XX(w)/ctf.Tm,YY(w)/ctf.Tm);
500  double ctfval=fabs(ctf.getValueAt());
501  if (ctfval<=minCTF)
502  return 0.0;
503  else
504  return 1.0/ctfval;
505  }
506  break;
507  case BFACTOR:
508  {
509  double R = absw / sampling_rate;
510  return exp( - (w1 / 4.) * R * R);
511  }
512  break;
513  case FSCPROFILE:
514  {
515  double R = absw / sampling_rate;
516  int idx=-1;
517  size_t imax=freqContFSC.size();
518  double *ptrFreq=&freqContFSC[0];
519  for (size_t i=0; i<imax; ++i)
520  if (ptrFreq[i]>R)
521  {
522  idx=(int)i;
523  break;
524  }
525  if (idx>=1)
526  {
527  double x0=freqContFSC[idx-1];
528  double x1=freqContFSC[idx];
529  double y0=FSC[idx-1];
530  double y1=FSC[idx];
531  return y0+(y1-y0)*(R-x0)/(x1-x0);
532  }
533  return 0;
534  }
535  case ASTIGMATISMPROFILE:
536  {
537  double R = absw / sampling_rate;
538  ctf.precomputeValues(R,0.0);
539  double phaseU=ctf.getPhaseAt();
540  ctf.precomputeValues(0.0,R);
541  double phaseV=ctf.getPhaseAt();
542  double diff=(phaseV-phaseU)*0.5;
543  return exp(-0.5*diff*diff/(w1*w1));
544  }
545  break;
546  default:
547  REPORT_ERROR(ERR_ARG_INCORRECT,"Unknown mask type");
548  }
549  return 0;
550 }

552 /* Generate mask ----------------------------------------------------------- */
553 void FourierFilter::generateMask(MultidimArray<double> &v)
554 {
555  if (FilterShape==SPARSIFY)
556  return;
557 
558  if (FilterShape==FSCPROFILE)
559  {
560  MetaData mdFSC(fnFSC);
561  mdFSC.getColumnValues(MDL_RESOLUTION_FREQ,freqContFSC);
562  mdFSC.getColumnValues(MDL_RESOLUTION_FRC,FSC);
563  }
564  // std::cout << "Generating mask " << do_generate_3dmask << std::endl;
565 
566  if (do_generate_3dmask)
567  {
568  transformer.setReal(v);
569  MultidimArray< std::complex<double> > Fourier;
570  transformer.getFourierAlias(Fourier);
571 
572  if (FilterShape==WEDGE || FilterShape==CONE)
573  {
574  maskFourier.initZeros(Fourier);
575  maskFourier.setXmippOrigin();
576  switch (FilterShape)
577  {
578  case WEDGE:
579  {
580  Matrix2D<double> A;
581  Euler_angles2matrix(rot,tilt,psi,A,false);
582  BinaryWedgeMask(maskFourier, t1, t2, A,true);
583  break;
584  }
585  case CONE:
586  BinaryConeMask(maskFourier, 90. - fabs(t1),INNER_MASK,true);
587  break;
588  }
589  }
590  else if (FilterShape==BINARYFILE)
591  {
592  Image<double> filter;
593  filter.read(fnFilter);
594  scaleToSize(BSPLINE3, maskFourierd, filter(), XSIZE(v), YSIZE(v), ZSIZE(v));
595  maskFourierd.resize(Fourier);
596  return;
597  }
598 
599  else
600  {
601  maskFourierd.initZeros(Fourier);
602  maskFourierd.setXmippOrigin();
603 
604  w.resizeNoCopy(3);
605  for (size_t k=0; k<ZSIZE(Fourier); k++)
606  {
607  FFT_IDX2DIGFREQ(k,ZSIZE(v),ZZ(w));
608  for (size_t i=0; i<YSIZE(Fourier); i++)
609  {
610  FFT_IDX2DIGFREQ(i,YSIZE(v),YY(w));
611  for (size_t j=0; j<XSIZE(Fourier); j++)
612  {
613  FFT_IDX2DIGFREQ(j,XSIZE(v),XX(w));
614  DIRECT_A3D_ELEM(maskFourierd,k,i,j)=maskValue(w);
615  }
616  }
617  }
618  }
619  }
620  else if (MULTIDIM_SIZE(v)<=1024*1024)
621  {
622  transformer.setReal(v);
