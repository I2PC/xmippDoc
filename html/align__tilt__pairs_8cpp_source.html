81 // Center one tilted image =====================================================
82 //#define DEBUG
83 bool ProgAlignTiltPairs::centerTiltedImage(const MultidimArray<double> &imgRefU,
84  bool flip,
85  double inPlaneU, double shiftXu, double shiftYu,
86  double alphaT, double alphaU,
87  double tilt, MultidimArray<double> &imgT, double &shiftX,
88  double &shiftY, CorrelationAux &auxCorr)
89 {
90  // Cosine stretching, store stretched image in imgT2DClass
91  MultidimArray<double> imgT2DClass;
92  Matrix2D<double> E, E2D, Mu2D, A2D;
93 
94  if (!do_stretch)
95  tilt = flip ? 180. : 0.;
96 
97  Euler_angles2matrix(flip ? alphaU : -alphaU, tilt, alphaT, E, true);
98 
99  rotation2DMatrix(inPlaneU, Mu2D, true);
100  MAT_ELEM(Mu2D,0,2) = flip ? shiftXu : -shiftXu;
101  MAT_ELEM(Mu2D,1,2) = -shiftYu;
102 
103  if (flip)
104  {
105  // We need to multiply the first column by -1, because
106  // in Xmipp the Mu2D order is: first flip (if necessary), then rot, then shift
107  //MAT_ELEM(Mu2D,0,0)*=-1;
108 
109  MAT_ELEM(Mu2D,1,0)*=-1;
110  MAT_ELEM(Mu2D,2,0)*=-1;
111 
112  // We need to multiply the first row by -1, as needed by the RCT theory
113  MAT_ELEM(Mu2D,0,1)*=-1;
114  MAT_ELEM(Mu2D,0,2)*=-1;
115  }
116 
117  E2D.initIdentity(3);
118  MAT_ELEM(E2D,0,0)=MAT_ELEM(E,0,0);
119  MAT_ELEM(E2D,0,1)=MAT_ELEM(E,0,1);
120  MAT_ELEM(E2D,1,0)=MAT_ELEM(E,1,0);
121  MAT_ELEM(E2D,1,1)=MAT_ELEM(E,1,1);
122  A2D = Mu2D * E2D.inv();
123 
124  applyGeometry(LINEAR, imgT2DClass, imgT, A2D, IS_NOT_INV, WRAP);
125 
126  // Calculate best shift
127  int max_shift_pixels=(int)(max_shift/100.0*XSIZE(imgT));
128  CorrelationAux aux;
129  double corr=bestShift(imgRefU, imgT2DClass, shiftX, shiftY, auxCorr, NULL, max_shift_pixels);
130  // double corr=bestShiftRealSpace(imgRefU, imgT2DClass, shiftX, shiftY, NULL, max_shift_pixels, 0.5);
131 
132 #ifdef DEBUG
133 
134  std::cout << "alphaU=" << alphaU << " inplaneU=" << inPlaneU << " tilt=" << tilt << " alphaT=" << alphaT << std::endl;
135  std::cout << "Best shift= " << shiftX << " " << shiftY << " corr=" << corr << std::endl;
136 #endif
137 
138  Matrix1D<double> vShift(2);
139  XX(vShift)=shiftX;
140  YY(vShift)=shiftY;
141  Matrix2D<double> Tt, Tt2D;
142  translation2DMatrix(vShift, Tt2D,true);
143  Tt=A2D.inv()*Tt2D.inv()*A2D;
144 
145  // Readjust shift
146  shiftX=MAT_ELEM(Tt,0,2);
147  shiftY=MAT_ELEM(Tt,1,2);
148  double shift = sqrt(shiftX * shiftX + shiftY * shiftY);
149 
150 //#define DEBUG_2
151 #ifdef DEBUG_2
152 
153  Image<double> save;
154  save() = imgT;
155  save.write("PPPtilted.xmp");
156  save() = imgRefU;
157  save.write("PPPuntiltedRef.xmp");
158  save() = imgT2DClass;
159  save.write("PPPtiltedAdjusted.xmp");
160  std::cout << "Corrected shift= " << shiftX << " " << shiftY << std::endl;
161  std::cout << "Press any key\n";
162  char c = getchar();
163 #endif
164 
165  return (shift < (double)max_shift_pixels) || corr<0;
166 }

181

190

193  Matrix2D<double> E, Tu, Tup;

199

207  row.getValue(MDL_IMAGE, fnUntilted);
208  row.getValue(MDL_IMAGE_TILTED, fnTilted);
209  imgT.read(fnTilted);
210  imgT().setXmippOrigin();
211 
212 
213 #ifdef DEBUG
214  {
215  Image<double> imgU;
216  imgU.read(fnUntilted);
217  imgU().setXmippOrigin();
218  Image<double> save;
219  save() = imgU();
220  save.write("PPPuntilted.xmp");
221  Matrix2D<double> E;
222  save.readApplyGeo(fnUntilted, row);
223  save.write("PPPuntiltedAligned.xmp");
224  geo2TransformationMatrix(row, E);
225  //rotation2DMatrix(-inPlaneU,E);
226  //save() = imgU();
227  //applyGeometry(LINEAR, save(), imgU(), E, IS_NOT_INV, WRAP);
228  //save.write("PPPuntiltedAligned2.xmp");
229  }
230 #endif
231 
232  // Get alignment parameters
233  double shiftXu, shiftYu;
234  row.getValue(MDL_ANGLE_Y, alphaU);
235  row.getValue(MDL_ANGLE_Y2, alphaT);
236  row.getValue(MDL_ANGLE_TILT, tilt);
237  row.getValue(MDL_SHIFT_X, shiftXu);
238  row.getValue(MDL_SHIFT_Y, shiftYu);
239 
240  if (flip)
241  {
242  tilt += 180;
243  minusInPlaneU = inPlaneU+alphaU;
244  }
245  else
246  minusInPlaneU=-(inPlaneU+alphaU);
247 
248  // Correct untilted alignment
249  Euler_angles2matrix( minusInPlaneU, tilt, alphaT, E, true);
250  XX(vShift) = shiftXu;
251  YY(vShift) = shiftYu;
252  translation3DMatrix(vShift, Tu);
253  Tup = E * Tu * E.inv();
254 
255  // Align tilt and untilted projections
256  double shiftX = 0, shiftY = 0;
257  bool enable = true;
258  if (max_shift > 0)
259  enable = centerTiltedImage(imgRef(), flip, inPlaneU, shiftXu, shiftYu,
260  alphaT, alphaU, tilt, imgT(), shiftX, shiftY, auxCorr);
261  if (!enable)
262  {
263  ++nDiscarded;
264  shiftX = shiftY = 0;
265  }
266 
267  // Write results
268  rowOut.setValue(MDL_IMAGE, fnTilted);
269  rowOut.setValue(MDL_ANGLE_ROT, minusInPlaneU);
270  rowOut.setValue(MDL_ANGLE_TILT, tilt);
271  rowOut.setValue(MDL_ANGLE_PSI, alphaT);
272  rowOut.setValue(MDL_SHIFT_X, -MAT_ELEM(Tup,0,3) + shiftX);
273  rowOut.setValue(MDL_SHIFT_Y, -MAT_ELEM(Tup,1,3) + shiftY);
274  rowOut.setValue(MDL_ENABLED, (int) enable);
275  mdOut.addRow(rowOut);
276 
277  setProgress(++imgno);
278  }
279 
280  if (verbose && nDiscarded)
281  std::cout << " Discarded " << nDiscarded
282  << " tilted images that shifted too much" << std::endl;
283 
284  // Write out selfile
285  mdOut.write(fnOut);
286 }
