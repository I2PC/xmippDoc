160 void ProgReconsART::run()
161 {
162  BasicARTParameters &artPrm = artRecons->artPrm;
163 
164  Image<double> vol_voxels;
165  GridVolume vol_basis;
166  // Configure time clock
167  time_config();
168 
169  struct timeval start_time, end_time;
170  long int init_usecs, process_usecs, finish_usecs;
171 
172  gettimeofday(&start_time, NULL);
173 
174  show();
175  // Produce side information and initial volume
176  artRecons->preProcess(vol_basis);
177  if (verbose > 0)
178  {
179  std::cout << " ---------------------------------------------------------------------" << std::endl;
180  std::cout << " Projections : " << artRecons->artPrm.numIMG << std::endl;
181  std::cout << " ---------------------------------------------------------------------" << std::endl;
182  }
183  // Show parameters and initiate history
184  artRecons->initHistory(vol_basis);
185 
186  gettimeofday(&end_time, NULL);
187 
188  init_usecs = (end_time.tv_sec-start_time.tv_sec)*1000000+(end_time.tv_usec-start_time.tv_usec);
189 
190  gettimeofday(&start_time,NULL);
191 
192  // Iterations
193  artRecons->iterations(vol_basis);
194 
195 
196  gettimeofday(&end_time,NULL);
197 
198  process_usecs = (end_time.tv_sec-start_time.tv_sec)*1000000+(end_time.tv_usec-start_time.tv_usec);
199 
200  gettimeofday(&start_time,NULL);
201 
202  // Finish iterations
203  artRecons->postProcess(vol_basis);
204 
205  // Write final volume
206  int Xoutput_volume_size=(artPrm.Xoutput_volume_size==0) ?
207  artPrm.projXdim:artPrm.Xoutput_volume_size;
208  int Youtput_volume_size=(artPrm.Youtput_volume_size==0) ?
209  artPrm.projYdim:artPrm.Youtput_volume_size;
210  int Zoutput_volume_size=(artPrm.Zoutput_volume_size==0) ?
211  artPrm.projXdim:artPrm.Zoutput_volume_size;
212 
213  // int min_distance = ceil( ( 2 * artPrm.grid_relative_size ) / artPrm.basis.blob.radius) + 1;
214 
215  artPrm.basis.changeToVoxels(vol_basis, &(vol_voxels()),
216  Zoutput_volume_size, Youtput_volume_size, Xoutput_volume_size,artPrm.threads);
217 
218  vol_voxels.write(artPrm.fn_out);
219 
220  if (artPrm.tell&TELL_SAVE_BASIS)
221  vol_basis.write(artPrm.fn_root+".basis");
222  artPrm.fh_hist->close();
223 
224  gettimeofday(&end_time,NULL);
225 
226  finish_usecs = (end_time.tv_sec-start_time.tv_sec)*1000000+(end_time.tv_usec-start_time.tv_usec);
227 
228  std::cout << "INIT_TIME: " << (double)init_usecs/(double)1000000 << std::endl;
229  std::cout << "PROCESS_TIME: " << (double)process_usecs/(double)1000000 << std::endl;
230  std::cout << "FINISH_TIME: " << (double)finish_usecs/(double)1000000 << std::endl;
231 
232 }
