83 void ProgImageResize::preProcess()
84 {
85  double factor=1.0;
86  double oxdim = xdimOut, oydim = ydimOut, ozdim = zdimOut;
87 
88  //If zdimOut greater than 1, is a volume and should apply transform
89  dim = (isVol = (zdimOut > 1)) ? 3 : 2;
90 
91  resizeFactor.resizeNoCopy(dim);
92  resizeFactor.initConstant(1.);
93 
94  if (fn_out.empty() && oroot.empty())
95  {
96  fn_out=fn_in+"_tmp";
97  fn_out=fn_out.addExtension(fn_in.getExtension());
98  temporaryOutput=true;
99  create_empty_stackfile=true;
100  }
101 
102  if (checkParam("--factor") || checkParam("--dim"))
103  {
104  //Calculate scale factor from images sizes and given dimensions
105  //this approach assumes that all images have equal size
106  if (checkParam("--dim"))
107  {
108  xdimOut = getIntParam("--dim", 0);
109  ydimOut = STR_EQUAL(getParam("--dim", 1), "x") ? xdimOut : getIntParam("--dim", 1);
110  XX(resizeFactor) = (double)xdimOut / oxdim;
111  YY(resizeFactor) = (double)ydimOut / oydim;
112  }
113  else
114  {
115  factor = getDoubleParam("--factor");
116  if (factor <= 0)
117  REPORT_ERROR(ERR_VALUE_INCORRECT,"Resize factor must be a positive number");
118  xdimOut = (int) (xdimOut * factor);
119  ydimOut = (int) (ydimOut * factor);
120  resizeFactor.initConstant(factor);
121  }
122 
123  //if scale factor is large splines s not the way to go, print a warning
124  if( fabs(1.0-XX(resizeFactor)) > 0.1 )
125  {
126  reportWarning("Do not apply large scale factor using B-splines "
127  "choose fourier option.");
128  }
129 
130  if (isVol)
131  {
132  if (checkParam("--dim"))
133  {
134  zdimOut = STR_EQUAL(getParam("--dim", 2), "x")
135  ? xdimOut : getIntParam("--dim", 2);
136  ZZ(resizeFactor) = (double)zdimOut / ozdim;
137  }
138  else
139  zdimOut = (int) (zdimOut * factor);
140  }
141  scale_type = RESIZE_FACTOR;
142  }
143  else if (checkParam("--fourier"))
144  {
145  // if (isVol)
146  // REPORT_ERROR(ERR_PARAM_INCORRECT, "The 'fourier' scaling type is only valid for images");
147  int oxdim = xdimOut, oydim = ydimOut, ozdim = zdimOut;
148  scale_type = RESIZE_FOURIER;
149 
150  xdimOut = getIntParam("--fourier", 0);
151  ydimOut = STR_EQUAL(getParam("--fourier", 1), "x") ? xdimOut : getIntParam("--fourier", 1);
152  if (isVol)
153  zdimOut = STR_EQUAL(getParam("--fourier", 2), "x") ? xdimOut : getIntParam("--fourier", 2);
154  else
155  zdimOut=1;
156  fourier_threads = getIntParam("--fourier", 3);
157  XX(resizeFactor) = (double)xdimOut / oxdim;
158  YY(resizeFactor) = (double)ydimOut / oydim;
159  if (isVol)
160  ZZ(resizeFactor) = (double)zdimOut / ozdim;
161  //Do not think this is true
162  // if (oxdim < xdimOut || oydim < ydimOut)
163  // REPORT_ERROR(ERR_PARAM_INCORRECT, "The 'fourier' scaling type can only be used for reducing size");
164  }
165  else if (checkParam("--pyramid"))
166  {
167  scale_type = RESIZE_PYRAMID_EXPAND;
168  pyramid_level = getIntParam("--pyramid", 0);
169  factor = (double)(pow(2.0, pyramid_level));
170  xdimOut = (size_t)(xdimOut*factor);
171  ydimOut = (size_t)(ydimOut*factor);
172  if (isVol)
173  zdimOut = (size_t)(zdimOut*factor);
174  if (pyramid_level < 0)
175  {
176  pyramid_level *= -1; //change sign, negative means reduce operation
177  scale_type = RESIZE_PYRAMID_REDUCE;
178  }
179  resizeFactor.initConstant(factor);
180  }
181 }

186 void ProgImageResize::processImage(const FileName &fnImg, const FileName &fnImgOut, const MDRow &rowIn, MDRow &rowOut)
187 {
188  double aux;
189  if (apply_geo)
190  {
191  SCALE_SHIFT(MDL_SHIFT_X, XX(resizeFactor));
192  SCALE_SHIFT(MDL_SHIFT_Y, YY(resizeFactor));
193  if (isVol)
194  SCALE_SHIFT(MDL_SHIFT_Z, ZZ(resizeFactor));
195  }
196  else
197  rowOut.resetGeo(false);
198 
199  img.read(fnImg);
200  img().setXmippOrigin();
201  imgOut.setDatatype(img.getDatatype());
202 
203  switch (scale_type)
204  {
205  case RESIZE_FACTOR:
206  //selfScaleToSize(splineDegree, img(), xdimOut, ydimOut, zdimOut);
207  scaleToSize(splineDegree, imgOut(), img(), xdimOut, ydimOut, zdimOut);
208  break;
209  case RESIZE_PYRAMID_EXPAND:
210  pyramidExpand(splineDegree, imgOut(), img(), pyramid_level);
211  //selfPyramidExpand(splineDegree, img(), pyramid_level);
212  break;
213  case RESIZE_PYRAMID_REDUCE:
214  pyramidReduce(splineDegree, imgOut(), img(), pyramid_level);
215  //selfPyramidReduce(splineDegree, img(), pyramid_level);
216  break;
217  case RESIZE_FOURIER:
218  selfScaleToSizeFourier(zdimOut, ydimOut, xdimOut, img(), fourier_threads);
219  img.write(fnImgOut);
220  return;
221  case RESIZE_NONE:
222  break;
223  }
224  imgOut.write(fnImgOut);
225 }
