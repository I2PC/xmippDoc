130  outMovieFn = self._getExtraPath(self._getOutputMovieName(movie))
131

133  dark = inputMovies.getDark()
134  gain = inputMovies.getGain()
135  # Get the number of frames and the range
136  # to be used for alignment and sum
137  x, y, n = movie.getDim()
138  a0, aN = self._getFrameRange(n, 'align')
139  gpuId = self.gpuList.get()

141  writeMovieMd(movie, inputMd, a0, aN, useAlignment=self.useAlignment)
142 
143  args = '-i %s ' % inputMd
144  args += '-o "%s" ' % self._getOutputShifts(movie)
145  args += ' --frameRange %d %d ' % (0, aN - a0)
146 
147  if dark:
148  args += '--dark %s ' % dark
149  if gain:
150  args += '--gain %s ' % gain
151  winSize = self.winSize.get()
152  doSaveMovie = self.doSaveMovie.get()
153  groupSize = self.groupSize.get()
154  args += ' --winSize %(winSize)d --groupSize %(groupSize)d ' % locals()
155 
156  if self.useGpu:
157  program = 'xmipp_movie_optical_alignment_gpu'
158  args += '--gpu %s ' % gpuId
159  else:
160  program = 'xmipp_movie_optical_alignment_cpu'
161 
162  # We should save the movie either if the user selected it (default)
163  # or if the PSD is going to be computed
164  if self.doSaveAveMic or self.doComputePSD:
165  args += '--oavg "%s" ' % outMicFn
166 
167  if self.doComputePSD:
168  args += '--oavgInitial %s ' % aveMic
169 
170  if doSaveMovie:
171  args += '--outMovie %s ' % outMovieFn
172 
173  roi = [self.cropOffsetX.get(), self.cropOffsetY.get(),
174  self.cropDimX.get(), self.cropDimY.get()]
175 
176  args += '--cropULCorner %d %d ' % (roi[0], roi[1])
177  args += '--cropDRCorner %d %d ' % (roi[0] + roi[2] -1,
178  roi[1] + roi[3] -1)
179 
180  if self.memory:
181  args += ' --inmemory'
182 
183  toDelete=[]
184  if self.doApplyDoseFilter:
185  pxSize = movie.getSamplingRate()
186  vol = movie.getAcquisition().getVoltage()
187  preExp, dose = self._getCorrectedDose(inputMovies)
188  args += ' --doseCorrection %f %f %f %f' %(dose, pxSize, vol, preExp)
189 
190  if self.applyDosePreAlign:
191  args += ' pre'
192  else:
193  args += ' post'
194 
195  if self.doSaveUnweightedMic:
196  outUnwtMicFn = self._getExtraPath(self._getOutputMicName(movie))
197  outUnwtMovieFn = self._getExtraPath(self._getOutputMovieUnWtName(movie))
198  args += ' --oUnc %s %s' % (outUnwtMicFn, outUnwtMovieFn)
199  toDelete.append(outUnwtMovieFn)
200 
201  try:
202  self.runJob(program, args)
203 
204  if self.doSaveAveMic:
205  if not exists(outMicFn):
206  raise Exception("Micrograph %s not produced after "

208

211  raise Exception("Movie %s not produced after running %s "

213 
214  if self.doComputePSD:

216  outputFnCorrected=outMicFn+'psd.png')
217  # If the micrograph was only saved for computing the PSD
