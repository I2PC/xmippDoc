34 void MissingWedge::removeWedge(MultidimArray<double> &V) const
35 {
36  Matrix2D<double> Epos, Eneg;
37  Euler_angles2matrix(rotPos,tiltPos,0.,Epos);
38  Euler_angles2matrix(rotNeg,tiltNeg,0.,Eneg);
39 
40  Matrix1D<double> freq(3), freqPos, freqNeg;
41  Matrix1D<int> idx(3);
42 
43  FourierTransformer transformer;
44  MultidimArray< std::complex<double> > Vfft;
45  transformer.FourierTransform(V,Vfft,false);
46 
47  FOR_ALL_ELEMENTS_IN_ARRAY3D(Vfft)
48  {
49  // Frequency in the coordinate system of the volume
50  VECTOR_R3(idx,j,i,k);
51  FFT_idx2digfreq(V,idx,freq);
52 
53  // Frequency in the coordinate system of the plane
54  freqPos=Epos*freq;
55  freqNeg=Eneg*freq;
56  if (ZZ(freqPos)<0 || ZZ(freqNeg)>0)
57  Vfft(k,i,j)=0;
58  }
59  transformer.inverseFourierTransform();
60 }

63 Steerable::Steerable(double sigma, MultidimArray<double> &Vtomograph,
64  double deltaAng, const std::string &filterType, const MissingWedge *_MW)
65 {
66  MW=_MW;
67  buildBasis(Vtomograph,sigma);
68 
69  // Choose a,b,c parameters as a function of the filterType
70  double a;
71  double b;
72  double c;
73  if (filterType == "wall") {
74  // for wall structures
75  a = -(1.0/4.0);
76  b = 5.0/4.0;
77  c = 5.0/2.0;
78  }
79  else{
80  // for filament structures
81  a = 1.0;
82  b = -(5.0/3.0);
83  c = 10.0/3.0;
84  }
85 
86  // Filter along tilt=0 and 180 => u=(1,0,0)
87  double u0=1;
88  double u1=0;
89  double u2=0;
90  FOR_ALL_ELEMENTS_IN_ARRAY3D(Vtomograph){
91  Vtomograph(k,i,j) = basis[0](k,i,j) * (a+b*u0*u0) +
92  basis[1](k,i,j) * (a+b*u1*u1) +
93  basis[2](k,i,j) * (a+b*u2*u2) +
94  c*(basis[3](k,i,j) * u0*u1 +
95  basis[4](k,i,j) * u0*u2 +
96  basis[5](k,i,j) * u1*u2);
97  }
98 
99  // Filter the rest of directions and keep the maximum
100  double Ntilt = round(180.0/deltaAng);
101  for (int i=1;i<Ntilt;i++){
102  double tilt = deltaAng*i;
103  double deltaRoti = deltaAng/SIND(tilt);
104  double NrotP = round(360.0/deltaRoti);
105  for (int j=0;j<NrotP;j++){
106  double rot = j*deltaRoti;
107  double u0 = SIND(rot)*COSD(tilt);
108  double u1 = SIND(rot)*SIND(tilt);
109  double u2 = COSD(rot);
110  FOR_ALL_ELEMENTS_IN_ARRAY3D(Vtomograph)
111  {
112  double filterval =
113  basis[0](k,i,j) * (a+b*u0*u0) +
114  basis[1](k,i,j) * (a+b*u1*u1) +
115  basis[2](k,i,j) * (a+b*u2*u2) +
116  c*(basis[3](k,i,j) * u0*u1 +
117  basis[4](k,i,j) * u0*u2 +
118  basis[5](k,i,j) * u1*u2);
119 
120  if(filterval>Vtomograph(k,i,j))
121  Vtomograph(k,i,j) = filterval;
122  }
123  }
124  }
125 }

140 void Steerable::singleFilter(const MultidimArray<double>& Vin,
141  MultidimArray<double> &hx1, MultidimArray<double> &hy1, MultidimArray<double> &hz1,
142  MultidimArray<double> &Vout){
143 
144  MultidimArray< std::complex<double> > H, Aux;
145  Vout.initZeros(Vin);
146 
147  // Filter in X
148  #define MINUS_ONE_POWER(n) (((n)%2==0)? 1:-1)
149  FourierTransformer transformer;
150  transformer.FourierTransform(hx1,H);
151 
152  FOR_ALL_ELEMENTS_IN_ARRAY1D(H)
153  H(i)*= MINUS_ONE_POWER(i);
154 
155  FourierTransformer transformer2;
156 
157  MultidimArray<double> aux(XSIZE(Vin));
158 
159  transformer2.setReal(aux);
160 
161  for (size_t k=0; k<ZSIZE(Vin); k++)
162  for (size_t i=0; i<YSIZE(Vin); i++)
163  {
164  for (size_t j=0; j<XSIZE(Vin); j++)
165  DIRECT_A1D_ELEM(aux,j)=DIRECT_A3D_ELEM(Vin,k,i,j);
166 
167  transformer2.FourierTransform( );
168  transformer2.getFourierAlias( Aux );
169  Aux*=H;
170  transformer2.inverseFourierTransform( );
171 
172  for (size_t j=0; j<XSIZE(Vin); j++)
173  DIRECT_A3D_ELEM(Vout,k,i,j)=XSIZE(aux)*DIRECT_A1D_ELEM(aux,j);
174  }
175 
176  // Filter in Y
177  transformer.FourierTransform(hy1,H);
178 
179  FOR_ALL_ELEMENTS_IN_ARRAY1D(H)
180  H(i)*= MINUS_ONE_POWER(i);
181 
182  aux.initZeros(YSIZE(Vin));
183  transformer2.setReal(aux);
184 
185  for (size_t k=0; k<ZSIZE(Vin); k++)
186  for (size_t j=0; j<XSIZE(Vin); j++)
187  {
188  for (size_t i=0; i<YSIZE(Vin); i++)
189  DIRECT_A1D_ELEM(aux,i)=DIRECT_A3D_ELEM(Vout,k,i,j);
190 
191  transformer2.FourierTransform( );

195 
196  for (size_t i=0; i<YSIZE(Vin); i++)

200  // Filter in Z
201 
202  transformer.FourierTransform(hz1,H);
203

206

208  transformer2.setReal(aux);
209

219  transformer2.inverseFourierTransform( );

222  DIRECT_A3D_ELEM(Vout,k,i,j)=XSIZE(aux)*DIRECT_A1D_ELEM(aux,k);
223  }
224 
225  // If Missing wedge
226  if (MW!=NULL)

230 /* Filter generation ------------------------------------------------------- */
231 void Steerable::generate1DFilters(double sigma,
232  const MultidimArray<double> &Vtomograph,
233  std::vector< MultidimArray<double> > &hx1,
234  std::vector< MultidimArray<double> > &hy1,
235  std::vector< MultidimArray<double> > &hz1){
236 
237  // Initialization
238  MultidimArray<double> aux;
239  aux.initZeros(XSIZE(Vtomograph));
240  aux.setXmippOrigin();
241  for (int i=0; i<6; i++) hx1.push_back(aux);
242 
243  aux.initZeros(YSIZE(Vtomograph));
244  aux.setXmippOrigin();
245  for (int i=0; i<6; i++) hy1.push_back(aux);
246 
247  aux.initZeros(ZSIZE(Vtomograph));
248  aux.setXmippOrigin();
249  for (int i=0; i<6; i++) hz1.push_back(aux);
250 
251  double sigma2=sigma*sigma;
252  double k1 = 1.0/pow((2.0*PI*sigma),(3.0/2.0));
253  double k2 = -1.0/(sigma2);
254 
255  FOR_ALL_ELEMENTS_IN_ARRAY1D(hx1[0])
256  {
257  double i2=i*i;
258  double g = -exp(-i2/(2.0*sigma2));
259  hx1[0](i) = k1*k2*g*(1.0-(i2/sigma2));
260  hx1[1](i) = k1*k2*g;
261  hx1[2](i) = k1*k2*g;
262  hx1[3](i) = k1*k2*k2*g*i;
263  hx1[4](i) = k1*k2*k2*g*i;
264  hx1[5](i) = k1*k2*k2*g;
265  }
266  FOR_ALL_ELEMENTS_IN_ARRAY1D(hy1[0])
267  {
268  double i2=i*i;
269  double g = -exp(-i2/(2.0*sigma2));
270  hy1[0](i) = g;
271  hy1[1](i) = g*(1.0-(i2/sigma2));
272  hy1[2](i) = g;
273  hy1[3](i) = g*i;
274  hy1[4](i) = g;
275  hy1[5](i) = g*i;
276  }
277  FOR_ALL_ELEMENTS_IN_ARRAY1D(hz1[0])
278  {
279  double i2=i*i;
280  double g = -exp(-i2/(2.0*sigma2));
281  hz1[0](i) = g;
282  hz1[1](i) = g;
283  hz1[2](i) = g*(1.0-(i2/sigma2));
284  hz1[3](i) = g;
285  hz1[4](i) = g*i;
286  hz1[5](i) = g*i;
287  }
288 }
