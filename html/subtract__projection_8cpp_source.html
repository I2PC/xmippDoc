104  void ProgSubtractProjection::POCSFourierAmplitudeProj(const MultidimArray<double> &A, MultidimArray< std::complex<double> > &FI,
105  double lambdapocs, const MultidimArray<double> &rQ, int Isize) const{
106  int Isize2 = Isize/2;
107  double Isizei = 1.0/Isize;
108  double wx;
109  double wy;
110  for (int i=0; i<YSIZE(A); ++i) {
111  FFT_IDX2DIGFREQ_FAST(i,Isize,Isize2,Isizei,wy)
112  double wy2 = wy*wy;
113  for (int j=0; j<XSIZE(A); ++j) {
114  FFT_IDX2DIGFREQ_FAST(j,Isize,Isize2,Isizei,wx)
115  double w = sqrt(wx*wx + wy2);
116  auto iw = (int)round(w*Isize);
117  double mod = std::abs(DIRECT_A2D_ELEM(FI,i,j));
118  if (mod>1e-6)
119  DIRECT_A2D_ELEM(FI,i,j)*=(((1-lambdapocs)+lambdapocs*DIRECT_A2D_ELEM(A,i,j))/mod)*DIRECT_MULTIDIM_ELEM(rQ,iw);
120  }
121  }
122  }

203  MultidimArray<double> computeRadialAvg(const Image<double> &img, MultidimArray<double> &radial_mean){
204  Image<double> imgRad;
205  imgRad = img;
206  FourierTransformer transformerRad;
207  MultidimArray< std::complex<double> > IFourierRad;
208  MultidimArray<double> IFourierMagRad;
209  transformerRad.completeFourierTransform(imgRad(),IFourierRad);
210  CenterFFT(IFourierRad, true);
211  FFT_magnitude(IFourierRad,IFourierMagRad);
212  // Compute radial average profile (1D)
213  IFourierMagRad.setXmippOrigin();
214  Matrix1D<int> center(2);
215  center.initZeros();
216  MultidimArray<int> radial_count;
217  radialAverage(IFourierMagRad, center, radial_mean, radial_count);
218  int my_rad;
219  FOR_ALL_ELEMENTS_IN_ARRAY3D(IFourierMagRad) {
220  my_rad = (int)floor(sqrt((double)(i * i + j * j + k * k)));
221  imgRad(k, i, j) = radial_mean(my_rad);
222  }
223  return radial_mean;
224  }

291  void ProgSubtractProjection::run() {
292  show();
293  V.read(fnVolR);
294  V().setXmippOrigin();
295  mdParticles.read(fnParticles);
296  maskVol = createMask(fnMaskVol, maskVol);
297  mask = createMask(fnMask, mask);
298  // Initialize Gaussian LPF to smooth mask
299  FilterG.FilterShape=REALGAUSSIAN;
300  FilterG.FilterBand=LOWPASS;
301  FilterG.w1=sigma;
302  // Initialize Gaussian LPF to threshold mask with ctf
303  FilterG2.FilterShape=REALGAUSSIAN;
304  FilterG2.FilterBand=LOWPASS;
305  FilterG2.w1=2;
306  // Initialize LPF to filter at desired resolution
307  Filter2.FilterBand=LOWPASS;
308  Filter2.FilterShape=RAISED_COSINE;
309  Filter2.raised_w=0.02;
310  Filter2.w1=cutFreq;
311  for (size_t i = 1; i <= mdParticles.size(); ++i) {
312  row = mdParticles.getRowVec(i);
313  readParticle(row);
314  struct Angles angles;
315  row.getValueOrDefault(MDL_ANGLE_ROT, angles.rot, 0);
316  row.getValueOrDefault(MDL_ANGLE_TILT, angles.tilt, 0);
317  row.getValueOrDefault(MDL_ANGLE_PSI, angles.psi, 0);
318  roffset.initZeros(2);
319  row.getValueOrDefault(MDL_SHIFT_X, roffset(0), 0);
320  row.getValueOrDefault(MDL_SHIFT_Y, roffset(1), 0);
321  roffset *= -1;
322  projectVolume(V(), P, (int)XSIZE(I()), (int)XSIZE(I()), angles.rot, angles.tilt, angles.psi, &roffset);
323  projectVolume(maskVol(), PmaskVol, (int)XSIZE(I()), (int)XSIZE(I()), angles.rot, angles.tilt, angles.psi, &roffset);
324  PmaskVolI = binarizeMask(PmaskVol);
325  FilterG.applyMaskSpace(PmaskVolI());
326  POCSmaskProj(PmaskVolI(), P());
327  POCSmaskProj(PmaskVolI(), I());
328  row.getValueOrDefault(MDL_IMAGE, fnPart, "no_filename");
329  Pctf = applyCTF(row, P);
330  struct Radial radial;
331  radial.meanI = computeRadialAvg(I, radial.meanI);
332  radial.meanP = computeRadialAvg(Pctf, radial.meanP);
333  radQuotient = computeRadQuotient(radQuotient, radial.meanI, radial.meanP);
334  percentileMinMax(I(), Imin, Imax);
335  transformer.FourierTransform(I(),IFourier,false);
336  FFT_magnitude(IFourier,IFourierMag);
337  transformer.FourierTransform(Pctf(),PFourierPhase,true);
338  extractPhaseProj(PFourierPhase);
339  for (int n=0; n<iter; ++n) {
340  runIteration();
341  if (cutFreq!=0) {
342  Filter2.generateMask(Pctf());
343  Filter2.do_generate_3dmask=true;
344  Filter2.applyMaskSpace(Pctf());
345  }
346  }
347  Image<double> IFiltered;
348  I.read(fnImage);
349  IFiltered() = I();
350  if (cutFreq!=0)
351  Filter2.applyMaskSpace(IFiltered());
352  projectVolume(mask(), Pmask, (int)XSIZE(I()), (int)XSIZE(I()), angles.rot, angles.tilt, angles.psi, &roffset);
353  Pmaskctf = applyCTF(row, Pmask);
354  Pmaskctf = thresholdMask(Pmaskctf);
355  PmaskInv = invertMask(Pmaskctf);
356  FilterG.w1=sigma;
357  FilterG.applyMaskSpace(Pmaskctf());
358  I = subtraction(I, P, PmaskInv, Pmaskctf);
359  writeParticle(int(i), I);
360  }
361  mdParticles.write(fnParticles);
362  }
