221  double ProgSubtractProjection::evaluateFitting(const MultidimArray< std::complex<double> > &y,
222  const MultidimArray< std::complex<double> > &yp) const{
223  double sumY = 0;
224  double sumY2 = 0;
225  double sumE2 = 0;
226  FOR_ALL_DIRECT_ELEMENTS_IN_MULTIDIMARRAY(y) {
227  double realyn = real(DIRECT_MULTIDIM_ELEM(y, n));
228  double imagyn = imag(DIRECT_MULTIDIM_ELEM(y, n));
229  double ereal = realyn - real(DIRECT_MULTIDIM_ELEM(yp, n));
230  double eimag = imagyn - imag(DIRECT_MULTIDIM_ELEM(yp, n));
231  sumE2 += ereal*ereal + eimag*eimag;
232  sumY += realyn + imagyn;
233  sumY2 += realyn*realyn + imagyn*imagyn;
234  }
235  auto meanY = sumY / (2.0*(double)MULTIDIM_SIZE(y));
236  auto varY = sumY2 / (2.0*(double)MULTIDIM_SIZE(y)) - meanY*meanY;
237  auto R2 = 1.0 - (sumE2/(2.0*(double)MULTIDIM_SIZE(y))) / varY;
238  return R2;
239  }

241 Matrix1D<double> ProgSubtractProjection::checkBestModel(MultidimArray< std::complex<double> > &PFourierf, const MultidimArray< std::complex<double> > &PFourierf0,
242  const MultidimArray< std::complex<double> > &PFourierf1, const MultidimArray< std::complex<double> > &IFourierf) const {
243  // Compute R2 coefficient for order 0 model (R20) and order 1 model (R21)
244  auto N = 2.0*(double)MULTIDIM_SIZE(PFourierf);
245  double R20adj = evaluateFitting(IFourierf, PFourierf0); // adjusted R2 for an order 0 model = R2
246  double R21 = evaluateFitting(IFourierf, PFourierf1);
247  double R21adj = 1.0 - (1.0 - R21) * (N - 1.0) / (N - 2.0); // adjusted R2 for an order 1 model -> p = 2
248  //Decide best fitting
249  Matrix1D<double> R2(2);
250  if (R21adj > R20adj) { // Order 1: T(w) = b01 + b1*wi
251  PFourierf = PFourierf1;
252  R2(0) = R21adj;
253  R2(1) = 1;
254  }
255  else { // Order 0: T(w) = b00
256  PFourierf = PFourierf0;
257  R2(0) = R20adj;
258  R2(1) = 0;
259  }
260  return R2;
261 }

263  void ProgSubtractProjection::run() {
264  // Read input volume, mask and particles metadata
265  show();
266  V.read(fnVolR);
267  V().setXmippOrigin();
268 
269  // Create 2D circular mask to avoid edge artifacts after wrapping
270  size_t Xdim;
271  size_t Ydim;
272  size_t Zdim;
273  size_t Ndim;
274  V.getDimensions(Xdim, Ydim, Zdim, Ndim);
275  cirmask().initZeros((int)Ydim, (int)Xdim);
276  cirmask().setXmippOrigin();
277  if (cirmaskrad == -1.0)
278  cirmaskrad = (double)XSIZE(V())/2;
279  RaisedCosineMask(cirmask(), cirmaskrad*0.8, cirmaskrad*0.9);
280  cirmask.write(formatString("%s/cirmask.mrc", fnProj.c_str()));
281 
282  // Read or create mask keep and compute inverse of mask keep (mask subtract)
283  createMask(fnMask, vM, ivM);
284  FOR_ALL_DIRECT_ELEMENTS_IN_MULTIDIMARRAY(V())
285  DIRECT_MULTIDIM_ELEM(V(),n) = DIRECT_MULTIDIM_ELEM(V(),n)*DIRECT_MULTIDIM_ELEM(ivM(),n);
286  mdParticles.read(fnParticles);
287 
288  // Initialize Gaussian LPF to smooth mask
289  FilterG.FilterShape=REALGAUSSIAN;
290  FilterG.FilterBand=LOWPASS;
291  FilterG.w1=sigma;
292 
293  // Initialize Fourier projectors
294  double cutFreq = sampling/maxResol;
295  projector = std::make_unique<FourierProjector>(V(), padFourier, cutFreq, xmipp_transformation::BSPLINE3);
296  std::unique_ptr<FourierProjector> projectorMask;
297  projectorMask = std::make_unique<FourierProjector>(vM(), padFourier, cutFreq, xmipp_transformation::BSPLINE3);
298 
299  // Read first particle
300  const auto sizeI = (int)XSIZE(I());
301  processParticle(1, sizeI, transformerP, transformerI);
302  const MultidimArray<double> &mPctf = Pctf();
303 
304  // Construct frequencies image
305  MultidimArray<int> wi;
306  wi.initZeros(PFourier);
307  Matrix1D<double> w(2);
308  for (int i=0; i<YSIZE(wi); i++) {
309  FFT_IDX2DIGFREQ(i,YSIZE(mPctf),YY(w))
310  for (int j=0; j<XSIZE(wi); j++) {
311  FFT_IDX2DIGFREQ(j,XSIZE(mPctf),XX(w))
312  DIRECT_A2D_ELEM(wi,i,j) = (int)round((sqrt(YY(w)*YY(w) + XX(w)*XX(w))) * (int)XSIZE(mPctf)); // indexes
313  }
314  }
315  int maxwiIdx;
316  if (limitfreq == 0)
317  maxwiIdx = (int)XSIZE(wi);
318  else
319  DIGFREQ2FFT_IDX(cutFreq, (int)YSIZE(PFourier), maxwiIdx)
320 
321  // Declare complex structures that will be used in the loop
322  FourierTransformer transformerIiM;
323  FourierTransformer transformerPiM;
324  MultidimArray< std::complex<double> > IiMFourier;
325  MultidimArray< std::complex<double> > PiMFourier;
326 
327  // For each particle in metadata:
328  for (size_t i = 1; i <= mdParticles.size(); ++i) {
329  // Initialize aux variable
330  disable = false;
331  // Project volume and process projections
332  processParticle(i, sizeI, transformerP, transformerI);
333  // Build projected and final masks
334  if (fnMask.isEmpty()) { // If there is no provided mask
335  M().initZeros(P());
336  // inverse mask (iM) is all 1s
337  iM = invertMask(M);
338  }
339  else { // If a mask has been provided
340  projectVolume(*projectorMask, Pmask, sizeI, sizeI, part_angles.rot, part_angles.tilt, part_angles.psi, ctfImage);
341  // Apply binarization, shift and gaussian filter to the projected mask
342  M = binarizeMask(Pmask);
343  selfTranslate(xmipp_transformation::LINEAR, M(), roffset, xmipp_transformation::DONT_WRAP);
344  FilterG.applyMaskSpace(M());
345  if (subtract) // If the mask contains the part to subtract: inverse the original mask
346  iM = M;
347  else // Compute inverse of original mask
348  iM = invertMask(M);
349  }
350 
351  // Compute estimation images: IiM = I*iM and PiM = P*iM
352  IiMFourier = computeEstimationImage(I(), iM(), transformerIiM);
353  PiMFourier = computeEstimationImage(Pctf(), iM(), transformerPiM);
354 
355  // Estimate transformation with model of order 0: T(w) = beta00 and model of order 1: T(w) = beta01 + beta1*w
356  MultidimArray<double> num0;
357  num0.initZeros(maxwiIdx+1);
358  MultidimArray<double> den0;
359  den0.initZeros(maxwiIdx+1);
360  Matrix2D<double> A1;
361  A1.initZeros(2,2);
362  Matrix1D<double> b1;
363  b1.initZeros(2);
364  FOR_ALL_DIRECT_ELEMENTS_IN_MULTIDIMARRAY(PiMFourier) {
365  int win = DIRECT_MULTIDIM_ELEM(wi, n);
366  if (win < maxwiIdx)
367  {
368  double realPiMFourier = real(DIRECT_MULTIDIM_ELEM(PiMFourier,n));
369  double imagPiMFourier = imag(DIRECT_MULTIDIM_ELEM(PiMFourier,n));
370  DIRECT_MULTIDIM_ELEM(num0,win) += real(DIRECT_MULTIDIM_ELEM(IiMFourier,n)) * realPiMFourier
371  + imag(DIRECT_MULTIDIM_ELEM(IiMFourier,n)) * imagPiMFourier;
372  DIRECT_MULTIDIM_ELEM(den0,win) += realPiMFourier*realPiMFourier + imagPiMFourier*imagPiMFourier;
373  A1(0,0) += realPiMFourier*realPiMFourier + imagPiMFourier*imagPiMFourier;
374  A1(0,1) += win*(realPiMFourier + imagPiMFourier);
375  A1(1,1) += 2*win;
376  b1(0) += real(DIRECT_MULTIDIM_ELEM(IiMFourier,n)) * realPiMFourier + imag(DIRECT_MULTIDIM_ELEM(IiMFourier,n)) * imagPiMFourier;
377  b1(1) += win*(real(DIRECT_MULTIDIM_ELEM(IiMFourier,n))+imag(DIRECT_MULTIDIM_ELEM(IiMFourier,n)));
378  }
379  }
380  A1(1,0) = A1(0,1);
381 
382  // Compute beta00 from order 0 model
383  double beta00 = num0.sum()/den0.sum();
384  if (nonNegative && beta00 < 0)
385  {
386  disable = true;
387  }
388 
389  // Apply adjustment order 0: PFourier0 = T(w) * PFourier = beta00 * PFourier
390  PFourier0 = PFourier;
391  FOR_ALL_DIRECT_ELEMENTS_IN_MULTIDIMARRAY(PFourier0)
392  DIRECT_MULTIDIM_ELEM(PFourier0,n) *= beta00;
393  PFourier0(0,0) = IiMFourier(0,0);
394 
395  // Compute beta01 and beta1 from order 1 model
396  PseudoInverseHelper h;
397  h.A = A1;
398  h.b = b1;
399  Matrix1D<double> betas1;
400  solveLinearSystem(h,betas1);
401  double beta01 = betas1(0);
402  double beta1 = betas1(1);
403 
404  // Apply adjustment order 1: PFourier1 = T(w) * PFourier = (beta01 + beta1*w) * PFourier
405  PFourier1 = PFourier;
406  FOR_ALL_DIRECT_ELEMENTS_IN_MULTIDIMARRAY(PFourier1)
407  DIRECT_MULTIDIM_ELEM(PFourier1,n) *= (beta01+beta1*DIRECT_MULTIDIM_ELEM(wi,n));
408  PFourier1(0,0) = IiMFourier(0,0);
409 
410  // Check best model
411  Matrix1D<double> R2adj = checkBestModel(PFourier, PFourier0, PFourier1, IFourier);
412  double beta0save;
413  double beta1save;
414  if (R2adj(1) == 0)
415  {
416  beta0save = beta00;
417  beta1save = 0;
418  }
419  else if (R2adj(1) == 1)
420  {
421  beta0save = beta01;
422  beta1save = beta1;
423  }
424 
425  // Create empty new image for output particle
426  MultidimArray<double> &mIdiff=Idiff();
427  mIdiff.initZeros(I());
428  mIdiff.setXmippOrigin();
429 
430  // Boosting of original particles
431  if (boost)
432  {
433  if (R2adj(1) == 0)
434  {
435  FOR_ALL_DIRECT_ELEMENTS_IN_MULTIDIMARRAY(IFourier)
436  DIRECT_MULTIDIM_ELEM(IFourier,n) /= beta00;
437  }
438  else if (R2adj(1) == 1)
439  {
440  FOR_ALL_DIRECT_ELEMENTS_IN_MULTIDIMARRAY(IFourier)
441  DIRECT_MULTIDIM_ELEM(IFourier,n) /= (beta01+beta1*DIRECT_MULTIDIM_ELEM(wi,n));
442  }
443  transformerI.inverseFourierTransform(IFourier, Idiff());
444  }
445 
446  // Subtraction
447  else
448  {
449  // Recover adjusted projection (P) in real space
450  transformerP.inverseFourierTransform(PFourier, P());
451  mIdiff.initZeros(I());
452  mIdiff.setXmippOrigin();
453  FOR_ALL_DIRECT_ELEMENTS_IN_MULTIDIMARRAY(mIdiff)
454  DIRECT_MULTIDIM_ELEM(mIdiff,n) = DIRECT_MULTIDIM_ELEM(I(),n)-DIRECT_MULTIDIM_ELEM(P(),n);
455  }
456  // Write particle
457  writeParticle(int(i), Idiff, R2adj(0), beta0save, beta1save);
458  }
459  // Write metadata
460  mdParticles.write(formatString("%s.xmd", fnOut.c_str()));
461  }
