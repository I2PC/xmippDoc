234 def objectToRow(obj, row, attrDict, extraLabels=[]):
235  """ This function will convert an EMObject into a Row.
236  Params:
237  obj: the EMObject instance (input)
238  row: the Row instance (output) -see emlib.metadata.utils.Row()-
239  attrDict: dictionary with the map between obj attributes(keys) and
240  row MDLabels in Xmipp (values).
241  extraLabels: a list with extra labels that could be included
242  as _xmipp_labelName
243  """
244  if obj.isEnabled():
245  enabled = 1
246  else:
247  enabled = -1
248  row.setValue(emlib.MDL_ENABLED, enabled)
249 
250  for attr, label in attrDict.items():
251  if hasattr(obj, attr):
252  valueType = getLabelPythonType(label)
253  value = getattr(obj, attr).get()
254  try:
255  row.setValue(label, valueType(value))
256  except Exception as e:
257  print(e)
258  print("Problems found converting metadata: ")
259  print("Label id = %s" % label)
260  print("Attribute = %s" % attr)
261  print("Value = %s" % value)
262  print("Value type = %s" % valueType)
263  raise e
264  row.setValue(label, valueType(getattr(obj, attr).get()))
265 
266  attrLabels = attrDict.values()
267 
268  for label in extraLabels:
269  attrName = prefixAttribute(emlib.label2Str(label))
270  if label not in attrLabels and hasattr(obj, attrName):
271  value = obj.getAttributeValue(attrName)
272  row.setValue(label, value)
