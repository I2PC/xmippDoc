99  double correction = 1;

124

165

170  mdPartial.getColumnValues(MDL_WEIGHT,P);
171 
172  for (size_t idx=0; idx< P.size();idx++)
173  {
174  if (P[idx] > 1)

176  num_images += 1.;
177  }
178  validation /= (num_images);
