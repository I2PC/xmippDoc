50 void FuzzyCMeans::train(FuzzyCodeBook& _xmippDS, TS& _examples) const
51 {
52 
53  // Defines verbosity
54 
55  int verbosity = listener->getVerbosity();
56  if (verbosity)
57  listener->OnReportOperation((std::string) "Training....\n");
58  if (verbosity == 1 || verbosity == 3)
59  listener->OnInitOperation(epochs);
60 
61  // Create auxiliar Codebook
62 
63  FuzzyCodeBook auxCB;
64 
65 
66  // Create auxiliar stuff
67 
68  unsigned numClusters = _xmippDS.size();
69  unsigned numVectors = _examples.size();
70  unsigned i, j, k;
71  double stopError = 0, auxError = 0;
72  double auxDist, auxProd, tmp, auxExp, auxSum;
73  unsigned t = 0; // Iteration index
74  FeatureVector zero(_xmippDS.theItems[0].size()) ;
75  fill(zero.begin(), zero.end(), 0.0);
76 
77 
78  // Initialize auxiliary Codebook
79 
80  auxCB = _xmippDS;
81 
82  // Set auxExp
83 
84  auxExp = 2 / (m - 1);
85 
86  // This is the main code of the algorithm. Iterates "epochs" times
87 
88 
89  stopError = epsilon + 1; // Initially must be higher
90 
91  while ((stopError > epsilon) && (t < epochs))
92  {
93 
94  // Update Membership matrix
95 
96  for (k = 0; k < numVectors; k++)
97  {
98 
99  auxProd = 1;
100  for (j = 0; j < numClusters; j++)
101  auxProd *= euclideanDistance(_xmippDS.theItems[j], _examples.theItems[k]);
102 
103  if (auxProd == 0.)
104  { // Apply k-means criterion (Data-CB) must be > 0
105  for (j = 0; j < numClusters; j ++)
106  if (euclideanDistance(_xmippDS.theItems[j], _examples.theItems[k]) == 0.)
107  _xmippDS.memb[k][j] = 1.0;
108  else
109  _xmippDS.memb[k][j] = 0.0;
110  }
111  else
112  {
113  for (i = 0; i < numClusters; i ++)
114  {
115  auxDist = 0;
116  for (j = 0; j < numClusters; j ++)
117  {
118  tmp = euclideanDistance(_xmippDS.theItems[i], _examples.theItems[k]) /
119  euclideanDistance(_xmippDS.theItems[j], _examples.theItems[k]);
120  auxDist += pow(tmp, auxExp);
121  } // for j
122  _xmippDS.memb[k][i] = (floatFeature) 1.0 / auxDist;
123  } // for i
124  } // if auxProd
125  } // for k
126 
127 
128  // Update code vectors (Cluster Centers)
129 
130  for (i = 0; i < numClusters; i++)
131  {
132  _xmippDS.theItems[i] = zero;
133  auxSum = 0;
134  for (k = 0; k < numVectors; k++)
135  {
136  _xmippDS.theItems[i] += (floatFeature) pow((double)(_xmippDS.memb[k][i]), m) * _examples.theItems[k];
137  auxSum += pow((double)(_xmippDS.memb[k][i]), m);
138  } // for i
139  _xmippDS.theItems[i] /= (floatFeature) auxSum;
140  } // for k
141 
142  // Compute stopping criterion
143  stopError = 0;
144  for (i = 0; i < numClusters; i ++)
145  {
146  auxError = euclideanDistance(_xmippDS.theItems[i], auxCB.theItems[i]);
147  stopError += auxError * auxError;
148  } // for i
149 
150 
151  // Update iteration index
152 
153  t++;
154  auxCB = _xmippDS;
155 
156  if (verbosity == 1 || verbosity == 3)
157  listener->OnProgress(t);
158  if (verbosity >= 2)
159  {
160  char s[100];
161  sprintf(s, "Iteration %d of %d. Code vectors variation: %g\n", t + 1, epochs, stopError);
162  listener->OnReportOperation((std::string) s);
163  }
164 
165  } // while
166 
167  if (verbosity == 1 || verbosity == 3)
168  listener->OnProgress(epochs);
169

330 double FuzzyCMeans::S(const FuzzyCodeBook& _xmippDS,
331  const TS& _examples) const
332 {
333 
334  std::vector< std::vector< floatFeature > > ICD; // Intercluster distance
335  std::vector< std::vector< floatFeature > > D; // Distance from each data to cluster centers
336 
337  unsigned i;
338  D.resize(_xmippDS.membClusters());
339  for (i = 0; i < _xmippDS.membClusters(); i++)
340  {
341  std::vector <floatFeature> d;
342  d.resize(_xmippDS.membVectors());
343  for (unsigned k = 0; k < _xmippDS.membVectors(); k++)
344  d[k] = (floatFeature)euclideanDistance(_xmippDS.theItems[i], _examples.theItems[k]);
345  D[i] = d;
346  } // for i
347 
348  ICD.resize(_xmippDS.membClusters());
349  for (i = 0; i < _xmippDS.membClusters(); i++)
350  {
351  std::vector <floatFeature> v;
352  v.resize(_xmippDS.membVectors());
353  for (unsigned j = 0; j < _xmippDS.membClusters(); j++)
354  v[j] = (floatFeature)euclideanDistance(_xmippDS.theItems[i], _xmippDS.theItems[j]);
355  ICD[i] = v;
356  } // for i
357 
358  floatFeature auxSum = 0;
359  for (i = 0; i < _xmippDS.membClusters(); i++)
360  for (unsigned k = 0; k < _xmippDS.membVectors(); k++)
361  auxSum += (floatFeature) pow((double)(D[i][k] * _xmippDS.memb[k][i]), (double)m);
362 
363  floatFeature auxMin = MAXFLOAT;
364  for (i = 0; i < _xmippDS.membClusters(); i++)
365  for (unsigned j = i + 1; j < _xmippDS.membClusters(); j++)
366  if (auxMin > ICD[i][j])
367  auxMin = ICD[i][j];
368 
369  double S = auxSum / (double)(_xmippDS.membVectors()) / (double)(auxMin);
370  return S;
371 
372 }
