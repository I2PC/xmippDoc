83 
84 
85  /******************* Assign appropriate value to size flag ******************/
86 
87  if (format == FOURIER)
88  col += 2; /* Special treatment for FFT format (see foutrans.c) */
89 
90  switch (format)
91  {
92  case NATURAL:
93  element_size = sizeof(BYTE);
94  pointer_size = sizeof(BYTE *);
95  break;
96  case INTFMT:
97  element_size = sizeof(UWORD);
98  pointer_size = sizeof(UWORD *);
99  break;
100  case LONGFMT:
101  element_size = sizeof(ULONG);
102  pointer_size = sizeof(ULONG *);
103  break;
104  case FLOATFMT:
105  case FOURIER:
106  element_size = sizeof(float);
107  pointer_size = sizeof(float *);
108  break;
109  default:
110  return NULL;
111  }
112 
113  row_alloc = ALLOC_SIZE / (col * element_size); /* No. of rows to alloc */
114  all_size = row_alloc * col * element_size;
115  tot_size = ((long) element_size) * ((long) row) * ((long) col);
116  no_blocks = tot_size / all_size;
117  rest_size = tot_size - ((long) no_blocks) * ((long) all_size);
118  rest_rows = rest_size / (col * element_size);
119 
120  /********************* Allocate base pointer ********************************/
121 
122  if ((temp = (char **) malloc(row * pointer_size)) == NULL)
123  return NULL; /* Not even this little bit of memory */
124 
125  /*********************** Allocate most blocks *******************************/
126 
127  j = 0;
128  for (i = 0; i < no_blocks; i++)
129  {
130  if ((aux = (char *)malloc((long)all_size)) == NULL)
131  {
132  for (j = 0; j < i; j++)
133  free(temp[j*row_alloc]);
134  free((char *) temp);
135  return NULL;
136  }
137  for (k = 0; k < row_alloc; k++, j++)
138  temp [j] = aux + k * col * element_size;
139  }
140 
141  /*************************** Alloc the last block ***************************/
142 
143  if (rest_size != 0)
144  {
145  if ((aux = (char *)malloc((long)rest_size)) == NULL)
146  {
147  for (j = 0; j < no_blocks; j++)
148  free(temp[j*row_alloc]);
149  free(temp);
150  return NULL;
151  }
152  for (k = 0; k < rest_rows; k++, j++)
153  temp [j] = aux + k * col * element_size;
154  }
155 
156  /************************* return OK pointer value **********************/
157 
158  return (void **)temp;

221 float conv1x(double y, double x)
222 /**************************************************************************/
223 /* Returns the value of the image at the point (y, x) using linear interp-*/
224 /* olation. For higher accuracy, use "conv3x" (available in 370 assembly */
225 /* languaje. */
226 /**************************************************************************/
227 { short i, j; /* Row and column */
228  float intfila1, intfila2; /* Partial row interpolations */
229  float escala; /* Scale factor */
230 
231  j = (short int) y; /* Trunc the x, y coordinates to short */
232  i = (short int) x;
233 
234  escala = y - j;
235  /* 1st row interpolation */
236  intfila1 = imagen[i][j] + escala * ((short)imagen[i][j+1] - (short)imagen[i][j]);
237  /* 2nd row interpolation */
238  intfila2 = imagen[i+1][j] + escala * ((short)imagen[i+1][j+1] - (short)imagen [i+1][j]);
239  /* Column interpolation */
240  return intfila1 + (x - i)*(intfila2 - intfila1);
241 }
