131 void ProgMovieFilterDose::applyDoseFilterToImage(
132  int Ydim, int Xdim,
133  const MultidimArray<std::complex<double> > &FFT1,
134  const double dose_start, const double dose_finish) {
135  std::complex<double> complex_zero = 0;
136  double x, y;
137  double yy;
138  double current_critical_dose, current_optimal_dose;
139  int sizeX_2 = Xdim / 2;
140  double ixsize = 1.0 / Xdim;
141  int sizeY_2 = Ydim / 2;
142  double iysize = 1.0 / Ydim;
143  for (long int i = 0; i < YSIZE(FFT1); i++) { //i->y,j->x xmipp convention is oposite summove
144  FFT_IDX2DIGFREQ_FAST(i, Ydim, sizeY_2, iysize, y);
145  yy = y * y;
146  for (long int j = 0; j < XSIZE(FFT1); j++) {
147  FFT_IDX2DIGFREQ_FAST(j, Xdim, sizeX_2, ixsize, x);
148 #ifdef DEBUG_FFT_FREQ
149  std::cerr << "(j+1,i+1)(x,y)=" << j +1 << "," << i + 1<< " "
150  << x << " " << y << " = "
151  << DIRECT_A2D_ELEM(FFT1, i, j) << std::endl;
152 #endif
153  if (i == 0 && j == 0)
154  //ROB: I do not understand this step
155  //It forces the origin to 0
156  //Why not keep the value?
157  current_critical_dose = critical_dose_at_dc;
158  else
159  current_critical_dose = criticalDose(
160  sqrt(x * x + yy) / pixel_size);
161  current_optimal_dose = optimalDoseGivenCriticalDose(
162  current_critical_dose);
163  if (abs(dose_finish - current_optimal_dose)
164  < abs(dose_start - current_optimal_dose)) {
165  DIRECT_A2D_ELEM(FFT1, i, j ) *= doseFilter(dose_finish,
166  current_critical_dose);
167  } else
168  DIRECT_A2D_ELEM(FFT1, i, j ) = complex_zero;
169 
170  }
171  }
