184 void CudaShiftCorrEstimator<T>::computeCorrelations2DOneToN(
185  std::complex<T> *h_inOut, bool center) {
186  bool isReady = (this->m_isInit && (AlignType::OneToN == this->m_type) && m_is_d_single_FD_loaded);
187 
188  if ( ! isReady) {
189  REPORT_ERROR(ERR_LOGIC_ERROR, "Not ready to execute. Call init() before");
190  }
191 
192  m_loadStream->set();
193  m_workStream->set();
194  auto loadStream = *(cudaStream_t*)m_loadStream->stream();
195  auto workStream = *(cudaStream_t*)m_workStream->stream();
196  // process signals in batches
197  for (size_t offset = 0; offset < this->m_settingsInv->fDim().n(); offset += this->m_settingsInv->batch()) {
198  // how many signals to process
199  size_t toProcess = std::min(this->m_settingsInv->batch(), this->m_settingsInv->fDim().n() - offset);
200 
201  // copy memory
202  gpuErrchk(cudaMemcpyAsync(
203  m_d_batch_FD,
204  h_inOut + offset * this->m_settingsInv->fDim().xy(),
205  toProcess * this->m_settingsInv->fBytesSingle(),
206  cudaMemcpyHostToDevice, loadStream));
207  m_loadStream->synch();
208 
209  auto dims = Dimensions(
210  this->m_settingsInv->fDim().x(),
211  this->m_settingsInv->fDim().y(),
212  1,
213  toProcess);
214  if (center) {
215  sComputeCorrelations2DOneToN<true>(*m_workStream, m_d_batch_FD, m_d_single_FD, dims);
216  } else {
217  sComputeCorrelations2DOneToN<false>(*m_workStream, m_d_batch_FD, m_d_single_FD, dims);
218  }
219  m_workStream->synch();
220 
221  // copy data back
222  gpuErrchk(cudaMemcpyAsync(
223  h_inOut + offset * this->m_settingsInv->fDim().xy(),
224  m_d_batch_FD,
225  toProcess * this->m_settingsInv->fBytesSingle(),
226  cudaMemcpyDeviceToHost, loadStream));
227  }

230 template<typename T>
231 void CudaShiftCorrEstimator<T>::loadThreadRoutine(T *others) {
232  m_loadStream->set();
233  auto lStream = *(cudaStream_t*)m_loadStream->stream();
234  for (size_t offset = 0; offset < this->m_settingsInv->fDim().n(); offset += this->m_settingsInv->batch()) {
235  // how many signals to process
236  size_t toProcess = std::min(this->m_settingsInv->batch(), this->m_settingsInv->fDim().n() - offset);
237  std::unique_lock<std::mutex> lk(*m_mutex);
238  // wait till the data is processed
239  while (m_isDataReady) {
240  m_cv->wait(lk);
241  }
242  T *src = others + offset * this->m_settingsInv->sDim().xy();
243  size_t bytes = toProcess * this->m_settingsInv->sBytesSingle();
244 
245  auto kind = m_loadStream->isGpuPointer(src)
246  ? cudaMemcpyDeviceToDevice
247  : cudaMemcpyHostToDevice;
248  // copy memory
249  gpuErrchk(cudaMemcpyAsync(
250  m_d_batch_SD_load,
251  src,
252  bytes,
253  kind, lStream));
254  // block until data is loaded
255  m_loadStream->synch();
256 
257  // notify processing stream that it can work
258  m_isDataReady = true;
259  m_cv->notify_one();
260  }

264 void CudaShiftCorrEstimator<T>::waitAndConvert() {
265  // block until data is loaded
266  // mutex will be freed once leaving this block
267  std::unique_lock<std::mutex> lk(*m_mutex);
268  while(!m_isDataReady) {
269  m_cv->wait(lk);
270  }
271  // perform FT
272  CudaFFT<T>::fft(*m_batchToFD, m_d_batch_SD_load, m_d_batch_FD);
273

275  m_workStream->synch();
276  m_isDataReady = false;

280 template<typename T>
281 void CudaShiftCorrEstimator<T>::computeShift2DOneToN(
282  T *others) {
283  bool isReady = (this->m_isInit && (AlignType::OneToN == this->m_type)
284  && m_is_d_single_FD_loaded && this->m_includingBatchFT);
285 
286  if ( ! isReady) {
287  REPORT_ERROR(ERR_LOGIC_ERROR, "Not ready to execute. Call init() before");
288  }
289  if ( ! GPU::isMemoryPinned(others)) {
290  REPORT_ERROR(ERR_LOGIC_ERROR, "Input memory has to be pinned (page-locked)");
291  }
292  if (this->m_allowDataOverwrite && ( ! m_workStream->isGpuPointer(others))) {
293  REPORT_ERROR(ERR_LOGIC_ERROR, "Incompatible parameters: allowDataOverwrite && 'others' data on host");
294  }
295 
296  m_workStream->set();
297  // make sure that all work (e.g. loading and processing of the reference) is done
298  m_loadStream->synch();
299  m_workStream->synch();
300  // start loading data at the background
301  m_isDataReady = false;
302  const bool createLocalCopy = ( ! this->m_allowDataOverwrite);
303  std::thread loadingThread;
304  if (createLocalCopy) {
305  loadingThread = std::thread(&CudaShiftCorrEstimator<T>::loadThreadRoutine, this, others);
306  }
307 
308  // reserve enough space for shifts
309  this->m_shifts2D.reserve(this->m_settingsInv->fDim().n());
310  // process signals
311  for (size_t offset = 0; offset < this->m_settingsInv->fDim().n(); offset += this->m_settingsInv->batch()) {
312  size_t batchOffset = createLocalCopy
313  ? offset
314  : std::min(offset, this->m_settingsInv->fDim().n() - this->m_settingsInv->batch());
315  // how many signals to process
316  size_t toProcess = std::min(
317  this->m_settingsInv->batch(),
318  this->m_settingsInv->fDim().n() - batchOffset);
319 
320  if (createLocalCopy) {
321  waitAndConvert();
322  } else {
323  T *in = others + (batchOffset * this->m_settingsInv->sDim().sizeSingle());
324  CudaFFT<T>::fft(*m_batchToFD, in, m_d_batch_FD);
325  }
326 
327  // compute shifts
328  auto shifts = computeShifts2DOneToN(
329  std::vector<GPU*>{m_workStream, m_loadStream},
330  m_d_batch_FD,
331  m_d_batch_SD_work,
332  m_d_single_FD,
333  this->m_settingsInv->createSubset(toProcess),
334  *m_batchToSD,
335  this->m_h_centers,
336  this->m_maxShift);
337 
338  size_t start = createLocalCopy
339  ? 0
340  : (offset - batchOffset);
341  // append shifts to existing results
342  this->m_shifts2D.insert(this->m_shifts2D.end(),
343  shifts.begin() + start,
344  shifts.end());
345  }
346  if (createLocalCopy) {
347  loadingThread.join();
348  }
349  // update state
350  this->m_is_shift_computed = true;
351 }

353 template<typename T>
354 std::vector<Point2D<float>> CudaShiftCorrEstimator<T>::computeShifts2DOneToN(
355  const std::vector<GPU*> &gpus,
356  std::complex<T> *d_othersF,
357  T *d_othersS,
358  std::complex<T> *d_ref,
359  const FFTSettingsNew<T> &settings,
360  cufftHandle plan,
361  T *h_centers,
362  size_t maxShift) {
363  // we need even input in order to perform the shift (in FD, while correlating) properly
364  assert(0 == (settings.sDim().x() % 2));
365  assert(0 == (settings.sDim().y() % 2));
366  assert(1 == settings.sDim().zPadded());
367  assert(2 == gpus.size());
368 
369  size_t centerSize = maxShift * 2 + 1;
370 
371  // make sure we have enough memory for centers of the correlation functions
372  assert(settings.fBytesBatch() >= (settings.sDim().n() * centerSize * centerSize * sizeof(T)));
373 
374  auto workGPU = gpus.at(0);
375  auto loadGPU = gpus.at(1);
376  loadGPU->set();
377  workGPU->set();
378  auto loadStream = *(cudaStream_t*)loadGPU->stream();
379  auto workStream = *(cudaStream_t*)workGPU->stream();
380 
381  // correlate signals and shift FT so that it will be centered after IFT
382  sComputeCorrelations2DOneToN<true>(*workGPU,
383  d_othersF, d_ref,
384  settings.fDim());
385 
386  // perform IFT
387  CudaFFT<T>::ifft(plan, d_othersF, d_othersS);
388 
389  // locate maxima
390  auto p_pos = (float*)d_othersF;
391  auto h_pos = (float*)h_centers;
392  ExtremaFinder::CudaExtremaFinder<T>::sFindMax2DAroundCenter(
393  *workGPU, settings.sDim(), d_othersS, p_pos, nullptr, maxShift);
394  workGPU->synch();
395  // copy data back
396  gpuErrchk(cudaMemcpyAsync(h_pos, p_pos,
397  settings.batch() * sizeof(float),
398  cudaMemcpyDeviceToHost, loadStream));
399  loadGPU->synch();
400 
401  // convert absolute 1D index to logical 2D index
402  // FIXME DS extract this to some utils
403  auto result = std::vector<Point2D<float>>();
404  result.reserve(settings.sDim().n());
405  for (size_t n = 0; n < settings.batch(); ++n) {
406  float y = (size_t)h_pos[n] / settings.sDim().x();
407  float x = (size_t)h_pos[n] % settings.sDim().x();
408  result.emplace_back(
409  x - settings.sDim().x() / 2,
410  y - settings.sDim().y() / 2);
411  }
412  return result;
413 }

435 template<typename T>
436 template<bool center>
437 void CudaShiftCorrEstimator<T>::sComputeCorrelations2DOneToN(
438  const GPU &gpu,
439  std::complex<T> *d_inOut,
440  const std::complex<T> *d_ref,
441  const Dimensions &dims) {
442  if (center) {
443  // we cannot assert xDim, as we don't know if the spatial size was even
444  assert(0 == (dims.y() % 2));
445  }
446  assert(0 < dims.x());
447  assert(0 < dims.y());
448  assert(1 == dims.z());
449  assert(0 < dims.n());
450 
451  // create threads / blocks
452  dim3 dimBlock(64, 1, 1);
453  dim3 dimGrid(
454  std::ceil(dims.x() / (float)dimBlock.x),
455  dims.n(), 1);
456  auto stream = *(cudaStream_t*)gpu.stream();
457  if (std::is_same<T, float>::value) {
458  computeCorrelations2DOneToNKernel<float2, center>
459  <<<dimGrid, dimBlock, 0, stream>>> (
460  (float2*)d_inOut, (float2*)d_ref,
461  dims.x(), dims.y(), dims.n());
462  } else if (std::is_same<T, double>::value) {
463  computeCorrelations2DOneToNKernel<double2, center>
464  <<<dimGrid, dimBlock, 0, stream>>> (
465  (double2*)d_inOut, (double2*)d_ref,
466  dims.x(), dims.y(), dims.n());
467  } else {
468  REPORT_ERROR(ERR_TYPE_INCORRECT, "Not implemented");
469  }
