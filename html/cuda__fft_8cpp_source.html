31 template<typename T>
32 void CudaFFT<T>::init(const HW &gpu, const FFTSettingsNew<T> &settings, bool reuse) {
33  bool canReuse = m_isInit
34  && reuse
35  && (m_settings->sBytesBatch() >= settings.sBytesBatch())
36  && (m_settings->fBytesBatch() >= settings.fBytesBatch());
37  bool mustAllocate = !canReuse;
38  if (mustAllocate) {
39  release();
40  }
41  // previous plan and settings has to be released,
42  // otherwise we will get GPU/CPU memory leak
43  release(m_plan);
44  delete m_settings;
45 
46  m_settings = new FFTSettingsNew<T>(settings);
47  try {
48  m_gpu = &dynamic_cast<const GPU&>(gpu);
49  } catch (std::bad_cast&) {
50  REPORT_ERROR(ERR_ARG_INCORRECT, "Instance of GPU expected");
51  }
52 
53  check();
54 
55  m_plan = createPlan(*m_gpu, *m_settings);
56  if (mustAllocate) {
57  // allocate input data storage
58  gpuErrchk(cudaMalloc(&m_d_SD, m_settings->sBytesBatch()));
59  if (m_settings->isInPlace()) {
60  // input data holds also the output
61  m_d_FD = (std::complex<T>*)m_d_SD;
62  } else {
63  // allocate also the output buffer
64  gpuErrchk(cudaMalloc(&m_d_FD, m_settings->fBytesBatch()));
65  }
66  }
67 
68  m_isInit = true;
69 }

136 template<typename T>
137 std::complex<T>* CudaFFT<T>::fft(const T *h_in,
138  std::complex<T> *h_out) {
139  auto isReady = m_isInit && m_settings->isForward();
140  if ( ! isReady) {
141  REPORT_ERROR(ERR_LOGIC_ERROR, "Not ready to perform Fourier Transform. "
142  "Call init() function first");
143  }
144 
145  // process signals in batches
146  for (size_t offset = 0; offset < m_settings->sDim().n(); offset += m_settings->batch()) {
147  // how many signals to process
148  size_t toProcess = std::min(m_settings->batch(), m_settings->sDim().n() - offset);
149 
150  // copy memory
151  gpuErrchk(cudaMemcpyAsync(
152  m_d_SD,
153  h_in + offset * m_settings->sDim().xyzPadded(),
154  toProcess * m_settings->sBytesSingle(),
155  cudaMemcpyHostToDevice, *(cudaStream_t*)m_gpu->stream()));
156 
157  fft(*m_plan, m_d_SD, m_d_FD);
158 
159  // copy data back
160  gpuErrchk(cudaMemcpyAsync(
161  h_out + offset * m_settings->fDim().xyzPadded(),
162  m_d_FD,
163  toProcess * m_settings->fBytesSingle(),
164  cudaMemcpyDeviceToHost, *(cudaStream_t*)m_gpu->stream()));
165  }
166  return h_out;
167 }

242 template<typename T>
243 template<typename F>
244 void CudaFFT<T>::manyHelper(const FFTSettingsNew<T> &settings, F function) {
245  auto n = std::array<int, 3>{(int)settings.sDim().z(), (int)settings.sDim().y(), (int)settings.sDim().x()};
246  int idist;
247  int odist;
248  cufftType type;
249  if (settings.isForward()) {
250  idist = settings.sDim().xyzPadded();
251  odist = settings.fDim().xyzPadded();
252  type = std::is_same<T, float>::value ? CUFFT_R2C : CUFFT_D2Z;
253  } else {
254  idist = settings.fDim().xyzPadded();
255  odist = settings.sDim().xyzPadded();
256  type = std::is_same<T, float>::value ? CUFFT_C2R : CUFFT_Z2D;
257  }
258  int rank = 3;
259  if (settings.sDim().z() == 1) rank--;
260  if ((2 == rank) && (settings.sDim().y() == 1)) rank--;
261 
262  int offset = 3 - rank;
263  function(rank, &n[offset], nullptr,
264  1, idist, nullptr, 1, odist, type, settings.batch());
265 }

286 template<typename T>
287 FFTSettingsNew<T> CudaFFT<T>::findMaxBatch(const FFTSettingsNew<T> &settings,
288  size_t maxBytes) {
289  size_t singleBytes = settings.sBytesSingle() + (settings.isInPlace() ? 0 : settings.fBytesSingle());
290  size_t batch = min((maxBytes / singleBytes), settings.batch()) + 1; // + 1 will be deducted in the while loop
291  while (batch > 1) {
292  batch--;
293  auto tmp = FFTSettingsNew<T>(settings.sDim(), batch, settings.isInPlace(), settings.isForward());
294  size_t totalBytes = CudaFFT<T>().estimateTotalBytes(tmp);
295  if (totalBytes <= maxBytes) {
296  return tmp;
297  }
298  }
299  REPORT_ERROR(ERR_GPU_MEMORY, "Estimated batch size is 0(zero). "
300  "This probably means you don't have enough GPU memory for even a single transformation.");
301 }

303 template<typename T>
304 core::optional<FFTSettingsNew<T>> CudaFFT<T>::findOptimal(GPU &gpu,
305  const FFTSettingsNew<T> &settings,
306  size_t reserveBytes, bool squareOnly, int sigPercChange,
307  bool crop, bool verbose) {
308  using cuFFTAdvisor::Tristate::TRUE;
309  using cuFFTAdvisor::Tristate::FALSE;
310  using core::optional;
311  size_t freeBytes = gpu.lastFreeBytes();
312  std::vector<cuFFTAdvisor::BenchmarkResult const *> *options =
313  cuFFTAdvisor::Advisor::find(10, gpu.device(), // FIXME DS this should be configurable
314  settings.sDim().x(), settings.sDim().y(), settings.sDim().z(), settings.sDim().n(),
315  TRUE, // use batch
316  std::is_same<T, float>::value ? TRUE : FALSE,
317  settings.isForward() ? TRUE : FALSE,
318  settings.isInPlace() ? TRUE : FALSE,
319  cuFFTAdvisor::Tristate::TRUE, // is real
320  sigPercChange, memoryUtils::MB(freeBytes - reserveBytes),
321  false, // allow transposition
322  squareOnly, crop);
323 
324  auto result = optional<FFTSettingsNew<T>>();
325  if (0 != options->size()) {
326  auto res = options->at(0);
327  auto optSetting = FFTSettingsNew<T>(
328  res->transform->X,
329  res->transform->Y,
330  res->transform->Z,
331  settings.sDim().n(),
332  res->transform->N / res->transform->repetitions,
333  settings.isInPlace(),
334  settings.isForward());
335  result = optional<FFTSettingsNew<T>>(optSetting);
336  }
337  if (verbose) {
338  if (result.has_value()) {
339  options->at(0)->printHeader(stdout); printf("\n");
340  options->at(0)->print(stdout); printf("\n");
341  } else {
342  std::cout << "No result obtained. Maybe too strict search?" << std::endl;
343  }
344  }
345  for (auto& it : *options) delete it;
346  delete options;
347  return result;
348 }
