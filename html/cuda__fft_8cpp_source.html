33 template<typename T>
34 void CudaFFT<T>::init(const HW &gpu, const FFTSettings<T> &settings, bool reuse) {
35  bool canReuse = m_isInit
36  && reuse
37  && (m_settings->sBytesBatch() >= settings.sBytesBatch())
38  && (m_settings->fBytesBatch() >= settings.fBytesBatch());
39  bool mustAllocate = !canReuse;
40  if (mustAllocate) {
41  release();
42  }
43  // previous plan and settings has to be released,
44  // otherwise we will get GPU/CPU memory leak
45  release(m_plan);
46  delete m_settings;
47 
48  m_settings = new FFTSettings<T>(settings);
49  try {
50  m_gpu = &dynamic_cast<const GPU&>(gpu);
51  } catch (std::bad_cast&) {
52  REPORT_ERROR(ERR_ARG_INCORRECT, "Instance of GPU expected");
53  }
54 
55  check();
56 
57  m_plan = createPlan(*m_gpu, *m_settings);
58  if (mustAllocate) {
59  // allocate input data storage
60  gpuErrchk(cudaMalloc(&m_d_SD, m_settings->sBytesBatch()));
61  if (m_settings->isInPlace()) {
62  // input data holds also the output
63  m_d_FD = (std::complex<T>*)m_d_SD;
64  } else {
65  // allocate also the output buffer
66  gpuErrchk(cudaMalloc(&m_d_FD, m_settings->fBytesBatch()));
67  }
68  }
69 
70  m_isInit = true;
71 }

138 template<typename T>
139 std::complex<T>* CudaFFT<T>::fft(const T *h_in,
140  std::complex<T> *h_out) {
141  auto isReady = m_isInit && m_settings->isForward();
142  if ( ! isReady) {
143  REPORT_ERROR(ERR_LOGIC_ERROR, "Not ready to perform Fourier Transform. "
144  "Call init() function first");
145  }
146 
147  // process signals in batches
148  for (size_t offset = 0; offset < m_settings->sDim().n(); offset += m_settings->batch()) {
149  // how many signals to process
150  size_t toProcess = std::min(m_settings->batch(), m_settings->sDim().n() - offset);
151 
152  // copy memory
153  gpuErrchk(cudaMemcpyAsync(
154  m_d_SD,
155  h_in + offset * m_settings->sDim().xyzPadded(),
156  toProcess * m_settings->sBytesSingle(),
157  cudaMemcpyHostToDevice, *(cudaStream_t*)m_gpu->stream()));
158 
159  fft(*m_plan, m_d_SD, m_d_FD);
160 
161  // copy data back
162  gpuErrchk(cudaMemcpyAsync(
163  h_out + offset * m_settings->fDim().xyzPadded(),
164  m_d_FD,
165  toProcess * m_settings->fBytesSingle(),
166  cudaMemcpyDeviceToHost, *(cudaStream_t*)m_gpu->stream()));
167  }
168  return h_out;
169 }

244 template<typename T>
245 template<typename F>
246 void CudaFFT<T>::manyHelper(const FFTSettings<T> &settings, F function) {
247  auto n = std::array<int, 3>{(int)settings.sDim().z(), (int)settings.sDim().y(), (int)settings.sDim().x()};
248  int idist;
249  int odist;
250  cufftType type;
251  if (settings.isForward()) {
252  idist = settings.sDim().xyzPadded();
253  odist = settings.fDim().xyzPadded();
254  type = std::is_same<T, float>::value ? CUFFT_R2C : CUFFT_D2Z;
255  } else {
256  idist = settings.fDim().xyzPadded();
257  odist = settings.sDim().xyzPadded();
258  type = std::is_same<T, float>::value ? CUFFT_C2R : CUFFT_Z2D;
259  }
260  int rank = 3;
261  if (settings.sDim().z() == 1) rank--;
262  if ((2 == rank) && (settings.sDim().y() == 1)) rank--;
263 
264  int offset = 3 - rank;
265  function(rank, &n[offset], nullptr,
266  1, idist, nullptr, 1, odist, type, settings.batch());
267 }

270 cufftHandle* CudaFFT<T>::createPlan(const GPU &gpu, const FFTSettings<T> &settings) {
271  if (settings.sElemsBatch() > std::numeric_limits<int>::max()) {
272  REPORT_ERROR(ERR_ARG_INCORRECT, "Too many elements for Fourier Transformation. "
273  "It would cause int overflow in the cuda kernel. Try to decrease batch size");
274  }
275  auto plan = new cufftHandle;
276  auto f = [&] (int rank, int *n, int *inembed,
277  int istride, int idist, int *onembed, int ostride,
278  int odist, cufftType type, int batch) {
279  gpuErrchkFFT(cufftPlanMany(plan, rank, n, inembed,
280  istride, idist, onembed, ostride,
281  odist, type, batch));
282  };
283  manyHelper(settings, f);
284  gpuErrchkFFT(cufftSetStream(*plan, *(cudaStream_t*)gpu.stream()));
285  return plan;

288 template<typename T>
289 FFTSettings<T> CudaFFT<T>::findMaxBatch(const FFTSettings<T> &settings,
290  size_t maxBytes) {
291  size_t singleBytes = settings.sBytesSingle() + (settings.isInPlace() ? 0 : settings.fBytesSingle());
292  size_t batch = min((maxBytes / singleBytes), settings.batch()) + 1; // + 1 will be deducted in the while loop
293  while (batch > 1) {
294  batch--;
295  auto tmp = FFTSettings<T>(settings.sDim(), batch, settings.isInPlace(), settings.isForward());
296  size_t totalBytes = CudaFFT<T>().estimateTotalBytes(tmp);
297  if (totalBytes <= maxBytes) {
298  return tmp;
299  }
300  }
301  REPORT_ERROR(ERR_GPU_MEMORY, "Estimated batch size is 0(zero). "
302  "This probably means you don't have enough GPU memory for even a single transformation.");
303 }

305 template<typename T>
306 FFTSettings<T> *CudaFFT<T>::findOptimal(const GPU &gpu,
307  const FFTSettings<T> &settings,
308  size_t reserveBytes, bool squareOnly, int sigPercChange,
309  bool crop, bool verbose) {
310  using cuFFTAdvisor::Tristate::TRUE;
311  using cuFFTAdvisor::Tristate::FALSE;
312  size_t freeBytes = gpu.lastFreeBytes();
313  std::vector<cuFFTAdvisor::BenchmarkResult const *> *options =
314  cuFFTAdvisor::Advisor::find(10, gpu.device(), // FIXME DS this should be configurable
315  settings.sDim().x(), settings.sDim().y(), settings.sDim().z(), settings.sDim().n(),
316  TRUE, // use batch
317  std::is_same<T, float>::value ? TRUE : FALSE,
318  settings.isForward() ? TRUE : FALSE,
319  settings.isInPlace() ? TRUE : FALSE,
320  cuFFTAdvisor::Tristate::TRUE, // is real
321  sigPercChange, memoryUtils::MB(freeBytes - reserveBytes),
322  false, // allow transposition
323  squareOnly, crop);
324 
325  FFTSettings<T> *result = nullptr;
326  if (0 != options->size()) {
327  auto res = options->at(0);
328  auto optSetting = FFTSettings<T>(
329  res->transform->X,
330  res->transform->Y,
331  res->transform->Z,
332  settings.sDim().n(),
333  res->transform->N / res->transform->repetitions,
334  settings.isInPlace(),
335  settings.isForward());
336  result = new FFTSettings<T>(optSetting);
337  }
338  if (verbose) {
339  if (nullptr != result) {
340  options->at(0)->printHeader(stdout); printf("\n");
341  options->at(0)->print(stdout); printf("\n");
342  } else {
343  std::cout << "No result obtained. Maybe too strict search?" << std::endl;
344  }
345  }
346  for (auto& it : *options) delete it;
347  delete options;
348  return result;
349 }

353  const FFTSettings<T> &settings,
354  size_t reserveBytes, bool squareOnly, int sigPercChange,
355  bool crop, bool verbose) {
