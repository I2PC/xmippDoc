248 double getScale(ImageInfo imgInf, size_t &xdim, size_t &ydim)
249 {
250  double scale = 0;
251  // If only xdim is passed, it is the higher allowable size, for any dimension
252  if (ydim == 0 && imgInf.adim.xdim < imgInf.adim.ydim)
253  {
254  ydim = xdim;
255  scale = ((double) ydim)/((double) imgInf.adim.ydim);
256  xdim = (int)(scale*imgInf.adim.xdim);
257  }
258  else
259  {
260  scale = ((double) xdim)/((double) imgInf.adim.xdim);
261  if (ydim == 0)
262  ydim = (int)(scale*imgInf.adim.ydim);
263  }
264  return scale;
265 }

401 void ImageGeneric::convert2Datatype(DataType _datatype, CastWriteMode castMode)
402 {
403  if (_datatype == datatype || _datatype == DT_Unknown)
404  return;
405 
406  ArrayDim aDim;
407  data->getDimensions(aDim);
408 
409  ImageBase * newImage;
410  MultidimArrayGeneric * newMAG;
411 
412 #define CONVERTTYPE(type) Image<type> *imT = new Image<type>; \
413  newImage = imT;\
414  newMAG = new MultidimArrayGeneric((MultidimArrayBase*) &(imT->data), _datatype);\
415  MultidimArray<type>* pMAG;\
416  newMAG->getMultidimArrayPointer(pMAG);\
417  if (castMode == CW_CAST)\
418  data->getImage(*pMAG);\
419  else\
420  {\
421  pMAG->resize(aDim);\
422  double min, max;\
423  data->computeDoubleMinMax(min, max);\
424  ((Image<double>*) image)->getCastConvertPageFromT(0, (char*)pMAG->data, _datatype, aDim.nzyxdim, min, max, castMode);\
425  }\
426 
427  SWITCHDATATYPE(_datatype, CONVERTTYPE)
428 
429 #undef CONVERTTYPE
430 
431  /* aDimFile must be set in order to movePointerTo can be used.
432  * If movePointerTo has been used before convert2Datatype, then
433  * only the pointed image/slice is converted, and the images/slices left
434  * are lost. This is why we set the new dimensions to the new ImageGeneric Object*/
435  newImage->setADimFile(aDim);
436 
437  clear();
438  datatype = _datatype;
439  image = newImage;
440  data = newMAG;
441 }

444 void ImageGeneric::reslice(AxisView face, ImageGeneric &imgOut)
445 {
446  ArrayDim aDim, aDimOut;
447  data->getDimensions(aDim);
448 
449  char axis;
450  bool reverse;
451 
452  aDimOut = aDim;
453 
454  if (face == VIEW_Y_NEG || face == VIEW_Y_POS)
455  {
456  axis = 'Y';
457  aDimOut.ydim = aDim.zdim;
458  aDimOut.zdim = aDim.ydim;
459  reverse = (face == VIEW_Y_NEG);
460  }
461  else if (face == VIEW_X_NEG || face == VIEW_X_POS)
462  {
463  axis = 'X';
464  aDimOut.xdim = aDim.zdim;
465  aDimOut.zdim = aDim.xdim;
466  reverse = (face == VIEW_X_NEG);
467  }
468  else
469  REPORT_ERROR(ERR_VALUE_INCORRECT, formatString("reslice: not supported view %d", (int)face));
470 
471  DataType dtype = getDatatype();
472  imgOut.setDatatype(dtype);
473  imgOut().resize(aDimOut, false);
474  imgOut.image->setADimFile(aDimOut);
475 
476  MultidimArrayGeneric imTemp;
477 
478  int index;
479 
480  for (size_t k = 0; k < aDimOut.zdim; k++)
481  {
482  imTemp.aliasSlice(MULTIDIM_ARRAY_GENERIC(imgOut), k);
483  index = k + (aDimOut.zdim - 1 - 2*k) * (int)reverse;
484  MULTIDIM_ARRAY_GENERIC(*this).getSlice(index, &imTemp, axis, !reverse);
485  }
486 
487 }
