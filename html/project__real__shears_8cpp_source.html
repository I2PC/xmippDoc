125 void convertAngles(double &phi, double &theta, double &psi)
126 {
127  Matrix2D<double> E;
128  Euler_angles2matrix(phi,theta,psi,E,false);
129  double A00=MAT_ELEM(E,0,0);
130  double A02=MAT_ELEM(E,0,2);
131  double A10=MAT_ELEM(E,1,0);
132  double A12=MAT_ELEM(E,1,2);
133  double A20=MAT_ELEM(E,2,0);
134  double A21=MAT_ELEM(E,2,1);
135  double A22=MAT_ELEM(E,2,2);
136 
137  double abs_cosay = sqrt(A22*A22+A21*A21);
138 
139  //Results angles
140  double ax, ay, az;
141 
142  if(abs_cosay > 0.0)
143  {
144  double sign_cosay;
145 
146  ax = atan2(-A21, A22);
147  az = atan2(-A10, A00);
148 
149  if(fabs(cos(ax)) == 0.0)
150  sign_cosay = SGN(-A21/sin(ax));
151  else
152  {
153  if (cos(ax) > 0.0)
154  sign_cosay = SGN(A22);
155  else
156  sign_cosay = -SGN(A22);
157  }
158 
159  ay = atan2(A20, sign_cosay * abs_cosay);
160  }
161  else
162  {
163  //Let's consider the matrix as a rotation around Z
164  if (SGN(A20) > 0.0)
165  {
166  ax = 0.0;
167  ay = PI/2.0;
168  az = atan2(A12, -A02);
169  }
170  else
171  {
172  ax = 0.0;
173  ay = -PI/2.0;
174  az = atan2(-A12, A02);
175  }
176  }
177 
178  phi = ax;
179  theta = ay;
180  psi = az;
181 }

185 void projectionRealShears2(MultidimArray<double> &CoefVolume,

199  Ki[2]=dMn(Binv,11);
200  Ki[3]=dMn(Binv,15);

205  K[2]=Ki[2];

217  Arg[3]=VEC_ELEM(BinvCscaled,3)*sc+K[3];

219  double h = Arg[arr[2]];
220

223  int m1 = (int)ceil(h-2.0);

246  }
