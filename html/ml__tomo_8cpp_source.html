389 ProgMLTomo::run()
390 {
391  double LL, sumw_allrefs, sumcorr;
392  bool converged;
393  std::vector<double> conv;
394  double wsum_sigma_noise, wsum_sigma_offset;
395  std::vector<MultidimArray<double> > wsumimgs; //3D

402 
403  produceSideInfo();
404  show();
405  produceSideInfo2();
406  Maux.resize(dim, dim, dim);
407  Maux.setXmippOrigin();
408 
409  // Loop over all iterations
410  for (int iter = istart; iter <= Niter; iter++)
411  {
412 
413  if (verbose)
414  std::cout << " Multi-reference refinement: iteration " << iter << " of "
415  << Niter << std::endl;
416  // Save old reference images
417  for (int refno = 0; refno < nr_ref; refno++)
418  Iold[refno]() = Iref[refno]();
419  //#define DEBUG5

421

432 #endif
433  // Integrate over all images
434  expectation(MDimg, Iref, iter, LL, sumcorr, wsumimgs, wsumweds,
