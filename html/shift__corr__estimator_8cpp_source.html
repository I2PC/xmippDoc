171 template<typename T>
172 template<bool CENTER>
173 void ShiftCorrEstimator<T>::sComputeCorrelations2DOneToN(
174  const HW &hw,
175  std::complex<T> *__restrict inOut,
176  const std::complex<T> *__restrict ref,
177  const Dimensions &__restrict dims) {
178  if (CENTER) {
179  // we cannot assert xDim, as we don't know if the spatial size was even
180  assert(0 == (dims.y() % 2));
181  }
182  assert(0 < dims.x());
183  assert(0 < dims.y());
184  assert(1 == dims.z());
185  assert(0 < dims.n());
186 
187  const size_t maxN = dims.n();
188  const size_t maxY = dims.y();
189  const size_t maxX = dims.x();
190 
191  for (size_t n = 0; n < maxN; ++n) {
192  size_t offsetN = n * dims.xyzPadded();
193  for (size_t y = 0; y < maxY; ++y) {
194  int centerCoeff = (0 == y % 2) ? 1 : -1;
195  size_t offsetY = y * dims.xPadded();
196  for (size_t x = 0; x < maxX; ++x) {
197  size_t destIndex = offsetN + offsetY + x;
198  auto r = ref[offsetY + x];
199  auto o = r * std::conj(inOut[destIndex]);
200  inOut[destIndex] = o;
201  if (CENTER) {
202  inOut[destIndex] *= centerCoeff;
203  centerCoeff *= -1;
204  }
205  }
206  }
207  }
208 }

210 template<typename T>
211 void ShiftCorrEstimator<T>::computeShift2DOneToN(
212  T *others) {
213  bool isReady = (this->m_isInit && (AlignType::OneToN == this->m_type)
214  && this->m_is_ref_FD_loaded && this->m_includingBatchFT);
215 
216  if ( ! isReady) {
217  REPORT_ERROR(ERR_LOGIC_ERROR, "Not ready to execute. Call init() before");
218  }
219 
220  // reserve enough space for shifts
221  this->m_shifts2D.reserve(this->m_settingsInv->fDim().n());
222  // process signals
223  for (size_t offset = 0; offset < this->m_settingsInv->fDim().n(); offset += this->m_settingsInv->batch()) {
224  // how many signals to process
225  size_t toProcess = std::min(this->m_settingsInv->batch(), this->m_settingsInv->fDim().n() - offset);
226 
227  T *batchStart;
228  if (toProcess == this->m_settingsInv->batch()) {
229  // we process whole batch, so we don't need to copy data
230  batchStart = others + offset * this->m_settingsInv->sDim().xyPadded();
231  } else {
232  assert(this->m_settingsInv->batch() <= this->m_settingsInv->sDim().n());
233  // less than 'batch' signals are left, so we need to process last 'batch'
234  // signals to avoid invalid memory access
235  batchStart = others
236  + (this->m_settingsInv->sDim().n() - this->m_settingsInv->batch())
237  * this->m_settingsInv->sDim().xy();
238  }
239 
240  // perform FT
241  FFTwT<T>::fft(m_batchToFD, batchStart, m_batch_FD);
242 
243  // compute shifts
244  auto shifts = computeShifts2DOneToN(
245  *m_cpu,
246  m_batch_FD,
247  m_batch_SD,
248  m_single_FD,
249  this->m_settingsInv->createBatch(), // always process whole batch, as we do it to avoid copying memory
250  m_batchToSD,
251  this->m_maxShift);
252 
253  // append shifts to existing results
254  this->m_shifts2D.insert(this->m_shifts2D.end(),
255  // in case of the last iteration, take only the shifts we actually need
256  shifts.begin() + this->m_settingsInv->batch() - toProcess,
257  shifts.end());
258  }
259 
260  // update state
261  this->m_is_shift_computed = true;
262 }

264 template<typename T>
265 std::vector<Point2D<float>> ShiftCorrEstimator<T>::computeShifts2DOneToN(
266  const CPU &cpu,
267  std::complex<T> *othersF,
268  T *othersS,
269  std::complex<T> *ref,
270  const FFTSettingsNew<T> &settings,
271  void *plan,
272  size_t maxShift) {
273  // we need even input in order to perform the shift (in FD, while correlating) properly
274  assert(0 == (settings.sDim().x() % 2));
275  assert(0 == (settings.sDim().y() % 2));
276  assert(1 == settings.sDim().zPadded());
277 
278  // correlate signals and shift FT so that it will be centered after IFT
279  sComputeCorrelations2DOneToN(cpu,
280  othersF, ref,
281  settings.fDim(), true);
282 
283  // perform IFT
284  FFTwT<T>::ifft(plan, othersF, othersS);
285 
286  // compute shifts
287  auto maxIndices = std::vector<float>(settings.sDim().n(), -1.f);
288  ExtremaFinder::SingleExtremaFinder<T>::sFindMax2DAroundCenter(cpu, settings.sDim(),
289  othersS, maxIndices.data(), nullptr, maxShift);
290  // convert absolute indices to 2D position
291  // FIXME DS extract this to some indexing utils or sth
292  int cX = settings.sDim().x() / 2;
293  int cY = settings.sDim().y() / 2;
294  auto result = std::vector<Point2D<float>>();
295  const int x = settings.sDim().x();
296  for (auto i : maxIndices) {
297  result.emplace_back(((int)i % x) - cX, ((int)i / x) - cY);
298  }
299  return result;
300 }
