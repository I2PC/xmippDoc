32 /* Switch among different datatypes.
33  *
34  * This macro replicates the code for the different data type options.
35  *
36  *@code
37  *
38  *#define MYFUNC(type) getSlice(k, *(MultidimArray<type>*)image, axis, n)
39  *
40  *SWITCHDATATYPE(datatype,MYFUNC)
41  *
42  *@endcode
43  */
44 #define SWITCHDATATYPE(datatype,OP) \
45  switch (datatype)\
46  {\
47  case DT_Double:\
48  {OP(double)};\
49  break;\
50  case DT_Float:\
51  {OP(float)};\
52  break;\
53  case DT_UInt:\
54  {OP(unsigned int)};\
55  break;\
56  case DT_Int:\
57  {OP(int)};\
58  break;\
59  case DT_Short:\
60  {OP(short)};\
61  break;\
62  case DT_UShort:\
63  {OP(unsigned short)};\
64  break;\
65  case DT_SChar:\
66  {OP(char)};\
67  break;\
68  case DT_UHalfByte:\
69  case DT_UChar:\
70  {OP(unsigned char)};\
71  break;\
72  case DT_Long:\
73  {OP(long)};\
74  break;\
75  case DT_ULong:\
76  {OP(unsigned long)};\
77  break;\
78  default:\
79  REPORT_ERROR(ERR_ARG_INCORRECT,"Do not know how to handle this type at this point");\
80  }
