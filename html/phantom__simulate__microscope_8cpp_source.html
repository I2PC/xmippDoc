31 void ProgSimulateMicroscope::readParams()
32 {
33  XmippMetadataProgram::readParams();
34 
35  fn_ctf = "";//initialize empty, force recalculation of first time
36  pmdIn = getInputMd();
37  CTFpresent=true;
38  if (checkParam("--ctf"))
39  {
40  //Fill the input metadata with the value of 'fn_ctf'
41  fn_ctf=getParam("--ctf");
42  MDConstGenerator generator(fn_ctf);
43  generator.label = MDL_CTF_MODEL;
44  generator.fill(*pmdIn);
45  }
46  else
47  {
48  if (pmdIn->containsLabel(MDL_CTF_MODEL))
49  {
50  //sort the images according to the ctf to avoid the recaculation of it

53  pmdIn->sort(md, MDL_CTF_MODEL);
54  }
55  else
56  CTFpresent=false;
57  }
58  after_ctf_noise = checkParam("--after_ctf_noise");
59  defocus_change = getDoubleParam("--defocus_change");
60  if (checkParam("--noise"))
61  {
62  sigma = getDoubleParam("--noise");
63  low_pass_before_CTF = getDoubleParam("--noise",1);
64  estimateSNR=false;
65  }
66  else if (checkParam("--noNoise"))
67  {
68  estimateSNR=0;
69  sigma=0;
70  low_pass_before_CTF=0.5;
71  }
72  else
73  {
74  targetSNR = getDoubleParam("--targetSNR");
75  low_pass_before_CTF=0.5;
76  estimateSNR=true;
77  }
78  downsampling = getDoubleParam("--downsampling");
79 
80 }

128 void ProgSimulateMicroscope::estimateSigma()
129 {
130  if (CTFpresent)
131  {
132  ctf.FilterBand = CTF;
133  ctf.ctf.enable_CTFnoise = false;
134  ctf.ctf.read(getParam("--ctf"));
135  ctf.ctf.changeSamplingRate(ctf.ctf.Tm*downsampling);
136  ctf.ctf.produceSideInfo();
137  }
138 
139  size_t N_stats = pmdIn->size();
140  std::cout << "N_stats=" << N_stats << std::endl;
141 
142  MultidimArray<double> proj_power(N_stats);
143  MultidimArray<double> proj_area(N_stats);
144  double power_avg=0., power_stddev, area_avg=0., area_stddev, avg, dummy;
145  if (verbose!=0)
146  {
147  std::cerr << "Estimating noise power for target SNR=" << targetSNR << std::endl;
148  init_progress_bar(N_stats);
149  }
150  FileName fnImg;
151  Image<double> proj;
152  size_t nImg=0;
153  FOR_ALL_OBJECTS_IN_METADATA(*pmdIn)
154  {
155  pmdIn->getValue(image_label, fnImg,__iter.objId);
156  proj.read(fnImg);
157  MultidimArray<double> mProj=proj();
158 
159  if (CTFpresent)
160  {
161  if (nImg == 0)
162  ctf.generateMask(mProj);
163  ctf.applyMaskSpace(mProj);
164  }
165 
166  // Compute projection area
167  DIRECT_A1D_ELEM(proj_area,nImg) = 0;
168  FOR_ALL_DIRECT_ELEMENTS_IN_MULTIDIMARRAY(mProj)
169  if (fabs(DIRECT_MULTIDIM_ELEM(mProj,n)) > 1e-6)
170  DIRECT_A1D_ELEM(proj_area,nImg)++;
171 
172  // Compute projection power
173  mProj.computeStats(avg, DIRECT_A1D_ELEM(proj_power,nImg), dummy, dummy);
174 
175  if (nImg++ % 30 == 0 && verbose!=0)
176  progress_bar(nImg);
177  }
178  progress_bar(N_stats);
179  Histogram1D hist_proj, hist_area;
180  compute_hist(proj_power, hist_proj, 300);
181  compute_hist(proj_area, hist_area, 300);
182  proj_power.computeStats(power_avg, power_stddev, dummy, dummy);
183  std::cout << "# Projection power average: " << power_avg << std::endl
184  << "# Projection power stddev: " << power_stddev << std::endl
185  << "# Projection percentil 2.5%: " << hist_proj.percentil(2.5) << std::endl
186  << "# Projection percentil 97.5%: " << hist_proj.percentil(97.5) << std::endl;
187  proj_area.computeStats(area_avg, area_stddev, dummy, dummy);
188  std::cout << "# Projection area average: " << area_avg << std::endl
189  << "# Projection area stddev: " << area_stddev << std::endl
190  << "# Area percentil 2.5%: " << hist_area.percentil(2.5) << std::endl
191  << "# Area percentil 97.5%: " << hist_area.percentil(97.5) << std::endl;
192 
193  sigma=sqrt(power_avg*power_avg*Xdim*Ydim / (targetSNR*area_avg));
194  std::cout << "Estimated sigma=" << sigma << std::endl;
195  updateCtfs();
196 }

218 void ProgSimulateMicroscope::updateCtfs()
219 {
220  double before_power = 0, after_power = 0;
221  sigma_after_CTF = sigma_before_CTF = 0;
222 
223  if (CTFpresent)
224  {
225  setupFourierFilter(ctf, false, before_power);
226  if (after_ctf_noise)
227  setupFourierFilter(after_ctf, true, after_power);
228  }
229 
230  // Compute noise balance
231  if ((after_power != 0 || before_power != 0) && sigma!=0)
232  {
233  double p = after_power / (after_power + before_power);
234  double K = 1 / sqrt(p * after_power + (1 - p) * before_power);
235  sigma_after_CTF = sqrt(p) * K * sigma;
236  sigma_before_CTF = sqrt(1 - p) * K * sigma;
237  }
238  else if (sigma != 0)
239  {
240  sigma_before_CTF = sigma;
241  sigma_after_CTF = 0;
242  }
243  else
244  sigma_before_CTF=sigma_after_CTF=0;
245 }

292 /* Apply ------------------------------------------------------------------- */
293 void ProgSimulateMicroscope::apply(MultidimArray<double> &I)
294 {
295  I.setXmippOrigin();
296  I.selfWindow(FIRST_XMIPP_INDEX(2*Ydim), FIRST_XMIPP_INDEX(2*Xdim),
297  LAST_XMIPP_INDEX(2*Ydim), LAST_XMIPP_INDEX(2*Xdim));
298 
299  // Add noise before CTF
300  MultidimArray<double> noisy;
301  noisy.resize(I);
302  noisy.initRandom(0, sigma_before_CTF, RND_GAUSSIAN);
303  if (low_pass_before_CTF < 0.5)
304  lowpass.applyMaskSpace(noisy);
305  I += noisy;
306 
307  if (CTFpresent)
308  {
309  // Check if the mask is a defocus changing CTF
310  // In that case generate a new mask with a random defocus
311  if (defocus_change != 0)
312  {
313  MultidimArray<double> aux;
314  ctf.ctf.DeltafU = defocusU * rnd_unif(1 - defocus_change / 100, 1 + defocus_change / 100);
315  ctf.ctf.DeltafV = defocusV *rnd_unif(1 - defocus_change / 100, 1 + defocus_change / 100);
316  aux.initZeros(2*Ydim, 2*Xdim);
317  ctf.generateMask(aux);
318  }
319 
320  // Apply CTF
321  ctf.applyMaskSpace(I);
322 
323  // Add noise after CTF
324  noisy.initRandom(0, sigma_after_CTF, RND_GAUSSIAN);
325  if (after_ctf_noise)
326  after_ctf.applyMaskSpace(noisy);
327  I += noisy;
328  }
329  I.selfWindow(FIRST_XMIPP_INDEX(Ydim), FIRST_XMIPP_INDEX(Xdim),
330  LAST_XMIPP_INDEX(Ydim), LAST_XMIPP_INDEX(Xdim));
331 }
