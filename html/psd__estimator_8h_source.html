53  for (size_t y = 0; y < settings.sDim().y(); ++y) {
54  for (size_t x = 0; x < settings.sDim().x(); ++x) {
55  bool mirror = x >= settings.fDim().x();
56  size_t xS = mirror
57  ? (settings.sDim().x() - x)

61  : y;
62  size_t indexD = y * settings.sDim().x() + x;
63  size_t indexS = yS * settings.fDim().x() + xS;
64  out[indexD] = func(mirror, in[indexS]);
