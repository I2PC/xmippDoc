107  def _pickMicrograph(self, mic, *args):
108  micPath = mic.getFileName()
109  # Get particle picking boxsize from the previous run
110  boxSize = self.particlePickingRun.outputCoordinates.getBoxSize()
111  modelRoot = self._getExtraPath('model')
112 
113  micName = removeBaseExt(micPath)
114  proceed = True
115  if self.micsToPick == MICS_SAMEASPICKING:
116  basePos = replaceBaseExt(micPath, "pos")
117  fnPos = self.particlePickingRun._getExtraPath(basePos)
118  if exists(fnPos):
119  blocks = emlib.getBlocksInMetaDataFile(fnPos)
120  copy = True
121  if 'header' in blocks:
122  mdheader = emlib.MetaData("header@" + fnPos)
123  state = mdheader.getValue(emlib.MDL_PICKING_MICROGRAPH_STATE,
124  mdheader.firstObject())
125  if state == "Available":
126  copy = False
127  if copy:
128  # Copy manual .pos file of this micrograph
129  copyFile(fnPos, self._getExtraPath(basename(fnPos)))
130  proceed = False
131 
132  if proceed:
133  args = "-i %s " % micPath
134  args += "--particleSize %d " % boxSize
135  args += "--model %s " % modelRoot
136  args += "--outputRoot %s " % self._getExtraPath(micName)
