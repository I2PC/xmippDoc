64 void ProgImageSSNR::preProcess()
65 {
66  size_t Xdim, Ydim, Zdim, Ndim;
67  getImageSize(*getInputMd(), Xdim, Ydim, Zdim, Ndim);
68  maskS.initZeros(Ydim,Xdim);
69  maskS.setXmippOrigin();
70 
71  if (R==-1)
72  R=0.5*Xdim-Rwidth;
73  RaisedCosineMask(maskS,R-Rwidth,R+Rwidth);
74  maskN=1-maskS;
75 
76  imin=(size_t)std::max(3.0,0.5*Xdim*(sampling/fmin));
77  imax=(size_t)std::min(Xdim-3.0,0.5*Xdim*(sampling/fmax));
78 }

129 void normalizeSSNR(MetaData &mdOut)
130 {
131  double maxSSNR=-1e38;
132  FOR_ALL_OBJECTS_IN_METADATA(mdOut)
133  {
134  double ssnr;
135  mdOut.getValue(MDL_CUMULATIVE_SSNR,ssnr,__iter.objId);
136  if (ssnr>maxSSNR)
137  maxSSNR=ssnr;
138  }
139 
140  if (maxSSNR>0)
141  {
142  double imaxSSNR=1/maxSSNR;
143  FOR_ALL_OBJECTS_IN_METADATA(mdOut)
144  {
145  double ssnr;
146  mdOut.getValue(MDL_CUMULATIVE_SSNR,ssnr,__iter.objId);
147  mdOut.setValue(MDL_WEIGHT_SSNR,ssnr*imaxSSNR,__iter.objId);
148  }
149  }

152 void ProgImageSSNR::postProcess()
153 {
154  if (ssnrcut>0)

156 
157  if (ssnrpercent>0)

160  getOutputMd()->getColumnValues(MDL_CUMULATIVE_SSNR,ssnr);
161  std::sort(ssnr.begin(),ssnr.end());
162  size_t idx=(size_t)(ssnrpercent/100.0*ssnr.size());
163  thresholdSSNR(*getOutputMd(),ssnr[idx]);
164  }
165 
166  if (normalizessnr)

168  if (fn_out!="")
169  getOutputMd()->write(fn_out);
