32 double evaluatePlane(double rot, double tilt,
33  const MultidimArray<double> *V, const MultidimArray<double> *Vmag,
34  double maxFreq, double planeWidth, int direction,
35  MultidimArray<double> *Vdraw=NULL,
36  bool setPos=false, double rotPos=0, double tiltPos=0)
37 {
38  if (rot<0 || rot>360 || tilt<-90 || tilt>90)
39  return 0;
40 
41  Matrix2D<double> E, Einv;
42  Euler_angles2matrix(rot,tilt,0.,E);
43  Einv=E.transpose();
44 
45  if (setPos)
46  {
47  Matrix2D<double> Epos;
48  Euler_angles2matrix(rotPos,tiltPos,0.,Epos);
49  double angle=acos(E(2,0)*Epos(2,0)+E(2,1)*Epos(2,1)+E(2,2)*Epos(2,2));
50  angle=RAD2DEG(angle);
51  if (fabs(angle)<20 || fabs(180-angle)<20)
52  return 0;
53  }
54 
55  size_t N=XMIPP_MAX(XSIZE(*Vmag),YSIZE(*Vmag)/2);
56  N=XMIPP_MAX(N,ZSIZE(*Vmag)/2);
57  double df=0.5/N;
58  Matrix1D<double> freq(3), freqp(3);
59  Matrix1D<int> idx(3);
60  double sumNeg=0, sumPos=0;
61  int Nneg=0, Npos=0;
62  double maxFreq2=maxFreq*maxFreq;
63  int iPlaneWidth=(int)ceil(planeWidth);
64  for (double ix=0; ix<=N; ix++)
65  {
66  XX(freq)=ix*df;
67  double fx2=XX(freq)*XX(freq);
68  if (fx2>maxFreq2)
69  continue;
70  for (double iy=-(int)N; iy<=N; iy++)
71  {
72  YY(freq)=iy*df;
73  double fx2fy2=fx2+YY(freq)*YY(freq);
74  if (fx2fy2>maxFreq2)
75  continue;
76  for (int iz=-iPlaneWidth; iz<=iPlaneWidth; iz++)
77  {
78  if (iz==0 || ix==0 || iy==0)
79  continue;
80 
81  // Frequency in the coordinate system of the plane
82  ZZ(freq)=iz*df;
83 
84  // Frequency in the coordinate system of the volume
85  SPEED_UP_temps012;
86  M3x3_BY_V3x1(freqp,Einv,freq);
87  bool inverted=false;
88  if (XX(freqp)<0)
89  {
90  XX(freqp)=-XX(freqp);
91  YY(freqp)=-YY(freqp);
92  ZZ(freqp)=-ZZ(freqp);
93  inverted=true;
94  }
95 
96  // Get the corresponding index
97  DIGFREQ2FFT_IDX(ZZ(freqp), ZSIZE(*V), ZZ(idx));
98  DIGFREQ2FFT_IDX(YY(freqp), YSIZE(*V), YY(idx));
99  DIGFREQ2FFT_IDX(XX(freqp), XSIZE(*V), XX(idx));
100  if (XX(idx) < STARTINGX(*Vmag) || XX(idx) > FINISHINGX(*Vmag) ||
101  YY(idx) < STARTINGY(*Vmag) || YY(idx) > FINISHINGY(*Vmag) ||
102  ZZ(idx) < STARTINGZ(*Vmag) || ZZ(idx) > FINISHINGZ(*Vmag))
103  continue;
104 
105  // Make the corresponding sums
106  bool negativeSum;
107  if (direction==1)
108  negativeSum=iz<0;
109  else
110  negativeSum=iz>0;
111  double val=A3D_ELEM(*Vmag,ZZ(idx),YY(idx),XX(idx));
112  if ((negativeSum && !inverted) || (!negativeSum && inverted)) // XOR
113  {
114  sumNeg+=val;
115  Nneg++;
116  if (Vdraw!=NULL)
117  (*Vdraw)(idx)=2*direction*val;
118  }
119  else
120  {
121  sumPos+=val;
122  Npos++;
123  if (Vdraw!=NULL)
124  (*Vdraw)(idx)=1.0/2.0*direction*val;
125  }
126  }
127  }
128  }
129  if (fabs(Nneg-Npos)/(0.5*(Nneg+Npos))>0.5)
130  // If there is a difference of more than 50%
131  return 1e38;
132  if (Nneg!=0)
133  sumNeg/=Nneg;
134  else
135  return 1e38;
136  if (Npos!=0)
137  sumPos/=Npos;
138  else
139  return 1e38;
140 
141  return -(sumPos-sumNeg);
142 }

201 void lookForPlane(const MultidimArray<double> *V, const MultidimArray<double> *Vmag,
202  double maxFreq, double planeWidth, int direction,
203  double &rot, double &tilt,
204  bool setPos=false, double rotPos=0, double tiltPos=0)
205 {
206  // Optimize with DE
207  int length=2;
208  int Npop=50;
209 
210  double min_allowed[] = {0,-90};
211  double max_allowed[] = {360,90};
212 
213  WedgeSolver * solver = new WedgeSolver(length,length*Npop,V,Vmag,maxFreq,planeWidth,direction);
214  solver->Setup( min_allowed, max_allowed, stBest2Bin, 0.5, 0.8);
215 
216  if (setPos)
217  {
218  solver->setPos=true;
219  solver->rotPos=rotPos;
220  solver->tiltPos=tiltPos;
221  }
222 
223  solver->Solve(50);
224 
225  double * bestSolution = solver->Solution();
226 
227  // Refine with Powell
228  Matrix1D<double> x(2), steps(2);
229  double fitness;
230  int iter;
231  steps.initConstant(1);
232  x(0)=bestSolution[0];
233  x(1)=bestSolution[1];
234  powellOptimizer(x,1,2,&wrapperFitnessDetectMissingWedge,solver,0.01,fitness,iter,steps,true);
235  rot=x(0);
236  tilt=x(1);
237 
238  delete solver;
239 
240  solver = NULL;
241 }

243 // Draw wedge --------------------------------------------------------------
244 void drawWedge(double rotPos, double tiltPos, double rotNeg, double tiltNeg,
245  const MultidimArray<double> *V,
246  const MultidimArray<double> *Vmag, MultidimArray<double> *Vdraw)
247 {
248  Matrix2D<double> Epos, Eneg;
249  Euler_angles2matrix(rotPos,tiltPos,0.,Epos);
250  Euler_angles2matrix(rotNeg,tiltNeg,0.,Eneg);
251 
252  Matrix1D<double> freq(3), freqPos, freqNeg;
253  Matrix1D<int> idx(3);
254  Vdraw->initZeros(*Vmag);
255  FOR_ALL_ELEMENTS_IN_ARRAY3D(*Vdraw)
256  {
257  // Frequency in the coordinate system of the volume
258  VECTOR_R3(idx,j,i,k);
259  FFT_idx2digfreq(*V,idx,freq);
260 
261  // Frequency in the coordinate system of the plane
262  freqPos=Epos*freq;
263  freqNeg=Eneg*freq;
264  if (ZZ(freqPos)<0 || ZZ(freqNeg)>0)
265  (*Vdraw)(k,i,j)+=1;
266  }

325 // Run ---------------------------------------------------------------------
326 void ProgDetectMissingWedge::run()
327 {
328  produceSideInfo();
329  FileName fn_root=V.name().withoutExtension();
330 
331  MultidimArray<double> * mdaV = &MULTIDIM_ARRAY(V);
332 
333  // Detect one of the planes
334  lookForPlane(mdaV, Vmag, maxFreq, planeWidth, 1, rotPos, tiltPos);
335 
336  Image<double> * Vdraw = new Image<double>();
337 
338  if (saveMarks)
339  {
340  (*Vdraw)()=(*Vmag);
341 
342  evaluatePlane(rotPos, tiltPos, mdaV, Vmag, maxFreq, planeWidth,
343  1, &(*Vdraw)());
344  }
345 
346  // Detect the other plane
347  lookForPlane(mdaV, Vmag, maxFreq, planeWidth, -1, rotNeg, tiltNeg, true, rotPos, tiltPos);
348 
349  if (saveMarks)
350  {
351  evaluatePlane(rotNeg, tiltNeg, mdaV, Vmag, maxFreq, planeWidth,
352  -1, &(*Vdraw)());
353  Vdraw->write(fn_root+"_marks.vol");
354  }
355 
356  if (saveMask)
357  {
358  Vdraw->clear();
359  drawWedge(rotPos, tiltPos, rotNeg, tiltNeg, mdaV, Vmag, &(*Vdraw)());
360  Vdraw->write(fn_root+"_mask.vol");
361  }
362 
363  std::cout << "Plane1: " << rotPos << " " << tiltPos << std::endl;
364  std::cout << "Plane2: " << rotNeg << " " << tiltNeg << std::endl;
365 
366  delete Vdraw;
367  delete Vmag;
368 }
