123  void produceSideInfo(int rank)
124  {
125  fnStack = fnRoot + ".stk";
126 
127  if (rank == 0)
128  fnStack.deleteFile();

130  // Read input selfile and reference
131  MetaData SF;
132  SF.read(fnSel);
133  FileName fnImg;
134  toClassify.reserve(SF.size());
135  FileName fnImageStack;
136  CorrelationAux aux;
137  RotationalCorrelationAux aux2;
138  MDRow row;
139  bool thereIsClassCount=SF.containsLabel(MDL_CLASS_COUNT);
140  bool firstSelected=false;
141  FOR_ALL_OBJECTS_IN_METADATA(SF)
142  {
143  SF.getRow(row,__iter.objId);
144  bool proceed=true;
145  if (thereIsClassCount)
146  {
147  size_t classCount;
148  row.getValue(MDL_CLASS_COUNT,classCount);
149  if (classCount==0)
150  proceed=false;
151  }
152  if (proceed)
153  {
154  toClassify.push_back(row);
155  row.getValue(MDL_IMAGE,fnImg);
156  if (!firstSelected)
157  {
158  row.getValue(MDL_IMAGE,fnImg);
159  lastImage.read(fnImg);
160  if (center)
161  centerImage(lastImage(),aux,aux2);
162  if (rank==0)
163  lastImage.write(fnStack,0,true,WRITE_APPEND);
164  fnImageStack.compose(1,fnStack);
165  if (rank==0)
166  {
167  row.setValue(MDL_IMAGE_ORIGINAL,fnImg);
168  row.setValue(MDL_MAXCC,1.0);
169  SFout.addRow(row);
170  }
171  firstSelected=true;
172  }
173  }
174  }
175  if (toClassify.size()==0)
176  return;
177  stillToDo.resizeNoCopy(toClassify.size());
178  stillToDo.initConstant(1);
179  VEC_ELEM(stillToDo,0)=0;
180 
181  // Prepare mask
182  mask.resize(lastImage());
183  mask.setXmippOrigin();
184  BinaryCircularMask(mask,XSIZE(lastImage())/2, INNER_MASK);
185  }

188  void chooseNextImage(int rank, int Nproc)
189  {
190  Image<double> bestImage, I;
191  Matrix2D<double> M;
192  double bestCorr=-1;
193  int bestIdx=-1;
194  int count=0;
195  FileName fnImageStack, fnImg;
196  AlignmentAux aux;
197  CorrelationAux aux2;
198  RotationalCorrelationAux aux3;
199  FOR_ALL_ELEMENTS_IN_MATRIX1D(stillToDo)
200  {
201  if (!VEC_ELEM(stillToDo,i))
202  continue;
203  if ((count+1)%Nproc!=rank)
204  {
205  ++count;
206  continue;
207  }
208  toClassify[i].getValue(MDL_IMAGE,fnImg);
209  I.read(fnImg);
210  I().setXmippOrigin();
211  double corr=alignImagesConsideringMirrors(lastImage(),I(),M,aux,aux2,aux3,&mask);
212  if (center)
213  centerImage(I(),aux2,aux3);
214  if (corr>bestCorr)
215  {
216  bestCorr=corr;
217  bestImage()=I();
218  bestIdx=i;
219  }
220  ++count;
221  }
222 
223  // Rank 0 receives from the other nodes their best image
224  double buffer[2];
225  if (rank==0)
226  {
227  MPI_Status status;
228  for (int n=1; n<Nproc; ++n)
229  {
230  MPI_Recv(buffer, 2, MPI_DOUBLE, n, 0, MPI_COMM_WORLD, &status);
231  if (buffer[1]>bestCorr)
232  {
233  bestIdx=(int)buffer[0];
234  bestCorr=buffer[1];
235  }
236  }
237  std::cout << "Images to go=" << stillToDo.sum()-1 << " current correlation= " << bestCorr << std::endl;
238  }
239  else
240  {
241  buffer[0]=bestIdx;
242  buffer[1]=bestCorr;
243  MPI_Send(buffer, 2, MPI_DOUBLE, 0, 0, MPI_COMM_WORLD);
244  }
245 
246  // Now rank 0 redistributes the best image
247  if (rank==0)
248  {
249  buffer[0]=bestIdx;
250  MPI_Bcast(buffer, 1, MPI_DOUBLE, 0, MPI_COMM_WORLD);
251  }
252  else
253  {
254  MPI_Bcast(buffer, 1, MPI_DOUBLE, 0, MPI_COMM_WORLD);
255  bestIdx=(int)buffer[0];
256  }
257 
258  // All compute the best image
259  toClassify[bestIdx].getValue(MDL_IMAGE,fnImg);
260  I.read(fnImg);
261  I().setXmippOrigin();
262  bestCorr=alignImagesConsideringMirrors(lastImage(),I(),M,aux,aux2,aux3,&mask);
263  if (center)
264  centerImage(I(),aux2,aux3);
265  lastImage()=bestImage()=I();
266 
267  if (rank==0)
268  {
269  int idxStack=SFout.size();
270  fnImageStack.compose(idxStack+1,fnStack);
271  bestImage.write(fnStack,idxStack,true,WRITE_APPEND);
272  MDRow &row=toClassify[bestIdx];
273  FileName fnImgOrig;
274  row.getValue(MDL_IMAGE,fnImgOrig);
275  row.setValue(MDL_IMAGE_ORIGINAL,fnImgOrig);
276  row.setValue(MDL_IMAGE,fnImageStack);
277  row.setValue(MDL_MAXCC,bestCorr);
278  SFout.addRow(row);
279  }
280  VEC_ELEM(stillToDo,bestIdx)=0;
281  }
