182  void
183  transposeInPlace(MultidimArray<T> &multidimArray, const std::array<int,4> &order)
184  {
185  // Creating new multidim array of the same size than the original
186  const std::array<size_t,4> sizes = {
187  NSIZE(multidimArray),
188  ZSIZE(multidimArray),
189  YSIZE(multidimArray),
190  XSIZE(multidimArray)
191  };
192  MultidimArray<T> result(
193  sizes[order[0]],
194  sizes[order[1]],
195  sizes[order[2]],
196  sizes[order[3]]
197  );
198 
199  // Performing transposition in a loop for every dimension
200  for (size_t n = 0; n < NSIZE(multidimArray); n++) {
201  for (size_t z = 0; z < ZSIZE(multidimArray); z++) {
202  for (size_t y = 0; y < YSIZE(multidimArray); y++) {
203  for (size_t x = 0; x < XSIZE(multidimArray); x++) {
204  // Defining array to access with the axis orders
205  const std::array<size_t,4> indices = {n, z, y, x};
206 
207  // Transposing element
208  DIRECT_NZYX_ELEM(result, indices[order[0]], indices[order[1]], indices[order[2]], indices[order[3]]) =
209  DIRECT_NZYX_ELEM(multidimArray, n, z, y, x);
210  }
211  }
212  }
213  }
214 
215  // Remapping pointers from original multidim array to transposed one
216  multidimArray = std::move(result);
217  }

501  CW_CONVERT) const

793  mirrorY(void)
794  {
795  T aux(0);
796  size_t Z, Y, X, N, Y2;
797 
798  X = XSIZE(data);
799  Y = YSIZE(data);
800  Z = ZSIZE(data);
801  N = NSIZE(data);
802  Y2 = Y / 2;
803  Y--;
804  for (size_t l = 0; l < N; ++l)
805  for (size_t k = 0; k < Z; ++k)
806  for (size_t i = 0; i < Y2; ++i)
807  for (size_t j = 0; j < X; ++j)
808  {
809  aux = DIRECT_NZYX_ELEM(data, l, k, i, j);
810  DIRECT_NZYX_ELEM(data, l, k, i, j) = DIRECT_NZYX_ELEM(data, l, k,
811  Y - i, j);
812  DIRECT_NZYX_ELEM(data, l, k, Y-i, j) = aux;
