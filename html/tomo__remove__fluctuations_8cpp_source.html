88 // Denoise image -----------------------------------------------------------
89 void ProgTomoRemoveFluctuations::run()
90 {
91  produceSideInfo();
92 
93  // Filter each line in the series
94  MultidimArray<double> &mV=V();
95  int maxPixel=CEIL(ZSIZE(mV)*maxFreq);
96  FourierTransformer transformer;
97  if (verbose>=1)
98  init_progress_bar(YSIZE(mV));
99  MultidimArray<double> line;
100  MultidimArray< std::complex<double> > lineFourier;
101  std::complex<double> zero=0;
102  line.initZeros(ZSIZE(mV));
103  for (size_t i=0; i<YSIZE(mV); i++)
104  {
105  for (size_t j=0; j<XSIZE(mV); j++)
106  {
107  // Get line
108  for (size_t k=0; k<ZSIZE(mV); k++)
109  DIRECT_A1D_ELEM(line,k)=DIRECT_A3D_ELEM(mV,k,i,j);
110 
111  // Fourier transform
112  transformer.setReal(line);
113  transformer.FourierTransform();
114 
115  // Filter
116  transformer.getFourierAlias(lineFourier);
117  for (size_t k=maxPixel; k<XSIZE(lineFourier); k++)
118  DIRECT_A1D_ELEM(lineFourier,k)=zero;
119 
120  // Inverse Fourier transform and back to the volume
121  transformer.inverseFourierTransform();
122  for (size_t k=0; k<ZSIZE(mV); k++)
123  DIRECT_A3D_ELEM(mV,k,i,j)=DIRECT_A1D_ELEM(line,k);
124  }
125  if (verbose>=1)
126  progress_bar(i+1);
127  }
128  if (verbose>=1)
129  progress_bar(YSIZE(mV));
130 
131  // Write the results
132  mV.setDimensions(XSIZE(mV),YSIZE(mV),1,ZSIZE(mV));
133  V.write(fnOut,ALL_IMAGES,true);
134 }
