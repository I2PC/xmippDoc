58 {

60 
61  //Mask generation
62  Image<double> aux;
63  double bg_mean;

66 #ifdef DEBUG
67 
68  aux.write("PPPmask_no_ero_03.vol");
69 #endif
70 
71  //Mask volume erosion to expand the mask boundaries
72  Vmask.initZeros(V());
73  erode3D(aux(),Vmask, 18,0,2);
74 
75  //Correction of some flaws produced in the edges of the mask volume
76  FOR_ALL_DIRECT_ELEMENTS_IN_ARRAY3D(Vmask)
77  if (k<=4 || i<=4 || j<=4 ||
78  k>=ZSIZE(Vmask)-4 || i>=YSIZE(Vmask)-4 || j>=XSIZE(Vmask)-4)
79  DIRECT_A3D_ELEM(Vmask,k,i,j)=1;
80 
81  aux()=Vmask;
82 #ifdef DEBUG
83 
84  aux.write("PPPmask_ero_03.vol");
85 #endif
86 
87  //Sobel edge detection applied to original volume
88  Image<double> Vedge;
89  computeEdges(V(),Vedge());
90 #ifdef DEBUG
91 
92  Vedge.write("PPPvolume_sobel_unmask_03.vol");
93 #endif
94 
95  //Masked volume generation
96  const MultidimArray<double> &mVedge=Vedge();
97  FOR_ALL_DIRECT_ELEMENTS_IN_MULTIDIMARRAY(mVedge)
98  if (DIRECT_MULTIDIM_ELEM(Vmask,n)==1)
99  DIRECT_MULTIDIM_ELEM(mVedge,n)=0;
100 #ifdef DEBUG
101 
102  Vedge.write("volume_sobel_mask_03.vol");
103 #endif
104 
105  double minval, maxval, avg, stddev;
106 
107  //Invert the mask to meet computeStats_within_binary_mask requirements
108  FOR_ALL_DIRECT_ELEMENTS_IN_MULTIDIMARRAY(Vmask)
109  if (DIRECT_MULTIDIM_ELEM(Vmask,n)==1)
110  DIRECT_MULTIDIM_ELEM(Vmask,n)=0;
111  else
112  DIRECT_MULTIDIM_ELEM(Vmask,n)=1;
113 
114  //Threshold is 3 times the standard deviation of unmasked pixel values
115  double thresh;
116  computeStats_within_binary_mask(Vmask,mVedge,minval, maxval, avg, stddev);
117  thresh=3*stddev;
118 
119  //Final edge volume generated by setting to 1 positions with values > threshold
120  Image<double> Vaux;
121  Vaux().initZeros(mVedge);
122  FOR_ALL_DIRECT_ELEMENTS_IN_MULTIDIMARRAY(mVedge)
123  if (DIRECT_MULTIDIM_ELEM(mVedge,n)>=thresh)
124  DIRECT_MULTIDIM_ELEM(Vaux(),n)=1;
125 
126 #ifdef DEBUG
127 
128  Vaux.write("volumen_bordes_definitivo_03.vol");
129 #endif
130 
131  const MultidimArray<double> &mVaux=Vaux();
132 
133  //Spline coefficient volume from original volume, to allow <1 step sizes
134  MultidimArray<double> Volcoeffs;
135  Volcoeffs.initZeros(V());
136 
137  produceSplineCoefficients(3,Volcoeffs,V());
138 
139  //Width parameter volume initialization
140  Image<double> widths;
141  widths().resizeNoCopy(V());
142  widths().initConstant(1e5);
143  double step=0.25;
144 
145  Matrix1D<double> direction(3);
146 
147  //Calculation of edge width for 10 different directions, if a smaller value is found for a different
148  //direction on a given position the former value is overwritten
149 
150  //Direction (1,0,0)
151  VECTOR_R3(direction,1,0,0);
152  edgeWidth(Volcoeffs, mVaux, widths(), direction, step);
153 
154  //Direction (0,1,0)
155  VECTOR_R3(direction,0,1,0);
156  edgeWidth(Volcoeffs, mVaux, widths(), direction, step);
157 
158  //Direction (0,0,1)
159  VECTOR_R3(direction,0,0,1);
160  edgeWidth(Volcoeffs, mVaux, widths(), direction, step);
161 
162  //Direction (1,1,0)
163  VECTOR_R3(direction,(1/sqrt(2)),(1/sqrt(2)),0);
164  edgeWidth(Volcoeffs, mVaux, widths(), direction, step);
165 
166  //Direction (1,0,1)
167  VECTOR_R3(direction,(1/sqrt(2)),0,(1/sqrt(2)));
168  edgeWidth(Volcoeffs, mVaux, widths(), direction, step);
169 
170  //Direction (0,1,1)
171  VECTOR_R3(direction,0,(1/sqrt(2)),(1/sqrt(2)));
172  edgeWidth(Volcoeffs, mVaux, widths(), direction, step);
173 
174  //Direction (1,1,1)
175  VECTOR_R3(direction,(1/sqrt(3)),(1/sqrt(3)),(1/sqrt(3)));
176  edgeWidth(Volcoeffs, mVaux, widths(), direction, step);
177 
178  //Direction (-1,1,1)
179  VECTOR_R3(direction,-(1/sqrt(3)),(1/sqrt(3)),(1/sqrt(3)));

181

187  VECTOR_R3(direction,(1/sqrt(3)),-(1/sqrt(3)),(1/sqrt(3)));

189 
190 #ifdef DEBUG
191 
192  std::cout << "width stats: ";
193  widths().printStats();
194  std::cout << std::endl;
195  widths.write("PPPwidths.vol");
196 #endif
197

199  std::cout << "Resolution ibw= " << ibw << std::endl;
200  if (fnOut!="")
201  widths.write(fnOut);
202 }

204 void ProgResolutionIBW::edgeWidth(const MultidimArray<double> &volCoeffs,
205  const MultidimArray<double> &edges,
206  MultidimArray <double>& widths, const Matrix1D<double> &dir,
207  double step) const
208 {
209  double forward_count, backward_count, slope;
210  Matrix1D<double> pos_aux_fw(3), pos_aux_bw(3), pos(3), pos_aux(3), next_pos(3), Kdir;
211 
212  Kdir=step*dir;
213 
214  //Visit all elements in volume
215  FOR_ALL_ELEMENTS_IN_ARRAY3D(edges)
216  {
217  //Check for border pixels
218  if (A3D_ELEM(edges,k,i,j)!=0)
219  {
220  //reset all counters
221  forward_count=0;
222  backward_count=0;
223  VECTOR_R3(pos_aux_fw,j,i,k);
224  pos_aux_bw=pos=pos_aux_fw;
225 
226  //find out if pixel magnitude grows or decreases
227  pos_aux=pos;
228  pos_aux+=dir;
229  double value_plus_dir=volCoeffs.interpolatedElementBSpline3D(XX(pos_aux),YY(pos_aux),ZZ(pos_aux));
230 
231  pos_aux=pos;
232  pos_aux-=dir;
233  double value_minus_dir=volCoeffs.interpolatedElementBSpline3D(XX(pos_aux),YY(pos_aux),ZZ(pos_aux));
234 
235  slope=value_plus_dir-value_minus_dir;
236 
237  double sign;
238  if (slope>0)
239  sign=1;
240  else
241  sign=-1;
242 
243  //current_pixel is multiplied by the sign, so only one condition is enough to detect an
244  //extremum no matter if the pixel values increase or decrease
245  double current_pixel=sign*volCoeffs.interpolatedElementBSpline3D
246  (XX(pos_aux_fw),YY(pos_aux_fw),ZZ(pos_aux_fw));
247 
248  double next_pixel;
249  bool not_found;
250 
251  //Search for local extremum ahead of the edge in the given direction
252  do
253  {
254  not_found=true;
255  next_pos=pos_aux_fw+Kdir;
256  next_pixel=sign*volCoeffs.interpolatedElementBSpline3D
257  (XX(next_pos),YY(next_pos),ZZ(next_pos));
258 
259  if(next_pixel>current_pixel)
260  {
261  current_pixel=next_pixel;
262  pos_aux_fw=next_pos;
263  forward_count++;
264  }
265  else
266  {
267  not_found=false;
268  }
269  }
270  while(not_found);
271 
272  current_pixel=sign*volCoeffs.interpolatedElementBSpline3D
273  (XX(pos_aux_bw),YY(pos_aux_bw),ZZ(pos_aux_bw));
274 
275  //Search for local extremum behind of the edge in the given direction
276  do
277  {
278  not_found=true;
279  next_pos=pos_aux_bw-Kdir;
280  next_pixel=sign*volCoeffs.interpolatedElementBSpline3D
281  (XX(next_pos),YY(next_pos),ZZ(next_pos));
282 
283  if(next_pixel<current_pixel)
284  {
285  current_pixel=next_pixel;
286  pos_aux_bw=next_pos;
287  backward_count++;
288  }
289  else
290  {
291  not_found=false;
292  }
293  }
294  while(not_found);
295 
296  //If the width found for this position is smaller than the one stores in edges volume
297  //before it is overwritten
298  if ((forward_count+backward_count)<A3D_ELEM(widths,k,i,j))
299  {
300  A3D_ELEM(widths,k,i,j)=forward_count+backward_count;
301  }
302  }
303  }
304 }

306 double ProgResolutionIBW::calculateIBW(MultidimArray <double>& widths) const
307 {
308  double total, count;
309  total=count=0;
310 
311  FOR_ALL_ELEMENTS_IN_ARRAY3D(widths)
312  {
313  double width=A3D_ELEM(widths,k,i,j);
314  if (width<1e4)
315  {
316  total+=width;
317  count++;
318  }
319  }
320  double avg = total/count;
