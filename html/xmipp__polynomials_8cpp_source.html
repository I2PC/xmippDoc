38 void PolyZernikes::create(const Matrix1D<int> & coef)
39 {
40  Matrix2D<int> * fMatT;
41 
42  int nMax=(int)VEC_XSIZE(coef);
43  for (int nZ = 0; nZ < nMax; ++nZ)
44  {
45  if (VEC_ELEM(coef,nZ) == 0)
46  {
47  fMatT = new Matrix2D<int>(1,1);
48  dMij(*fMatT,0,0)=0;
49  //*fMatT = 0;
50  }
51  else
52  {
53  // Note that the paper starts in n=1 and we start in n=0
54  int n = (size_t)ZERNIKE_ORDER(nZ);
55  int l = 2*nZ-n*(n+2);
56 
57  Matrix2D<int> testM(n+1,n+1);
58  fMatT = new Matrix2D<int>(n+1,n+1);
59 
60  int p = (l>0);
61  int q = ((n % 2 != 0) ? (abs(l)-1)/2 : (( l > 0 ) ? abs(l)/2-1 : abs(l)/2 ) );
62  l = abs(l); //We want the positive value of l
63  int m = (n-l)/2;
64 
65  for (int i = 0; i <= q; ++i)
66  {
67  int K1=binom(l,2*i+p);
68  for (int j = 0; j <= m; ++j)
69  {
70  int factor = ( (i+j)%2 ==0 ) ? 1 : -1 ;
71  int K2=factor * K1 * fact(n-j)/(fact(j)*fact(m-j)*fact(n-m-j));
72  for (int k = 0; k <= (m-j); ++k)
73  {
74  int ypow = 2 * (i+k) + p;
75  int xpow = n - 2 * (i+j+k) - p;
76  dMij(*fMatT,xpow,ypow) +=K2 * binom(m-j,k);
77  }
78  }
79  }
80  }
81 
82  fMatV.push_back(*fMatT);
83  delete fMatT;
84  }
