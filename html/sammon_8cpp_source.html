56 void Sammon::operator()(const In& in, Out& out)
57 {
58  // clean the mapped space
59  out.clear();
60  out.calibrated(in.calibrated());
61  // initialization of mapped space
62  std::random_device rd; //Will be used to obtain a seed for the random number engine
63  std::mt19937 gen(rd()); //Standard mersenne_twister_engine seeded with rd()
64  std::uniform_real_distribution<> dist(-0.5, 0.5);
65  FeatureVector v(mapped);
66  unsigned i;
67  for (i = 0; i < in.size(); i++)
68  {
69  generate(v.begin(), v.end(), [&]{ return dist(gen); });
70  transform(v.begin(), v.end(), v.begin(),
71  bind2nd(std::divides<double>(), norm(v)));
72  out.add(v, in.theTargets[i]);
73  }
74 
75  // calculate distances in original space
76  std::vector<floatFeature> distances(in.size() *(in.size() - 1) / 2);
77  std::vector<floatFeature>::iterator distance = distances.begin();
78  for (i = 1; i < in.size(); i++)
79  for (unsigned j = 0; j < i; j++)
80  *distance++ = max(0.001, euclideanDistance(in.theItems[i], in.theItems[j]));
81 
82  // centroids of mapped samples
83  std::vector<floatFeature> centroid(mapped);
84 
85  // first derivative and second derivative of mapping error
86  std::vector<floatFeature> dE(in.size());
87  std::vector<floatFeature> d2E2(in.size());
88 
89  // copy of the samples for each pattern loop p
90  std::vector<std::vector<floatFeature> > out2(in.size(), std::vector<floatFeature>(mapped));
91  int p;
92  unsigned q;
93 
94  verbosity = listener->getVerbosity();
95  if (verbosity)
96  listener->OnReportOperation((std::string) "mapping....\n");
97  if (verbosity == 1 || verbosity == 3)
98  listener->OnInitOperation(num_iterations);
99 
100 
101  for (unsigned iteration = 0; iteration < num_iterations; iteration++)
102  {
103  // reset centroids
104  fill(centroid.begin(), centroid.end(), 0.0);
105 
106  // for each pattern ...
107  for (p = 0; p < (int)in.size(); p++)
108  {
109  // reset error
110  fill(dE.begin(), dE.end(), 0.0);
111  fill(d2E2.begin(), d2E2.end(), 0.0);
112 
113  for (int j = 0; j < (int)in.size(); j++)
114  if (j != p)
115  {
116  unsigned m;
117  if (j < p)
118  m = p * (p - 1) / 2 + j;
119  else
120  m = j * (j - 1) / 2 + p;
121 
122  // distance between two mapped samples
123  double old_dist = distances[m];
124  double new_dist = euclideanDistance(out.theItems[p], out.theItems[j]);
125  double dist_diff = old_dist - new_dist;
126  double dist_prod = old_dist * new_dist;
127 
128  // calculate delta for each vector in mapped space
129  for (q = 0; q < mapped; q++)
130  {
131  double out_diff = out.theItems[p][q] - out.theItems[j][q];
132 
133  if (dist_prod != 0.0)
134  {
135  // first order derivative
136  dE[q] += (dist_diff / dist_prod) * out_diff;
137  // second order derivative
138  d2E2[q] += (1.0 / dist_prod) * (dist_diff - (out_diff * out_diff / new_dist) * (1.0 + dist_diff / new_dist));
139  }
140  }
141  }
142 
143  // adapt each patterm in the mapped space
144  for (q = 0; q < mapped; q++)
145  // adjust the q-th dimension of the p-th mapped vector
146  centroid[q] += out2[p][q] =
147  out.theItems[p][q] + learning_rate * dE[q] / fabs(d2E2[q]);
148  }
149 
150  // shift the mapped vectors to their centroid
151  for (p = 0; p < (int)out.size(); p++)
152  for (q = 0; q < mapped; q++)
153  out.theItems[p][q] = out2[p][q] - (centroid[q] / out.size());
154 
155  // shift the mapped vectors to their centroid
156  for (p = 0; p < (int)out.size(); p++)
157  for (q = 0; q < mapped; q++)
158  out.theItems[p][q] = out2[p][q] - (centroid[q] / out.size());
159 
160

162  listener->OnProgress(iteration);
163 
164  if ((verbosity > 1) || (iteration == num_iterations - 1))
165  {
166  // Calculates sammon stress (error in distances)
167 
168  stress = 0.;

171  for (int pp = 0; pp < (int)in.size(); pp++)
172  for (int jj = 0; jj < pp; jj++)

176  double ee = d - euclideanDistance(out.theItems[pp], out.theItems[jj]);
177  stress += (ee * ee / d);
178  mutual++;
179  }
