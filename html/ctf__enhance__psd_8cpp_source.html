111 void ProgCTFEnhancePSD::applyFilter(MultidimArray<double> &PSD)
112 {
113  // Take the logarithm
114  FOR_ALL_ELEMENTS_IN_ARRAY2D(PSD)
115  A2D_ELEM(PSD, i, j) = log10(1 + A2D_ELEM(PSD, i, j));
116 
117  // Remove single outliers
118  CenterFFT(PSD, true);
119  MultidimArray<double> aux;
120  medianFilter3x3(PSD, aux);
121  PSD = aux;
122 
123  // Reject other outliers
124  reject_outliers(PSD, 2);
125 
126  // Band pass filter
127  FourierFilter Filter;
128  Filter.FilterShape = RAISED_COSINE;
129  Filter.FilterBand = BANDPASS;
130  Filter.w1 = filter_w1;
131  Filter.w2 = filter_w2;
132  Filter.raised_w = decay_width;
133  PSD.setXmippOrigin();
134  Filter.generateMask(PSD);
135  Filter.applyMaskSpace(PSD);
136  STARTINGX(PSD) = STARTINGY(PSD) = 0;
137  CenterFFT(PSD, false);
138 
139  // Mask the input PSD
140  MultidimArray<int> mask;
141  mask.resize(PSD);
142  Matrix1D<double> freq(2); // Frequencies for Fourier plane
143  double limit0_2 = mask_w1;
144  limit0_2 = limit0_2 * limit0_2;
145  double limitF_2 = mask_w2;
146  limitF_2 = limitF_2 * limitF_2;
147  for (int i = STARTINGY(PSD); i <= FINISHINGY(PSD); ++i)
148  {
149  FFT_IDX2DIGFREQ(i, YSIZE(PSD), YY(freq));
150  double freqy2 = YY(freq) * YY(freq);
151  for (int j = STARTINGX(PSD); j <= FINISHINGX(PSD); ++j)
152  {
153  FFT_IDX2DIGFREQ(j, XSIZE(PSD), XX(freq));
154  double freq2 = XX(freq) * XX(freq) + freqy2;
155  if (freq2 < limit0_2 || freq2 > limitF_2)
156  A2D_ELEM(PSD, i, j) = 0;
157  else
158  A2D_ELEM(mask, i, j) = 1;
159  }
160  }
161 
162  //Compute the mean and the standard deviation under a tighter mask
163  //close to the border and normalize the PSD image
164  MultidimArray<int> tighterMask;
165  tighterMask.resizeNoCopy(PSD);
166  limit0_2 = mask_w2 * 0.9;
167  limit0_2 = limit0_2 * limit0_2;
168  limitF_2 = mask_w2;
169  limitF_2 = limitF_2 * limitF_2;
170  for (int i = STARTINGY(PSD); i <= FINISHINGY(PSD); ++i)
171  {
172  FFT_IDX2DIGFREQ(i, YSIZE(PSD), YY(freq));
173  double freqy2 = YY(freq) * YY(freq);
174  for (int j = STARTINGX(PSD); j <= FINISHINGX(PSD); ++j)
175  {
176  FFT_IDX2DIGFREQ(j, XSIZE(PSD), XX(freq));
177  double freq2 = XX(freq) * XX(freq) + freqy2;
178  A2D_ELEM(tighterMask, i, j) = freq2 > limit0_2 && freq2 < limitF_2;
179  }
180  }
181 
182  double avg, stddev;
183  PSD.computeAvgStdev_within_binary_mask(tighterMask, avg, stddev);
184  double istddev = 1.0 / stddev;
185  FOR_ALL_ELEMENTS_IN_ARRAY2D(PSD)
186  if (A2D_ELEM(mask, i, j))
187  A2D_ELEM(PSD, i, j) = (A2D_ELEM(PSD, i, j) - avg) * istddev;
188 
189  // Mask again
190  limit0_2 = mask_w1;
191  limit0_2 = limit0_2 * limit0_2;
192  limitF_2 = mask_w2 * 0.9;
193  limitF_2 = limitF_2 * limitF_2;
194  for (int i = STARTINGY(PSD); i <= FINISHINGY(PSD); ++i)
195  {
196  FFT_IDX2DIGFREQ(i, YSIZE(PSD), YY(freq));
197  double freqy2 = YY(freq) * YY(freq);
198  for (int j = STARTINGX(PSD); j <= FINISHINGX(PSD); ++j)
199  {
200  FFT_IDX2DIGFREQ(j, XSIZE(PSD), XX(freq));
201  double freq2 = XX(freq) * XX(freq) + freqy2;
202  if (freq2 < limit0_2 || freq2 > limitF_2)
203  A2D_ELEM(PSD, i, j) = 0;
204  }
205  }
206 
207  CenterFFT(PSD, true);
208 }
