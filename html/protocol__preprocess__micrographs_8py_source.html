339  def _insertStepsForMicrograph(self, inputMic, outputMic):
340  self.params['inputMic'] = inputMic
341  self.params['outputMic'] = outputMic
342  self.lastStepId = None
343  self.prerequisites = [] # First operation should not depend on nothing before
344  # Crop
345  self.__insertOneStep(self.doCrop, "xmipp_transform_window",
346  " -i %(inputMic)s --crop %(cropPixels)d -v 0")
347  # Take logarithm
348  self.__insertOneStep(self.doLog, "xmipp_transform_filter",
349  " -i %(inputMic)s --log --fa %(logA)f --fb %(logB)f --fc %(logC)f")
350  # Remove bad pixels
351  self.__insertOneStep(self.doRemoveBadPix, "xmipp_transform_filter",
352  " -i %(inputMic)s --bad_pixels outliers %(stddev)f -v 0")
353  # Invert
354  self.__insertOneStep(self.doInvert, "xmipp_image_operate",
355  "-i %(inputMic)s --mult -1")
356  # Downsample
357  self.__insertOneStep(self.doDownsample, "xmipp_transform_downsample",
358  "-i %(inputMic)s --step %(downFactor)f --method fourier")
359  # Denoise
360  self.__insertOneStep(self.doDenoise, "xmipp_transform_filter",
361  "-i %(inputMic)s --denoiseTV --maxIterTV %(maxIterTV)d")
362  # Smooth
363  self.__insertOneStep(self.doSmooth, "xmipp_transform_filter",
364  "-i %(inputMic)s --fourier real_gaussian %(sigmaConvolution)f")
365  # Highpass
366  self.__insertOneStep(self.doHighPass, "xmipp_transform_filter",
367  "-i %(inputMic)s --fourier high_pass %(highCutoff)f %(highRaised)f")
368  # Lowpass
369  self.__insertOneStep(self.doLowPass, "xmipp_transform_filter",
370  "-i %(inputMic)s --fourier low_pass %(lowCutoff)f %(lowRaised)f")
371  # Normalize
372  self.__insertOneStep(self.doNormalize, "xmipp_transform_normalize",
373  "-i %(inputMic)s --method OldXmipp")
374  return self.lastStepId

376  def __insertOneStep(self, condition, program, arguments):
377  """Insert operation if the condition is met.
378  Possible conditions are: doDownsample, doCrop...etc"""
379  if condition.get():
380  # If the input micrograph and output micrograph differss,
381  # add the -o option
382  if self.params['inputMic'] != self.params['outputMic']:
383  arguments += " -o %(outputMic)s"
384  # Insert the command with the formatted parameters
385  self.lastStepId = self._insertRunJobStep(program, arguments % self.params,
386  prerequisites=self.prerequisites)
387  self.prerequisites = [self.lastStepId] # next should depend on this step
388  # Update inputMic for next step as outputMic
389  self.params['inputMic'] = self.params['outputMic']
