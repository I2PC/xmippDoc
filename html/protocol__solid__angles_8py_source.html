252  def classifyOneGroup(self, projNumber, projMdBlock, projRef,
253  mdClasses, mdImages):
254  """ Classify one of the neighbourhood groups if not empty.
255  Class information will be stored in output metadata: mdOut
256  """
257  blockSize = md.getSize(projMdBlock)
258  Nclasses = self.directionalClasses.get()
259  Nlevels = int(math.ceil(math.log(Nclasses) / math.log(2)))
260 
261  # Skip projection directions with not enough images to
262  # create a given number of classes
263  if blockSize / Nclasses < 10:
264  return
265 
266  fnDir = self._getExtraPath("direction_%s" % projNumber)
267  makePath(fnDir)
268 
269  # Run CL2D classification for the images assigned to one direction
270  args = "-i %s " % projMdBlock
271  args += "--odir %s " % fnDir
272  args += "--ref0 %s --iter %d --nref %d " % (
273  projRef, self.cl2dIterations, Nclasses)
274  args += "--distance correlation --classicalMultiref "
275  args += "--maxShift %f " % self.maxShift
276  try:
277  self.runJob("xmipp_classify_CL2D", args)
278  except:
279  return
280 
281  # After CL2D the stk and xmd files should be produced
282  classesXmd = join(fnDir, "level_%02d/class_classes.xmd" % Nlevels)
283  classesStk = join(fnDir, "level_%02d/class_classes.stk" % Nlevels)
284 
285  # Let's check that the output was produced
286  if not exists(classesStk):
287  return
288 
289  # Run align of the class average and the projection representative
290  fnAlignRoot = join(fnDir, "classes")
291  args = "-i %s " % classesStk
292  args += "--ref %s " % projRef
293  args += " --oroot %s --iter 1" % fnAlignRoot
294  self.runJob("xmipp_image_align", args, numberOfMpi=1)
295 
296  # Apply alignment
297  args = "-i %s_alignment.xmd --apply_transform" % fnAlignRoot
298  self.runJob("xmipp_transform_geometry", args, numberOfMpi=1)
299 
300  for classNo in range(1, Nclasses + 1):
301  localImagesMd = emlib.MetaData("class%06d_images@%s"
302  % (classNo, classesXmd))
303 
304  # New class detected
305  self.classCount += 1
306  # Check which images have not been assigned yet to any class
307  # and assign them to this new class
308  for objId in localImagesMd:
309  imgId = localImagesMd.getValue(emlib.MDL_ITEM_ID, objId)
310  # Add images not classify yet and store their class number
311  if imgId not in self.classImages:
312  self.classImages.add(imgId)
313  newObjId = mdImages.addObject()
314  mdImages.setValue(emlib.MDL_ITEM_ID, imgId, newObjId)
315  mdImages.setValue(emlib.MDL_REF2, self.classCount,
316  newObjId)
317 
318  newClassId = mdClasses.addObject()
319  mdClasses.setValue(emlib.MDL_REF, projNumber, newClassId)
320  mdClasses.setValue(emlib.MDL_REF2, self.classCount, newClassId)
321  mdClasses.setValue(emlib.MDL_IMAGE, "%d@%s" %
322  (classNo, classesStk), newClassId)
323  mdClasses.setValue(emlib.MDL_IMAGE1, projRef, newClassId)
324  mdClasses.setValue(emlib.MDL_CLASS_COUNT, localImagesMd.size(),
325  newClassId)
