175  // Bring both angles to a normalized set
176  rot1 = realWRAP(rot1, -180, 180);
177  tilt1 = realWRAP(tilt1, -180, 180);
178  psi1 = realWRAP(psi1, -180, 180);
179 
180  rot2 = realWRAP(rot2, -180, 180);
181  tilt2 = realWRAP(tilt2, -180, 180);
182  psi2 = realWRAP(psi2, -180, 180);
183 
184  // Apply rotations to find the minimum distance angles

191  angular_distance += distp;
192 
193  // Compute angular difference

195  tilt_diff(i) = tilt1 - tilt2p;
196  psi_diff(i) = psi1 - psi2p;
197  vec_diff(i) = distp;
198  X_diff(i) = X1 - X2;
199  Y_diff(i) = Y1 - Y2;
200  shift_diff(i) = sqrt(X_diff(i)*X_diff(i)+Y_diff(i)*Y_diff(i));
201  shift_distance += shift_diff(i);
202 
203  // Fill the output result
204  if (fillOutput)
205  {
206  //output[0]=rot1;
207  row.setValue(MDL_ANGLE_ROT, rot1);
208  //output[1]=rot2p;
209  row.setValue(MDL_ANGLE_ROT2, rot2p);
210  //output[2]=rot_diff(i);
211  row.setValue(MDL_ANGLE_ROT_DIFF, rot_diff(i));
212  //output[3]=tilt1;
213  row.setValue(MDL_ANGLE_TILT, tilt1);
214  //output[4]=tilt2p;
215  row.setValue(MDL_ANGLE_TILT2, tilt2p);
216  //output[5]=tilt_diff(i);
217  row.setValue(MDL_ANGLE_TILT_DIFF,tilt_diff(i) );
218  //output[6]=psi1;
219  row.setValue(MDL_ANGLE_PSI, psi1);
220  //output[7]=psi2p;
221  row.setValue(MDL_ANGLE_PSI2, psi2);
222  //output[8]=psi_diff(i);
223  row.setValue(MDL_ANGLE_PSI_DIFF, psi_diff(i));
224  //output[9]=distp;
225  if (set==1)
226  row.setValue(MDL_ANGLE_DIFF, distp);
227  else
228  row.setValue(MDL_ANGLE_DIFF2, distp);
229  //output[10]=X1;
230  row.setValue(MDL_SHIFT_X,X1);
231  //output[11]=X2;
232  row.setValue(MDL_SHIFT_X2, X2);
233  //output[12]=X_diff(i);
234  row.setValue(MDL_SHIFT_X_DIFF, X_diff(i));
235  //output[13]=Y1;
236  row.setValue(MDL_SHIFT_Y, Y1);
237  //output[14]=Y2;
238  row.setValue(MDL_SHIFT_Y2, Y2);
239  //output[15]=Y_diff(i);
240  row.setValue(MDL_SHIFT_Y_DIFF, Y_diff(i));
241  //output[16]=shift_diff(i);
242  row.setValue(MDL_SHIFT_DIFF,shift_diff(i));
243 
244  id = DF_out.addRow(row);
245  //id = DF_out.addObject();
246  DF_out.setValue(MDL_IMAGE,fnImg,id);
247  //DF_out.setValue(MDL_ANGLE_COMPARISON,output, id);
248  }
249 
250  i++;
251  }
252  if (0 == i) {
253  REPORT_ERROR(ERR_NUMERICAL, "i is zero (0), which would lead to division by zero");
254  }
255  angular_distance /= i;
256  shift_distance /=i;
257 
258  if (fillOutput)
259  {
260  DF_out.write(fn_out + ".xmd");
261  Histogram1D hist;
262  compute_hist(vec_diff, hist, 0, 180, 180);
263  hist.write(fn_out + "_vec_diff_hist.txt",MDL_ANGLE_DIFF,MDL_COUNT);
264  compute_hist(shift_diff, hist, 20);
265  hist.write(fn_out + "_shift_diff_hist.txt",MDL_SHIFT_DIFF,MDL_COUNT);
266  if (verbose==2)
267  {
268  compute_hist(rot_diff, hist, 100);
269  hist.write(fn_out + "_rot_diff_hist.txt");
270  compute_hist(tilt_diff, hist, 100);
271  hist.write(fn_out + "_tilt_diff_hist.txt");
272  compute_hist(psi_diff, hist, 100);
273  hist.write(fn_out + "_psi_diff_hist.txt");
274  compute_hist(X_diff, hist, 20);
275  hist.write(fn_out + "_X_diff_hist.txt");
276  compute_hist(Y_diff, hist, 20);
277  hist.write(fn_out + "_Y_diff_hist.txt");
278  }
279  }
280 
281  std::cout << "Global angular distance = " << angular_distance << std::endl;
282  std::cout << "Global shift distance = " << shift_distance << std::endl;
283 }

285 void ProgAngularDistance::computeWeights()
286 {
287  MetaData DF1sorted, DF2sorted, DFweights;
288  MDLabel label=MDL::str2Label(idLabel);
289  DF1sorted.sort(DF1,label);
290  DF2sorted.sort(DF2,label);
291  std::vector<MDLabel> labels;
292  labels.push_back(label);
293  labels.push_back(MDL_ANGLE_ROT);
294  labels.push_back(MDL_ANGLE_TILT);
295  labels.push_back(MDL_ANGLE_PSI);
296  labels.push_back(MDL_SHIFT_X);
297  labels.push_back(MDL_SHIFT_Y);
298  labels.push_back(MDL_FLIP);
299  DF1sorted.keepLabels(labels);
300  DF2sorted.keepLabels(labels);
301  MDLabel angleDiffLabel, shiftDiffLabel, weightLabel;
302  switch (set)
303  {
304  case 1:
305  angleDiffLabel=MDL_ANGLE_DIFF;
306  shiftDiffLabel=MDL_SHIFT_DIFF;
307  weightLabel=MDL_WEIGHT_JUMPER;
308  break;
309  case 2:
310  angleDiffLabel=MDL_ANGLE_DIFF2;
311  shiftDiffLabel=MDL_SHIFT_DIFF2;
312  weightLabel=MDL_WEIGHT_JUMPER2;
313  break;
314  case 0:
315  angleDiffLabel=MDL_ANGLE_DIFF0;
316  shiftDiffLabel=MDL_SHIFT_DIFF0;
317  weightLabel=MDL_WEIGHT_JUMPER0;
318  break;
319  }
320 
321  // for(MDIterator __iter(__md); __iter.hasNext(); __iter.moveNext())
322  MDIterator iter1(DF1sorted), iter2(DF2sorted);
323  std::vector< Matrix1D<double> > ang1, ang2;
324  Matrix1D<double> rotTiltPsi(5);
325  size_t currentId;
326  bool anotherImageIn2=iter2.hasNext();
327  size_t id1, id2;
328  bool mirror;
329  while (anotherImageIn2)
330  {
331  ang1.clear();
332  ang2.clear();
333 
334  // Take current id
335  DF2sorted.getValue(label,currentId,iter2.objId);
336 
337  // Grab all the angles in DF2 associated to this id
338  bool anotherIteration=false;
339  do
340  {
341  DF2sorted.getValue(label,id2,iter2.objId);
342  anotherIteration=false;
343  if (id2==currentId)
344  {
345  DF2sorted.getValue(MDL_ANGLE_ROT,XX(rotTiltPsi),iter2.objId);
346  DF2sorted.getValue(MDL_ANGLE_TILT,YY(rotTiltPsi),iter2.objId);
347  DF2sorted.getValue(MDL_ANGLE_PSI,ZZ(rotTiltPsi),iter2.objId);
348  DF2sorted.getValue(MDL_SHIFT_X,VEC_ELEM(rotTiltPsi,3),iter2.objId);
349  DF2sorted.getValue(MDL_SHIFT_Y,VEC_ELEM(rotTiltPsi,4),iter2.objId);
350  DF2sorted.getValue(MDL_FLIP,mirror,iter2.objId);
351  if (mirror)
352  {
353  double rotp, tiltp, psip;
354  Euler_mirrorY(XX(rotTiltPsi),YY(rotTiltPsi),ZZ(rotTiltPsi), rotp, tiltp, psip);
355  XX(rotTiltPsi)=rotp;
356  YY(rotTiltPsi)=tiltp;
357  ZZ(rotTiltPsi)=psip;
358  }
359  ang2.push_back(rotTiltPsi);
360  iter2.moveNext();
361  if (iter2.hasNext())
362  anotherIteration=true;
363  }
364  } while (anotherIteration);
365 
366  // Advance Iter 1 to catch Iter 2
367  double N=0, cumulatedDistance=0, cumulatedDistanceShift=0;
368  size_t newObjId=0;
369  if (iter1.objId>0)
370  {
371  DF1sorted.getValue(label,id1,iter1.objId);
372  while (id1<currentId && iter1.hasNext())
373  {
374  iter1.moveNext();
375  if (iter1.hasNext())
376  DF1sorted.getValue(label,id1,iter1.objId);
377  }
378 
379  // If we are at the end of DF1, then we did not find id1 such that id1==currentId
380  if (!iter1.hasNext())
381  break;
382 
383  // Grab all the angles in DF1 associated to this id
384  anotherIteration=false;
385  do
386  {
387  DF1sorted.getValue(label,id1,iter1.objId);
388  anotherIteration=false;
389  if (id1==currentId)
390  {
391  DF1sorted.getValue(MDL_ANGLE_ROT,XX(rotTiltPsi),iter1.objId);
392  DF1sorted.getValue(MDL_ANGLE_TILT,YY(rotTiltPsi),iter1.objId);
393  DF1sorted.getValue(MDL_ANGLE_PSI,ZZ(rotTiltPsi),iter1.objId);
394  DF1sorted.getValue(MDL_SHIFT_X,VEC_ELEM(rotTiltPsi,3),iter1.objId);
395  DF1sorted.getValue(MDL_SHIFT_Y,VEC_ELEM(rotTiltPsi,4),iter1.objId);
396  DF1sorted.getValue(MDL_FLIP,mirror,iter1.objId);
397  if (mirror)
398  {
399  double rotp, tiltp, psip;
400  Euler_mirrorY(XX(rotTiltPsi),YY(rotTiltPsi),ZZ(rotTiltPsi), rotp, tiltp, psip);
401  XX(rotTiltPsi)=rotp;
402  YY(rotTiltPsi)=tiltp;
403  ZZ(rotTiltPsi)=psip;
404  }
405  ang1.push_back(rotTiltPsi);
406  iter1.moveNext();
407  if (iter1.hasNext())
408  anotherIteration=true;
409  }
410  } while (anotherIteration);
411 
412  // Process both sets of angles
413  cumulatedDistance=0;
414  N=0;
415  for (size_t i=0; i<ang2.size(); ++i)
416  {
417  const Matrix1D<double> &anglesi=ang2[i];
418  double rot2=XX(anglesi);
419  double tilt2=YY(anglesi);
420  double psi2=ZZ(anglesi);
421  double bestDistance=1e38, bestShiftDistance=1e38;
422  for (size_t j=0; j<ang1.size(); ++j)
423  {
424  const Matrix1D<double> &anglesj=ang1[j];
425  double rot1=XX(anglesj);
426  double tilt1=YY(anglesj);
427  double psi1=ZZ(anglesj);
428  double dist = SL.computeDistance(rot1, tilt1, psi1, rot2, tilt2, psi2, false,
429  check_mirrors, object_rotation);
430  if (dist<bestDistance)
431  {
432  bestDistance=dist;
433  bestShiftDistance=fabs(VEC_ELEM(anglesi,3)-VEC_ELEM(anglesj,3))+fabs(VEC_ELEM(anglesi,4)-VEC_ELEM(anglesj,4));
434  }
435  }
436  bestShiftDistance*=0.5;
437  if (bestDistance<360)
438  {
439  cumulatedDistance+=bestDistance;
440  cumulatedDistanceShift+=bestShiftDistance;
441  N++;
442  }
443  }
444  newObjId=DFweights.addObject();
445  DFweights.setValue(label,currentId,newObjId);
446  }
447  else
448  N=0;
449 
450  if (N>0)
451  {
452  double meanDistance=cumulatedDistance/ang2.size();
453  DFweights.setValue(angleDiffLabel,meanDistance,newObjId);
454  double meanDistanceShift=cumulatedDistanceShift/ang2.size();
455  DFweights.setValue(shiftDiffLabel,meanDistanceShift,newObjId);
456  ang1.clear();
457  ang2.clear();
458  }
459  else
460  if (newObjId>0)
461  {
462  DFweights.setValue(angleDiffLabel,-1.0,newObjId);
463  DFweights.setValue(shiftDiffLabel,-1.0,newObjId);
464  ang1.clear();
465  ang2.clear();
466  }
467  anotherImageIn2=iter2.hasNext();
468  }
469  if (ang2.size()>0)
470  {
471  size_t newObjId=DFweights.addObject();
472  DFweights.setValue(label,currentId,newObjId);
473  DFweights.setValue(angleDiffLabel,-1.0,newObjId);
474  DFweights.setValue(shiftDiffLabel,-1.0,newObjId);
475  }
476 
477  // If there are more images in MD1 than in MD2, set the last images to 0
478  while (iter2.hasNext())
479  {
480  size_t newObjId=DFweights.addObject();
481  DFweights.setValue(label,currentId,newObjId);
482  DFweights.setValue(angleDiffLabel,-1.0,newObjId);
483  DFweights.setValue(shiftDiffLabel,-1.0,newObjId);
484  iter2.moveNext();
485  }
486 
487  // Calculate the deviation with respect to angleDiff=0 of the angular distances
488  std::vector<double> angleDistances, shiftDistances;
489  DFweights.getColumnValues(angleDiffLabel,angleDistances);
490  DFweights.getColumnValues(shiftDiffLabel,shiftDistances);
491 
492  double sigma2=0, sigma2D=0, N=0;
493  for (size_t i=0; i<angleDistances.size(); ++i)
494  {
495  double d=angleDistances[i];
496  if (d>0)
497  {
498  sigma2+=d*d;
499  d=shiftDistances[i];
500  sigma2D+=d*d;
501  ++N;
502  }
503  }
504  sigma2/=N;
505  sigma2=std::max(minSigma*minSigma,sigma2);
506  std::cout << "Sigma of angular distances=" << sqrt(sigma2) << std::endl;
507  sigma2D/=N;
508  sigma2D=std::max(minSigmaD*minSigmaD,sigma2D);
509  std::cout << "Sigma of shift distances=" << sqrt(sigma2D) << std::endl;
510 
511  // Adjust the jumper weights according to a Gaussian
512  double isigma2=-0.5/sigma2;
513  double isigma2D=-0.5/sigma2D;
514  FOR_ALL_OBJECTS_IN_METADATA(DFweights)
515  {
516  double d;
517  double weight=1.0;
518  DFweights.getValue(angleDiffLabel,d,__iter.objId);
519  if (d>0)
520  weight=exp(d*d*isigma2);
521  else
522  weight=0.5;
523 
524  if (minSigmaD>0)
525  {
526  DFweights.getValue(shiftDiffLabel,d,__iter.objId);
527  if (d>0)
528  weight*=exp(d*d*isigma2D);
529  else
530  weight*=0.5;
531  }
532  DFweights.setValue(weightLabel,weight,__iter.objId);
533  }
534 
535  // Transfer these weights to the DF2 metadata
536  MetaData DF2weighted;
537  if (DF2.containsLabel(angleDiffLabel))
538  DF2.removeLabel(angleDiffLabel);
539  if (DF2.containsLabel(angleDiffLabel))
540  DF2.removeLabel(angleDiffLabel);
541  if (DF2.containsLabel(weightLabel))
542  DF2.removeLabel(weightLabel);
543  DF2weighted.join1(DF2,DFweights,label,INNER);
544  FOR_ALL_OBJECTS_IN_METADATA(DF2weighted)
545  {
546  double d;
547  DF2weighted.getValue(angleDiffLabel,d,__iter.objId);
548  if (d<0)
549  {
550  // DF2weighted.setValue(MDL_ENABLED,-1,__iter.objId);
551  DF2weighted.setValue(angleDiffLabel,0.0,__iter.objId);
552  }
553  DF2weighted.getValue(shiftDiffLabel,d,__iter.objId);
554  if (d<0)
555  {
556  // DF2weighted.setValue(MDL_ENABLED,-1,__iter.objId);
557  DF2weighted.setValue(shiftDiffLabel,0.0,__iter.objId);
558  }
