99 BasicMemManager::Record *BasicMemManager::find(size_t bytes, MemType type)
100 {
101  Record *res = nullptr;
102  const auto maxBytes = static_cast<size_t>(bytes * 1.1f); // if we overflow size_t, too bad ...
103  for (auto &b : memoryBlocks)
104  {
105  if (b.available && (type == b.type))
106  {
107  if (b.bytes == bytes)
108  {
109  return &b;
110  }
111  else if ((b.bytes >= bytes) && (b.bytes <= maxBytes))
112  {
113  if ((nullptr == res) || (res->bytes > b.bytes))
114  {
115  res = &b;
116  }
117  }
118  }
119  }
120  return res;
121 }
