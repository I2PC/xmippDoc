171 void ProgAdjustVolume::apply(MultidimArray<float> &out)
172 {
173  // Compute the average power and average value of all the projections
174  double sum = 0, sum2 = 0, N = 0;
175  std::cerr << "Computing first estimate of the linear transformation ...\n";
176  int imgno = SF.size();
177  init_progress_bar(imgno);
178  int i = 0;
179 
180  int projXdim, projYdim;
181  if(SF.isEmpty())
182  {
183  std::cerr << "Empty inputFile File\n";
184  exit(1);
185  }
186  Image<double> I;
187  FOR_ALL_OBJECTS_IN_METADATA(SF)
188  {
189  // Read image
190  I.readApplyGeo(SF,__iter.objId);
191  projXdim = XSIZE(I());
192  projYdim = YSIZE(I());
193 
194  // Compute the image statistics
195  double avg=0., stddev=0., min, max;
196  I().computeStats(avg, stddev, min, max);
197  double Ni = projXdim * projYdim;
198  sum += avg;
199  sum2 += stddev * stddev;
200  N += Ni;
201 
202  // End of loop
203  i++;
204  if (i % 10 == 0)
205  progress_bar(i);
206  }
207  progress_bar(imgno);
208  std::cout << std::endl;
209 
210  // Statistics of the volume
211  double avg0=0., stddev0=0., min0, max0;
212  V.computeStats(avg0, stddev0, min0, max0);
213 
214  // First guess of the transformation parameters a*(x-vm)+b
215  // r is the average length of a ray
216  // The equations to solve are
217  // v_i follows currently N(avg0,stddev0)
218  // we want it to follow N(avgF,stddevF)
219  // such that sum_{i=1}^r{v_i} follows N(avg_pict,stddev_pict)
220  // On the other hand we know that
221  // sum_{i=1}^r{v_i} follows N(r avgF, sqrt(r)*stddevF)
222  // Thus,
223  // avgF=avg_pict/r
224  // stddevF=stddev_pict/sqrt(r)
225  // the equations of the transformation are
226  // y=ax+b
227  // avg_y =a*avg_x+b=avgF
228  // stddev_y =a*stddev_y=stddevF
229  // Therefore
230  // a=stddevF/stddev0
231  // b=avgF-a*avg0
232  double r = pow(MULTIDIM_SIZE(V), 1.0 / 3.0);
233  double avg_pict = sum / imgno;
234  double stddev_pict = sqrt(sum2 / imgno);
235  double avgF = avg_pict / r;
236  double stddevF = stddev_pict / sqrt(r);
237  double a = stddevF / stddev0;
238  double b = avgF - a * avg0;
239  std::cout << "First Linear transformation: y=" << a << "*x+" << b << std::endl;
240 
241  // Optimize
242  if (optimize)
243  {
244  Matrix1D<double> p(2), steps(2);
245  p(0) = a;
246  p(1) = b;
247  steps.initConstant(1);
248  double ftol = 0.01, fret;
249  int iter;
250  globalAdjustVolumeProg = this;
251  powellOptimizer(p, 1, 2, &projectionMismatching, NULL,
252  ftol, fret, iter, steps, true);
253  a = p(0);
254  b = p(1);
255  }
256 
257  // Apply the transformation
258 // out = V;
259  FOR_ALL_ELEMENTS_IN_ARRAY3D(V) out(k, i, j) = a * V(k, i, j) + b;
260 }
