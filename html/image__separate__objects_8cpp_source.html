65  void run()
66  {

69  Image<double> I, label;
70  I.read(fn_in);
71  int object_no;
72  if (ZSIZE(I())==1)
73  object_no=labelImage2D(I(), label());
74  else
75  object_no=labelImage3D(I(), label());

77  {
78  I() = label();
79  MultidimArray<double> &Im=MULTIDIM_ARRAY(I);
80  FOR_ALL_DIRECT_ELEMENTS_IN_MULTIDIMARRAY(Im)
81  {
82  DIRECT_MULTIDIM_ELEM(Im,n) = (DIRECT_MULTIDIM_ELEM(Im,n) == o);
83  if (invert)
84  DIRECT_MULTIDIM_ELEM(Im,n) = 1 - DIRECT_MULTIDIM_ELEM(Im,n);
85  }
86  double number_elements = I().sum();
