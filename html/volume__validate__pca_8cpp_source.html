172 void ProgVolumeValidationPCA::evaluate()
173 {
174  MetaData md, md2;
175  MDRow row;
176  FileName imag;
177  double rot, tilt, psi, rot2, tilt2, psi2;
178  std::vector<double> error;
179  size_t id, id2;
180 
181  md.read(fnClasses);
182  std::stringstream ss;
183 
184  MDMultiQuery query;
185  MetaData auxMetadata;
186  double distance = 0;
187 
188  FOR_ALL_OBJECTS_IN_METADATA(md)
189  {
190  id = __iter.objId;
191 
192  md.getRow(row, id);
193  row.getValue(MDL_IMAGE, imag);
194  row.getValue(MDL_ANGLE_ROT, rot);
195  row.getValue(MDL_ANGLE_TILT,tilt);
196  row.getValue(MDL_ANGLE_PSI,psi);
197 
198  for (int index=0; index<NVols;index++)
199  {
200  ss << index;
201  fnAngles=fnClasses.removeAllExtensions()+ss.str();
202  fnAngles+=".xmd";
203  ss.str(std::string());
204  md2.read(fnAngles);
205  MDValueEQ eq(MDL_IMAGE, imag);
206  query.addAndQuery(eq);
207  auxMetadata.importObjects(md2, eq);
208 
209  for(MDIterator __iter2(auxMetadata); __iter2.hasNext(); __iter2.moveNext())
210  {
211  id2 = __iter2.objId;
212 
213  auxMetadata.getRow(row, id2);
214  row.getValue(MDL_ANGLE_ROT, rot2);
215  row.getValue(MDL_ANGLE_TILT,tilt2);
216  row.getValue(MDL_ANGLE_PSI,psi2);
217  distance = Euler_distanceBetweenAngleSets(rot,tilt, psi,rot2, tilt2, psi2, true);
218  error.push_back(distance);
219  }
220  }
221  }
222 
223  double sum = std::accumulate(error.begin(), error.end(), 0.0);
224  double mean = sum / error.size();
225  double sq_sum = std::inner_product(error.begin(), error.end(), error.begin(), 0.0);
226  double stdev = std::sqrt(sq_sum / error.size() - mean * mean);
227 
228  std::cout << std::endl;
229  std::cout << "stdev of errors : " << stdev << std::endl;
230  std::cout << "mean of errors : " << mean << std::endl;
231 
232 }
