44 template <typename T>
45 T PhantomMovie<T>::bilinearInterpolation(const MultidimArray<T> &src, float x, float y) const
46 {
47  auto x_center = static_cast<float>(src.xdim) / 2.f;
48  auto y_center = static_cast<float>(src.ydim) / 2.f;
49  const auto xdim = static_cast<float>(src.xdim);
50  x += x_center;
51  y += y_center;
52  auto xf = std::floor(x);
53  auto xc = std::ceil(x);
54  auto yf = std::floor(y);
55  auto yc = std::ceil(y);
56  auto xw = x - xf;
57  auto yw = y - yf;
58  auto vff = src.data[static_cast<size_t>(xdim * yf + xf)];
59  auto vfc = src.data[static_cast<size_t>(xdim * yc + xf)];
60  auto vcf = src.data[static_cast<size_t>(xdim * yf + xc)];
61  auto vcc = src.data[static_cast<size_t>(xdim * yc + xc)];
62  return vff * (1.f - xw) * (1.f - yw) + vcf * xw * (1.f - yw) + vfc * (1.f - xw) * yw + vcc * xw * yw;
63 }

65 template <typename T>
66 void PhantomMovie<T>::displace(float &x, float &y, size_t n) const
67 {
68  auto x_shift = options.skipShift ? 0 : shiftX(params.req_size.n() - n - 1); // 'reverse' the order (see doc)
69  auto y_shift = options.skipShift ? 0 : shiftY(params.req_size.n() - n - 1); // 'reverse' the order (see doc)
70  if (options.skipBarrel)
71  {
72  x += x_shift;
73  y += y_shift;
74  }
75  else
76  {
77  auto x_center = static_cast<float>(params.req_size.x()) / 2.f;
78  auto y_center = static_cast<float>(params.req_size.y()) / 2.f;
79  auto k1 = dispParams.k1_start + static_cast<float>(n) * (dispParams.k1_end - dispParams.k1_start) / (static_cast<float>(params.req_size.n()) - 1);
80  auto k2 = dispParams.k2_start + static_cast<float>(n) * (dispParams.k2_end - dispParams.k2_start) / (static_cast<float>(params.req_size.n()) - 1);
81  auto y_norm = (y - y_center + (options.shiftAfterBarrel ? 0 : y_shift)) / y_center;
82  auto x_norm = (x - x_center + (options.shiftAfterBarrel ? 0 : x_shift)) / x_center;
83  auto r_out = sqrt(x_norm * x_norm + y_norm * y_norm);
84  auto r_out_2 = r_out * r_out;
85  auto r_out_4 = r_out_2 * r_out_2;
86  auto scale = (1 + k1 * r_out_2 + k2 * r_out_4);
87  x = (x_norm * scale * x_center) + x_center + (options.shiftAfterBarrel ? x_shift : 0);
88  y = (y_norm * scale * y_center) + y_center + (options.shiftAfterBarrel ? y_shift : 0);
89  }
90 }

126 template <typename T>
127 MultidimArray<T> PhantomMovie<T>::findWorkSize() const
128 {
129  auto x_max_shift = 0.f;
130  auto y_max_shift = 0.f;
131  const auto x = static_cast<float>(params.req_size.x());
132  const auto y = static_cast<float>(params.req_size.y());
133  for (size_t n = 0; n < params.req_size.n(); ++n)
134  {
135  auto x_0 = 0.f;
136  auto y_0 = 0.f;
137  auto x_n = x - 1.f;
138  auto y_n = y - 1.f;
139  displace(x_0, y_0, n); // top left corner
140  displace(x_n, y_n, n); // bottom right corner
141  // barrel deformation moves to center, so we need to read outside of the edge - [0,0] will be negative and [n-1,n-1] will be bigger than that
142  auto x_shift = std::max(std::abs(x_0), x_n - x - 1.f);
143  auto y_shift = std::max(std::abs(y_0), y_n - y - 1.f);
144  x_max_shift = std::max(x_max_shift, x_shift);
145  y_max_shift = std::max(y_max_shift, y_shift);
146  }
147  // new size must incorporate 'gaps' on both sides, min values should be negative, max values positive
148  // the shift might not be uniform, so the safer solution is to have the bigger gap on both sides
149  auto x_new = params.req_size.x() + 2 * static_cast<size_t>(std::ceil(x_max_shift));
150  auto y_new = params.req_size.y() + 2 * static_cast<size_t>(std::ceil(y_max_shift));
151  std::cout << "Due to displacement, working with frames of size [" << x_new << ", " << y_new << "%]\n";
152  return MultidimArray<T>(1, 1, static_cast<int>(y_new), static_cast<int>(x_new));
153 }

180 template <typename T>
181 template <bool SKIP_DOSE>
182 void PhantomMovie<T>::generateMovie(const MultidimArray<T> &refFrame) const
183 {
184  MultidimArray<T> frame(1, 1, static_cast<int>(params.req_size.y()), static_cast<int>(params.req_size.x()));
185  params.fn_out.deleteFile();
186  std::mt19937 gen(content.seed);
187 
188  auto genFrame = [&frame, &refFrame, &gen, this](auto n)
189  {
190  float x_center = static_cast<float>(frame.xdim) / 2.f;
191  float y_center = static_cast<float>(frame.ydim) / 2.f;
192  std::cout << "Processing frame " << n << std::endl;
193  for (size_t y = 0; y < params.req_size.y(); ++y) {
194  for (size_t x = 0; x < params.req_size.x(); ++x) {
195  auto x_tmp = static_cast<float>(x);
196  auto y_tmp = static_cast<float>(y);
197  displace(x_tmp, y_tmp, n);
198  // move coordinate system to center - [0, 0] will be in the center of the frame
199  auto val = bilinearInterpolation(refFrame, x_tmp - x_center, y_tmp - y_center);
200  if (!SKIP_DOSE) {
201  auto dist = std::poisson_distribution<int>(val * content.dose);
202  val = dist(gen);
203  }
204  frame.data[y * frame.xdim + x] = val;
205  }
206  }
207  };
208 
209  for (size_t n = 0; n < params.req_size.n(); ++n)
210  {
211  genFrame(n);
212  Image<T> tmp(frame);
213  tmp.write(params.fn_out, n + 1, true, WRITE_REPLACE);
214  }
215 }
