106 template<typename T>
107 T PhantomMovie<T>::bilinearInterpolation(Image<T>& src, T x, T y)
108 {
109  T xf = std::floor(x);
110  T xc = std::ceil(x);
111  T yf = std::floor(y);
112  T yc = std::ceil(y);
113  T xw = x - xf;
114  T yw = y - yf;
115  T vff = getValue(src, xf, yf);
116  T vfc = getValue(src, xf, yc);
117  T vcf = getValue(src, xc, yf);
118  T vcc = getValue(src, xc, yc);
119  return vff * ((T)1 - xw) * ((T)1 - yw)
120  + vcf * xw * ((T)1 - yw)
121  + vfc * ((T)1 - xw) * yw
122  + vcc * xw * yw;
123 }

125 template<typename T>
126 void PhantomMovie<T>::addShiftBarrelDeformation(Image<T> &movie)
127 {
128  // Temporal frame with original data
129  Image<T> tmp = Image<T>(xdim, ydim);
130  T x_center = xdim / (T)2;
131  T y_center = ydim / (T)2;
132  for (size_t n = 0; n < ndim; ++n) {
133  size_t framePixels = xdim * ydim;
134  size_t frameBytes = framePixels * sizeof(T);
135  memcpy(tmp.data.data, &(movie.data[n * framePixels]), frameBytes);
136  T k1 = k1_start + n * (k1_end - k1_start) / (ndim-1);
137  T k2 = k2_start + n * (k2_end - k2_start) / (ndim-1);
138  std::cout << "k1 = " << k1 << " k2 = " << k2 << std::endl;
139  T x_shift = skipShift ? 0 : shiftX(ndim - n - 1); // 'reverse' the order (see doc)
140  T y_shift = skipShift ? 0 : shiftY(ndim - n - 1); // 'reverse' the order (see doc)
141  if (!skipShift) std::cout << "shiftX = " << x_shift << " shiftY = " << y_shift << std::endl;
142  for (size_t y = 0; y < ydim; ++y) {
143  T y_norm = ((T)y - y_center + (shiftAfterBarrel ? 0 : y_shift)) / y_center;
144  for (size_t x = 0; x < xdim; ++x) {
145  T x_norm = ((T)x - x_center + (shiftAfterBarrel ? 0 : x_shift)) / x_center;
146  T r_out = sqrt(x_norm*x_norm + y_norm*y_norm);
147  T r_out_2 = r_out * r_out;
148  T r_out_4 = r_out_2 * r_out_2;
149  T scale = (1 + k1 * r_out_2 + k2 * r_out_4);
150  T x_new = (x_norm * scale * x_center) + x_center
151  + (shiftAfterBarrel ? x_shift : 0);
152  T y_new = (y_norm * scale * y_center) + y_center
153  + (shiftAfterBarrel ? y_shift : 0);
154  size_t index = n * ydim * xdim + y * xdim + x;
155  movie.data[index] = bilinearInterpolation(tmp, x_new, y_new);
156  }
157  }
158  }
