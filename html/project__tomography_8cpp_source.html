51 void ProgProjectTomography::run()
52 {
53 
54  Projection proj;
55  MetaDataVec projMD;
56 
57  randomize_random_generator();
58 
59  TomoProjectSideInfo side;
60  side.produceSideInfo(projParam);
61 
62  // Project
63  int expectedNumProjs = FLOOR((projParam.tiltF-projParam.tilt0)/projParam.tiltStep);
64  int numProjs=0;
65 
66  std::cerr << "Projecting ...\n";
67  if (!(projParam.show_angles))
68  init_progress_bar(expectedNumProjs);
69 
70  projMD.setComment("True rot, tilt and psi; rot, tilt, psi, X and Y shifts applied");
71  double tRot,tTilt,tPsi,rot,tilt,psi;
72  FileName fn_proj; // Projection name
73  int idx = 1;
74  size_t objId;
75 
76  for (double angle=projParam.tilt0; angle<=projParam.tiltF; angle+=projParam.tiltStep)
77  {
78  if (projParam.singleProjection)
79  fn_proj = projParam.fnOut;
80  else
81  fn_proj.compose(idx, projParam.fnRoot + ".stk");
82 
83  // Choose Center displacement ........................................
84  double shiftX = rnd_gaus(projParam.Ncenter_avg, projParam.Ncenter_dev);
85  double shiftY = rnd_gaus(projParam.Ncenter_avg, projParam.Ncenter_dev);
86  Matrix1D<double> inPlaneShift(3);
87  VECTOR_R3(inPlaneShift,shiftX,shiftY,0);
88 
89  projParam.calculateProjectionAngles(proj,angle, 0,inPlaneShift);
90 
91  // Really project ....................................................
92  if (!projParam.only_create_angles)
93  projectVolumeOffCentered(side.phantomVol(), proj,
94  projParam.proj_Ydim, projParam.proj_Xdim);
95 
96  // Add noise in angles and voxels ....................................
97  proj.getEulerAngles(tRot, tTilt,tPsi);
98 
99  rot = tRot + rnd_gaus(projParam.Nangle_avg, projParam.Nangle_dev);
100  tilt = tTilt + rnd_gaus(projParam.Nangle_avg, projParam.Nangle_dev);
101  psi = tPsi + rnd_gaus(projParam.Nangle_avg, projParam.Nangle_dev);
102 
103  proj.setEulerAngles(rot,tilt,psi);
104 
105  objId = projMD.addObject();
106  if (!projParam.only_create_angles)
107  {
108  proj.write(fn_proj, ALL_IMAGES, !projParam.singleProjection, WRITE_REPLACE);
109  projMD.setValue(MDL_IMAGE,fn_proj,objId);
110  }
111 
112  projMD.setValue(MDL_ANGLE_ROT,rot,objId);
113  projMD.setValue(MDL_ANGLE_TILT,tilt,objId);
114  projMD.setValue(MDL_ANGLE_PSI,psi,objId);
115  projMD.setValue(MDL_ANGLE_ROT2,tRot,objId);
116  projMD.setValue(MDL_ANGLE_TILT2,tTilt,objId);
117  projMD.setValue(MDL_ANGLE_PSI2,tPsi,objId);
118  projMD.setValue(MDL_SHIFT_X,shiftX,objId);
119  projMD.setValue(MDL_SHIFT_Y,shiftY,objId);
120 
121  IMGMATRIX(proj).addNoise(projParam.Npixel_avg, projParam.Npixel_dev, "gaussian");
122 
123  // Save ..............................................................
124  if (projParam.show_angles)
125  std::cout << idx << "\t" << tRot << "\t"
126  << tTilt << "\t" << tPsi << std::endl;
127  else if ((expectedNumProjs % XMIPP_MAX(1, numProjs / 60))
128  == 0)
129  progress_bar(numProjs);
130 
131  numProjs++;
132  idx++;
133  }
134  if (!projParam.show_angles)
135  progress_bar(expectedNumProjs);
136 
137  if (!projParam.singleProjection)
138  {
139  projMD.setComment("Angles rot,tilt and psi contain noisy projection angles and rot2,tilt2 and psi2 contain actual projection angles");
140  projMD.write(projParam.fnRoot + ".sel");
141  }
142 
143  return;
144 }
