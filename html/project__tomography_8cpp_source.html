55 void ProgProjectTomography::run()
56 {
57 
58  Projection proj;
59  MetaData projMD;
60 
61  randomize_random_generator();
62 
63  TomoProjectSideInfo side;
64  side.produceSideInfo(projParam);
65 
66  // Project
67  int expectedNumProjs = FLOOR((projParam.tiltF-projParam.tilt0)/projParam.tiltStep);
68  int numProjs=0;
69 
70  std::cerr << "Projecting ...\n";
71  if (!(projParam.show_angles))
72  init_progress_bar(expectedNumProjs);
73 
74  projMD.setComment("True rot, tilt and psi; rot, tilt, psi, X and Y shifts applied");
75  double tRot,tTilt,tPsi,rot,tilt,psi;
76  FileName fn_proj; // Projection name
77  int idx = 1;
78  size_t objId;
79 
80  for (double angle=projParam.tilt0; angle<=projParam.tiltF; angle+=projParam.tiltStep)
81  {
82  if (projParam.singleProjection)
83  fn_proj = projParam.fnOut;
84  else
85  fn_proj.compose(idx, projParam.fnRoot + ".stk");
86 
87  // Choose Center displacement ........................................
88  double shiftX = rnd_gaus(projParam.Ncenter_avg, projParam.Ncenter_dev);
89  double shiftY = rnd_gaus(projParam.Ncenter_avg, projParam.Ncenter_dev);
90  Matrix1D<double> inPlaneShift(3);
91  VECTOR_R3(inPlaneShift,shiftX,shiftY,0);
92 
93  projParam.calculateProjectionAngles(proj,angle, 0,inPlaneShift);
94 
95  // Really project ....................................................
96  if (!projParam.only_create_angles)
97  projectVolumeOffCentered(side.phantomVol(), proj,
98  projParam.proj_Ydim, projParam.proj_Xdim);
99 
100  // Add noise in angles and voxels ....................................
101  proj.getEulerAngles(tRot, tTilt,tPsi);
102 
103  rot = tRot + rnd_gaus(projParam.Nangle_avg, projParam.Nangle_dev);
104  tilt = tTilt + rnd_gaus(projParam.Nangle_avg, projParam.Nangle_dev);
105  psi = tPsi + rnd_gaus(projParam.Nangle_avg, projParam.Nangle_dev);
106 
107  proj.setEulerAngles(rot,tilt,psi);
108 
109  objId = projMD.addObject();
110  if (!projParam.only_create_angles)
111  {
112  proj.write(fn_proj, ALL_IMAGES, !projParam.singleProjection, WRITE_REPLACE);
113  projMD.setValue(MDL_IMAGE,fn_proj,objId);
114  }
115 
116  projMD.setValue(MDL_ANGLE_ROT,rot,objId);
117  projMD.setValue(MDL_ANGLE_TILT,tilt,objId);
118  projMD.setValue(MDL_ANGLE_PSI,psi,objId);
119  projMD.setValue(MDL_ANGLE_ROT2,tRot,objId);
120  projMD.setValue(MDL_ANGLE_TILT2,tTilt,objId);
121  projMD.setValue(MDL_ANGLE_PSI2,tPsi,objId);
122  projMD.setValue(MDL_SHIFT_X,shiftX,objId);
123  projMD.setValue(MDL_SHIFT_Y,shiftY,objId);
124 
125  IMGMATRIX(proj).addNoise(projParam.Npixel_avg, projParam.Npixel_dev, "gaussian");
126 
127  // Save ..............................................................
128  if (projParam.show_angles)
129  std::cout << idx << "\t" << tRot << "\t"
130  << tTilt << "\t" << tPsi << std::endl;
131  else if ((expectedNumProjs % XMIPP_MAX(1, numProjs / 60))
132  == 0)
133  progress_bar(numProjs);
134 
135  numProjs++;
136  idx++;
137  }
138  if (!projParam.show_angles)
139  progress_bar(expectedNumProjs);
140 
141  if (!projParam.singleProjection)
142  {
143  projMD.setComment("Angles rot,tilt and psi contain noisy projection angles and rot2,tilt2 and psi2 contain actual projection angles");
144  projMD.write(projParam.fnRoot + ".sel");
145  }
146 
147  return;
148 }
