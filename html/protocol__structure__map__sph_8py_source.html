35 def mds(d, dimensions=2):
36  """
37  Multidimensional Scaling - Given a matrix of interpoint distances,
38  find a set of low dimensional points that have similar interpoint
39  distances.
40  """
41 
42  (n, n) = d.shape
43  E = (-0.5 * d ** 2)
44 
45  # Use mat to get column and row means to act as column and row means.
46  Er = np.mat(np.mean(E, 1))
47  Es = np.mat(np.mean(E, 0))
48 
49  # From Principles of Multivariate Analysis: A User's Perspective (page 107).
50  F = np.array(E - np.transpose(Er) - Es + np.mean(E))
51 
52  [U, S, V] = np.linalg.svd(F)
53 
54  Y = U * np.sqrt(S)
55 
56  return (Y[:, 0:dimensions], S)
