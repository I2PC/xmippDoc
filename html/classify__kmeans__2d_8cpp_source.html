193  // return ID of nearest center (uses euclidean distance)
194  int getIDNearestCenter(Point point)
195  {
196  double sum = 0.0, min_dist;
197  int id_cluster_center = 0;
198 
199  for (int i = 0; i < total_values; i++)
200  sum += ((clusters[0].getCentralValue(i) - point.getValue(i)) *
201  (clusters[0].getCentralValue(i) - point.getValue(i)));
202 
203  min_dist = sqrt(sum);
204 
205  for (int i = 1; i < K; i++)
206  {
207  double dist;
208  sum = 0.0;
209 
210  for (int j = 0; j < total_values; j++)
211  sum += ((clusters[i].getCentralValue(j) - point.getValue(j)) *
212  (clusters[i].getCentralValue(j) - point.getValue(j)));
213 
214  dist = sqrt(sum);
215 
216  if (dist < min_dist)
217  {
218  min_dist = dist;
219  id_cluster_center = i;
220  }
221  }
222 
223  return id_cluster_center;
224  }

275  }
276  }

278 
279  int iter = 1;
280  while(true)
281  {
282  bool done = true;
283 
284  // when total_points is larger than threshold value maxObjects
285  // we only assign labels to point based on nearest center
286  if ((maxObjects != -1) && (total_points > maxObjects))
287  {
288  for (int i = 0; i < total_points; i++)
289  {
290  int nearest_center = getIDNearestCenter(points[i]);
291  points[i].setCluster(nearest_center);
292  clusters[nearest_center].addPoint(points[i]);
293  }
294  }
295  else // perform k-means clustering
296  {
297  // associates each point to the nearest center
298  for (int i = 0; i < total_points; i++)
299  {
300  int old_cluster = points[i].getCluster();
301  int nearest_center = getIDNearestCenter(points[i]);
302

307 
308  points[i].setCluster(nearest_center);
309  clusters[nearest_center].addPoint(points[i]);
310  done = false;
311  }
312  }
313 
314  // recalculating the center of each cluster
315  for (int i = 0; i < K; i++)
316  {
317  for (int j = 0; j < total_values; j++)

320  double sum = 0.0;
321 
322  if (total_points > 0)
323  {
324  for (int p = 0; p < total_points; p++)
325  sum += clusters[i].getPoint(p).getValue(j);
326 
327  clusters[i].setCentralValue(j, sum / total_points);
328  }
329  }
330  }
331  }
332  if (done == true || iter >= maxIterations) break;
333  iter++;
334  }

413  srand (time(NULL));
414 
415  // reading new images from input file
416  SF.read(fnSel);
417  FOR_ALL_OBJECTS_IN_METADATA(SF)
418  {
419  SF.getValue(MDL_IMAGE, fnImg,__iter.objId);
420  I.read(fnImg);
421  I().setXmippOrigin();
422  centerImageTranslationally(I(), aux);
423  fv.clear();
424  ef.extractEntropy(I(), Imasked(), fv);
425  //ef.extractZernike(I(), fv);
426  //ef.extractLBP(I(), fv);
427  //ef.extractVariance(I(), fv);
428  //ef.extractGranulo(I(), fv);
429  //ef.extractRamp(I(), fv);
430  //ef.extractHistDist(I(), fv);
431  fvs.push_back(fv);
432  }
433 
434  double min_item, max_item;
435  for(int i = 0; i < fv.size(); i++)
436  {
437  fv_temp.clear();
438  for (int j = 0; j < fvs.size(); j++)
439  fv_temp.push_back(fvs[j][i]);
440 
441  max_item = *std::max_element(fv_temp.begin(), fv_temp.end());
442  min_item = *std::min_element(fv_temp.begin(), fv_temp.end());
443  for (int j = 0; j < fvs.size(); j++)
444  fvs[j][i] = ((fvs[j][i] - min_item)) / (max_item - min_item);
445  }
446 
447  int allItems = 0;
448  FOR_ALL_OBJECTS_IN_METADATA(SF)
449  {
450  allItems++;
451  Point p(allItems, fvs.front());
452  points.push_back(p);
453  fvs.erase(fvs.begin());
454  }
455 
456  // preparing all the paths to external files
457  std::size_t extraPath = fnSel.find_last_of("/");
458  fnOut = fnSel.substr(0, extraPath+1) + fnOut.c_str();
459  fnClusters = fnSel.substr(0, extraPath+1) + fnClusters.c_str();
460  fnPoints = fnSel.substr(0, extraPath+1) + fnPoints.c_str();
461  fnallDone = fnSel.substr(0, extraPath+1) + "allDone.xmd";
462 
463  // loading all the stored points from file (their count is unknown here)
464  std::vector<double> fv_load;
465  std::fstream savedPoints(fnPoints.c_str());
466  std::string line;
467  while (savedPoints.good())
468  {
469  std::getline(savedPoints, line);
470  if (line.size() < 2) break;
471  allItems++;
472  std::stringstream ss(line);
473  fv_load.clear();
474  double point_value;
475  for (int j = 0; j < fv.size(); j++)
476  {
477  ss >> point_value;
478  fv_load.push_back(point_value);
479  }
480  Point p(allItems, fv_load);
481  points.push_back(p);
482  }
483 
484  // performing k-means clustering
485  KMeans kmeans(K, allItems, fv.size(), allItems, maxObjects);
486  clusters = kmeans.run(points, fnClusters, fnPoints);
487 
488  // for cycle writing output file
489  for (int i = 0; i < clusters.size(); i++)
490  {
491  size_t total_points_cluster = clusters[i].getTotalPoints();
492 
493  size_t ii = MDsummary.addObject();
494  MDsummary.setValue(MDL_REF, i+1, ii);
495  MDsummary.setValue(MDL_CLASS_COUNT, total_points_cluster, ii);
496 
497  std::ostringstream clusterValues;
498  clusterValues << "[";
499  for (int j = 0; j < fv.size()-1; j++)
500  clusterValues << clusters[i].getCentralValue(j) << ", ";
501  clusterValues << clusters[i].getCentralValue(fv.size()-1) << "]";
502 
503  MDsummary.setValue(MDL_KMEANS2D_CENTROID, clusterValues.str(), ii);
504  MDsummary.write(formatString("classes@%s", fnOut.c_str()), MD_APPEND);
505  MDclass.clear();
506 
507  std::ifstream f(fnallDone.c_str());
508  if (f.good()) MDallDone.read(fnallDone);
509 
510  for (int j = 0; j < total_points_cluster; j++)
511  {
512  MDallDone.getRow(row, clusters[i].getPoint(j).getID());
513  size_t recId = MDclass.addRow(row);
514  MDclass.setValue(MDL_REF, i+1, recId);
515  }
516  MDclass.write(formatString("class%06d_images@%s", i+1,
517  fnOut.c_str()), MD_APPEND);
518  }
