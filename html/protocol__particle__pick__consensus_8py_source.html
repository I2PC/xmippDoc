47 class XmippProtConsensusPicking(ProtParticlePicking):
48  """
49  Protocol to estimate the agreement between different particle picking
50  algorithms. The protocol takes several Sets of Coordinates calculated
51  by different programs and/or different parameter settings. Let's say:
52  we consider N independent pickings. Then, a coordinate is considered
53  to be a correct particle if M pickers have selected the same particle
54  (within a radius in pixels specified in the form).
55 
56  If you want to be very strict, then set M=N; that is, a coordinate
57  represents a particle if it has been selected by all particles (this
58  is the default behaviour). Then you may relax this condition by setting
59  M=N-1, N-2, ...
60 
61  If you want to be very flexible, set M=1, in this way it suffices that
62  1 picker has selected the coordinate to be considered as a particle. Note
63  that in this way, the cleaning of the dataset has to be performed by other
64  means (screen particles, 2D and 3D classification, ...).
65  """

302 def consensusWorker(coords, consensus, consensusRadius, posFn, jaccFn=None,
303  mode=PICK_MODE_LARGER):
304  """ Worker for calculate the consensus of N picking algorithms of
305  M_n coordinates each one.
306 
307  coords: Array of N numpy arrays of M_n coordinates each one.
308  consensus: Minimum number of votes to get a consensus coordinate
309  consensusRadius: Tolerance to see two coordinates as the same (in pixels)
310  posFn: Where to write the consensus coordinates
311  jaccFN: Where to write the Jaccard index per micrograph
312  """
313  if len(coords) == 1: # self consensus (remove duplicates)
314  N0 = 0
315  firstInput = 0
316  else: # in regular consensus all first coords are directly added to allCoords
317  N0 = coords[0].shape[0]
318  firstInput = 1
319 
320  # initializing arrays
321  Ninputs = len(coords)
322  Ncoords = sum([x.shape[0] for x in coords])
323  allCoords = np.zeros([Ncoords, 2])
324  votes = np.zeros(Ncoords)
325 
326  inAllMicrographs = consensus <= 0 or consensus >= Ninputs
327 
328  # if nothing in the first and it should be in all, nothing to do
329  if (not all([coords[idx].shape[0] for idx in range(Ninputs)])
330  and inAllMicrographs):
331  print("Returning from worker: doing AND consensus and, at least, one "
332  "picker is empty for this micrograph (%s)." % posFn)
333  return
334 
335  # Add all the first coordinates to 'allCoords' and 'votes' lists
336  if N0 > 0:
337  allCoords[0:N0, :] = coords[0]
338  votes[0:N0] = 1
339 
340  # Add the rest of coordinates to 'allCoords' and 'votes' lists
341  Ncurrent = N0
342  for n in range(firstInput, Ninputs):
343  for coord in coords[n]:
344  if Ncurrent > 0:
345  dist = np.sum((coord - allCoords[0:Ncurrent]) ** 2, axis=1)
346  imin = np.argmin(dist)
347  if sqrt(dist[imin]) < consensusRadius:
348  newCoord = (votes[imin] * allCoords[imin,] + coord) / (
349  votes[imin] + 1)
350  allCoords[imin,] = newCoord
351  votes[imin] += 1
352  else:
353  allCoords[Ncurrent, :] = coord
354  votes[Ncurrent] = 1
355  Ncurrent += 1
356  else:
357  allCoords[Ncurrent, :] = coord
358  votes[Ncurrent] = 1
359  Ncurrent += 1
360 
361  # Select those in the consensus
362  if consensus <= 0 or consensus > Ninputs:
363  consensus = Ninputs
364  elif not isinstance(consensus, int):
365  consensus = consensus.get()
366 
367  if mode==PICK_MODE_LARGER:
368  consensusCoords = allCoords[votes >= consensus, :]
369  else:
370  consensusCoords = allCoords[votes == consensus, :]
371 
372  try:
373  if jaccFn:
374  jaccardIdx = float(len(consensusCoords)) / (
375  float(len(allCoords)) / Ninputs)
376  # COSS: Possible problem with concurrent writes
377  with open(jaccFn, "a") as fhJaccard:
378  fhJaccard.write("%s \t %f\n" % (posFn, jaccardIdx))
379  except Exception as exc:
380  print("Some error occurred during Jaccard index calculation or "
381  "writing it's file. Maybe a concurrence issue:\n%s" % exc)
382  # Write the consensus file only if there
383  # are some coordinates (size > 0)
384  if consensusCoords.size:
385  np.savetxt(posFn, consensusCoords)
