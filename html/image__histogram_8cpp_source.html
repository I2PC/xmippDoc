97  void run()
98  {
99  MDIterator iter(mdIn);
100  image.readApplyGeo(mdIn, iter.objId);
101  image().setXmippOrigin();
102  double dummy;
103 
104  // Generate mask if necessary
105  if (apply_mask)
106  {
107  mask.generate_mask(image());
108  maskArray = mask.get_binary_mask();
109  if (automatic_range)
110  computeStats_within_binary_mask(maskArray, image(), m, M, dummy, dummy);
111  }
112  else if (automatic_range)
113  image().computeDoubleMinMax(m, M);
114 
115  if (automatic_range)
116  {
117  m *= 1.5;
118  M *= 1.5;
119  }
120 
121  if (apply_mask)
122  compute_hist_within_binary_mask(maskArray, image(), histb, m, M, StepsNo);
123  else
124  compute_hist(image(), histb, m, M, StepsNo);
125 
126  while (iter.moveNext())
127  {
128  image.readApplyGeo(mdIn, iter.objId);
129  image().setXmippOrigin();
130  // Compute histogram ----------------------------------------------------
131  if (apply_mask)
132  compute_hist_within_binary_mask(maskArray, image(), hist, m, M, StepsNo);
133  else
134  compute_hist(image(), hist, m, M, StepsNo);
135  histb += hist;
136  }
137 
138  if (do_normalize)
139  histb *= 1.0 / (histb.sum() * histb.step_size);
140 
141  if (!fn_out.empty())
142  histb.write(fn_out);
143  else
144  std::cout << histb;
145  }
