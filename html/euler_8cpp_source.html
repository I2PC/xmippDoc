152 void Euler::extract(const Matrix2D<double> &M)
153 {
154  int i,j,k;
155  angleOrder(i,j,k);
156  if (_initialRepeated)
157  {
158  // Extract the first angle, x.
159  //
160 
161  //x = Math::atan2(M[j][i], M[k][i]);
162  x = atan2(dMij(M,j,i), dMij(M,k,i));
163 
164  //
165  // Remove the x rotation from M, so that the remaining
166  // rotation, N, is only around two axes, and gimbal lock
167  // cannot occur.
168  //
169 
170  //Vec3<T> r (0, 0, 0);
171  Matrix1D <double> r;
172  r.initZeros(3);
173  VEC_ELEM(r,i) = (_parityEven? -x: x);
174 
175  Matrix2D<double> N(4,4);
176  N.initIdentity();
177  eulerRotate(N,r);
178  N = N * M;
179 
180  //
181  // Extract the other two angles, y and z, from N.
182  //
183 
184  double sy = sqrt (dMij(N,j,i)*dMij(N,j,i) +
185  dMij(N,k,i)*dMij(N,k,i) );
186  y = atan2 (sy, dMij(N,i,i));
187  z = atan2 (dMij(N,j,k),dMij(N,j,j));
188  }
189  else
190  {
191  //
192  // Extract the first angle, x.
193  //
194  // x = Math<T>::atan2 (M[j][k], M[k][k]);
195  x = atan2 (dMij(M,j,k),dMij(M,k,k));
196  //
197  // Remove the x rotation from M, so that the remaining
198  // rotation, N, is only around two axes, and gimbal lock
199  // cannot occur.
200  //
201 
202  Matrix1D <double> r;
203  r.initZeros(3);
204  VEC_ELEM(r,i) = (_parityEven? -x: x);
205  //std::cerr << "DEBUG_ROB, r:" << r << std::endl;
206  Matrix2D<double> N(4,4);
207  N.initIdentity();
208  eulerRotate(N,r);
209  //std::cerr << "DEBUG_ROB, N:" << N << std::endl;
210  N = N * M;
211  //std::cerr << "DEBUG_ROB, NN:" << N << std::endl;
212 
213  //
214  // Extract the other two angles, y and z, from N.
215  //
216  // T cy = Math<T>::sqrt (N[i][i]*N[i][i] + N[i][j]*N[i][j]);
217  double cy = sqrt (dMij(N,i,i)*dMij(N,i,i) +
218  dMij(N,i,j)*dMij(N,i,j) );
219  // y = Math<T>::atan2 (-N[i][k], cy);
220  // z = Math<T>::atan2 (-N[j][i], N[j][j]);
221  y = atan2 (-dMij(N,i,k),cy);
222  z = atan2 (-dMij(N,j,i),dMij(N,j,j));
223 
224  }
225 
226  if (!_parityEven)
227  //*this *= -1;
228  {
229  vec3 *= -1;
230  x *=-1;
231  y *=-1;
232  z *=-1;
233  }
234 
235  if (!_frameStatic)
236  {
237  double t = x;
238  x = z;
239  z = t;
240  }
241

244 void Euler::toMatrix(Matrix2D<double>& M) const
245 {
246  int i,j,k;
247  angleOrder(i,j,k);
248 
249  Matrix1D<double> angles;
250 
251  if ( _frameStatic )
252  angles = vectorR3(x,y,z);
253  else
254  angles=vectorR3(z,y,x);
255 
256  if ( !_parityEven )
257  {
258  angles *= -1.0;
259  }
260 
261  double ci = cos(XX(angles));
262  double cj = cos(YY(angles));
263  double ch = cos(ZZ(angles));
264  double si = sin(XX(angles));
265  double sj = sin(YY(angles));
266  double sh = sin(ZZ(angles));
267 
268  double cc = ci*ch;
269  double cs = ci*sh;
270  double sc = si*ch;
271  double ss = si*sh;
272 
273  M.initIdentity(4);
274 
275  if ( _initialRepeated )
276  {
277  dMij(M,i,i) = cj;
278  dMij(M,j,i) = sj*si;
279  dMij(M,k,i) = sj*ci;
280 
281  dMij(M,i,j) = sj*sh;
282  dMij(M,j,j) = -cj*ss+cc;
283  dMij(M,k,j) = -cj*cs-sc;
284 
285  dMij(M,i,k) = -sj*ch;
286  dMij(M,j,k) = cj*sc+cs;
287  dMij(M,k,k) = cj*cc-ss;
288  }
289  else
290  {
291  dMij(M,i,i) = cj*ch;
292  dMij(M,j,i) = sj*sc-cs;
293  dMij(M,k,i) = sj*cc+ss;
294 
295  dMij(M,i,j) = cj*sh;
296  dMij(M,j,j) = sj*ss+cc;
297  dMij(M,k,j) = sj*cs-sc;
298 
299  dMij(M,i,k) = -sj;
300  dMij(M,j,k) = cj*si;
301  dMij(M,k,k) = cj*ci;
302  }
303 }
