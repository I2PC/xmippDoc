65  void readColumn(MDLabel &column, bool &automatic, double &m, double &M, int &steps, bool second=false)
66  {
67  const char* colStr = second ? "--col2" : "--col";
68  const char* rangeStr = second ? "--range2" : "--range";
69  const char* stepStr = second ? "--steps2" : "--steps";
70 
71  column = MDL::str2Label(getParam(colStr));
72  steps = getIntParam(stepStr);

78  automatic = false;

80 
81  // Check if valid columns where provided
82  if (column == MDL_UNDEFINED)
83  REPORT_ERROR(ERR_MD_UNDEFINED, "Metadata Histogram: Column for histogram not valid");

85  REPORT_ERROR(ERR_MD_BADTYPE, "Metadata Histogram: Column type for histogram should be double");
