375  void precomputeValues(double X)
376  {
377  precomputed.ang = 0;
378  precomputed.u2 = X * X;
379  precomputed.u = sqrt(precomputed.u2);
380  precomputed.u3 = precomputed.u2 * precomputed.u;
381  precomputed.u4 = precomputed.u2 * precomputed.u2;
382  precomputed.u_sqrt = sqrt(precomputed.u);
383 
384  if (fabs(X) < XMIPP_EQUAL_ACCURACY)
385  precomputed.deltaf=0;

392  * in particular term defocus and twofold axial astigmatism
393  * take into account that a1 and a2 are the coefficient
394  * of the zernike polynomials difference of defocus at 0
395  * and at 45 degrees. In this case a2=0
396  */
397  precomputed.deltaf= Defocus;//(defocus_average + defocus_deviation*cos_ellipsoid_ang_2);
