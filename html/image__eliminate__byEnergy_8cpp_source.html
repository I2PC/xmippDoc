62 /* Process image ------------------------------------------------------------- */
63 void ProgEliminateByEnergy::processImage(const FileName &fnImg, const FileName &fnImgOut, const MDRow &rowIn, MDRow &rowOut)
64 {
65  rowOut = rowIn;
66  Image<double> I;
67  I.read(fnImg);
68  double avg, stddev;
69  I().computeAvgStdev(avg,stddev);
70  double sigma2=stddev*stddev;
71 
72  double z=(sigma2/sigma20-1);
73  double zalpha=fabs(icdf_gauss(confidence));
74  if (z>zalpha || sigma2<minSigma2 || !std::isfinite(stddev) || fabs(avg)>sigma20/9.0)
75  rowOut.setValue(MDL_ENABLED,-1);
76  else
77  rowOut.setValue(MDL_ENABLED,1);
78 }
