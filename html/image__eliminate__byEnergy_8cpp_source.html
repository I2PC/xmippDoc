63 /* Process image ------------------------------------------------------------- */
64 void ProgEliminateByEnergy::processImage(const FileName &fnImg, const FileName &fnImgOut, const MDRow &rowIn, MDRow &rowOut)
65 {
66  rowOut = rowIn;
67  Image<double> I;
68  I.read(fnImg);
69  double avg, stddev;
70  I().computeAvgStdev(avg,stddev);
71  double sigma2=stddev*stddev;
72 
73  double z=(sigma2/sigma20-1);
74  double zalpha=fabs(icdf_gauss(confidence));
75  if (z>zalpha || sigma2<minSigma2 || !std::isfinite(stddev) || fabs(avg)>sigma20/9.0)
76  rowOut.setValue(MDL_ENABLED,-1);
77  else
78  rowOut.setValue(MDL_ENABLED,1);
79 }
