66 void ProgPdbReduce::reduceNumberPseudoatoms()
67 {
68  PDBRichPhantom pdb;
69  if(thresh != 0.0)
70  pdb.read(fn_volume, -1.0, thresh);
71  else
72  {
73  pdb.read(fn_volume, num);
74  std::sort(pdb.intensities.rbegin(), pdb.intensities.rend());
75  thresh = pdb.intensities.at(num);
76  pdb.atomList.clear();
77  pdb.remarks.clear();
78  pdb.read(fn_volume, -1.0, thresh);
79  }
80  pdb.write(fn_out);
81 }
