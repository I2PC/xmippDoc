132 void ProgImageRotationalPCA::produceSideInfo()
133 {
134  time_config();
135  MetaData MDin(fnIn);
136 
137  if (maxNimgs > 0)
138  selectPartFromMd(MDin);
139 
140  Nimg = MDin.size();
141  size_t Ydim, Zdim, Ndim;
142  getImageSize(MDin, Xdim, Ydim, Zdim, Ndim);

169  }

171  Matrix2D<double> &W = Wnode[0];
172  FileName fnMatrixF(fnRoot + "_matrixF.raw");
173  FileName fnMatrixH(fnRoot + "_matrixH.raw");
174 
175  if (IS_MASTER)

177  // F (#images*#shifts*#angles) x (#eigenvectors+2)*(its+1)

182  fnMatrixH.createEmptyFileWithGivenLength(
183  Nimg * 2 * Nangles * Nshifts * (Neigen + 2) * sizeof(double));
184

188  }
189

191

193  Nimg * 2 * Nangles * Nshifts);

195

233 void threadApplyT(ThreadArgument &thArg)
234 {
235  ProgImageRotationalPCA *self=(ProgImageRotationalPCA *) thArg.workClass;
236  //MpiNode *node=self->node;
237  int rank = self->rank;
238  ThreadTaskDistributor *taskDistributor=self->taskDistributor;
239  std::vector<size_t> &objId=self->objId;
240  MetaData &MD=self->MD[thArg.thread_id];
241 
242  Image<double> &I=self->I[thArg.thread_id];
243  MultidimArray<double> &Iaux=self->Iaux[thArg.thread_id];
244  Matrix2D<double> &Wnode=self->Wnode[thArg.thread_id];
245  Matrix2D<double> &Hblock=self->Hblock[thArg.thread_id];
246  Matrix2D<double> &A=self->A[thArg.thread_id];
247  Matrix2D<double> &H=self->H;
248  MultidimArray< unsigned char > &mask=self->mask;
249  Wnode.initZeros(self->Npixels,MAT_XSIZE(H));
250 
251  const int unroll=8;
252  const int jmax=(MAT_XSIZE(Wnode)/unroll)*unroll;
253 
254  size_t first, last;
255  if (IS_MASTER && thArg.thread_id==0)
256  {
257  std::cout << "Applying T ...\n";
258  init_progress_bar(objId.size());
259  }
260  while (taskDistributor->getTasks(first, last))
261  {
262  for (size_t idx=first; idx<=last; ++idx)
263  {
264  // Read image
265  I.readApplyGeo(MD,objId[idx]);
266  MultidimArray<double> &mI=I();
267 
268  // Locate the corresponding index in Matrix H
269  // and copy a block in memory to speed up calculations
270  size_t Hidx=idx*2*self->Nangles*self->Nshifts;
271  memcpy(&MAT_ELEM(Hblock,0,0),&MAT_ELEM(H,Hidx,0),MAT_XSIZE(Hblock)*MAT_YSIZE(Hblock)*sizeof(double));
272 
273  // For each rotation, shift and mirror
274  int block_idx=0;
275  for (int mirror=0; mirror<2; ++mirror)
276  {
277  if (mirror)
278  {
279  mI.selfReverseX();
280  mI.setXmippOrigin();
281  }
282  for (double psi=0; psi<360; psi+=self->psi_step)
283  {
284  rotation2DMatrix(psi,A,true);
285  for (double y=-self->max_shift_change; y<=self->max_shift_change; y+=self->shift_step)
286  {
287  MAT_ELEM(A,1,2)=y;
288  for (double x=-self->max_shift_change; x<=self->max_shift_change; x+=self->shift_step, ++block_idx)
289  {
290  MAT_ELEM(A,0,2)=x;
291 
292  // Rotate and shift image
293  applyGeometry(1,Iaux,mI,A,IS_INV,true);
294 
295  // Update Wnode
296  int i=0;
297  FOR_ALL_DIRECT_ELEMENTS_IN_MULTIDIMARRAY(Iaux)
298  {
299  if (DIRECT_MULTIDIM_ELEM(mask,n)==0)
300  continue;
301  double pixval=DIRECT_MULTIDIM_ELEM(Iaux,n);
302  double *ptrWnode=&MAT_ELEM(Wnode,i,0);
303  double *ptrHblock=&MAT_ELEM(Hblock,block_idx,0);
304  for (int j=0; j<jmax; j+=unroll, ptrHblock+=unroll, ptrWnode+=unroll)
305  {
306  (*(ptrWnode )) +=pixval*(*ptrHblock );
307  (*(ptrWnode+1)) +=pixval*(*(ptrHblock+1));
308  (*(ptrWnode+2)) +=pixval*(*(ptrHblock+2));
309  (*(ptrWnode+3)) +=pixval*(*(ptrHblock+3));
310  (*(ptrWnode+4)) +=pixval*(*(ptrHblock+4));
311  (*(ptrWnode+5)) +=pixval*(*(ptrHblock+5));
312  (*(ptrWnode+6)) +=pixval*(*(ptrHblock+6));
313  (*(ptrWnode+7)) +=pixval*(*(ptrHblock+7));
314  }
315  for (size_t j=jmax; j<MAT_XSIZE(Wnode); ++j, ptrHblock+=1, ptrWnode+=1)
316  (*(ptrWnode )) +=pixval*(*ptrHblock );
317  ++i;
318  }
319  }
320  }
321  }
322  }
323  }
324  if (IS_MASTER && thArg.thread_id==0)
325  progress_bar(last);
326  }

468 // QR =====================================================================
469 int ProgImageRotationalPCA::QR()
470 {
471  size_t jQ=0;
472  Matrix1D<double> qj1, qj2;
473  int iBlockMax=MAT_XSIZE(F)/4;
474 
475  for (size_t j1=0; j1<MAT_YSIZE(F); j1++)
476  {
477  F.getRow(j1,qj1);
478  // Project twice in the already established subspace
479  // One projection should be enough but Gram-Schmidt suffers
480  // from numerical problems
481  for (int it=0; it<2; it++)
482  {
483  for (size_t j2=0; j2<jQ; j2++)
484  {
485  F.getRow(j2,qj2);
486 
487  // Compute dot product
488  double s12=qj1.dotProduct(qj2);
489 
490  // Subtract the part of qj2 from qj1
491  double *ptr1=&VEC_ELEM(qj1,0);
492  const double *ptr2=&VEC_ELEM(qj2,0);
493  for (int i=0; i<iBlockMax; i++)
494  {
495  (*ptr1++)-=s12*(*ptr2++);
496  (*ptr1++)-=s12*(*ptr2++);
497  (*ptr1++)-=s12*(*ptr2++);
498  (*ptr1++)-=s12*(*ptr2++);
499  }
500  for (size_t i=iBlockMax*4; i<MAT_XSIZE(F); ++i)
501  (*ptr1++)-=s12*(*ptr2++);
502  }
503  }
504 
505  // Keep qj1 in Q if it has enough norm
506  double Rii=qj1.module();
507  if (Rii>1e-14)
508  {
509  // Make qj1 to be unitary and store in Q
510  qj1/=Rii;
511  F.setRow(jQ++,qj1);
512  }
513  }
514  return jQ;
515 }

570 // Run ====================================================================
571 void ProgImageRotationalPCA::run()
572 {
573  show();
574  produceSideInfo();
575 
576  // Compute matrix F:
577  // Set H pointing to the first block of F
578  applyTt();// H=Tt(W)
579  copyHtoF(0);
580  for (int it=0; it<Nits; it++)
581  {
582  applyT(); // W=T(H)
583  applyTt();// H=Tt(W)
584  copyHtoF(it+1);
585  }
586  H.clear();
587  clearHbuffer();
588 
589  // QR decomposition of matrix F
590  int qrDim;
591 
592  comunicateQrDim(qrDim);
