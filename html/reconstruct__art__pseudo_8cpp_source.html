36 void project_Pseudo(const std::vector< Matrix1D<double> > &atomPosition,
37  std::vector<double> &atomWeight, double sigma,
38  MultidimArray<double> &proj, MultidimArray<double> &norm_proj,
39  Matrix2D<double> &Euler, double shiftX, double shiftY,
40  const std::vector<double> &lambda,
41  const std::vector< Matrix2D<double> > &NMA,
42  int direction, const Matrix1D<double> &gaussianProjectionTable,
43  const Matrix1D<double> &gaussianProjectionTable2)
44 {
45  // Project all pseudo atoms ............................................
46  int nmax=atomPosition.size();
47  Matrix1D<double> actprj(3);
48  double sigma4=4*sigma;
49  Matrix1D<double> actualAtomPosition;
50  int lambdaSize=lambda.size();
51  for (int n=0; n<nmax; n++)
52  {
53  actualAtomPosition=atomPosition[n];
54  double weight=atomWeight[n];
55  for (int mode=0; mode<lambdaSize; mode++)
56  {
57  const Matrix2D<double> &NMAmode=NMA[mode];
58  double lambdam=lambda[mode];
59  FOR_ALL_ELEMENTS_IN_MATRIX1D(actualAtomPosition)
60  VEC_ELEM(actualAtomPosition,i)+=
61  lambdam*MAT_ELEM(NMAmode,n,i);
62  }
63 
64  Uproject_to_plane(actualAtomPosition, Euler, actprj);
65  XX(actprj)+=shiftX;
66  YY(actprj)+=shiftY;
67 #ifdef DEBUG
68 
69  bool condition = true;
70  if (condition)
71  {
72  std::cout << "Projecting point " << atomPositions[n].transpose() << std::endl;
73  std::cout << "Vol there = " << atomWeight[n] << std::endl;
74  std::cout << " Center of the basis proj (2D) " << XX(actprj) << "," << YY(actprj) << std::endl;
75  }
76 #endif
77 
78  // Search for integer corners for this basis
79  int XX_corner1 = CEIL(XMIPP_MAX(STARTINGX(proj), XX(actprj) - sigma4));
80  int YY_corner1 = CEIL(XMIPP_MAX(STARTINGY(proj), YY(actprj) - sigma4));
81  int XX_corner2 = FLOOR(XMIPP_MIN(FINISHINGX(proj), XX(actprj) + sigma4));
82  int YY_corner2 = FLOOR(XMIPP_MIN(FINISHINGY(proj), YY(actprj) + sigma4));
83 
84 #ifdef DEBUG
85 
86  if (condition)
87  {
88  std::cout << "Clipped and rounded Corner 1 " << XX_corner1
89  << " " << YY_corner1 << " " << std::endl;
90  std::cout << "Clipped and rounded Corner 2 " << XX_corner2
91  << " " << YY_corner2 << " " << std::endl;
92  }
93 #endif
94 
95  // Check if the basis falls outside the projection plane
96  if (XX_corner1 <= XX_corner2 && YY_corner1 <= YY_corner2)
97  {
98  double vol_corr=0;
99 
100  // Effectively project this basis
101  for (int y = YY_corner1; y <= YY_corner2; y++)
102  {
103  double y_diff2=y-YY(actprj);
104  y_diff2=y_diff2*y_diff2;
105  for (int x = XX_corner1; x <= XX_corner2; x++)
106  {
107  double x_diff2=x-XX(actprj);
108  x_diff2=x_diff2*x_diff2;
109  double r=sqrt(x_diff2+y_diff2);
110  double didx=r*1000;
111  int idx=ROUND(didx);
112  double a = VEC_ELEM(gaussianProjectionTable,idx);
113  double a2 = VEC_ELEM(gaussianProjectionTable2,idx);
114 #ifdef DEBUG
115 
116  if (condition)
117  std::cout << "=" << a << " , " << a2;
118 #endif
119 
120  if (direction==FORWARD)
121  {
122  A2D_ELEM(proj, y, x) += weight * a;
123  A2D_ELEM(norm_proj, y, x) += a2;
124 #ifdef DEBUG
125 
126  if (condition)
127  {
128  std::cout << " proj= " << A2D_ELEM(proj, y, x)
129  << " norm_proj=" << A2D_ELEM(norm_proj, y, x) << std::endl;
130  std::cout.flush();
131  }
132 #endif
133 
134  }
135  else
136  {
137  vol_corr += A2D_ELEM(norm_proj, y, x) * a;
138 #ifdef DEBUG
139 
140  if (condition)
141  {
142  std::cout << " corr_img= " << A2D_ELEM(norm_proj, y, x)
143  << " correction=" << vol_corr << std::endl;
144  std::cout.flush();
145  }
146 #endif
147 
148  }
149  }
150  }
151 
152  if (direction==BACKWARD)
153  {
154  atomWeight[n] += vol_corr;
155 #ifdef DEBUG
156 
157  if (condition)
158  {
159  std::cout << "\nFinal value at ( " << n << ") = "
160  << atomWeight[n] << std::endl;
161  }
162 #endif
163 
164  }
165  } // If not collapsed
166  }

227 void ProgARTPseudo::produceSideInfo()
228 {
229  DF.read(fnDoc);
230  std::ifstream fhPseudo;
231  fhPseudo.open(fnPseudo.c_str());
232  if (!fhPseudo)
233  REPORT_ERROR(ERR_IO_NOTEXIST,fnPseudo);
234  while (!fhPseudo.eof())
235  {
236  std::string line;
237  getline(fhPseudo, line);
238  if (line.length() == 0)
239  continue;
240  if (line.substr(7,13)=="fixedGaussian" && sigma<0)
241  {
242  std::vector < std::string> results;
243  splitString(line," ",results);
244  sigma=textToFloat(results[2]);
245  sigma/=sampling;
246  }
247  else if (line.substr(0,4)=="ATOM")
248  {
249  Matrix1D<double> v(3);
250  v(0)=textToFloat(line.substr(30,8));
251  v(1)=textToFloat(line.substr(38,8));
252  v(2)=textToFloat(line.substr(46,8));
253  v/=sampling;
254  atomPosition.push_back(v);
255  atomWeight.push_back(0);
256  }
257  }
258  fhPseudo.close();
259 
260  double sigma4=4*sigma;
261  gaussianProjectionTable.resize(CEIL(sigma4*sqrt(2)*1000));
262  FOR_ALL_ELEMENTS_IN_MATRIX1D(gaussianProjectionTable)
263  gaussianProjectionTable(i)=gaussian1D(i/1000.0,sigma);
264  gaussianProjectionTable*=gaussian1D(0,sigma);
265  gaussianProjectionTable2=gaussianProjectionTable;
266  gaussianProjectionTable2*=gaussianProjectionTable;
267 
268  // NMA
269  if (!fnNMA.empty())
270  {
271  MetaData DFNMA(fnNMA);
272  DFNMA.removeDisabled();
273  FOR_ALL_OBJECTS_IN_METADATA(DFNMA)
274  {
275  Matrix2D<double> mode;
276  mode.initZeros(atomPosition.size(),3);
277  FileName fnMode;
278  DFNMA.getValue( MDL_NMA_MODEFILE, fnMode,__iter.objId);
279  mode.read(fnMode);
280  NMA.push_back(mode);
281  }
282  }
283 }

285 void ProgARTPseudo::run()
286 {
287  show();
288  produceSideInfo();
289  Image<double> Iexp;
290  for (int it=0; it<Nit; it++)
291  {
292  double itError=0;
293  FOR_ALL_OBJECTS_IN_METADATA(DF)
294  {
295  FileName fnExp;
296  DF.getValue( MDL_IMAGE, fnExp,__iter.objId);
297  double rot;
298  DF.getValue( MDL_ANGLE_ROT, rot,__iter.objId);
299  double tilt;
300  DF.getValue( MDL_ANGLE_TILT, tilt,__iter.objId);
301  double psi;
302  DF.getValue( MDL_ANGLE_PSI, psi,__iter.objId);
303  double shiftX;
304  DF.getValue( MDL_SHIFT_X, shiftX,__iter.objId);
305  double shiftY;
306  DF.getValue( MDL_SHIFT_Y, shiftY,__iter.objId);
307  std::vector<double> lambda;
308  if (NMA.size()>0)
309  DF.getValue( MDL_NMA, lambda,__iter.objId);
310 
311  Iexp.read(fnExp);
312  Iexp().setXmippOrigin();
313  itError+=ART_single_step(Iexp(),rot,tilt,psi,-shiftX,-shiftY,lambda);
314  }
315  if (DF.size()>0)
316  itError/=DF.size();
317  std::cerr << "Error at iteration " << it << " = " << itError << std::endl;
318  }
319  writePseudo();

322 void ProgARTPseudo::writePseudo()
323 {
324  // Convert from pseudoatoms to volume
325  Image<double> V;
326  size_t objId = DF.firstObject();
327  FileName fnExp;
328  DF.getValue( MDL_IMAGE, fnExp, objId);
329  Image<double> I;
330  I.read(fnExp,HEADER);
331  V().resize(XSIZE(I()),XSIZE(I()),XSIZE(I()));
332  V().setXmippOrigin();
333 
334  int nmax=atomPosition.size();
335  double sigma4=4*sigma;
336  for (int n=0; n<nmax; n++)
337  {
338  int XX_corner1 = CEIL(XMIPP_MAX(STARTINGX(V()), XX(atomPosition[n]) - sigma4));
339  int YY_corner1 = CEIL(XMIPP_MAX(STARTINGY(V()), YY(atomPosition[n]) - sigma4));
340  int ZZ_corner1 = CEIL(XMIPP_MAX(STARTINGY(V()), ZZ(atomPosition[n]) - sigma4));
341  int XX_corner2 = FLOOR(XMIPP_MIN(FINISHINGX(V()), XX(atomPosition[n]) + sigma4));
342  int YY_corner2 = FLOOR(XMIPP_MIN(FINISHINGY(V()), YY(atomPosition[n]) + sigma4));
343  int ZZ_corner2 = FLOOR(XMIPP_MIN(FINISHINGY(V()), ZZ(atomPosition[n]) + sigma4));
344  if (XX_corner1 <= XX_corner2 && YY_corner1 <= YY_corner2 &&
345  ZZ_corner1 <= ZZ_corner2)
346  {
347  for (int z = ZZ_corner1; z <= ZZ_corner2; z++)
348  for (int y = YY_corner1; y <= YY_corner2; y++)
349  for (int x = XX_corner1; x <= XX_corner2; x++)
350  V(z,y,x)+=atomWeight[n]*
351  gaussian1D(z-ZZ(atomPosition[n]),sigma)*
352  gaussian1D(y-YY(atomPosition[n]),sigma)*
353  gaussian1D(x-XX(atomPosition[n]),sigma);
354  }
355  }
356  V.write(fnRoot+".vol");
357 
358  // Histogram of the intensities
359  MultidimArray<double> intensities(atomWeight);
360  Histogram1D hist;
361  compute_hist(intensities, hist, 100);
362  hist.write(fnRoot+"_intensities.hist");
363 }

365 double ProgARTPseudo::ART_single_step(const MultidimArray<double> &Iexp,
366  double rot, double tilt, double psi, double shiftX, double shiftY,
367  const std::vector<double> &lambda)
368 {
369  MultidimArray<double> Itheo, Icorr, Idiff;
370  Itheo.initZeros(Iexp);
371  Icorr.initZeros(Iexp);
372  Matrix2D<double> Euler;
373  Euler_angles2matrix(rot, tilt, psi, Euler);
374  project_Pseudo(atomPosition, atomWeight, sigma, Itheo, Icorr,
375  Euler, shiftX, shiftY, lambda, NMA, FORWARD,
376  gaussianProjectionTable, gaussianProjectionTable2);
377  Idiff.initZeros(Iexp);
378 
379  double mean_error=0;
380  FOR_ALL_DIRECT_ELEMENTS_IN_ARRAY2D(Iexp)
381  {
382  // Compute difference image and error
383  DIRECT_A2D_ELEM(Idiff, i, j) = DIRECT_A2D_ELEM(Iexp, i, j) - DIRECT_A2D_ELEM(Itheo, i, j);
384  mean_error += DIRECT_A2D_ELEM(Idiff, i, j) * DIRECT_A2D_ELEM(Idiff, i, j);
385 
386  // Compute the correction image
387  DIRECT_A2D_ELEM(Icorr, i, j) = XMIPP_MAX(DIRECT_A2D_ELEM(Icorr, i, j), 1);
388  DIRECT_A2D_ELEM(Icorr, i, j) =
389  lambdaART * DIRECT_A2D_ELEM(Idiff, i, j) / DIRECT_A2D_ELEM(Icorr, i, j);
390  }
391  mean_error /= YXSIZE(Iexp);
392 
393  project_Pseudo(atomPosition, atomWeight, sigma, Itheo, Icorr,
394  Euler, shiftX, shiftY, lambda, NMA, BACKWARD,
395  gaussianProjectionTable, gaussianProjectionTable2);
396  return mean_error;
