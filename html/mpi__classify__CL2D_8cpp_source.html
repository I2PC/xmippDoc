928 void CL2D::lookNode(MultidimArray<double> &I, int oldnode, int &newnode,

940  bestAssignment.likelihood = bestAssignment.corr = 0;
941  size_t objId = bestAssignment.objId;
942  for (int q = 0; q < Q; q++)
943  {

965  }
966  }
967  else

971  }
972  }

988  bestImg = Iaux;
989  bestAssignment = assignment;
990  }
991  }
992  }

995  newnode = bestq;
996  bestAssignment.objId = objId;
997

1001  std::cout << " New node=" << newnode << std::endl;
1002 #endif
1003  if (newnode != -1)
1004  {
1005  P[newnode]->updateProjection(I, bestAssignment);
1006  if (!prm->classicalMultiref)
1007  for (int q = 0; q < Q; q++)
1008  if (q != newnode && corrList(q) > 0)
1009  P[q]->updateNonProjection(corrList(q));
1010  }
