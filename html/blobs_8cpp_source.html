41 double kaiser_value(double r, double a, double alpha, int m)
42 {
43  double rda, rdas, arg, w;
44 
45  rda = r / a;
46  if (rda <= 1.0)
47  {
48  rdas = rda * rda;
49  arg = alpha * sqrt(1.0 - rdas);
50  if (m == 0)
51  {
52  w = bessi0(arg) / bessi0(alpha);
53  }
54  else if (m == 1)
55  {
56  w = sqrt (1.0 - rdas);
57  if (alpha != 0.0)
58  w *= bessi1(arg) / bessi1(alpha);
59  }
60  else if (m == 2)
61  {
62  w = sqrt (1.0 - rdas);
63  w = w * w;
64  if (alpha != 0.0)
65  w *= bessi2(arg) / bessi2(alpha);
66  }
67  else if (m == 3)
68  {
69  w = sqrt (1.0 - rdas);
70  w = w * w * w;
71  if (alpha != 0.0)
72  w *= bessi3(arg) / bessi3(alpha);
73  }
74  else if (m == 4)
75  {
76  w = sqrt (1.0 - rdas);
77  w = w * w * w *w;
78  if (alpha != 0.0)
79  w *= bessi4(arg) / bessi4(alpha);
80  }
81  else
82  REPORT_ERROR(ERR_VALUE_INCORRECT, "m out of range in kaiser_value()");
83 
84  }
85  else
86  w = 0.0;
87 
88  return w;
89 }

92 /* Value of line integral through Kaiser-Bessel radial function
93  (n >=2 dimensions) at distance s from center of function.
94  Parameter m = 0, 1, or 2. */
95 double kaiser_proj(double s, double a, double alpha, int m)
96 {
97  double sda, sdas, w, arg, p;
98 
99  sda = s / a;
100  sdas = sda * sda;
101  w = 1.0 - sdas;
102  if (w > 1.0e-10)
103  {
104  arg = alpha * sqrt(w);
105  if (m == 0)
106  {
107  if (alpha == 0.0)
108  p = 2.0 * a * sqrt(w);
109  else
110  p = (2.0 * a / alpha) * sinh(arg) / bessi0(alpha);
111 
112  }
113  else if (m == 1)
114  {
115  if (alpha == 0.0)
116  p = 2.0 * a * w * sqrt(w) * (2.0 / 3.0);
117  else
118  p = (2.0 * a / alpha) * sqrt(w) * (cosh(arg) - sinh(arg) / arg)
119  / bessi1(alpha);
120 
121  }
122  else if (m == 2)
123  {
124  if (alpha == 0.0)
125  p = 2.0 * a * w * w * sqrt(w) * (8.0 / 15.0);
126  else
127  p = (2.0 * a / alpha) * w *
128  ((3.0 / (arg * arg) + 1.0) * sinh(arg) - (3.0 / arg) * cosh(arg)) / bessi2(alpha);
129  }
130  else
131  REPORT_ERROR(ERR_VALUE_INCORRECT, "m out of range in kaiser_proj()");
132 
133  }
134  else
135  p = 0.0;
136

138 }

141 double kaiser_Fourier_value(double w, double a, double alpha, int m)
142 {
143  if (m != 2 && m !=0)

148  if (2.*PI*a*w > alpha)

152  return pow(2.*PI, 3. / 2.)*pow(a, 3.)*pow(alpha, 2.)*bessi3_5(sigma)

392  retval.alpha = best_alpha;
