509 static void fourier_corr(int bw, double *coeffr_hi, double *coeffi_hi,
510  double *coeffr_lo, double *coeffi_lo, double *ddd, double *coef_corr){
511 /* computes the Fourier coefficients of the correlation function */
512 /* transposing the blocks of data */
513 
514  int l, p, q, r, tr, zi, zii, size, tsize, isigp, isigr;
515  double temp_r, temp_i, tempd;
516  unsigned long index, indexn, init, ind1, ind3, indexa, indexan, indexb, indexbn,
517  indexmod, indexnmod, size2, tsize2;
518 
519  size=2*bw; tsize=2*size;
520  size2=size*size;
521  tsize2=2*size2;
522 
523  for(l=0;l<bw;l++){ /* compute T(p,q,r) for p>=0 and q>=0 */
524  init=l*(4*l*l-1)/3;
525  for(p=0;p<=l;p++){
526  ind1=seanindex(p,l,bw);
527  indexa=p*size2;
528  for(r=-l;r<0;r++){ /* r<0 */
529  ind3=seanindex(r,l,bw);
530  indexb=indexa+r+size;
531  temp_r = coeffr_lo[ind1]*coeffr_hi[ind3]+coeffi_lo[ind1]*coeffi_hi[ind3];
532  temp_i =-coeffr_lo[ind1]*coeffi_hi[ind3]+coeffi_lo[ind1]*coeffr_hi[ind3];
533  for(q=0;q<=l;q++){
534  tempd = ddd[init+(p+l)*(2*l+1)+(q+l)] * ddd[init+(q+l)*(2*l+1)+(r+l)];
535  index=2*(indexb+q*size);
536  *(coef_corr + index) += temp_r * tempd;
537  *(coef_corr + index + 1) += temp_i * tempd;
538  }
539  }
540  for(r=0;r<=l;r++){ /* r>=0 */
541  ind3=seanindex(r,l,bw);
542  indexb=indexa+r;
543  temp_r = coeffr_lo[ind1]*coeffr_hi[ind3]+coeffi_lo[ind1]*coeffi_hi[ind3];
544  temp_i =-coeffr_lo[ind1]*coeffi_hi[ind3]+coeffi_lo[ind1]*coeffr_hi[ind3];
545  for(q=0;q<=l;q++){
546  tempd = ddd[init+(p+l)*(2*l+1)+(q+l)] * ddd[init+(q+l)*(2*l+1)+(r+l)];
547  index=2*(indexb+q*size);

551  }
552  }

557  isigp = -isigp; /* parity of p */
558  zi=(indexa+bw+size2)<<1;

605 /*====================================================================*/
606 static void wigner(int bw, double theta, double *ddd){
607  /* computes the d-functions for argument theta, for degrees 0 through bw-1 */
608  /* the matrices are returned in the (3D) piramid ddd */
609  /* Reference: T. Risbo, Journal of Geodesy (1996) 70:383-396 */
610 
611  double p, q, pc, qc, temp, *d, *dd;
612  int size, index, i, j2, k, l, hdeg;
613  unsigned long init;
614  double fact1, fact2, fact3, fact4;
615 
616  size=2*bw;
617 
618  d = (double *) malloc(size*size*sizeof(double));
619  dd = (double *) malloc(size*size*sizeof(double));
620 
621  if( (d == NULL) || (dd == NULL) ){
622  perror("Error in allocating memory");
623  exit(1);
624  }
625 
626  p=sin(theta/2); q=cos(theta/2); /* Cayley-Klein parameters */
627  pc=p; qc=q;
628 
629  *(ddd+0)=1.0; /* d-matrix of degree 0 */
630 
631  /* d-matrix of degree 1/2 */
632  *(d+0)=q;
633  *(d+1)=p;
634  *(d+1*size+0)=-pc;
635  *(d+1*size+1)=qc;
636 
637  for(l=1;l<bw;l++){ /* l is the degree of the next d-matrix to be saved in ddd */
638  j2=(l<<1)-1; /* j2 will be twice the degree of the d-matrix about to be computed */
639  for(hdeg=0;hdeg<2;hdeg++){
640  fact1=q/ ++j2;
641  fact2=pc/j2;
642  fact3=p/j2;
643  fact4=qc/j2;
644  for(i=0;i<=j2+1;i++)
645  for(k=0;k<=j2+1;k++)
646  *(dd+i*size+k)=0.0;
647  for(i=0;i<j2;i++)
648  for(k=0;k<j2;k++){
649  index=i*size+k;
650  temp=*(d+index);
651  *(dd+index) += sqrt((j2-i)*(j2-k))*temp*fact1;
652  *(dd+index+size) += -sqrt((i+1)*(j2-k))*temp*fact2;
653  *(dd+index+1) += sqrt((j2-i)*(k+1)) *temp*fact3;
654  *(dd+index+size+1) += sqrt((i+1) *(k+1)) *temp*fact4;
655  }
656  for(i=0;i<=j2;i++) /* move dd to d */
657  for(k=0;k<=j2;k++)
658  *(d+i*size+k)=*(dd+i*size+k);
659  if(hdeg==0){ /* if degree is integer, copy d to the proper location in ddd */
660  init=l*(4*l*l-1)/3;
661  for(i=0;i<=j2;i++)
662  for(k=0;k<=j2;k++)
663  *(ddd+init+i*(j2+1)+k) = *(d+i*size+k);
664  }
665  if(l==bw-1) break; /* for l=bw-1 do hdeg=0 only */
666  }
667  }
668  free(d); free(dd);
669 }

681 static double distance_eulers(unsigned long i, unsigned long j, double *corr[]){
682  /* returns the distance between the rotations given by the Euler angles in
683  corr[1,2,3][i] and corr[1,2,3][j] */
684 
685  double matrix1[3][3],matrix2[3][3];
686  int k, l;
687  double temp, trace;
688 
689  get_rot_matrix(matrix1,corr[1][i],corr[2][i],corr[3][i]);
690  get_rot_matrix(matrix2,corr[1][j],corr[2][j],corr[3][j]);
691 
692 
693  trace=0.0; /* trace(matrix1*transpose(matrix2)) */
694  for(k=0;k<3;k++)
695  for(l=0;l<3;l++)
696  trace += matrix1[k][l]*matrix2[k][l];
697 
698  temp=0.5*(trace-1.0);
699  if(temp>=1.0)
700  return 0.0;
701  if(temp<=-1.0)
702  return M_PI;
703  return acos(temp);
704 }
