240 template<typename T>
241 void AProgMovieAlignmentCorrelation<T>::createLPF(T Ts, MultidimArray<T> &filter) {
242  // from formula
243  // e^(-1/2 * (omega^2 / sigma^2)) = 1/2; omega = Ts / max_resolution
244  // sigma = Ts / max_resolution * sqrt(1/-2log(1/2))
245  // c = sqrt(1/-2log(1/2))
246  // sigma = Ts / max_resolution * c
247  const size_t length = filter.xdim;
248  T iX = 1 / (T)length;
249  T sigma = (Ts * getC()) / maxResForCorrelation;
250  for (size_t x = 0; x < length; ++x) {
251  T w = x * iX;
252  filter.data[x] = (exp(-0.5*(w*w)/(sigma*sigma)));
253  }
254 }

256 template<typename T>
257 void AProgMovieAlignmentCorrelation<T>::scaleLPF(const MultidimArray<T>& lpf,
258  const Dimensions &dims, MultidimArray<T>& result) {
259  Matrix1D<T> w(2);
260  FOR_ALL_DIRECT_ELEMENTS_IN_ARRAY2D(result)
261  {
262  FFT_IDX2DIGFREQ(i, dims.y(), YY(w));
263  FFT_IDX2DIGFREQ(j, dims.x(), XX(w));
264  T wabs = w.module();
265  A2D_ELEM(result, i, j) = lpf.interpolatedElement1D(wabs * dims.x());
266  }

269 template<typename T>
270 void AProgMovieAlignmentCorrelation<T>::computeTotalShift(int iref, int j,
271  const Matrix1D<T> &shiftX, const Matrix1D<T> &shiftY, T &totalShiftX,
272  T &totalShiftY) {
273  totalShiftX = totalShiftY = 0;
274  if (iref < j) {
275  for (int jj = j - 1; jj >= iref; --jj) {
276  totalShiftX -= shiftX(jj);
277  totalShiftY -= shiftY(jj);
278  }
279  } else if (iref > j) {
280  for (int jj = j; jj <= iref - 1; ++jj) {
281  totalShiftX += shiftX(jj);

287 template<typename T>
288 int AProgMovieAlignmentCorrelation<T>::findReferenceImage(size_t N,
289  const Matrix1D<T>& shiftX, const Matrix1D<T>& shiftY) {
290  int bestIref = -1;
291  // Choose reference image as the minimax of shifts
292  T worstShiftEver = std::numeric_limits<T>::max();
293  for (int iref = 0; iref < N; ++iref) {
294  T worstShift = -1;
295  for (int j = 0; j < N; ++j) {
296  T totalShiftX, totalShiftY;
297  computeTotalShift(iref, j, shiftX, shiftY, totalShiftX,
298  totalShiftY);
299  if (fabs(totalShiftX) > worstShift)
300  worstShift = fabs(totalShiftX);
301  }
302  if (worstShift < worstShiftEver) {
303  worstShiftEver = worstShift;
304  bestIref = iref;
305  }
306  }
307  return bestIref;
308 }

334  // from formula
335  // e^(-1/2 * (omega^2 / sigma^2)) = 1/2; omega = Ts / max_resolution
336  // sigma = Ts / max_resolution * sqrt(1/-2log(1/2))

341 template<typename T>
342 T AProgMovieAlignmentCorrelation<T>::getTsPrime() {
343  // from formula
344  // e^(-1/2 * (omega^2 / sigma^2)) = 1/2; omega = Ts / max_resolution
345  // sigma = Ts / max_resolution * sqrt(1/-2log(1/2))
346  // c = sqrt(1/-2log(1/2))
347  // sigma = Ts / max_resolution * c
348  // then we want to find a resolution at 4 sigma (because values there will be almost zero anyway)
349  // omega4 = 4 * sigma -> Ts/R4 = 4 Ts / max_resolution * c
350  // R4 = max_resolution / (4 * c)
351  // new pixel size Ts' = R4 / 2 (to preserve Nyquist frequency)
352  T TsPrime = maxResForCorrelation / (8 * getC());
353  return TsPrime;
354 }

446 template<typename T>
447 AlignmentResult<T> AProgMovieAlignmentCorrelation<T>::computeAlignment(
448  Matrix1D<T> &bX, Matrix1D<T> &bY, Matrix2D<T> &A,
449  const core::optional<size_t> &refFrame, size_t N, int verbose) {
450  // now get the estimated shift (from the equation system)
451  // from each frame to successive frame
452  Matrix1D<T> shiftX, shiftY;
453  EquationSystemSolver::solve(bX, bY, A, shiftX, shiftY, verbose, solverIterations);
454  // prepare result
455  AlignmentResult<T> result {.refFrame = refFrame ?
456  refFrame.value() :
457  this->findReferenceImage(N, shiftX, shiftY)};
458  result.shifts.reserve(N);
459  // compute total shift in respect to reference frame
460  for (size_t i = 0; i < N; ++i) {
461  T x, y;
462  this->computeTotalShift(result.refFrame, i, shiftX, shiftY, x, y);
463  result.shifts.emplace_back(x, y);
464  }
465  return result;
466 }

507 template<typename T>
508 void AProgMovieAlignmentCorrelation<T>::storeResults(
509  const LocalAlignmentResult<T> &alignment) {
510  if (fnOut.isEmpty()) {
511  return;
512  }
513  if ( ! alignment.bsplineRep) {
514  REPORT_ERROR(ERR_VALUE_INCORRECT,
515  "Missing BSpline representation. This should not happen. Please contact developers.");
516  }
517  // Store average
518  std::vector<double> shifts;
519  for (auto &&p : alignment.shifts) {
520  int tileCenterX = p.first.rec.getCenter().x;
521  int tileCenterY = p.first.rec.getCenter().y;
522  int tileIdxT = p.first.id_t;
523  auto shift = BSplineHelper::getShift(
524  alignment.bsplineRep.value(), alignment.movieDim,
525  tileCenterX, tileCenterY, tileIdxT);
526  auto globalShift = alignment.globalHint.shifts.at(tileIdxT);
527  shifts.emplace_back(hypot(shift.first - globalShift.x, shift.second - globalShift.y));
528  }
529  MetaData mdIref;
530  size_t id = mdIref.addObject();
531  // Store confidence interval
532  std::sort(shifts.begin(), shifts.end(), std::less<double>());
533  size_t indexL = shifts.size() * 0.025;
534  size_t indexU = shifts.size() * 0.975;
535  mdIref.setValue(MDL_LOCAL_ALIGNMENT_CONF_2_5_PERC, shifts.at(indexL), id);
536  mdIref.setValue(MDL_LOCAL_ALIGNMENT_CONF_97_5_PERC, shifts.at(indexU), id);
537  // Store patches
538  mdIref.setValue(MDL_LOCAL_ALIGNMENT_PATCHES,
539  std::vector<size_t>{localAlignPatches.first, localAlignPatches.second}, id);
540  // Store coefficients
541  std::vector<double> tmpX;
542  std::vector<double> tmpY;
543  size_t size = alignment.bsplineRep.value().getCoeffsX().size();
544  for (int i = 0; i < size; ++i) {
545  tmpX.push_back(alignment.bsplineRep.value().getCoeffsX()(i));
546  tmpY.push_back(alignment.bsplineRep.value().getCoeffsY()(i));
