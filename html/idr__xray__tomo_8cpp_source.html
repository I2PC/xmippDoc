129 void ProgIDRXrayTomo::preRun()
130 {
131  psf.read(fnPSF);
132  psf.verbose = verbose;
133  psf.nThr = nThr;
134 
135  psf.calculateParams(sampling, sampling, psfThr);
136 
137  // Threads stuff
138  barrier = new Barrier(nThr);
139  //Create threads to start working
140  thMgr = new ThreadManager(nThr);
141 
142  // Reading the input projections. Tilt angle values are needed
143  projMD.read(fnInputProjMD);
144 
145  // Setting the intermediate filenames
146  fnInterProjs = fnRootInter + "_inter_projs";
147  fnInterProjs += (reconsMethod == RECONS_TOMO3D)? ".mrc" : ".stk";
148 
149  fnInterProjsMD = fnInterProjs.replaceExtension("xmd");
150  fnInterAngles = fnRootInter + "_angles.txt";
151 
152  // We make sure to delete intermediate files
153  fnInterProjs.deleteFile();
154  fnInterProjsMD.deleteFile();
155 
156  FileName fnProjs;
157  projMD.getValue(MDL_IMAGE, fnProjs, projMD.firstObject());
158  // We generate the MD of temporary projections to be used for further reconstructions
159  interProjMD = projMD;
160  interProjMD.replace(MDL_IMAGE, fnProjs.removePrefixNumber(), fnInterProjs);
161  interProjMD.write(fnInterProjsMD);
162 
163  String arguments = formatString("-i %s -o %s -s",
164  fnInputProjMD.c_str(), fnInterProjs.c_str());
165  ProgConvImg conv;
166  conv.read(arguments);
167  conv.run();
168 
169  // Saving angles in plain text to pass to tomo3D
170  if ( reconsMethod == RECONS_TOMO3D )
171  {
172  std::vector<MDLabel> desiredLabels;
173  desiredLabels.push_back(MDL_ANGLE_TILT);
174  projMD.writeText(fnInterAngles, &desiredLabels);
175  }
176 
177  if (fnStart.empty()) // if initial volume is not passed,then we must calculate it
178  {
179  // Projections must be in an MRC stack to be passed
180  if ( reconsMethod == RECONS_TOMO3D )
181  reconstruct(fnInterProjsMD, fnOutVol);
182  else
183  {
184  reconstruct(fnInputProjMD, fnOutVol);
185  phantom.read(fnOutVol);
186  }
187  }
188  else
189  phantom.read(fnStart);
190 }

192 void ProgIDRXrayTomo::run()
193 {
194  preRun();
195 
196  Projection proj, stdProj;
197  Image<double> fixedProj, prevFProj;
198  MultidimArray<double> mFixedProj, mPrevFProj;
199 
200 
201  double rot, tilt, psi;
202 
203  double lambda = VEC_ELEM(lambda_list, 0);
204 
205  FileName fnProj;
206 
207  for (int iter = 0; iter < itNum; iter++)
208  {
209  std::cout << "== Iteration " << iter << std::endl;
210  std::cout << "Projecting ..." <<std::endl;
211 
212  // Fix the reconstructed volume to physical density values
213  double factor = 1/sampling;
214  MULTIDIM_ARRAY(phantom.iniVol) *= factor;
215  // remove negative values
216  forcePositive(MULTIDIM_ARRAY(phantom.iniVol));
217 
218  initProgress(projMD.size());
219  size_t imgNo = 1; // Image number
220  double meanError = 0.;
221 
222  FOR_ALL_OBJECTS_IN_METADATA(projMD)
223  {
224  projMD.getValue(MDL_IMAGE , fnProj, __iter.objId);
225  projMD.getValue(MDL_ANGLE_ROT , rot, __iter.objId);
226  projMD.getValue(MDL_ANGLE_TILT, tilt, __iter.objId);
227  projMD.getValue(MDL_ANGLE_PSI , psi, __iter.objId);
228  proj.setAngles(rot, tilt, psi);
229 
230  XrayRotateAndProjectVolumeOffCentered(phantom, psf, proj, stdProj, YSIZE(MULTIDIM_ARRAY(phantom.iniVol)),
231  XSIZE(MULTIDIM_ARRAY(phantom.iniVol)));
232 
233  /* Image normalization to optimize for reconstruction.
234  * To correct for self-attenuation (without considering 3DPSF)
235  * I = -ln(Iab)
236  */
237 // FOR_ALL_DIRECT_ELEMENTS_IN_MULTIDIMARRAY(MULTIDIM_ARRAY(proj))
238 // dAi(MULTIDIM_ARRAY(proj),n) = -log(1. - dAi(MULTIDIM_ARRAY(proj),n));
239 
240 
241  fixedProj.read(fnProj);
242  mFixedProj.alias(MULTIDIM_ARRAY(fixedProj));
243 
244  FOR_ALL_DIRECT_ELEMENTS_IN_MULTIDIMARRAY(mFixedProj)
245  dAi(mFixedProj, n) = (log(1. - dAi(MULTIDIM_ARRAY(proj),n)) - log(dAi(mFixedProj, n)))*lambda + dAi(MULTIDIM_ARRAY(stdProj),n);
246 
247  prevFProj.read(fnInterProjs, DATA, imgNo); // To calculate meanError
248  mPrevFProj.alias(MULTIDIM_ARRAY(prevFProj));
249 
250  // Write the refined projection
251  fixedProj.write(fnInterProjs, imgNo, true, WRITE_REPLACE);
252 
253  // debug stuff //
254  if (verbose > 5)
255  {
256  proj.write(fnRootInter + "_debug_proj.stk", imgNo , true, WRITE_REPLACE);
257  stdProj.write(fnRootInter + "_debug_std_proj.stk", imgNo , true, WRITE_REPLACE);
258 
259  FOR_ALL_DIRECT_ELEMENTS_IN_MULTIDIMARRAY(mFixedProj)
260  dAi(mFixedProj, n) = dAi(MULTIDIM_ARRAY(stdProj),n) - dAi(MULTIDIM_ARRAY(proj),n)*lambda ;
261 
262  fixedProj.write(fnRootInter + "_debug_diff_proj.stk", imgNo , true, WRITE_REPLACE);
263  }
264 
265  FOR_ALL_DIRECT_ELEMENTS_IN_MULTIDIMARRAY(mFixedProj)
266  meanError += fabs(dAi(mPrevFProj, n) - dAi(mFixedProj, n));
267 
268  ++imgNo;
269  // Update progress bar
270  setProgress();
271  }
272  // Once calculated all fixed projection, let reconstruct again
273  endProgress();
274 
275  meanError /= (NZYXSIZE(mFixedProj)*projMD.size());
276 
277  std::cout << "Mean error = " << meanError <<std::endl;
278 
279  std::cout << "Reconstructing ..." << std::endl;
280  reconstruct(fnInterProjsMD, fnOutVol);
281 
282  if ( (iter < itNum-1) && (reconsMethod != RECONS_TOMO3D) ) // Since we process the output of tomo3d, this is already read
283  phantom.read(fnOutVol);
284 
285  }
286 
287  // Finish progress bar
288 
289  if (reconsMethod == RECONS_TOMO3D) // Trick to save the processed output of tomo3d
290  phantom.iniVol.write(fnOutVol);
291 
292 }

294 void ProgIDRXrayTomo::reconstruct(const FileName &fnProjsMD, const FileName &fnVol)
295 {
296  if (reconsMethod == RECONS_TOMO3D)
297  {
298  MetaData MD(fnProjsMD);
299  FileName fnProjs;
300  MD.getValue(MDL_IMAGE, fnProjs, MD.firstObject());
301 
302  reconsTomo3D(fnInterAngles, fnProjs.removePrefixNumber(), fnVol);
303  phantom.read(fnVol);
304  MULTIDIM_ARRAY(phantom.iniVol).resetOrigin();
305  MULTIDIM_ARRAY(phantom.iniVol).reslice(VIEW_Y_NEG);
306  //FIXME: this is a temporary fixing to match the reconstructed volume with phantom
307  double factor = 1/13.734;
308  MULTIDIM_ARRAY(phantom.iniVol) *= factor;
309  }
310  else
311  {
312  reconsProgram = createReconsProgram(fnProjsMD, fnVol);
313  reconsProgram->run();
314  delete reconsProgram;
315  }
