333 void MultireferenceAligneability::calc_sumw(const size_t num, double & sumw)
334 {
335  size_t trials=500;
336  double xRan,yRan,zRan;
337  double x,y;
338  double * xRanArray = new double[num];
339  double * yRanArray = new double[num];
340  double * zRanArray = new double[num];
341 
342  sumw=0;
343 
344  for (size_t n=0; n < trials; n++)
345  {
346  for (size_t nS=0; nS<num; nS++)
347  {
348  /*
349  x = sin(tilt*PI/180)*cos(rot*PI/180);
350  y = sin(tilt*PI/180)*sin(rot*PI/180);
351  z = std::abs(cos(tilt*PI/180));
352  */
353 
354  //http://mathworld.wolfram.com/SpherePointPicking.html
355  x = 2*(double(std::rand())-RAND_MAX/2)/RAND_MAX;
356  y = 2*(double(std::rand())-RAND_MAX/2)/RAND_MAX;
357 
358  while (x*x+y*y >= 1 )
359  {
360  x = 2*(std::rand()-RAND_MAX/2)/RAND_MAX;
361  y = 2*(std::rand()-RAND_MAX/2)/RAND_MAX;
362  }
363 
364  xRan = 2*x*std::sqrt(1-x*x-y*y);
365  yRan = 2*y*std::sqrt(1-x*x-y*y);
366  zRan = std::abs(1-2*(x*x+y*y));
367 
368  xRanArray[nS] = xRan;
369  yRanArray[nS] = yRan;
370  zRanArray[nS] = zRan;
371  }
372 
373  double WRan = 0.;
374  for (size_t nS1=0; nS1<num; nS1++)
375  {
376  double a;
377  for (size_t nS2=0; nS2<num; nS2++)
378  {
379  a = std::abs(std::acos(xRanArray[nS1]*xRanArray[nS2]+yRanArray[nS1]*yRanArray[nS2]+zRanArray[nS1]*zRanArray[nS2]));
380  if ( (a != 0))
381  WRan += a;
382  }
383  }
384 
385  sumw += WRan;
386  }
387 
388  sumw /= trials;
389  delete[] xRanArray;
390  delete[] yRanArray;
391  delete[] zRanArray;
392 }
