386 void AutoParticlePicking2::add2Dataset(int flagNegPos)
387 {
388  MultidimArray<double> vec;
389  MultidimArray<double> staticVec;
390  MultidimArray<double> avg;
391  MultidimArray<double> pcaBase;
392  MultidimArray<double> pcaRBase;
393 
394  int steps;
395  int yDataSet=YSIZE(dataSet);
396 
397  if (!flagNegPos)
398  steps=NSIZE(positiveInvariatnStack)/num_correlation;
399  else
400  steps=NSIZE(negativeInvariatnStack)/num_correlation;
401  // Resize the dataset for the new data
402  dataSet.resize(1,1,yDataSet+steps,num_features);
403  classLabel.resize(1,1,1,YSIZE(dataSet));
404  for (size_t n=yDataSet;n<XSIZE(classLabel);n++)
405  {
406  if (!flagNegPos)
407  classLabel(n)=1;
408  else
409  classLabel(n)=2;
410  }
411  // Here we take each channel of the particle and try to project it
412  // on related PCA basis. So we first do it for first channel and obtain
413  // all the features for all particles and then move to the next channel.
414  for (int i=0;i<num_correlation;i++)
415  {
416  avg.aliasImageInStack(pcaModel,i*(NPCA+1));
417  avg.resize(1,1,XSIZE(avg)*YSIZE(avg));
418  for (int j=0;j<NPCA;j++)
419  {
420  pcaBase.aliasImageInStack(pcaModel,i*(NPCA+1)+1+j);
421  pcaBase.resize(1,1,1,XSIZE(pcaBase)*YSIZE(pcaBase));
422  for (int k=0;k<steps;k++)
423  {
424  vec.resize(1 , NangSteps, NRsteps);
425  if (!flagNegPos)
426  positiveInvariatnStack.getImage(k*num_correlation+i, vec);
427  else
428  negativeInvariatnStack.getImage(k*num_correlation+i, vec);
429  vec.resize(1, 1, XSIZE(vec)*YSIZE(vec));
430  vec-=avg;
431  DIRECT_A2D_ELEM(dataSet,k+yDataSet,j+(i*NPCA))=PCAProject(pcaBase,vec);
432  }
433  }
434  }
435 
436  // Obtain the statics for each particle
437  for (int i=0;i<steps;i++)
438  {
439  vec.resize(1 , particle_size+1, particle_size+1);
440  if (!flagNegPos)
441  positiveParticleStack.getImage(i, vec);
442  else
443  negativeParticleStack.getImage(i, vec);
444  vec.resize(1,1,1,XSIZE(vec)*YSIZE(vec));
445  extractStatics(vec,staticVec);
446  for (int j=0;j<12;j++)
447  DIRECT_A2D_ELEM(dataSet,i+yDataSet,j+num_correlation*NPCA)=DIRECT_A1D_ELEM(staticVec,j);
448  // Project each particles on rotational PCA basis.
449  for (int j=0;j<NRPCA;j++)
450  {
451  pcaRBase.aliasImageInStack(pcaRotModel,j);
452  pcaRBase.resize(1,1,1,XSIZE(pcaRBase)*YSIZE(pcaRBase));
453  DIRECT_A2D_ELEM(dataSet,i+yDataSet,num_correlation*NPCA+12+j)=PCAProject(pcaRBase,vec);
454  }
455  }
