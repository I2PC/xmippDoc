198 void ProgConvImg::preProcess()
199 {
200  if (type == "auto")
201  {
202  if (oroot.empty())
203  {
204  /* It is stack if extension is stack compatible, or if --oroot is not passed
205  * and there are more than one image. Same for volumes.
206  * Firstly, we must check extensions, then stack and volume sizes.*/
207  if (( mdInSize > 1 || zdimOut > 1 ) && fn_out.hasStackExtension())
208  type = "stk";
209  else if (( mdInSize > 1 || zdimOut > 1 ) && fn_out.hasVolumeExtension()) // if it is volume
210  type = "vol";
211  else if (mdInSize > 1 || appendToStack) // If --append we suppose output is stack
212  type = "stk";
213  else if (zdimOut > 1) // if it is volume
214  type = "vol";
215  else
216  type = "img";
217  }
218  else
219  type = "img";
220  }
221 
222  // Set write mode
223  if (single_image && fn_out.isInStack()) // Replace a single image in a stack
224  {
225  type = "img";
226  writeMode = WRITE_REPLACE;
227  }
228  else if (type == "stk" && appendToStack)
229  writeMode = WRITE_APPEND;
230  else
231  {
232  writeMode = WRITE_OVERWRITE;
233  delete_output_stack = true;
234  }
235 
236  if (delete_output_stack)
237  {
238  FileName fn_stack_plain = fn_out.removeFileFormat();
239  fn_stack_plain.deleteFile();
240  delete_output_stack = false;
241  }
242 
243  create_empty_stackfile = (create_empty_stackfile && !appendToStack);
244 
245  convMode = MD2MD;
246 
247  if (!single_image && type == "vol")
248  {
249  convMode = MD2VOL;
250 
251  if (zdimOut != 1)
252  REPORT_ERROR(ERR_MULTIDIM_DIM,
253  "Only 2D images can be converted into volumes");
254  imOut = new ImageGeneric(datatypeOut);
255  imOut->mapFile2Write(xdimOut, ydimOut, mdInSize, fn_out);
256  k = 0;
257  }
258  else if (single_image)
259  {
260  // If --append is set, or fn_out is in a stack, then it is supposed not to convert VOL2MD
261  if ( zdimOut > 1 && !(type == "vol" || fn_out.isInStack() || appendToStack))
262  {
263  convMode = VOL2MD;
264  single_image = false;
265 
266  MetaData * md = getInputMd();
267  md->clear();
268 
269  FileName fnTemp;
270 
271  /* Fill mdIn to allow XmippMetaDataProgram create the fnImgOut,
272  but not used to read input images. Input volume is read here. */
273  for (k = 1; k <= zdimOut; k++)
274  {
275  fnTemp.compose(k, fn_in);
276  size_t id = md->addObject();
277  md->setValue(MDL_IMAGE, fnTemp, id);
278  md->setValue(MDL_ENABLED, 1, id);
279  }
280  imIn.read(fn_in, DATA, ALL_IMAGES, true);
281  imOut = new ImageGeneric(datatypeOut);
282  k = 0; // Reset to zero to select the slices when working with volumes
283  if (!fn_out.empty())
284  {
285  if (depth.empty())
286  depth = "%"+datatype2Str(datatypeOut);
287  createEmptyFile(fn_out+depth, xdimOut, ydimOut, 1, zdimOut, true, WRITE_OVERWRITE, swap);
288  }
289  }
290  }
291  else if (create_empty_stackfile)
292  {
293  if (depth.empty())
294  depth = "%"+datatype2Str(datatypeOut);
295  createEmptyFile(fn_out+depth, xdimOut, ydimOut, zdimOut, mdInSize, true, WRITE_OVERWRITE, swap);
296  }
297  create_empty_stackfile = false;
298 }//function preprocess

300 void ProgConvImg::processImage(const FileName &fnImg, const FileName &fnImgOut, const MDRow &rowIn, MDRow &rowOut)
301 {
302  switch(convMode)
303  {
304  case MD2MD:
305  {
306  FileName tempName, _fnImgOut;
307  if (fnImg == fnImgOut)
308  {
309  tempName.initUniqueName("tempConvert_XXXXXX");
310  _fnImgOut = tempName + ":" + fnImgOut.getExtension();
311  }
312  else
313  _fnImgOut= fnImgOut;
314 
315  imIn.read(fnImg, DATA, ALL_IMAGES, true);
316  imIn.write(_fnImgOut+depth, ALL_IMAGES, type == "stk", writeMode, castMode, swap);
317 
318  if ((fnImg == fnImgOut) && (rename(tempName.c_str(),fnImgOut.c_str())!=0))
319  REPORT_ERROR(ERR_IO, formatString("ProgConvImg:: Error renaming the file from %s to %s.",tempName.c_str(), fnImgOut.c_str()));
320 
321  break;
322  }
323  case MD2VOL:
324  {
325  imIn.read(fnImg,DATA, ALL_IMAGES,true);
326  imOut->data->setSlice(k++,imIn.data);
327  break;
328  }
329  case VOL2MD:
330  {
331  imIn.data->getSlice(k++,imOut->data);
332  imOut->write(fnImgOut+depth, ALL_IMAGES, type == "stk", writeMode, castMode, swap);
333  break;
334  }
335  }
336 }//function processImage
