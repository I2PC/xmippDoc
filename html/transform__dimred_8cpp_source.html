163 void ProgTransformDimRed::extractRandomProjections()
164 {
165  //number of grids in each axis
166  int num = numGrids;
167 
168  size_t numParticles = SFin.size();
169  std::vector<double> dimredProj;
170  std::vector< Matrix1D< double > > coor;
171  Matrix1D<double> dummy;
172  dummy.resizeNoCopy(numParticles);
173  for (int n=0; n<outputDim; ++n)
174  coor.push_back(dummy);
175 
176  // Keep the reduced coefficients
177  size_t i=0;
178  FOR_ALL_OBJECTS_IN_METADATA(SFin)
179  {
180  SFin.getValue(MDL_DIMRED,dimredProj,__iter.objId);
181  for (int n=0; n<outputDim; ++n)
182  VEC_ELEM(coor[n],i)=dimredProj[n];
183  i++;
184  }
185 
186  // We obtain the max and min coordinate values
187  std::vector<double> minCoor, maxCoor;
188  for (int n=0; n<outputDim; ++n)
189  {
190  double minval, maxval;
191  coor[n].computeMinMax(minval, maxval);
192  minCoor.push_back(minval);
193  maxCoor.push_back(maxval);
194 
195  }
196 
197  Matrix2D <int> squares;
198  squares.resizeNoCopy(numParticles, (int)pow(num+1,outputDim));
199 
200  Matrix1D <int> numElems;
201  numElems.resizeNoCopy(MAT_XSIZE(squares));
202 
203  for (size_t i=0; i<numParticles; i++)
204  {
205  int index=0;
206  for (int n=(outputDim-1); n>=0; --n)
207  {
208  index*=num;
209  //std::cout << "minCoor : " << minCoor[n] << std::endl;
210  //std::cout << "maxCoor : " << maxCoor[n] << std::endl;
211  //std::cout << "VEC_ELEM(coor[n],i) : " << VEC_ELEM(coor[n],i) << std::endl;
212 
213  index+=(int)floor(( ( VEC_ELEM(coor[n],i) - minCoor[n]) / (maxCoor[n] - minCoor[n]))*num);
214  }
215 
216 
217  MAT_ELEM(squares,VEC_ELEM(numElems,index),index)=i;
218  VEC_ELEM(numElems,index)++;
219  }
220 
221  int numElemFirstRow = 0;
222  for (size_t k=0; k < MAT_XSIZE(squares); k++)
223  if (VEC_ELEM(numElems,k)!=0)
224  numElemFirstRow++;
225 
226  std::vector<Matrix1D<double> > selectedCoor;
227  dummy.resizeNoCopy(numElemFirstRow);
228  for (int n=0; n<outputDim; ++n)
229  selectedCoor.push_back(dummy);
230 
231  //Metadata with the well sampled projection and random projections assigned
232  MetaData SFout;
233  int indx = 0;
234  randomize_random_generator();
235  std::vector<String> fnImg;
236  SFin.getColumnValues(MDL_IMAGE,fnImg);
237  for (size_t k=0; k <MAT_XSIZE(squares); k++)
238  {
239  if (MAT_ELEM(squares,0,k)!=0)
240  {
241  int randomN = round((rnd_unif(0,1))*VEC_ELEM(numElems,k));
242  for (int n=0; n<outputDim; ++n)
243  VEC_ELEM(selectedCoor[n],indx)=VEC_ELEM(coor[n],MAT_ELEM(squares,randomN,k));
244  size_t id=SFout.addObject();
245  SFout.setValue(MDL_IMAGE,fnImg[MAT_ELEM(squares,randomN,k)],id);
246  SFout.setValue(MDL_ANGLE_ROT,(rnd_unif(0,360)-180),id);
247  SFout.setValue(MDL_ANGLE_TILT,(rnd_unif(0,180)),id);
248  SFout.setValue(MDL_ANGLE_TILT,(rnd_unif(0,180)),id);
249  SFout.setValue(MDL_ANGLE_PSI,(rnd_unif(0,360)),id);
250  indx++;
251  }
252  }
253 
254  std::cout << " fnRandomSampling : " << fnRandomSampling << std::endl;
255  SFout.write(fnRandomSampling);
256 }
