110  auto k = [d_V, kd_x, kd_probXLessThanx, kprobStep, k_volume_size] __device__ (int index) {
111  int i = 0;
112  for (T p = kprobStep / 2; p < 1; p += kprobStep, i++) {
113  int idx = static_cast<int>(round(p * k_volume_size));
114  kd_probXLessThanx[i] = p;
115  kd_x[i] = d_V[idx];
116  }
117  };

120 }
