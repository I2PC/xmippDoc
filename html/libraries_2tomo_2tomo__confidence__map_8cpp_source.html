399 void ProgTomoConfidecenceMap::estimateNoiseStatistics(MultidimArray<float> &noiseMap,
400  MultidimArray<float> &noiseVarianceMap, MultidimArray<float> &noiseMeanMap,
401  int boxsize, Matrix2D<float> &thresholdMatrix_mean, Matrix2D<float> &thresholdMatrix_std)
402 {
403  std::cout << "Estimating noise statistics ...." << std::endl;
404  noiseVarianceMap.initZeros(noiseMap);
405  noiseMeanMap.initZeros(noiseMap);
406 
407 // std::cout << "Analyzing local noise" << std::endl;
408 
409  int Nx = Xdim/boxsize;
410  int Ny = Ydim/boxsize;
411 
412  Matrix2D<double> noiseStdMatrix, noiseMeanMatrix;
413  noiseStdMatrix.initZeros(Ny, Nx);
414  noiseMeanMatrix = noiseStdMatrix;
415 
416  // For the spline regression. A minimum of 8 points are considered
417  int lX=std::min(8,Nx-2), lY=std::min(8,Ny-2);
418  WeightedLeastSquaresHelper helperStd, helperMean;
419  helperStd.A.initZeros(Nx*Ny,lX*lY);
420  helperStd.b.initZeros(Nx*Ny);
421  helperStd.w.initZeros(Nx*Ny);
422  helperStd.w.initConstant(1);
423  helperMean = helperStd;
424 
425 
426  double hX = Xdim / (double)(lX-3);
427  double hY = Ydim / (double)(lY-3);
428 
429  if ( (Xdim<boxsize) || (Ydim<boxsize) )
430  std::cout << "Error: The tomogram in x-direction or y-direction is too small" << std::endl;
431 
432  std::vector<float> noiseVector(1);
433  // std::vector<double> x,y,t;
434 
435  int xLimit, yLimit, xStart, yStart;
436  int startX, startY, startZ, finishX, finishY, finishZ;
437  startX = 0;
438  startY = 0;
439  startZ = 0;
440  finishX = Xdim;
441  finishY = Ydim;
442  finishZ = Zdim;
443 
444  long N;
445  int idxBox=0;
446 
447  for (int X_boxIdx=0; X_boxIdx<Nx; ++X_boxIdx)
448  {
449  if (X_boxIdx==Nx-1)
450  {
451  xStart = startX + X_boxIdx*boxsize;
452  xLimit = finishX;
453  }
454  else
455  {
456  xStart = startX + X_boxIdx*boxsize;
457  xLimit = startX + (X_boxIdx+1)*boxsize;
458  }
459 
460  for (int Y_boxIdx=0; Y_boxIdx<Ny; ++Y_boxIdx)
461  {
462  if (Y_boxIdx==Ny-1)
463  {
464  yStart = startY + Y_boxIdx*boxsize;
465  yLimit = finishY;
466  }
467  else
468  {
469  yStart = startY + Y_boxIdx*boxsize;
470  yLimit = startY + (Y_boxIdx+1)*boxsize;
471  }
472 
473  N = 0;
474  long n = 0;
475  float sum2=0, sum=0;
476 
477  for (int k = startZ; k<finishZ; k++)
478  {
479  for (int i = yStart; i<yLimit; i++)
480  {
481  for (int j = xStart; j<xLimit; j++)
482  {
483  if (n%257 == 0)
484  {
485  float aux = A3D_ELEM(noiseMap, k, i, j);
486  sum += aux;
487  sum2 += aux*aux;
488  noiseVector.push_back( aux );
489  N++;
490  } //we take one voxel each 257 (prime number) points to reduce noise data
491  n++;
492  }
493  }
494  }
495  double std2, meanValue;
496  meanValue = (double ) noiseVector[size_t(noiseVector.size()*0.5)];
497  std2 = sqrt( (double) (sum2/N - (sum/N)*(sum/N)));
498 
499  std::sort(noiseVector.begin(),noiseVector.end());
500  MAT_ELEM(noiseMeanMatrix, Y_boxIdx, X_boxIdx) = meanValue;
501  MAT_ELEM(noiseStdMatrix, Y_boxIdx, X_boxIdx) = std2;
502 
503  double tileCenterY=0.5*(yLimit+yStart)-startY; // Translated to physical coordinates
504  double tileCenterX=0.5*(xLimit+xStart)-startX;
505  // Construction of the spline equation system
506  long idxSpline=0;
507  for(int controlIdxY = -1; controlIdxY < (lY - 1); ++controlIdxY)
508  {
509  double tmpY = Bspline03((tileCenterY / hY) - controlIdxY);
510 
511  VEC_ELEM(helperMean.b,idxBox)=MAT_ELEM(noiseMeanMatrix, Y_boxIdx, X_boxIdx);
512  VEC_ELEM(helperStd.b,idxBox)=MAT_ELEM(noiseStdMatrix, Y_boxIdx, X_boxIdx);
513 
514  if (tmpY == 0.0)
515  {
516  idxSpline+=lX;
517  continue;
518  }
519 
520  for(int controlIdxX = -1; controlIdxX < (lX - 1); ++controlIdxX)
521  {
522  double tmpX = Bspline03((tileCenterX / hX) - controlIdxX);
523  MAT_ELEM(helperStd.A, idxBox,idxSpline) = tmpY * tmpX;
524  MAT_ELEM(helperMean.A, idxBox,idxSpline) = tmpY * tmpX;
525  idxSpline+=1;
526  }
527  }
528  // x.push_back(tileCenterX);
529  // y.push_back(tileCenterY);
530  // t.push_back(MAT_ELEM(noiseMeanMatrix, Y_boxIdx, X_boxIdx));
531  // t.push_back(MAT_ELEM(noiseStdMatrix, Y_boxIdx, X_boxIdx));
532  noiseVector.clear();
533  idxBox+=1;
534  }
535  }
536 
537 
538  // Spline coefficients
539  Matrix1D<double> cij_std, cij_mean;
540  weightedLeastSquares(helperStd, cij_std);
541  weightedLeastSquares(helperMean, cij_mean);
542 
543  thresholdMatrix_mean.initZeros(Ydim, Xdim);
544  thresholdMatrix_std.initZeros(Ydim, Xdim);
545 
546  for (int i=0; i<Ydim; ++i)
547  {
548  for (int j=0; j<Xdim; ++j)
549  {
550  long idxSpline=0;
551 
552  for(int controlIdxY = -1; controlIdxY < (lY - 1); ++controlIdxY)
553  {
554  double tmpY = Bspline03((i / hY) - controlIdxY);
555 
556  if (tmpY == 0.0)
557  {
558  idxSpline+=lX;
559  continue;
560  }
561 
562  double xContrib_mean=0.0;
563  double xContrib_std=0.0;
564  for(int controlIdxX = -1; controlIdxX < (lX - 1); ++controlIdxX)
565  {
566  double tmpX = Bspline03((j / hX) - controlIdxX);
567  xContrib_mean += VEC_ELEM(cij_mean, idxSpline) * tmpX;// *tmpY;
568  xContrib_std += VEC_ELEM(cij_std, idxSpline) * tmpX;// *tmpY;
569  idxSpline+=1;
570  }
571  MAT_ELEM(thresholdMatrix_mean, i,j)+= (float) xContrib_mean*tmpY;
572  MAT_ELEM(thresholdMatrix_std, i,j)+= (float) xContrib_std*tmpY;
573  }
574  }
575  }
576 }

580 void ProgTomoConfidecenceMap::FDRcontrol(MultidimArray<float> &significanceMap)
581 {
582  int nelems = Xdim*Ydim;
583 
584  long n=0;
585  for (int k=0; k<Zdim; ++k)
586  {
587  std::vector<float> pValVector;
588  for (int i=0; i<Ydim; ++i)
589  {
590  for (int j=0; j<Xdim; ++j)
591  {
592  pValVector.push_back( DIRECT_MULTIDIM_ELEM(significanceMap, n) );
593  ++n;
594  }
595  }
596 
597  long nn = 0;
598  for (auto idx: sort_indexes(pValVector))
599  {
600  if ( pValVector[idx] <= fdr *(nn+1)/nelems)
601  {
602 
603  DIRECT_MULTIDIM_ELEM(significanceMap, k + idx) = 0;
604  }
605  ++nn;
606  }
607 
608  pValVector.clear();
609  }
