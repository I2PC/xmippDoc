45 template <typename T>
46 size_t CUDAFlexAlignCorrelate<T>::estimateBytesAlloc(bool alloc) {
47  const auto settings = getSettings();
48  auto bIT = CudaFFT<T>().estimatePlanBytes(settings);
49  auto bSignals = settings.sBytesBatch();
50  auto bFTs = settings.fBytesBatch();
51  auto bBuffer1 = settings.fBytesSingle() * mParams.bufferSize;
52  auto bBuffer2 = (mParams.dim.n() == mParams.bufferSize) ? 0 : bBuffer1;
53  auto bIndices = settings.batch() * sizeof(float); // index of the maxima
54  auto bPositions = settings.batch() * 2 * sizeof(float); // 2D position of the maxima
55  if (alloc) {
56  mIT = CudaFFT<T>::createPlan(mGpu, settings);
57  d_imgs = reinterpret_cast<T*>(BasicMemManager::instance().get(bSignals, MemType::CUDA));
58  d_ffts = reinterpret_cast<std::complex<T>*>(BasicMemManager::instance().get(bFTs, MemType::CUDA));
59  d_fftBuffer1 = reinterpret_cast<std::complex<T>*>(BasicMemManager::instance().get(bBuffer1, MemType::CUDA));
60  d_fftBuffer2 = reinterpret_cast<std::complex<T>*>(BasicMemManager::instance().get(bBuffer2, MemType::CUDA));
61  d_indices = reinterpret_cast<float*>(BasicMemManager::instance().get(bIndices, MemType::CUDA));
62  d_positions = reinterpret_cast<float*>(BasicMemManager::instance().get(bPositions, MemType::CUDA));
63  }
64  return bIT + bSignals + bFTs + bBuffer1 + bBuffer2 + bIndices + bPositions;
65 }

67 template<typename T>
68 void CUDAFlexAlignCorrelate<T>::copyInRightOrder(T* h_pos, bool isWithin,
69  int iStart, int iStop, int jStart, int jStop, size_t jSize,
70  size_t offset1, size_t offset2) {
71  constexpr size_t coordinates = 2; // we copy 2D coordinates
72  const auto signals = mParams.dim.n();
73  size_t counter = 0;
74  bool ready = false;
75  auto stream = *(cudaStream_t*)mGpu.stream();
76  for (int i = iStart; i <= iStop; i++) {
77  for (int j = isWithin ? i + 1 : 0; j < jSize; j++) {
78  if (!ready) {
79  ready = true;
80  j = jStart;
81  continue; // skip first iteration
82  }
83  if (ready) {
84  size_t actualI = offset1 + i;
85  size_t actualJ = offset2 + j;
86  size_t toCopy = (i == iStop) ? jStop - j + 1 : jSize - j;
87  // imagine correlation in layers, correlation of 0th img with other is first layer, 1st with other is second etc
88  // compute sum of images in complete layers
89  size_t imgsInPreviousLayers = (((signals - 1) + (signals - actualI)) * (actualI)) / 2;
90  size_t imgsInCurrentLayer = actualJ - actualI - 1;
91  gpuErrchk(cudaMemcpyAsync(h_pos + (coordinates * (imgsInPreviousLayers + imgsInCurrentLayer)),
92  d_positions + (counter * coordinates),
93  toCopy * coordinates * sizeof(float),
94  cudaMemcpyDeviceToHost, stream));
95  counter += toCopy;
96  break; // skip to next outer iteration
97  }
98  if ((iStop == i) && (jStop == j)) {
99  return;
100  }
101  }
102  }
103 }

105 template<typename T>
106 void CUDAFlexAlignCorrelate<T>::run(std::complex<T> *h_FTs, float *h_pos, float radius) {
107  if (!mReady)
108  REPORT_ERROR(ERR_VALUE_NOTSET, "Instance has not been initialized");
109  const auto maxRad = std::sqrt(static_cast<float>(mParams.dim.x() * mParams.dim.x() + mParams.dim.y() * mParams.dim.y()));
110  if (radius > maxRad)
111  REPORT_ERROR(ERR_ARG_INCORRECT, "Search radius too big");
112 
113  mGpu.set();
114 
115  const auto settings = getSettings();
116  const auto singleFTPixels = settings.fDim().xy();
117  const auto singleFTBytes = settings.fBytesSingle();
118 
119  const auto signals = mParams.dim.n();
120  const auto buffer2Size = signals == mParams.bufferSize ? 0 : mParams.bufferSize;
121  auto buffer1Offset = 0L;
122 
123  auto stream = *(cudaStream_t*)mGpu.stream();
124  do {
125  size_t buffer1ToCopy = std::min(mParams.bufferSize, signals - buffer1Offset);
126  size_t inputOffsetBuffer1 = buffer1Offset * singleFTPixels;
127  gpuErrchk(cudaMemcpyAsync(d_fftBuffer1, h_FTs + inputOffsetBuffer1,
128  buffer1ToCopy * singleFTBytes, cudaMemcpyHostToDevice, stream));
129 
130  // compute inter-buffer correlations
131  computeCorrelations(d_fftBuffer1, buffer1ToCopy, buffer1Offset,
132  d_fftBuffer1, buffer1ToCopy, buffer1Offset,
133  h_pos, radius);
134  size_t buffer2Offset = buffer1Offset + buffer1ToCopy;
135  while (buffer2Offset < signals) {
136  // copy other buffer
137  size_t buffer2ToCopy = std::min(buffer2Size, signals - buffer2Offset);
138  size_t inputOffsetBuffer2 = buffer2Offset * singleFTPixels;
139  gpuErrchk(cudaMemcpyAsync(d_fftBuffer2, h_FTs + inputOffsetBuffer2,
140  buffer2ToCopy * singleFTBytes, cudaMemcpyHostToDevice, stream));
141 
142  computeCorrelations(d_fftBuffer1, buffer1ToCopy, buffer1Offset,
143  d_fftBuffer2, buffer2ToCopy, buffer2Offset,
144  h_pos, radius);
145  buffer2Offset += buffer2ToCopy;
146  }
147  buffer1Offset += buffer1ToCopy;
148  } while (buffer1Offset < signals);
149  gpuErrchk(cudaPeekAtLastError());
150 }

152 template<typename T>
153 void CUDAFlexAlignCorrelate<T>::computeCorrelations(
154  std::complex<T> *d_in1, size_t in1Size, size_t in1Offset,
155  std::complex<T> *d_in2, size_t in2Size, size_t in2Offset,
156  T *h_pos, float maxDist) {
157  const auto isWithin = d_in1 == d_in2; // correlations are computed within the same buffer
158  const auto settings = getSettings();
159  auto stream = *(cudaStream_t*)mGpu.stream();
160  dim3 dimBlock(BLOCK_DIM, BLOCK_DIM);
161  dim3 dimGridCorr(ceil(settings.fDim().x()/(float)dimBlock.x), ceil(settings.fDim().y()/(float)dimBlock.y));
162 
163  auto counter = 0L;
164  int origI = 0;
165  int origJ = isWithin ? 0 : -1; // kernel must skip first iteration
166  for (int i = 0; i < in1Size; i++) {
167  for (int j = isWithin ? i + 1 : 0; j < in2Size; j++) {
168  counter++;
169  bool isLastIIter = isWithin ? (i == in1Size - 2) : (i == in1Size -1);
170  if (counter == settings.batch() || (isLastIIter && (j == in2Size -1)) ) {
171  // kernel must perform last iteration
172  // compute correlation from input buffers. Result are FFT signals with low frequencies in the middle
173  if (std::is_same<T, float>::value) {
174  correlate<<<dimGridCorr, dimBlock, 0, stream>>>(
175  reinterpret_cast<float2*>(d_in1), reinterpret_cast<float2*>(d_in2),
176  reinterpret_cast<float2*>(d_ffts), settings.fDim().x(), settings.fDim().y(),
177  isWithin, origI, i, origJ, j, in2Size);
178  } else {
179  throw std::logic_error("unsupported type");
180  }
181  // convert FD to SD
182  cudaMemsetAsync(d_imgs, 0, settings.sBytesBatch(), stream); // clean it to ensure that IFFT is properly computed
183  CudaFFT<T>::ifft(*reinterpret_cast<cufftHandle *>(mIT), d_ffts, d_imgs);
184  cudaMemsetAsync(d_indices, 0, counter * sizeof(float), stream); // clean it
185  // look for maxima - results are indices of the max position
186  ExtremaFinder::CudaExtremaFinder<T>::sFindMax2DAroundCenter(mGpu, settings.sDim().copyForN(counter), d_imgs, d_indices, nullptr, maxDist);
187  // now convert indices to float positions - we reuse the same memory block, but since we had images there, we should have more then enough space for that
188  cudaMemsetAsync(d_positions, 0, counter * sizeof(float) * 2, stream); // clean it
189  ExtremaFinder::CudaExtremaFinder<T>::sRefineLocation(mGpu, settings.sDim().copyForN(counter), d_indices, d_positions, d_imgs);
190  // send results to host
191  copyInRightOrder(h_pos, isWithin, origI, i, origJ, j, in2Size, in1Offset, in2Offset);
192  origI = i;
193  origJ = j;
194  counter = 0;
195  }
196  }
197  }
