33 // struct is used for compatibility with OpenCL / C for Cuda
34 /* Instance of this struct represent a 2D array of arbitrary type.
35  * Data are being stored as a dynamic 2D array (i.e. not continuous block of memory).
36  * Access to elements will be the fastest if traversing in Y -> X order. */
37 template<typename T>
38 struct Array2D {
39 public:
40  /* Empty constructor */
41  Array2D() : xSize(0), ySize(0), data(NULL) {};
42  /* Constructor, allocates the data immediately */
43  Array2D(int xSize, int ySize) :
44  xSize(xSize), ySize(ySize) {
45  data = new T*[ySize];
46  for (int y = 0; y < ySize; y++) {
47  data[y] = new T[xSize];
48  for (int x = 0; x < xSize; x++) {
49  data[y][x] = (T) 0;
50  }
51  }
52  #if DEBUG
53  std::cout << "Array2D created (" << getXSize() << "x" << getYSize()
54  << ") at " << data << std::endl;
55  #endif
56  }
57 
58  ~Array2D() {
59  for (int y = 0; y < ySize; y++) {
60  delete[] data[y];
61  }
62  delete[] data;
63  #if DEBUG
64  std::cout << "Array2D deleted (" << getXSize() << "x" << getYSize()
65  << ") at " << data << std::endl;
66  #endif
67  data = 0;
68  }
69 
70  /* Method to access elements of the array */
71  T& operator()(int x, int y) const {
72  #if DEBUG
73  if (0 > x || x >= xSize)
74  std::cout << "Array2D " << data << " X=" << x << " out of range [0.." << getXSize() << ")\n";
75  if (0 > y || y >= ySize)
76  std::cout << "Array2D " << data << " Y=" << y << " out of range [0.." << getYSize() << ")\n";
77  std::cout << std::flush;
78  #endif
79  return data[y][x];
80  }
81

83  return xSize;
84  }

88  }

90  bool inRange(int x, int y) const {
91  return inRangeX(x) && inRangeY(y);
92  }
93

95  return (x >= 0) && (x < xSize);
96  }
97

99  return (y >= 0) && (y < ySize);
100  }
101

103  return data[y];
104  }

109 };
