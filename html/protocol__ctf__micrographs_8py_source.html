211  if self.doInitialCTF:

213  prevValues = (self.ctfDict[micName] if micName in self.ctfDict
214  else self.getSinglePreviousParameters(mic.getObjId()))
215

219  else:
220  ma = self._params['maxDefocus']
221  mi = self._params['minDefocus']
222  localParams['defocusU'] = (ma + mi) / 2
223  localParams['defocus_range'] = (ma - mi) / 2
224

226  localParams['phaseShift0'] = self._params['phaseShift0']
227

233  finalName = micFn
234  def _getFn(key):
235  return self._getFileName(key, micBase=micBase, root=micDir)
236  localParams['root'] = _getFn('prefix')
237  downsampleList = self._calculateDownsampleList(mic.getSamplingRate())
238  try:
239  for i, downFactor in enumerate(downsampleList):
240  # Downsample if necessary
241  if downFactor != 1:
242  # Replace extension by 'mrc' cause there are some formats that
243  # cannot be written (such as dm3)
244  baseFn = pwutils.replaceBaseExt(micFn, 'mrc')

258  localParams['ctfmodelSize'] = self.windowSize.get()/2
259 
260  # Update _params dictionary with mic and micDir
261  localParams['micFn'] = finalName
262  localParams['samplingRate'] = mic.getSamplingRate() * downFactor
263 
264  # CTF estimation with Xmipp
265  params = self._args % localParams
266  params += " --downSamplingPerformed %f" % downFactor
267 
268  if not self.doInitialCTF:
269  params += " --selfEstimation "
270  else:
271  if not self.doOptimizeDefocus:
272  params += " --noDefocus --defocusV %(defocusV)f --azimuthal_angle %(defocusAngle)f" %\
273  localParams
274 
275  if not self.skipBorders.get():
276  params += " --skipBorders 0"
277 
278  self.runJob(self._program, params)
279 
280  # Check the quality of the estimation and reject it necessary
281  good = self.evaluateSingleMicrograph(mic)
282  if good:
283  break
284  for key in ['ctfParam', 'psd', 'enhanced_psd', 'ctfmodel_halfplane',
285  'ctfmodel_quadrant', 'ctf']:
286  pwutils.moveFile(_getFn(key), self._getExtraPath())
287 
288  except Exception as ex:
289  sys.stderr.write("xmipp_ctf_estimate_from_micrograph has " \
290  "failed with micrograph %s" % finalName)
