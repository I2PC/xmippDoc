86 void GPU::pinMemory(const void *h_mem, size_t bytes,
87  unsigned int flags) {
88  if (isMemoryPinned(h_mem)
89  && (isMemoryPinned((char*)h_mem + bytes - 1))) {
90  return;
91  }
92  assert(0 == cudaHostRegisterDefault); // default value should be 0
93  // check that it's aligned properly to the beginning of the page
94  if (0 != ((size_t)h_mem % 4096)) {
95  // otherwise the cuda-memcheck and cuda-gdb tends to randomly crash (confirmed on cuda 8 - cuda 10)
96  REPORT_ERROR(ERR_PARAM_INCORRECT, "Only pointer aligned to the page size can be registered");
97  }
98  // we remove const, but we don't change the data
99  gpuErrchk(cudaHostRegister(const_cast<void*>(h_mem), bytes, flags));
100 }
