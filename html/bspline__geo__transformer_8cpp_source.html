106 T *BSplineGeoTransformer<T>::interpolate(const std::vector<float> &matrices) {
107  bool isReady = this->isInitialized()
108  && this->isSrcSet();
109  if ( ! isReady) {
110  REPORT_ERROR(ERR_LOGIC_ERROR, "Instance is either not initialized or the 'original' has not been set.");
111  }
112  const Dimensions dims = this->getSettings().dims;
113  const size_t n = dims.n();
114  const size_t z = dims.z();
115  const size_t y = dims.y();
116  const size_t x = dims.x();
117 
118  auto futures = std::vector<std::future<void>>();
119 
120  auto workload = [&](int id, size_t signalId){
121  size_t offset = signalId * dims.sizeSingle();
122  auto in = MultidimArray<T>(1, z, y, x, const_cast<T*>(m_src + offset)); // removing const, but data should not be changed
123  auto out = MultidimArray<T>(1, z, y, x, m_dest.get() + offset);
124  in.setXmippOrigin();
125  out.setXmippOrigin();
126  // compensate the movement
127  Matrix2D<double> m(3,3);
128  const float *f = matrices.data() + (9 * signalId);
129  for (int i = 0; i < 9; ++i) {
130  m.mdata[i] = f[i];
131  }
132  applyGeometry(LINEAR, out, in, m, true, DONT_WRAP);
133  };
134 
135  for (size_t i = 0; i < n; ++i) {
136  futures.emplace_back(m_threadPool.push(workload, i));
137  }
138  for (auto &f : futures) {
139  f.get();
140  }
141  return m_dest.get();
142 }
