35 SparseMatrix2D::SparseMatrix2D(std::vector<SparseElement> &_elements, int _Nelements)
36 {
37  //First of all, we sort the elements by rows and then by columns
38  std::sort(_elements.begin(),_elements.end());
39 
40  size_t ln = _elements.size();
41  N = _Nelements;
42 
43  values.resizeNoCopy(ln);
44  jIdx.resizeNoCopy(ln);
45  iIdx.resizeNoCopy(N);
46 
47  int actualRow = -1;
48  int i = 0; // Iterator for the vectors "values" and "jIdx"
49 
50  for (size_t k=0 ; k< ln ; ++k )// Iterator for vector of SparseElemets
51  {
52  if(_elements.at(k).value != 0.0) // Searching that there isn't any zero value
53  {
54  DIRECT_MULTIDIM_ELEM(values,i) = _elements.at(k).value;
55  DIRECT_MULTIDIM_ELEM(jIdx,i) = _elements.at(k).j +1;
56 
57  int rse = _elements.at(k).i;
58  while( rse > actualRow )
59  {
60  actualRow++;
61  if( rse == actualRow )
62  // The first element in row number "x" is in position iIdx(x-1)
63  DIRECT_MULTIDIM_ELEM(iIdx,actualRow) = i+1;
64 
65  else
66  // If there isn't any nonzero value on this row, the value in this vector is 0
67  DIRECT_MULTIDIM_ELEM(iIdx,actualRow) = 0;
68  }
69  ++i;
70  }
71  }

190 void SparseMatrix2D::multMM(const SparseMatrix2D &X, SparseMatrix2D &Y)
191 {
192  size_t nnz=XSIZE(X.values);
193 
194  Y.N = X.N;
195  Y.values.initZeros(nnz);
196  Y.jIdx.initZeros(nnz);
197  Y.iIdx.initZeros(N);
198 
199  size_t nnzY = 0;
200  int Nelems = X.N;
201  for(int row = 0; row < Nelems ; ++row){
202  int firstElemRow = 0;
203  for(int col = 0; col < Nelems ; ++col){
204  double aij = 0.0;
205  for(int k = 0; k < Nelems ; ++k ){

208  double val2=X.getElemIJ(k,col);
209  aij += val * val2;
210  }
211  }
212 
213  if(aij != 0.0){ // If there is a nonzero element in (row,col) position, we include that in the matrix
214  if(firstElemRow == 0)
215  firstElemRow = nnzY +1;
216  DIRECT_MULTIDIM_ELEM(Y.values,nnzY) = aij;
217  DIRECT_MULTIDIM_ELEM(Y.jIdx,nnzY) = col+1;
218  ++nnzY;
219  }
220  }

222  DIRECT_MULTIDIM_ELEM(Y.iIdx,row) = firstElemRow;

230  * The matrix D is diagonal.
231  *
232  * | a11 a12 a13| | d11 0 0 | | a11*d1 a12*d1 a13*d1|
233  * | a21 a22 a23| * | 0 d22 0 | = | a21*d2 a22*d2 a23*d2|
234  * | a31 a32 a33| | 0 0 a33 | | a31*d3 a32*d3 a33*d3|
235  *

237  * Vector Values is the same as the non diagonal matrix multiply each row by each d_row

241  */
242 void SparseMatrix2D::multMMDiagonal(const MultidimArray<double> &D, SparseMatrix2D &Y)
243 {
244  int actualRow = N-1;
245 
246  Y=*this;
247 
248  // We see where is the first element in the next row
249  int until = DIRECT_MULTIDIM_ELEM(iIdx,actualRow) -1;
250 
251  // Obtain the value that we have to multiply by the first row.
252  double dx = DIRECT_MULTIDIM_ELEM(D,actualRow);
253  size_t nnz=XSIZE(values);
254  for(int i = nnz-1; i >= 0 ; --i)
255  {
256  // Yij = Aij * Dii / Y = A => Yij = Yij * Dii
257  DIRECT_MULTIDIM_ELEM(Y.values,i) *= dx;

259  if(until == i && i > 0){

263  --actualRow;
