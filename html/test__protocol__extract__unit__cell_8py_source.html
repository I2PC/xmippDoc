294 def generate(sym='I2n5', mode='xmipp', suffix="_i2", offset=0.):
295  offset = math.radians(offset)
296  symPreffix = sym[:1]
297  symSuffix = sym[1:]
298  if mode == 'xmipp':
299  suffix += ".feat"
300  else:
301  suffix += ".bild"
302 
303  (fd, filename) = mkstemp(suffix=suffix)
304  f = os.fdopen(fd, "w")
305 
306  if mode == 'xmipp':
307  f.write("""# Phantom description file, (generated with phantom help)
308 # General Volume Parameters:
309 # Xdim Ydim Zdim Background_Density Scale
310  3 3 3 0 60
311 # Feature Parameters:
312 #Type +/= Density X_Center Y_Center Z_Center
313 """)
314  else:
315  f.write(""".scale 60
316 .comment system of coordinates
317 .color 1 0 0
318 .arrow 0 0 0 2 0 0 0.025
319 .color 0 1 0
320 .arrow 0 0 0 0 2 0 0.025
321 .color 0 0 1
322 .arrow 0 0 0 0 0 2 0.025
323 .comment vertices
324 .color red
325 """)
326  if symPreffix == 'I':
327  generate_ico(symSuffix, mode, f)
328  elif symPreffix == 'C':
329  generate_cyclic(int(symSuffix), offset, mode, f)
330  elif symPreffix == 'D':
331  generate_dihedral(int(symSuffix), offset, mode, f)
332  elif symPreffix == 'T':
333  generate_tetrahedral(mode, f)
334  elif symPreffix == 'O':
335  generate_octahedral(mode, f)
336 
337  f.close()
338  return filename
