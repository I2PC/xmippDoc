102  double affine_fitness_individual(double *p)
103  {
104  // Check limits
105  if (!showMode)
106  FOR_ALL_ELEMENTS_IN_MATRIX1D(minAllowed)
107  if (p[i]<minAllowed(i) || p[i]>maxAllowed(i))
108  return 1e20;
109 
110  // Separate solution
111  Matrix2D<double> A12, A21;
112  A12.initIdentity(3);
113  A12(0,0)=p[0];
114  A12(0,1)=p[1];
115  A12(0,2)=p[4];
116  A12(1,0)=p[2];
117  A12(1,1)=p[3];
118  A12(1,2)=p[5];
119 
120  A21=A12.inv();
121 
122  // Check it is approximately a rotation
123  if (!showMode && checkRotation)
124  {
125  // Check that the eigenvalues are close to 1
126  std::complex<double> a=A12(0,0);
127  std::complex<double> b=A12(0,1);
128  std::complex<double> c=A12(1,0);
129  std::complex<double> d=A12(1,1);
130  std::complex<double> eig1=(a+d)/2.0+sqrt(4.0*b*c+(a-d)*(a-d))/2.0;
131  std::complex<double> eig2=(a+d)/2.0-sqrt(4.0*b*c+(a-d)*(a-d))/2.0;
132  double abs_eig1=abs(eig1);
133  double abs_eig2=abs(eig2);
134 
135  if (abs_eig1<0.95 || abs_eig1>1.05)
136  return 1e20*ABS(abs_eig1-1);
137  if (abs_eig2<0.95 || abs_eig2>1.05)
138  return 1e20*ABS(abs_eig2-1);
139 
140  // Compute the eigenvalues of A*A^t disregarding the shifts
141  // A*A^t=[a b; c d]
142 
143  a=A12(0,0)*A12(0,0)+A12(0,1)*A12(0,1);
144  b=A12(0,0)*A12(1,0)+A12(0,1)*A12(1,1);
145  c=b;
146  d=A12(1,0)*A12(1,0)+A12(1,1)*A12(1,1);
147  eig1=(a+d)/2.0+sqrt(4.0*b*c+(a-d)*(a-d))/2.0;
148  eig2=(a+d)/2.0-sqrt(4.0*b*c+(a-d)*(a-d))/2.0;
149  abs_eig1=abs(eig1);
150  abs_eig2=abs(eig2);
151 
152  if (abs_eig1<0.95 || abs_eig1>1.05)
153  return 1e20*ABS(abs_eig1-1);
154  if (abs_eig2<0.95 || abs_eig2>1.05)
155  return 1e20*ABS(abs_eig2-1);
156  if (abs(c)>0.05)
157  return 1e20*ABS(abs(c));
158  }
159 
160  // For each pyramid level
161  double dist=0;
162  Matrix2D<double> A12level=A12;
163  Matrix2D<double> A21level=A21;
164  for (size_t level=0; level<I1.size(); level++)
165  {
166  const MultidimArray<double> &Mask1_level=*(Mask1[level]);
167  const MultidimArray<double> &Mask2_level=*(Mask2[level]);
168 
169  // Produce the transformed images
170  MultidimArray<double> transformedI1, transformedI2;
171  applyGeometry(LINEAR,transformedI1,*(I1[level]),A12level,IS_NOT_INV,DONT_WRAP);
172  applyGeometry(LINEAR,transformedI2,*(I2[level]),A21level,IS_NOT_INV,DONT_WRAP);
173 
174  // Produce masks for the comparison
175  MultidimArray<int> maskInTheSpaceOf1, maskInTheSpaceOf2;
176  MultidimArray<double> maskAux;
177  applyGeometry(LINEAR,maskAux,Mask1_level,A12level,IS_NOT_INV,DONT_WRAP);
178  maskInTheSpaceOf2.initZeros(YSIZE(maskAux),XSIZE(maskAux));
179  maskInTheSpaceOf2.setXmippOrigin();
180  FOR_ALL_ELEMENTS_IN_ARRAY2D(maskAux)
181  {
182  if (A2D_ELEM(maskAux,i,j)>0.5)
183  A2D_ELEM(maskInTheSpaceOf2,i,j)=(int)A2D_ELEM(Mask2_level,i,j);
184  }
185 
186  maskAux.initZeros();
187  applyGeometry(LINEAR,maskAux,Mask2_level,A21level,IS_NOT_INV,DONT_WRAP);
188  maskInTheSpaceOf1.initZeros(YSIZE(maskAux),XSIZE(maskAux));
189  maskInTheSpaceOf1.setXmippOrigin();
190  FOR_ALL_ELEMENTS_IN_ARRAY2D(maskAux)
191  {
192  if (A2D_ELEM(maskAux,i,j)>0.5)
193  A2D_ELEM(maskInTheSpaceOf1,i,j)=(int)A2D_ELEM(Mask1_level,i,j);
194  }
195 
196  // Compare the two images
197  double distLevel=0.5*(
198  1-correlationIndex(transformedI1,*(I2[level]),&maskInTheSpaceOf2)+
199  1-correlationIndex(transformedI2,*(I1[level]),&maskInTheSpaceOf1));
200 
201  if (showMode)
202  {
203  Image<double> save;
204  save()=*(I1[level]);
205  save.write("PPPimg1.xmp");
206  save()=*(I2[level]);
207  save.write("PPPimg2.xmp");
208  save()=*(Mask1[level]);
209  save.write("PPPmask1.xmp");
210  save()=*(Mask2[level]);
211  save.write("PPPmask2.xmp");
212  save()=transformedI1;
213  save.write("PPPTransformedImg1.xmp");
214  save()=transformedI2;
215  save.write("PPPTransformedImg2.xmp");
216  typeCast(maskInTheSpaceOf1,save());
217  save.write("PPPmaskInTheSpaceOf1.xmp");
218  typeCast(maskInTheSpaceOf2,save());
219  save.write("PPPmaskInTheSpaceOf2.xmp");
220  save()=*(I1[level])-transformedI2;
221  save.write("PPPDiffImg1.xmp");
222  save()=*(I2[level])-transformedI1;
223  save.write("PPPDiffImg2.xmp");
224  std::cout << "Level=" << level << "\nA12level=\n" << A12level << "A21level=\n" << A21level << std::endl;
225  std::cout << "distLevel=" << distLevel << std::endl;
226  std::cout << "Do you like the alignment? (y/n)\n";

231  dist=-1;

235  dist+=distLevel;
236

238  A12level(1,2)/=2;
239  A21level(0,2)/=2;
240  A21level(1,2)/=2;
241  }
242  dist/=I1.size();
243 
244  return dist;
245  }

253 void setupAffineFitness(AffineFitness &fitness, const MultidimArray<double> &I1,
254  const MultidimArray<double> &I2, int maxShift, bool isMirror,
255  bool checkRotation, int pyramidLevel)
256 {
257  fitness.checkRotation=checkRotation;
258 
259  // Set images
260  int level=0;
261  MultidimArray<double> I1aux=I1;
262  MultidimArray<double> I2aux=I2;
263 
264  // Remove the borders
265  int borderY=CEIL(0.1*YSIZE(I1));
266  int borderX=CEIL(0.1*XSIZE(I1));
267  I1aux.selfWindow(STARTINGY(I1)+borderY,STARTINGX(I1)+borderX,
268  FINISHINGY(I1)-borderY,FINISHINGX(I1)-borderX);
269  I2aux.selfWindow(STARTINGY(I1)+borderY,STARTINGX(I1)+borderX,
270  FINISHINGY(I1)-borderY,FINISHINGX(I1)-borderX);
271 
272  MultidimArray<double> Mask1;
273  Mask1.initZeros(I1aux);
274  FOR_ALL_ELEMENTS_IN_ARRAY2D(I1aux)
275  if (A2D_ELEM(I1,i,j)!=0)
276  A2D_ELEM(Mask1,i,j)=1;
277 
278  MultidimArray<double> Mask2;
279  Mask2.initZeros(I2aux);
280  FOR_ALL_ELEMENTS_IN_ARRAY2D(I2aux)
281  if (A2D_ELEM(I2,i,j)!=0)
282  A2D_ELEM(Mask2,i,j)=1;
283 
284  do
285  {
286  // Push back the current images
287  I1aux.setXmippOrigin();
288  I2aux.setXmippOrigin();
289  Mask1.setXmippOrigin();
290  Mask2.setXmippOrigin();
291 
292  MultidimArray<double> *dummy=NULL;
293  dummy=new MultidimArray<double>;
294  *dummy=I1aux;
295  fitness.I1.push_back(dummy);
296  dummy=new MultidimArray<double>;
297  *dummy=I2aux;
298  fitness.I2.push_back(dummy);
299  dummy=new MultidimArray<double>;
300  *dummy=Mask1;
301  fitness.Mask1.push_back(dummy);
302  dummy=new MultidimArray<double>;
303  *dummy=Mask2;
304  fitness.Mask2.push_back(dummy);
305 
306  // Prepare for next level
307  level++;
308  if (pyramidLevel>=level)
309  {
310  selfScaleToSize(LINEAR,I1aux,YSIZE(I1aux)/2,XSIZE(I1aux)/2);
311  selfScaleToSize(LINEAR,I2aux,YSIZE(I2aux)/2,XSIZE(I2aux)/2);
312  selfScaleToSize(LINEAR,Mask1,YSIZE(Mask1)/2,XSIZE(Mask1)/2);
313  selfScaleToSize(LINEAR,Mask2,YSIZE(Mask2)/2,XSIZE(Mask2)/2);
314  FOR_ALL_ELEMENTS_IN_ARRAY2D(Mask1)
315  Mask1(i,j)=(Mask1(i,j)>0.5)? 1:0;
316  FOR_ALL_ELEMENTS_IN_ARRAY2D(Mask2)
317  Mask2(i,j)=(Mask2(i,j)>0.5)? 1:0;
318  }
319  }
320  while (level<=pyramidLevel);
321 
322  // Set limits for the affine matrices
323  // Order: 1->2: 4 affine params+2 translations
324  // Order: 2->1: 4 affine params+2 translations
325  fitness.minAllowed.resize(6);
326  fitness.maxAllowed.resize(6);
327 
328  // Scale factors
329  fitness.minAllowed(0)=fitness.minAllowed(3)=0.9;
330  fitness.maxAllowed(0)=fitness.maxAllowed(3)=1.1;
331  if (isMirror)
332  {
333  fitness.minAllowed(3)=-1.1;
334  fitness.maxAllowed(3)=-0.9;
335  }
336 
337  // Rotation factors
338  fitness.minAllowed(1)=fitness.minAllowed(2)=-0.2;
339  fitness.maxAllowed(1)=fitness.maxAllowed(2)= 0.2;
340 
341  // Shifts
342  fitness.minAllowed(4)=fitness.minAllowed(5)=-maxShift;
343  fitness.maxAllowed(4)=fitness.maxAllowed(5)= maxShift;
344 }

346 static pthread_mutex_t globalAffineMutex = PTHREAD_MUTEX_INITIALIZER;
347 double computeAffineTransformation(const MultidimArray<unsigned char> &I1,
348  const MultidimArray<unsigned char> &I2, int maxShift, int maxIterDE,
349  Matrix2D<double> &A12, Matrix2D<double> &A21, bool show,
350  double thresholdAffine, bool globalAffine, bool isMirror,
351  bool checkRotation, int pyramidLevel)
352 {
353  try
354  {
355  AffineFitness affy;
356  MultidimArray<double> I1d, I2d;
357  typeCast(I1, I1d);
358  typeCast(I2, I2d);
359  setupAffineFitness(affy, I1d, I2d, maxShift, isMirror, checkRotation,
360  pyramidLevel);
361 
362  // Return result
363  double cost;
364 
365  // Optimize with differential evolution
366  Matrix1D<double> A(6);
367  A(0)=A(3)=1;
368  if (isMirror)
369  A(3)*=-1;
370  if (MAT_XSIZE(A12)==0)
371  {
372  if (globalAffine)
373  {
374  // Exhaustive search
375  double bestCost=2;
376  double stepX=(affy.maxAllowed(4)-affy.minAllowed(4))/40.0;
377  double stepY=(affy.maxAllowed(5)-affy.minAllowed(5))/40.0;
378  for (double shiftY=affy.minAllowed(5); shiftY<=affy.maxAllowed(5); shiftY+=stepY)
379  for (double shiftX=affy.minAllowed(4); shiftX<=affy.maxAllowed(4); shiftX+=stepX)
380  {
381  A(4)=shiftX;
382  A(5)=shiftY;
383  double cost=affy.affine_fitness_individual(MATRIX1D_ARRAY(A));
384  if (cost<bestCost)
385  bestCost=cost;
386  }
387  }
388  else
389  {
390  // Initialize with cross correlation
391  double tx, ty;
392  pthread_mutex_lock( &globalAffineMutex );
393  CorrelationAux aux;
394  if (!isMirror)
395  bestShift(I1d,I2d,tx,ty,aux);
396  else
397  {
398  MultidimArray<double> auxI2d=I2d;
399  auxI2d.selfReverseY();
400  STARTINGX(auxI2d)=STARTINGX(I2d);
401  STARTINGY(auxI2d)=STARTINGY(I2d);
402  bestShift(I1d,auxI2d,tx,ty,aux);
403  ty=-ty;
404  }
405  pthread_mutex_unlock( &globalAffineMutex );
406  A(4)=-tx;
407  A(5)=-ty;
408  }
409 
410  // Optimize with Powell
411  Matrix1D<double> steps(A);
412  steps.initConstant(1);
413  int iter;
414  powellOptimizer(A, 1, VEC_XSIZE(A),
415  AffineFitness::Powell_affine_fitness_individual, &affy, 0.005,
416  cost, iter, steps, false);
417 
418  // Separate solution
419  A12.initIdentity(3);
420  A12(0,0)=A(0);
421  A12(0,1)=A(1);
422  A12(0,2)=A(4);
423  A12(1,0)=A(2);
424  A12(1,1)=A(3);
425  A12(1,2)=A(5);
426 
427  A21=A12.inv();
428  }
429 
430  if (show)
431  {
432  affy.showMode=true;
433  Matrix1D<double> p(6);
434  p(0)=A12(0,0);
435  p(1)=A12(0,1);
436  p(4)=A12(0,2);
437  p(2)=A12(1,0);
438  p(3)=A12(1,1);
439  p(5)=A12(1,2);
440  cost = affy.affine_fitness_individual(MATRIX1D_ARRAY(p));
441  affy.showMode=false;
442  }
443 
444  return cost;
445  }
446  catch (XmippError &XE)
447  {
448  std::cout << XE;
449  exit(1);
450  }
451 }
