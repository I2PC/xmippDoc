31 {
32  if (N < 2 || N > 9)
33  REPORT_ERROR(ERR_VALUE_INCORRECT, "integrateNewtonCotes: N must be greater than 1");

35  Matrix1D<double> fx(N);

37  fx(i) = (*f)(a + i * h);
38  switch (N)
39  {
40  case 2:
