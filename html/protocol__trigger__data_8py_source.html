39 class XmippProtTriggerData(EMProtocol):
40  """
41  Waits until certain number of images is prepared and then
42  send them to output.
43  It can be done in 3 ways:
44  - If *Send all particles to output?*' is _No_:
45  Once the number of images is reached, a setOfImages is returned and
46  the protocols finished (ending the streaming from this point).
47  - If *Send all particles to output?*' is _Yes_ and:
48  - If *Split particles to multiple sets?* is _Yes_:
49  Multiple closed outputs will be returned as soon as
50  the number of images is reached.
51  - If *Split particles to multiple sets?* is _No_:
52  Only one output is returned and it is growing up in batches of
53  a certain number of images (completely in streaming).
54  """
55  _label = 'trigger data'
56  _lastUpdateVersion = VERSION_2_0
