229  PyTuple_SetItem(angles, 0, PyFloat_FromDouble(rot));
230  PyTuple_SetItem(angles, 1, PyFloat_FromDouble(tilt));
231  PyTuple_SetItem(angles, 2, PyFloat_FromDouble(psi));
232  PyList_SetItem(retval,0,angles);
233 
234  Matrix2D<double> L(4, 4), R(4, 4), E, Ep;
235  Euler_angles2matrix(rot,tilt,psi,E,false);
236  for (int isym = 0; isym < symlist.symsNo(); isym++)
237  {
238  symlist.getMatrices(isym, L, R);
239  R.resize(3, 3);
240  L.resize(3, 3);
241 
242  Ep=L*E*R;
243  Euler_matrix2angles(Ep,rot,tilt,psi);
244 
245  angles = PyTuple_New(3);
246  PyTuple_SetItem(angles, 0, PyFloat_FromDouble(rot));
247  PyTuple_SetItem(angles, 1, PyFloat_FromDouble(tilt));
248  PyTuple_SetItem(angles, 2, PyFloat_FromDouble(psi));
249  PyList_SetItem(retval,isym+1,angles);
250  }
251 
252  return retval;
