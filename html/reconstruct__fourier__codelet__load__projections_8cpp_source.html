38 static void loadMatrices(const MDRow& row, Matrix2D<double>& shiftMatrix, Matrix2D<double>& orientationMatrix) {
39  if (!row.containsLabel(MDL_TRANSFORM_MATRIX)) {
40  geo2TransformationMatrix(row, shiftMatrix, /*only_apply_shifts*/ true);
41  } else {
42  //TODO This ignores only_apply_shifts!
43  String matrixStr;
44  row.getValue(MDL_TRANSFORM_MATRIX, matrixStr);
45  string2TransformationMatrix(matrixStr, shiftMatrix, 3);
46  }
47 
48  // Compute the coordinate axes associated to this projection
49  double rot = 0, tilt = 0, psi = 0;
50  row.getValue(MDL_ANGLE_ROT, rot);
51  row.getValue(MDL_ANGLE_TILT, tilt);
52  row.getValue(MDL_ANGLE_PSI, psi);
53  Euler_angles2matrix(rot, tilt, psi, orientationMatrix);
54  orientationMatrix = orientationMatrix.transpose();
55 }

151  }
152 
153  projectionIndex++;
154  }
