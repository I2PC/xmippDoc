140 // Produce side information ================================================
141 void ProgAngularDiscreteAssign::preProcess()
142 {
143  // Read input reference image names
144  SF_ref.read(fn_ref);
145  size_t refYdim, refXdim, refZdim, refNdim;
146  getImageSize(SF_ref,refYdim, refXdim, refZdim, refNdim);
147  if (refYdim != NEXT_POWER_OF_2(refYdim) || refXdim != NEXT_POWER_OF_2(refXdim))
148  REPORT_ERROR(ERR_MULTIDIM_SIZE,
149  "reference images must be of a size that is power of 2");
150 
151  // Produce side info of the angular distance computer
152  distance_prm.fn_ang1 = distance_prm.fn_ang2 = "";
153  distance_prm.fn_sym = fn_sym;
154  distance_prm.produce_side_info();
155 
156  // Read the angle file
157  rot.resize(SF_ref.size());
158  tilt.resize(SF_ref.size());
159  int i = 0;
160  FOR_ALL_OBJECTS_IN_METADATA(SF_ref)
161  {
162  SF_ref.getValue(MDL_ANGLE_ROT, rot[i],__iter.objId);
163  SF_ref.getValue(MDL_ANGLE_TILT, tilt[i],__iter.objId);
164  i++;
165  }
166 
167  // Build mask for subbands
168  Mask_no.resize(refYdim, refXdim);
169  Mask_no.initConstant(-1);
170 
171  if (smax == -1)
172  smax = Get_Max_Scale(refYdim) - 3;
173  SBNo = (smax - smin + 1) * 3 + 1;
174  SBsize.resize(SBNo);
175 
176  Mask Mask(INT_MASK);
177  Mask.type = BINARY_DWT_CIRCULAR_MASK;
178  Mask.R1 = CEIL((double)refXdim / 2.0);
179  Mask.resize(refYdim, refXdim);
180 
181  int m = 0, s;
182  for (s = smax; s >= smin; s--)
183  {
184  for (int q = 0; q <= 3; q++)
185  {
186  if (q == 0 && s != smax)
187  continue;
188  Mask.smin = s;
189  Mask.smax = s;
190  Mask.quadrant = Quadrant2D(q);
191  Mask.generate_mask();
192 
193  const MultidimArray<int> imask2D=Mask.get_binary_mask();
194  FOR_ALL_DIRECT_ELEMENTS_IN_ARRAY2D(imask2D)
195  if (DIRECT_A2D_ELEM(imask2D, i, j))
196  {
197  Mask_no(i, j) = m;
198  SBsize(m)++;
199  }
200 
201  m++;
202  }
203  }
204 
205  // Produce library
206  produce_library();
207 
208  // Save a little space
209  SF_ref.clear();
210 }

219 // Produce library -----------------------------------------------------------
220 void ProgAngularDiscreteAssign::produce_library()
221 {
222  Image<double> I;
223  int number_of_imgs = SF_ref.size();
224  set_DWT_type(DAUB12);
225 
226  // Create space for all the DWT coefficients of the library
227  Matrix1D<int> SBidx(SBNo);
228  for (int m = 0; m < SBNo; m++)
229  {
230  MultidimArray<double> *subband = new MultidimArray<double>;
231  subband->resize(number_of_imgs, SBsize(m));
232  library.push_back(subband);
233  }
234  library_power.initZeros(number_of_imgs, SBNo);
235 
236  if (verbose)
237  {
238  std::cerr << "Generating reference library ...\n";
239  init_progress_bar(number_of_imgs);
240  }
241  int n = 0, nstep = XMIPP_MAX(1, number_of_imgs / 60); // For progress bar
242  FOR_ALL_OBJECTS_IN_METADATA(SF_ref)
243  {
244  I.readApplyGeo(SF_ref,__iter.objId);
245  library_name.push_back(I.name());
246 
247  // Make and distribute its DWT coefficients in the different PCA bins
248  I().statisticsAdjust(0., 1.);
249  DWT(I(), I());
250  SBidx.initZeros();
251  FOR_ALL_ELEMENTS_IN_ARRAY2D(Mask_no)
252  {
253  int m = Mask_no(i, j);
254  if (m != -1)
255  {
256  double coef = I(i, j), coef2 = coef * coef;
257  (*library[m])(n, SBidx(m)++) = coef;
258  for (int mp = m; mp < SBNo; mp++)
259  library_power(n, mp) += coef2;
260  }
261  }
262 
263  // Prepare for next iteration
264  if (++n % nstep == 0 && verbose)
265  progress_bar(n);
266  }
267  if (verbose)
268  progress_bar(SF_ref.size());
269 }

301 // Refine candidate list ---------------------------------------------------
302 void ProgAngularDiscreteAssign::refine_candidate_list_with_correlation(
303  int m,
304  Matrix1D<double> &dwt,
305  bool *candidate_list, std::vector<double> &cumulative_corr,
306  Matrix1D<double> &x_power, std::vector<double> &sumxy,
307  double th)
308 {
309  int dimp = SBsize(m);
310  int imax = rot.size();
311  const MultidimArray<double> &library_m = *(library[m]);
312  std::vector<double> sortedCorr;
313  sortedCorr.reserve(imax);
314  for (int i = 0; i < imax; i++)
315  {
316  if (candidate_list[i])
317  {
318  double sumxyp = 0.0;
319  // Loop unrolling
320  unsigned long jmax=4*(dimp/4);
321  for (int j = 0; j < dimp; j+=4)
322  {
323  int j_1=j+1;
324  int j_2=j+2;
325  int j_3=j+3;
326  sumxyp += VEC_ELEM(dwt,j) * DIRECT_A2D_ELEM(library_m,i, j) +
327  VEC_ELEM(dwt,j_1) * DIRECT_A2D_ELEM(library_m,i, j_1) +
328  VEC_ELEM(dwt,j_2) * DIRECT_A2D_ELEM(library_m,i, j_2) +
329  VEC_ELEM(dwt,j_3) * DIRECT_A2D_ELEM(library_m,i, j_3);
330  }
331  for (int j = jmax+1;j<dimp;++j)
332  sumxyp += VEC_ELEM(dwt,j) * DIRECT_A2D_ELEM(library_m,i, j);
333 
334  sumxy[i] += sumxyp;
335 
336  double corr = sumxy[i] / sqrt(DIRECT_A2D_ELEM(library_power,i, m) *
337  VEC_ELEM(x_power,m));
338  cumulative_corr[i] = corr;
339  sortedCorr.push_back(corr);
340 
341  if (tell & TELL_ROT_TILT)
342  {
343  std::cout << "Candidate " << i << " corr= " << cumulative_corr[i]
344  << " rot= " << rot[i] << " tilt= " << tilt[i] << std::endl;
345  }
346  }
347  }
348  std::sort(sortedCorr.begin(),sortedCorr.end());
349  int idx=(int)floor(sortedCorr.size()*(1-th/100.0));
350 
351  double corr_th = sortedCorr[idx];
352 
353  // Remove all those projections below the threshold
354  for (int i = 0; i < imax; i++)
355  if (candidate_list[i])
356  candidate_list[i] = (cumulative_corr[i] >= corr_th);
357 
358  // Show the percentil used
359  if (tell & TELL_ROT_TILT)
360  std::cout << "# Percentil " << corr_th << std::endl << std::endl;
361 }

363 // Predict rot and tilt ----------------------------------------------------
364 double ProgAngularDiscreteAssign::predict_rot_tilt_angles(Image<double> &I,
365  double &assigned_rot, double &assigned_tilt, int &best_ref_idx)
366 {
367  if (XSIZE(I()) != NEXT_POWER_OF_2(XSIZE(I())) ||
368  YSIZE(I()) != NEXT_POWER_OF_2(YSIZE(I())))
369  REPORT_ERROR(ERR_MULTIDIM_SIZE,
370  "experimental images must be of a size that is power of 2");
371 
372  // Build initial candidate list
373  bool* candidate_list=new bool[rot.size()];
374  std::vector<double> cumulative_corr;
375  std::vector<double> sumxy;
376  build_ref_candidate_list(I, candidate_list, cumulative_corr, sumxy);
377 
378  // Make DWT of the input image and build vectors for comparison
379  std::vector<Matrix1D<double> * > Idwt;
380  Matrix1D<double> x_power(SBNo);
381  x_power.initZeros();
382  Matrix1D<int> SBidx(SBNo);
383  SBidx.initZeros();
384  for (int m = 0; m < SBNo; m++)
385  {
386  Matrix1D<double> *subband = new Matrix1D<double>;
387  subband->resize(SBsize(m));
388  Idwt.push_back(subband);
389  }
390 
391  I().statisticsAdjust(0., 1.);
392  DWT(I(), I());
393  FOR_ALL_DIRECT_ELEMENTS_IN_ARRAY2D(Mask_no)
394  {
395  int m = DIRECT_A2D_ELEM(Mask_no,i, j);
396  if (m != -1)
397  {
398  double coef = DIRECT_A2D_ELEM(IMGMATRIX(I), i, j), coef2 = coef * coef;
399  (*(Idwt[m]))(SBidx(m)++) = coef;
400  for (int mp = m; mp < SBNo; mp++)
401  VEC_ELEM(x_power,mp) += coef2;
402  }
403  }
404 
405  // Measure correlation for all possible PCAs
406  // These variables are used to compute the correlation at a certain
407  // scale
408  for (int m = 0; m < SBNo; m++)
409  {
410  // Show image name
411  if (tell & TELL_ROT_TILT)
412  std::cout << "# " << I.name() << " m=" << m
413  << " current rot=" << I.rot()
414  << " current tilt=" << I.tilt() << std::endl;
415  refine_candidate_list_with_correlation(m, *(Idwt[m]),
416  candidate_list, cumulative_corr,
417  x_power, sumxy, th_discard);
418  }
419 
420  // Select the maximum
421  int best_i = -1;
422  bool first = true;
423  int N_max = 0;
424  int imax = rot.size();
425  for (int i = 0; i < imax; i++)
426  if (candidate_list[i])
427  {
428  if (first)
429  {
430  best_i = i;
431  first = false;
432  N_max = 1;
433  }
434  else if (cumulative_corr[i] > cumulative_corr[best_i])
435  best_i = i;
436  else if (cumulative_corr[i] == cumulative_corr[best_i])
437  N_max++;
438  }
439 
440  if (N_max == 0)
441  {
442  if (verbose)
443  std::cerr << "Predict_angles: Empty candidate list for image "
444  << I.name() << std::endl;
445  assigned_rot = I.rot();
446  assigned_tilt = I.tilt();
447  return 0;
448  }
449 
450  // There are several maxima, choose one randomly
451  if (N_max != 1)
452  {
453  int selected = FLOOR(rnd_unif(0, 3));
454  for (int i = 0; i < imax && selected >= 0; i++)
455  if (candidate_list[i])
456  if (cumulative_corr[i] == cumulative_corr[best_i])
457  {
458  best_i = i;
459  selected--;
460  }
461  }
462 
463  // Free asked memory
464  for (int m = 0; m < SBNo; m++)
465  delete Idwt[m];
466 
467  assigned_rot = rot[best_i];
468  assigned_tilt = tilt[best_i];
469  best_ref_idx = best_i;
470  delete [] candidate_list;
471  return cumulative_corr[best_i];
472 }

474 // Evaluate candidates by correlation ----------------------------------------
475 double ProgAngularDiscreteAssign::evaluate_candidates(
476  const std::vector<double> &vscore,
477  const std::vector<int> &candidate_idx,
478  std::vector<double> &candidate_rate, double weight)
479 {
480  // Compute maximum and minimum of correlations
481  int imax = vscore.size();
482  double min_score, max_score;
483  min_score = max_score = vscore[0];
484  for (int i = 1; i < imax; i++)
485  {
486  if (vscore[i] < min_score)
487  min_score = vscore [i];
488  if (vscore[i] > max_score)
489  max_score = vscore [i];
490  }
491 
492  // Divide the correlation segment in as many pieces as candidates
493  double score_step = (max_score - min_score) / 10;
494 
495  int jmax = candidate_idx.size();
496  for (int j = 0; j < jmax; j++)
497  {
498  int i = candidate_idx[j];
499  int points;
500  if (score_step != 0)
501  points = FLOOR((vscore[i] - min_score) / score_step);
502  else
503  points = 10;
504  if (tell & TELL_PSI_SHIFT)
505  std::cout << "Candidate (" << i << ") score=" << vscore[i]
506  << " points=" << points << std::endl;
507  candidate_rate[j] += weight * points;
508  }
509 
510  if (tell & TELL_PSI_SHIFT)
511  std::cout << "Evaluation:" << candidate_rate << std::endl
512  << "Threshold for obtaining a 7 in score: "
513  << min_score + 7*score_step << std::endl;
514  return min_score + 7*score_step;
515 }

636  group_rate.push_back(temp_group_rate);
637  if (temp_group_rate > best_group_rate)
638  {
639  best_g = g;
640  best_group_rate = group_rate[g];
641  }
642  }
643 
644  // Check that there are not two groups with the same score
645  int groups_with_max_rate = 0;
646  for (size_t g = 0; g < groups.size(); g++)
647  if (group_rate[g] == best_group_rate)
648  groups_with_max_rate++;
649 #ifdef DEBUG
650 
651  if (groups_with_max_rate > 1)
652  {
653  std::cout << "There are two groups with maximum rate\n";
654  }
655 #endif
656

658  int best_j=0;
659  double best_rate = -1e38;
660  for (size_t j = 0; j < groups[best_g].size(); j++)
661  {
662 #ifdef NEVER_DEFINED
663  // Select the best with the rate
664  if (candidate_rate[groups[best_g][j]] > best_rate)
665  {
666  best_j = j;
667  best_rate = candidate_rate[groups[best_g][j]];
668  }
669 #endif
670  // Select the best with the scoreelation
671  if (vscore[candidate_idx[groups[best_g][j]]] > best_rate)
672  {
673  best_j = j;
674  best_rate = vscore[candidate_idx[groups[best_g][j]]];
675  }
676  }
677 
678  // Check that there are not two images with the same rate
679  int images_with_max_rate = 0;
680  for (size_t j = 0; j < groups[best_g].size(); j++)
681 #ifdef NEVER_DEFINED
682  // Select the best with the rate
683  if (candidate_rate[groups[best_g][j]] == best_rate)
684  images_with_max_rate++;
685 #endif
686  // Select the best with scoreelation
687  if (vscore[candidate_idx[groups[best_g][j]]] == best_rate)
688  images_with_max_rate++;
689  if (images_with_max_rate > 1)

697  candidate_rate[groups[best_g][j]] == best_rate)

699  best_j = j;
700  best_score = vscore[candidate_idx[groups[best_g][j]]];
701  }
702  }
703  }
704  return groups[best_g][best_j];
705  }
