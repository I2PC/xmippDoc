58 /* Recurrence coefficents for L2-normed associated Legendre
59  recurrence. When using these coeffs, make sure that
60  inital Pmm function is also L2-normed */
61 /* l represents degree, m is the order */
62 
63 double L2_an(int m,
64  int l)
65 {
66  return (sqrt((((double) (2*l+3))/((double) (2*l+1))) *
67  (((double) (l-m+1))/((double) (l+m+1)))) *
68  (((double) (2*l+1))/((double) (l-m+1))));
69 
70 }

72 /* note - if input l is zero, need to return 0 */
73 double L2_cn(int m,
74  int l)
75 {
76  if (l != 0) {
77  return (-1.0 *
78  sqrt((((double) (2*l+3))/((double) (2*l-1))) *
79  (((double) (l-m+1))/((double) (l+m+1))) *
80  (((double) (l-m))/((double) (l+m)))) *
81  (((double) (l+m))/((double) (l-m+1))));
82  }
83  else
84  return 0.0;
85 
86 }

92 double L2_cn_inv(int m,
93  int l)
94 {
95  double dl, dm;
96 
97  dl = (double) l;
98  dm = (double) m;
99 
100  return ( -(1.0 + (1. - 2. * dm)/(dm + dl)) *
101  sqrt( ((-1. + 2.*dl)/(3. + 2.*dl)) *
102  ((dl + dl*dl + dm + 2.*dl*dm + dm*dm)/
103  (dl + dl*dl - dm - 2.*dl*dm + dm*dm)) )
104  );
105 
106 }

232 /************************************************************************/
233 /* L2 normed Pmm. Expects input to be the order m, an array of
234  evaluation points arguments of length n, and a result vector of length n */
235 /* The norming constant can be found in Sean's PhD thesis */
236 /* This has been tested and stably computes Pmm functions thru bw=512 */
237 
238 void Pmm_L2( int m,
239  double *eval_pts,
240  int n,
241  double *result)
242 {
243  int i;
244  double md, id, mcons;
245 
246  id = (double) 0.0;
247  md = (double) m;
248  mcons = sqrt(md + 0.5);
249 
250  for (i=0; i<m; i++) {
251  mcons *= sqrt((md-(id/2.0))/(md-id));
252  id += 1.0;
253  }
254  if (m != 0 )
255  mcons *= pow(2.0,-md/2.0);
256  if ((m % 2) != 0) mcons *= -1.0;
257 
258  for (i=0; i<n; i++)
259  result[i] = mcons * pow(sin(eval_pts[i]),((double) m));
260 
261 }

266  associated Legendre functions. The coeffs array should contain
267  bw - m coefficients ordered from zeroth degree to bw-1, and eval_pts
268  should be an array of the arguments (arccos) of the desired
269  evaluation points of length 2*bw. Answer placed
270  in result (and has length 2*bw).

273 
274  workspace needs to be of size 14 * bw
275  */
276 /************************************************************************/

310  prev[i] = 1.0;

312  result[i] = coeffs[0] * prev[i];

328  splat = coeffs[i+1];
329  for (j=0; j<n; j++)

331  memcpy(prevprev, prev, sizeof(double) * n);
332  memcpy(prev, temp4, sizeof(double) * n);
