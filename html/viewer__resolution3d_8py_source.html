107  def _adjustPoints(self, data):
108  x=data.getXData()
109  if x[0]>x[1]:
110  aux=x[1]
111  x[1]=x[0]
112  x[0]=aux
113  X=[]
114  Y=[]
115  for objId in self.md:
116  f=self.md.getValue(emlib.MDL_RESOLUTION_FREQ2,objId)
117  if f>=x[0] and f<=x[1]:
118  X.append(f)
119  Y.append(self.md.getValue(emlib.MDL_RESOLUTION_LOG_STRUCTURE_FACTOR,objId))
120  X=np.array(X)
121  Y=np.array(Y)
122  A=np.array([np.ones(X.size), X.T])
123  beta=np.linalg.lstsq(A.T,Y)[0]
124  y = [beta[0]+beta[1]*xi for xi in x]
125  Bfactor = -4*beta[1]
126 
127  # Update data
128  data.getPoint(0).setX(x[0])
129  data.getPoint(0).setY(y[0])
130  data.getPoint(1).setX(x[1])
131  data.getPoint(1).setY(y[1])
132  data.bfactor=Bfactor
133 
134  f = open(self.protocol._getPath('bfactor.txt'), 'w')
135  print >> f, x[0], y[0], x[1], y[1], Bfactor
136  f.close()
