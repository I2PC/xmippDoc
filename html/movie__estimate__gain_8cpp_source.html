65 void ProgMovieEstimateGain::produceSideInfo()
66 {
67  if (fnIn.getExtension()=="mrc")
68  fnIn+=":mrcs";
69  mdIn.read(fnIn);
70  mdIn.removeDisabled();
71  if (mdIn.size()==0)
72  exit(0);
73  Image<double> Iframe;
74  FileName fnFrame;
75  mdIn.getValue(MDL_IMAGE,fnFrame,mdIn.firstObject());
76  Iframe.read(fnFrame);
77  Xdim=XSIZE(Iframe());
78  Ydim=YSIZE(Iframe());
79 
80  columnH.initZeros(Ydim,Xdim);
81  rowH.initZeros(Ydim,Xdim);
82  if (fnGain!="")
83  {
84  ICorrection.read(fnGain);
85  if (YSIZE(ICorrection())!=Ydim || XSIZE(ICorrection())!=Xdim)
86  REPORT_ERROR(ERR_MULTIDIM_SIZE,"The gain image and the movie do not have the same dimensions");
87  }
88  else
89  {
90  ICorrection().resizeNoCopy(Ydim,Xdim);
91  ICorrection().initConstant(1);
92  }
93 
94  sumObs.initZeros(Ydim,Xdim);
95 
96  int iFrame=0;
97  FOR_ALL_OBJECTS_IN_METADATA(mdIn)
98  {
99  if (iFrame%frameStep==0)
100  {
101  mdIn.getValue(MDL_IMAGE,fnFrame,__iter.objId);
102  Iframe.read(fnFrame);
103  MultidimArray<double> mIframe=Iframe();
104  MultidimArray<double> mICorrection=ICorrection();
105  FOR_ALL_DIRECT_ELEMENTS_IN_MULTIDIMARRAY(mIframe)
106  DIRECT_MULTIDIM_ELEM(mIframe,n) /= DIRECT_MULTIDIM_ELEM(mICorrection,n); sumObs+=Iframe();
107  }
108  iFrame++;
109  }
110  sumObs*=2;
111 
112  // Initialize sigma values
113  for (double sigma=0; sigma<=maxSigma; sigma+=sigmaStep)
114  listOfSigmas.push_back(sigma);
115 
116  for (size_t i=0; i<listOfSigmas.size(); ++i)
117  {
118  int jmax=ceil(3*listOfSigmas[i]);
119  listOfWidths.push_back(jmax);
120  double *weights=new double[jmax+1];
121  double K=1;
122  if (listOfSigmas[i]>0)
123  K=-0.5/(listOfSigmas[i]*listOfSigmas[i]);
124  for (int j=0; j<=jmax; ++j)
125  weights[j]=exp(K*j*j);
126  listOfWeights.push_back(weights);
127  }
128 }

298 void ProgMovieEstimateGain::constructSmoothHistogramsByColumn(const double *listOfWeights, int width)
299 {
300 
301  smoothColumnH.initZeros(columnH);
302 
303  for (size_t j=0; j<XSIZE(columnH); ++j)
304  {
305  double sumWeightsC = 0;
306  for(int k = -width; k<=width; ++k)
307  {
308  if (j+k<0 || j+k>=XSIZE(columnH))
309  continue;
310  double actualWeightC = listOfWeights[abs(k)];
311  sumWeightsC += actualWeightC;
312  for (size_t i=0; i<Ydim; ++i)
313  DIRECT_A2D_ELEM(smoothColumnH,i,j) += actualWeightC * DIRECT_A2D_ELEM(columnH,i,j+k);
314  }
315 
316  double iSumWeightsC=1/sumWeightsC;
317  for (size_t i=0; i<Ydim; ++i)
318  DIRECT_A2D_ELEM(smoothColumnH,i,j) *= iSumWeightsC;
319  }
320 
321  if (singleReference)
322  {
323  // Compute the average of all column histograms
324  for (size_t i=0; i<Ydim; ++i)
325  for (size_t j=1; j<Xdim; ++j)
326  DIRECT_A2D_ELEM(smoothColumnH,i,0)+=DIRECT_A2D_ELEM(smoothColumnH,i,j);
327 
328  double iXdim=1.0/Xdim;
329  for (size_t i=0; i<Ydim; ++i)
330  {
331  DIRECT_A2D_ELEM(smoothColumnH,i,0)*=iXdim;
332  double aux=DIRECT_A2D_ELEM(smoothColumnH,i,0);
333  for (size_t j=1; j<Xdim; ++j)
334  DIRECT_A2D_ELEM(smoothColumnH,i,j)=aux;
335  }
336  }
