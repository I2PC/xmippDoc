210 void ImageBase::write(const FileName &name, size_t select_img, bool isStack,
211  int mode, CastWriteMode castMode, int _swapWrite)
212 {
213  const FileName &fname = (name.empty()) ? filename : name;
214 
215  if (mmapOnWrite && mappedSize > 0)
216  {
217  bool hasTempFile = !tempFilename.empty();
218  if (hasTempFile && fname.isInStack())
219  mmapOnWrite = !(mmapOnRead = true); // We change mmap mode from write to read to allow writing the image into a stack.
220  else
221  {
222  if (_swapWrite > 0)
223  REPORT_ERROR(ERR_ARG_INCORRECT, "Cannot swap endianness on writing if file is already mapped.");
224  munmapFile();

226  REPORT_ERROR(ERR_IO, formatString("Error renaming file '%s' to '%s'.", tempFilename.c_str(), fname.c_str()));

229  }
230  // Swap the endianness of the image file when writing
231  swapWrite = _swapWrite;

235  */
236  if ( fname.isInStack() && mode == WRITE_OVERWRITE)

490 ImageFHandler* ImageBase::openFile(const FileName &name, int mode) const
491 {
492  if (name.empty())
493  REPORT_ERROR(ERR_PARAM_INCORRECT, "ImageBase::openFile Cannot open an empty Filename.");
494 
495  ImageFHandler* hFile = new ImageFHandler;
496  FileName fileName, headName = "";
497  FileName ext_name = name.getFileFormat();
498 
499  // Remove image number and block name
500  fileName = name.removeAllPrefixes();
501 
502  fileName = fileName.removeFileFormat();
503 
504  size_t found = fileName.find_first_of("%");
505  if (found!=String::npos)
506  fileName = fileName.substr(0, found) ;
507 
508  bool exist = fileName.exists();
509  bool sizeZero = true;
510 
511  if (exist)
512  sizeZero = fileName.getFileSize() <= 0;
513  hFile->exist = exist && !sizeZero;
514  hFile->mode = mode;
515 
516  String wmChar;
517 
518  switch (mode)
519  {
520  case WRITE_READONLY:
521  if (!exist)
522  REPORT_ERROR(ERR_IO_NOTEXIST, formatString("Cannot access file '%s'. It doesn't exist", name.c_str()));
523  else if (sizeZero)
524  REPORT_ERROR(ERR_IO_SIZE, formatString("Cannot read zero size file '%s'.", name.c_str()));
525 
526  wmChar = "r";
527  break;
528  case WRITE_OVERWRITE:
529  wmChar = "w";
530  break;
531  case WRITE_APPEND:
532  case WRITE_REPLACE:
533  wmChar = (hFile->exist) ? "r+" : "w+";
534  break;
535 
536  }
537 
538  if (ext_name.contains("tif"))
539  {
540  TIFFSetWarningHandler(NULL); // Switch off warning messages
541  if ((hFile->tif = TIFFOpen(fileName.c_str(), wmChar.c_str())) == NULL)
542  REPORT_ERROR(ERR_IO_NOTOPEN,"rwTIFF: There is a problem opening the TIFF file.");
543  hFile->fimg = NULL;
544  hFile->fhed = NULL;
545  }
546  else if (ext_name.contains("hdf") || ext_name.contains("h5"))
547  {
548  if ((hFile->fhdf5 = H5Fopen(fileName.c_str(), H5F_ACC_RDONLY, H5P_DEFAULT)) == -1 )
549  REPORT_ERROR(ERR_IO_NOTOPEN,"ImageBase::openFile: There is a problem opening the HDF5 file.");
550  // hFile->fimg = NULL;
551 
552  if ( (hFile->fimg = fopen(fileName.c_str(), wmChar.c_str())) == NULL )
553  {
554  if (errno == EACCES)
555  REPORT_ERROR(ERR_IO_NOPERM,formatString("Image::openFile: permission denied when opening %s",fileName.c_str()));
556  else
557  REPORT_ERROR(ERR_IO_NOTOPEN,formatString("Image::openFile cannot open: %s", fileName.c_str()));
558  }
559 
560  hFile->fhed = NULL;
561  hFile->tif = NULL;
562  }
563  else
564  {
565  hFile->tif = NULL;
566 
567  if (ext_name.contains("img") || ext_name.contains("hed"))
568  {
569  fileName = fileName.withoutExtension();
570  headName = fileName.addExtension("hed");
571  fileName = fileName.addExtension("img");
572  }
573  else if (ext_name.contains("raw"))
574  {
575  if (mode != WRITE_READONLY || fileName.addExtension("inf").exists() )
576  {
577  headName = fileName.addExtension("inf");
578  ext_name = "inf";
579  }
580  else
581  ext_name = "raw";
582  }
583  else if (ext_name.contains("inf"))
584  {
585  headName = fileName;
586  fileName = fileName.withoutExtension();
587  ext_name = "inf";
588  }
589 
590  // Open image file
591  if ( (hFile->fimg = fopen(fileName.c_str(), wmChar.c_str())) == NULL )
592  {
593  if (errno == EACCES)
594  REPORT_ERROR(ERR_IO_NOPERM,formatString("Image::openFile: permission denied when opening %s",fileName.c_str()));
595  else
596  REPORT_ERROR(ERR_IO_NOTOPEN,formatString("Image::openFile cannot open: %s", fileName.c_str()));
597  }
598 
599 
600  if (headName != "")
601  {
602  if ((hFile->fhed = fopen(headName.c_str(), wmChar.c_str())) == NULL )
603  {
604  if (errno == EACCES)
605  REPORT_ERROR(ERR_IO_NOPERM,formatString("Image::openFile: permission denied when opening %s",headName.c_str()));
606  else
607  REPORT_ERROR(ERR_IO_NOTOPEN,formatString("Image::openFile cannot open: %s",headName.c_str()));
608  }
609 
610  }
611  else
612  hFile->fhed = NULL;
613 
614  }
615  hFile->fileName = fileName;
616  hFile->headName = headName;
617  hFile->ext_name = ext_name;
618 
619  return hFile;
620 }

625 void ImageBase::closeFile(ImageFHandler* hFile) const
626 {
627  FileName ext_name, fileName;
628  FILE* fimg, *fhed;
629  TIFF* tif;
630  hid_t fhdf5;
631 
632  if (hFile != NULL)
633  {
634  fileName = hFile->fileName;
635  ext_name = hFile->ext_name;
636  fimg = hFile->fimg;
637  fhed = hFile->fhed;
638  tif = hFile->tif;
639  fhdf5 = hFile->fhdf5;
640  }
641  else
642  {
643  fileName = filename;
644  ext_name = filename.getFileFormat();
645  fimg = this->fimg;
646  fhed = this->fhed;
647  tif = this->tif;
648  fhdf5 = this->fhdf5;
649 
650  }
651 
652  if (ext_name.contains("tif"))
653  {
654  TIFFClose(tif);
655  /* Since when creating a TIFF file without adding an image the file is 8 bytes

657  * to suppose that under 8 bytes this is empty.
658  */
659  if (fileName.getFileSize() < 9)
660  filename.deleteFile();
661  }
662  else if (ext_name.contains("hdf") || ext_name.contains("h5"))
663  {
664  H5Fclose(fhdf5);

666  REPORT_ERROR(ERR_IO_NOCLOSED,(String)"Can not close image file "+ filename);
667  }
668  else

671  REPORT_ERROR(ERR_IO_NOCLOSED,(String)"Can not close image file "+ filename);
672 
673  if (fhed != NULL && fclose(fhed) != 0 )
674  REPORT_ERROR(ERR_IO_NOCLOSED,(String)"Can not close header file of "
675  + filename);
676  }
677  delete hFile;

719 #undef DEBUG

724  "name="<<name <<std::endl;

733  //Set the file pointer at beginning

735  fseek(fimg, 0, SEEK_SET);
736  if (fhed != NULL)

738 
739  if (ext_name.contains("spi") || ext_name.contains("xmp") ||
740  ext_name.contains("stk") || ext_name.contains("vol"))
741  err = readSPIDER(select_img);
742  else if (ext_name.contains("mrcs")||ext_name.contains("st"))//mrc stack MUST go BEFORE plain MRC
743  err = readMRC(select_img,true);
744  else if (ext_name.contains("mrc")||ext_name.contains("map"))//mrc

746  else if (ext_name.contains("img") || ext_name.contains("hed"))//

748  else if (ext_name.contains("ser"))//TIA
749  err = readTIA(select_img,false);
750  else if (ext_name.contains("dm3"))//DM3
751  err = readDM3(select_img,false);
752  else if (ext_name.contains("dm4"))//DM4
753  err = readDM4(select_img,false);
754  else if (ext_name.contains("ems"))//EM stack
755  err = readEM(select_img, true);
756  else if (ext_name.contains("em"))//EM
757  err = readEM(select_img);
758  else if (ext_name.contains("pif"))//PIF

760  else if (ext_name.contains("inf"))//RAW with INF file
