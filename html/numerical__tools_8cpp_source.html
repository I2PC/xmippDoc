337 /* Regularized least squares ----------------------------------------------- */
338 void regularizedLeastSquare(const Matrix2D< double >& A,
339  const Matrix1D< double >& d, double lambda,
340  const Matrix2D< double >& G, Matrix1D< double >& x)
341 {
342  int Nx=A.Xdim(); // Number of variables
343 
344  Matrix2D<double> X(Nx,Nx); // X=(A^t * A +lambda *G^t G)
345  // Compute A^t*A
346  FOR_ALL_ELEMENTS_IN_MATRIX2D(X)
347  // Compute the dot product of the i-th and j-th columns of A
348  for (size_t k=0; k<A.Ydim(); k++)
349  X(i,j) += A(k,i) * A(k,j);
350 
351  // Compute lambda*G^t*G
352  if (G.Xdim()==0)
353  for (int i=0; i<Nx; i++)
354  X(i,i)+=lambda;
355  else
356  FOR_ALL_ELEMENTS_IN_MATRIX2D(X)
357  // Compute the dot product of the i-th and j-th columns of G
358  for (size_t k=0; k<G.Ydim(); k++)
359  X(i,j) += G(k,i) * G(k,j);
360 
361  // Compute A^t*d
362  Matrix1D<double> Atd(Nx);
363  FOR_ALL_ELEMENTS_IN_MATRIX1D(Atd)
364  // Compute the dot product of the i-th column of A and d
365  for (size_t k=0; k<A.Ydim(); k++)
366  Atd(i) += A(k,i) * d(k);
367 
368  // Compute the inverse of X
369  Matrix2D<double> Xinv;
370  X.inv(Xinv);
371 
372  // Now multiply Xinv * A^t * d
373  x.initZeros(Nx);
374  FOR_ALL_ELEMENTS_IN_MATRIX2D(Xinv)
375  x(i) += Xinv(i,j) * Atd(j);
376 }
