30 double rfs_arch_cost_function(starpu_task * task, starpu_perfmodel_arch * arch, unsigned nimpl) {
31  auto* data = (rfs_data*)starpu_sched_ctx_get_policy_data(task->sched_ctx);
32  assert(data);
33 
34  size_t size = 0;
35  if (task->cl->model && task->cl->model->size_base) {
36  size = task->cl->model->size_base(task, nimpl);
37  }
38 
39  double estimate = 0;
40  for (int device = 0; device < arch->ndevices; ++device) {
41  double newEstimate = data->timing.estimate(task->cl, nimpl, arch->devices[device].type, arch->devices[device].devid, size);
42  estimate = STARPU_MAX(estimate, newEstimate);
43  }
44 
45  return estimate;
46 }
