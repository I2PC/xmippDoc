162 void Basis::produceSideInfo(const Grid &grid)
163 {
164  switch (type)
165  {
166  case (blobs):
167  {
168  int subsampling = (VolPSF == NULL )? BLOB_SUBSAMPLING : PIXEL_SUBSAMPLING;
169 
170  footprint_blob(blobprint, blob, subsampling);
171  sum_on_grid = sum_blob_Grid(blob, grid, D);
172  blobprint() /= sum_on_grid;
173 
174  if (VolPSF != NULL)
175  {
176  // let adjust to the same resolution and size both blobprint and VolPSF
177  // selfScaleToSize(LINEAR, *VolPSF, XSIZE(*VolPSF)*BLOB_SUBSAMPLING,
178  // YSIZE(*VolPSF)*BLOB_SUBSAMPLING, ZSIZE(*VolPSF));
179 
180  blobprint().setXmippOrigin();
181 
182  if (XSIZE(blobprint()) < XSIZE(*VolPSF))
183  blobprint().selfWindow(STARTINGY(*VolPSF),
184  STARTINGX(*VolPSF),
185  STARTINGY(*VolPSF)+YSIZE(*VolPSF)-1,
186  STARTINGX(*VolPSF)+XSIZE(*VolPSF)-1);
187  else if (XSIZE(blobprint()) > XSIZE(*VolPSF))
188  VolPSF->selfWindow(STARTINGY(blobprint()),
189  STARTINGX(blobprint()),
190  STARTINGY(blobprint())+YSIZE(blobprint())-1,
191  STARTINGX(blobprint())+XSIZE(blobprint())-1);
192 
193  // creation of the 3D blobprint from convolution of blobprint and PSF
194  ImageOver footprintT;
195  footprintT.init(*VolPSF, subsampling);
196 
197  convolutionFFTStack(*VolPSF, blobprint(), footprintT());
198 
199  blobprint = footprintT;
200  }
201 
202  blobprint2() = blobprint();
203  blobprint2() *= blobprint();
204  break;
205  }
206  case (voxels): sum_on_grid = 1;
207  break;
208  case (splines): sum_on_grid = sum_spatial_Bspline03_Grid(grid);
209  break;
210  }
211 
212 #ifdef DEBUG
213  std::cout << "Sum of a basis on the grid=" << sum_on_grid << std::endl;
214  std::cout << "D\n" << D << std::endl;
215  ImageXmipp save;
216  save() = blobprint();
217  save.write("footprint.xmp");
218 #endif
219 }

329 double Basis::valueAt(const Matrix1D<double> & r) const
330 {
331  double module_r;
332  switch (type)
333  {
334  case (blobs):
335  {
336  module_r = sqrt(XX(r) * XX(r) + YY(r) * YY(r) + ZZ(r) * ZZ(r));
337  return blob_val(module_r, blob);
338  break;
339  }
340  case (voxels):
341  {
342  if (-0.5 <= XX(r) && XX(r) < 0.5 &&
343  -0.5 <= YY(r) && YY(r) < 0.5 &&
344  -0.5 <= ZZ(r) && ZZ(r) < 0.5)
345  return 1.0;
346  else
347  return 0.0;
348  break;
349  }
350  case (splines):
351  {
352  if (-2 <= XX(r) && XX(r) < 2 &&
353  -2 <= YY(r) && YY(r) < 2 &&
354  -2 <= ZZ(r) && ZZ(r) < 2)
355  return spatial_Bspline03LUT(r);
356  else
357  return 0.0;
358  break;
359  }
360  }
361  return 0.0;
362 }

364 double Basis::projectionAt(const Matrix1D<double> & u, const Matrix1D<double> & r) const
365 {
366  const double p0 = 1.0 / (2 * PIXEL_SUBSAMPLING) - 0.5;
367  const double pStep = 1.0 / PIXEL_SUBSAMPLING;
368  const double pAvg = 1.0 / (PIXEL_SUBSAMPLING * PIXEL_SUBSAMPLING);
369  double module_r, px, py;
370  int i, j;
371  switch (type)
372  {
373  case (blobs):
374  module_r = sqrt(XX(r) * XX(r) + YY(r) * YY(r) + ZZ(r) * ZZ(r));
375  return blob_proj(module_r, blob);
376  break;
377  case (voxels):
378  {
379  double retval = 0;
380  ZZ(aux) = ZZ(r);
381  for (i = 0, px = p0; i < PIXEL_SUBSAMPLING; i++, px += pStep)
382  {
383  XX(aux) = XX(r) + px;
384  for (j = 0, py = p0; j < PIXEL_SUBSAMPLING; j++, py += pStep)
385  {
386  YY(aux) = YY(r) + py;
387  retval += intersection_unit_cube(u, aux);
388  }
389  }
390  return retval*pAvg;
391  break;
392  }
393  case (splines):
394  return spatial_Bspline03_proj(r, u);
395  break;
396  }
397  return 0.0;
398 }
