111  def extractUnit(self):
112  sym = self.symmetryGroup.get()
113  print("sym:_____________________________________________", sym)
114  if sym == XMIPP_CYCLIC:

117  sym = "%s%d" % \
118  (XMIPP_SYM_NAME[XMIPP_DIHEDRAL_X][:1], self.symmetryOrder)
119  elif sym == XMIPP_TETRAHEDRAL:
120  sym = "%s" % XMIPP_SYM_NAME[XMIPP_TETRAHEDRAL]
121  elif sym == XMIPP_OCTAHEDRAL:
122  sym = "%s" % XMIPP_SYM_NAME[XMIPP_OCTAHEDRAL]
123  elif sym >= XMIPP_I222 and sym <= XMIPP_In25r :
124  sym = XMIPP_SYM_NAME[sym]

126  inFileName = self.inputVolumes.get().getFileName()
