70 void sum_floats(float4 * target, float4 * source, uint length) {
71  uint i = blockIdx.x * blockDim.x + threadIdx.x;
72  if (i < length) {
73  target[i].x += source[i].x;
74  target[i].y += source[i].y;
75  target[i].z += source[i].z;
76  target[i].w += source[i].w;
77  }

80 inline void sum_floats_on_cuda(uintptr_t targetCudaPtr, uintptr_t sourceCudaPtr, size_t noOfFloats) {
81  // Note: When initializing data in reconstruct_fourier_starpu.cpp#run(),
82  // buffer size was padded to multiple of 4, so we can leverage that.
83 
84  assert(noOfFloats % 4 == 0);
85  size_t noOfFloat4 = noOfFloats / 4;
86 
87  dim3 threadsPerBlock(512);
88  // Rounding up, we don't know whether noOfFloats is also a multiple of 512 (unlikely), additional checking is done in kernel
89  dim3 numBlocks(static_cast<unsigned int>((noOfFloat4 + threadsPerBlock.x - 1) / threadsPerBlock.x));
90 
91  sum_floats<<<numBlocks, threadsPerBlock, 0, starpu_cuda_get_local_stream()>>>((float4*) targetCudaPtr, (float4*)(sourceCudaPtr), noOfFloat4);
92  gpuErrchk(cudaPeekAtLastError());
93  // gpuErrchk(cudaStreamSynchronize(starpu_cuda_get_local_stream())); disabled because codelet is async
94 }
