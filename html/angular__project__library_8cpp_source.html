194 {
195  Projection P;
196  FileName fn_proj;
197  double rot,tilt,psi;
198  int mySize;

204  numberStepsPsi = (int) (359.99999/psi_sampling);
205  mySize *= numberStepsPsi;
206  }
207 
208  if (verbose)
209  init_progress_bar(mySize);
210  int myCounter=0;
211 
212 
213  for (double mypsi=0;mypsi<360;mypsi += psi_sampling)
214  for (int i=0;i<my_init;i++)
215  myCounter++;
216 
217 // if (shears && XSIZE(inputVol())!=0 && VShears==NULL)
218 // VShears=new RealShearsInfo(inputVol());
219  if (projType == SHEARS && XSIZE(inputVol())!=0 && Vshears==NULL)
220  Vshears=new RealShearsInfo(inputVol());
221  if (projType == FOURIER && XSIZE(inputVol())!=0 && Vfourier==NULL)
222  Vfourier=new FourierProjector(inputVol(),
223  paddFactor,
224  maxFrequency,
225  BSplineDeg);
226 
227  for (double mypsi=0;mypsi<360;mypsi += psi_sampling)
228  {
229  for (int i=my_init;i<=my_end;i++)
230  {
231  if (verbose)
232  progress_bar(i-my_init);
233  psi= mypsi+ZZ(mysampling.no_redundant_sampling_points_angles[i]);
234  tilt= YY(mysampling.no_redundant_sampling_points_angles[i]);
235  rot= XX(mysampling.no_redundant_sampling_points_angles[i]);
236 
237 // if (shears)
238 // projectVolume(*VShears, P, Ydim, Xdim, rot,tilt,psi);
239 // else
240 // projectVolume(inputVol(), P, Ydim, Xdim, rot,tilt,psi);
241  if (projType == SHEARS)
242  projectVolume(*Vshears, P, Ydim, Xdim, rot, tilt, psi);
243  else if (projType == FOURIER)
244  projectVolume(*Vfourier, P, Ydim, Xdim, rot, tilt, psi);
245  else if (projType == REALSPACE)
246  projectVolume(inputVol(), P, Ydim, Xdim, rot, tilt, psi);
247 
248 
249  P.setEulerAngles(rot,tilt,psi);

252  }

264  //only rank 0

292  //mpi_barrier here

295  //symmetry_file = symmetry + ".sym";
296  //SL.readSymmetryFile(symmetry_file)

298  // Then we remove point redundant due to sampling symmetry

300  mysampling.removeRedundantPoints(symmetry, sym_order);
301 
302  //=========================
303  //======================
304  //recompute symmetry with neigh symmetry
305  // If uncomment neighbour are not OK. BE CAREFUL
306 #define BREAKSIMMETRY
