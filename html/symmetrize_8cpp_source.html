32 void ProgSymmetrize::readParams()
33 {
34  XmippMetadataProgram::readParams();
35  fn_sym = getParam("--sym");
36  doMask = false;
37  if (checkParam("--mask_in"))
38  {
39  fn_Maskin = getParam("--mask_in");
40  doMask = true;
41  }
42  helical=(fn_sym=="helical");
43  dihedral=(fn_sym=="dihedral");
44  helicalDihedral=(fn_sym=="helicalDihedral");
45  if (helical || helicalDihedral)
46  {
47  Ts=getDoubleParam("--sampling");
48  zHelical=getDoubleParam("--helixParams",0)/Ts;
49  rotHelical=DEG2RAD(getDoubleParam("--helixParams",1));
50  rotPhaseHelical=DEG2RAD(getDoubleParam("--helixParams",2));
51  }
52  do_not_generate_subgroup = checkParam("--no_group");
53  wrap = !checkParam("--dont_wrap");
54  sum = checkParam("--sum");
55  heightFraction = getDoubleParam("--heightFraction");
56  splineOrder = getIntParam("--spline");
57 }

109 /* Symmetrize ------------------------------------------------------- */
110 void symmetrizeVolume(const SymList &SL, const MultidimArray<double> &V_in,
111  MultidimArray<double> &V_out, int spline,
112  bool wrap, bool do_outside_avg, bool sum, bool helical, bool dihedral, bool helicalDihedral,
113  double rotHelical, double rotPhaseHelical, double zHelical, double heightFraction,
114  const MultidimArray<double> * mask)
115 {
116  Matrix2D<double> L(4, 4), R(4, 4); // A matrix from the list
117  MultidimArray<double> V_aux;
118  Matrix1D<double> sh(3);
119  double avg = 0.;
120 
121  if (do_outside_avg)
122  {
123  MultidimArray<int> mask1;
124  mask1.resizeNoCopy(V_in);
125  mask1.setXmippOrigin();
126  size_t rad = XMIPP_MIN(XSIZE(V_in), YSIZE(V_in));
127  rad = XMIPP_MIN(rad, ZSIZE(V_in));
128  BinaryCircularMask(mask1, rad / 2, OUTSIDE_MASK);
129  double dum;
130  computeStats_within_binary_mask(mask1, V_in, dum, dum, avg, dum);
131  }
132  V_out = V_in;
133 
134  if (!helical && !dihedral && !helicalDihedral)
135  {
136  MultidimArray<double> Bcoeffs;
137  MultidimArray<double> *BcoeffsPtr=NULL;
138  if (spline==3)
139  {
140  produceSplineCoefficients(BSPLINE3, Bcoeffs, V_in);
141  BcoeffsPtr=&Bcoeffs;
142  }
143  for (int i = 0; i < SL.symsNo(); i++)
144  {
145  SL.getMatrices(i, L, R);
146  /*FIXME: I do not think this make sense since V_aux is empty. ROB
147  SL.getShift(i, sh);
148  R(3, 0) = sh(0) * XSIZE(V_aux);
149  R(3, 1) = sh(1) * YSIZE(V_aux);
150  R(3, 2) = sh(2) * ZSIZE(V_aux);
151  */
152  applyGeometry(spline, V_aux, V_in, R.transpose(), IS_NOT_INV, wrap, avg, BcoeffsPtr);
153 
154  if ( mask==NULL)
155  arrayByArray(V_out, V_aux, V_out, '+');
156  else // op1, op2 , result
157  selfArrayByArrayMask(V_in, V_aux, V_out, '+', mask);
158  }
159 
160  if (!sum)
161  arrayByScalar(V_out, 1.0/(SL.symsNo() + 1.0f), V_out, '*');
162  }
163  else if (helical)
164  symmetry_Helical(V_out,V_in,zHelical,rotHelical,rotPhaseHelical,NULL,false,heightFraction);
165  else if (helicalDihedral)
166  {
167  symmetry_Helical(V_out,V_in,zHelical,rotHelical,rotPhaseHelical,NULL,true,heightFraction);
168  MultidimArray<double> Vrotated;
169  rotate(spline,Vrotated,V_out,180.0,'X',WRAP);
170  V_out+=Vrotated;
171  V_out*=0.5;
172  }
173  else if (dihedral)
174  {
175  int zmax=(int)(0.1*ZSIZE(V_in));
176  symmetry_Dihedral(V_out,V_in,1,-zmax,zmax,0.5);
177  }
178 }

180 void symmetrizeImage(int symorder, const MultidimArray<double> &I_in,
181  MultidimArray<double> &I_out, int spline,
182  bool wrap, bool do_outside_avg, bool sum,
183  const MultidimArray<double> * mask)
184 {
185  double avg = 0.;
186  if (do_outside_avg)
187  {
188  MultidimArray<int> mask1;
189  mask1.resizeNoCopy(I_in);
190  mask1.setXmippOrigin();
191  int rad = XMIPP_MIN(XSIZE(I_in), YSIZE(I_in));
192  BinaryCircularMask(mask1, rad / 2, OUTSIDE_MASK);
193  double dum;
194  computeStats_within_binary_mask(mask1, I_in, dum, dum, avg, dum);
195  }
196  I_out = I_in;
197  MultidimArray<double> rotatedImg;
198  if ( (mask)!=NULL)
199  REPORT_ERROR(ERR_NOT_IMPLEMENTED,"mask symmetrization not implemented for images");
200  for (int i = 1; i < symorder; i++)
201  {
202  rotate(spline, rotatedImg, I_in, 360.0 / symorder * i,'Z',wrap,avg);
203  I_out += rotatedImg;
204  }
205  if (!sum)
206  I_out *= 1.0/symorder;
207 }

226 void ProgSymmetrize::processImage(const FileName &fnImg, const FileName &fnImgOut, const MDRow &rowIn, MDRow &rowOut)
227 {
228  Image<double> Iin;
229  Image<double> Iout;
230  Image<double> mask;
231  MultidimArray<double> *mmask=NULL;
232 
233  Iin.readApplyGeo(fnImg, rowIn);
234  Iin().setXmippOrigin();
235  Iout().resize(Iin());
236 
237  if (doMask)
238  {
239  mask.read(fn_Maskin);
240  mmask=&(mask());
241  }
242 
243  if (ZSIZE(Iin())==1)
244  {
245  if (helical)
246  REPORT_ERROR(ERR_ARG_INCORRECT,"Helical symmetrization is not meant for images");
247  if (symorder!=-1)
248  symmetrizeImage(symorder,Iin(),Iout(),splineOrder,wrap,!wrap,sum,mmask);
249  else
250  REPORT_ERROR(ERR_ARG_MISSING,"The symmetry order is not valid for images");
251  }
252  else
253  {
254  if (SL.symsNo()>0 || helical || dihedral || helicalDihedral)
255  {
256  symmetrizeVolume(SL,Iin(),Iout(),splineOrder,wrap,!wrap,
257  sum,helical,dihedral,helicalDihedral,rotHelical,rotPhaseHelical,zHelical,heightFraction,mmask);
258  }
259  else
260  REPORT_ERROR(ERR_ARG_MISSING,"The symmetry description is not valid for volumes");
261  }
262  Iout.write(fnImgOut);
263 }
