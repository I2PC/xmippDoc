65 /* Recurrence coefficents for L2-normed associated Legendre
66  recurrence. When using these coeffs, make sure that
67  inital Pmm function is also L2-normed */
68 /* l represents degree, m is the order */

70 double ns_L2_an(int m, int l)
71 {
72  return (sqrt((((double) (2*l+3))/((double) (2*l+1))) *
73  (((double) (l-m+1))/((double) (l+m+1)))) *
74  (((double) (2*l+1))/((double) (l-m+1))));
75 }

83 double ns_L2_cn(int m, int l)
84 {
85  if (l != 0) {
86  return (-1.0 *
87  sqrt((((double) (2*l+3))/((double) (2*l-1))) *

89  (((double) (l-m))/((double) (l+m)))) *
90  (((double) (l+m))/((double) (l-m+1))));
91  }
92  else
93  return 0.0;

206 /************************************************************************/
207 /* L2 normed Pmm. Expects input to be the order m, an array of
208  evaluation points arguments of length n, and a result vector of length n */
209 /* The norming constant can be found in Sean's PhD thesis */
210 /* This has been tested and stably computes Pmm functions thru bw=512 */
211 
212 void ns_Pmm_L2(int m,
213  double *eval_pts,
214  int n,
215  double *result)
216 {
217  int i;
218  double md, id, mcons;
219 
220  id = (double) 0.0;
221  md = (double) m;
222  mcons = sqrt(md + 0.5);
223 
224  for (i=0; i<m; i++) {
225  mcons *= sqrt((md-(id/2.0))/(md-id));
226  id += 1.0;
227  }
228  if (m != 0 )
229  mcons *= pow(2.0,-md/2.0);
230  if ((m % 2) != 0) mcons *= -1.0;
231 
232  for (i=0; i<n; i++)
233  result[i] = mcons * pow(sin(eval_pts[i]),((double) m));
234 
235 }

256 void Naive_Synthesize(int bw,
257  int m,
258  double *coeffs,
259  double *result,
260  double *workspace)
261 {
262  double *prev, *prevprev, *temp1, *temp2, *temp3, *temp4, *x_i, *eval_args;
263  double t1;
264  int i, j;
265 
266  prevprev = workspace;
267  prev = prevprev + (2*bw);
268  temp1 = prev + (2*bw);
269  temp2 = temp1 + (2*bw);
270  temp3 = temp2 + (2*bw);
271  temp4 = temp3 + (2*bw);
272  x_i = temp4 + (2*bw);
273  eval_args = x_i + (2*bw);
274 
275 
276  /* get the evaluation nodes */
277  ns_EvalPts(2*bw,x_i);
278  ns_ArcCosEvalPts(2*bw,eval_args);
279 
280 
281  /* set initial values of first two Pmls */
282  for (i=0; i<2*bw; i++)
283  prevprev[i] = 0.0;
284  if (m == 0) {
285  for (i=0; i<2*bw; i++) {
286  prev[i] = 1.0;
287  }
288  }
289  else
290  ns_Pmm_L2(m, eval_args, 2*bw, prev);
291 
292  /* this may be useful for generating Gmls later but
293  removed for now
294 
295  if ((m % 2) == 1) {
296  for (i=0; i<2*bw; i++)
297  prev[i] /= sin(eval_args[i]);
298  }
299  */
300 
301  /* make sure result is zeroed out */
302  for (i=0; i<(2*bw); i++)
303  result[i] = 0.0;
304 
305  /* add in Pmm contribution */
306 
307  if (coeffs[0] != 0.0)
308  {
309  t1 = coeffs[0];
310  for (j=0; j<(2*bw); j++)
311  result[j] += t1 * prev[j];
312  }
313 
314  /* now generate remaining pmls while synthesizing function */
315 
316  for (i=0; i<bw-m-1; i++) {
317  ns_vec_mul(ns_L2_cn(m,m+i),prevprev,temp1,2*bw);
318  ns_vec_pt_mul(prev, x_i, temp2, 2*bw);
319  ns_vec_mul(ns_L2_an(m,m+i), temp2, temp3, 2*bw);
320  ns_vec_add(temp3, temp1, temp4, 2*bw); /* temp4 now contains P(m,m+i+1) */
321 
322  /* add in contribution */
323  if (coeffs[i+1] != 0.0)
324  {
325  t1 = coeffs[i+1];
326  for (j=0; j<(2*bw); j++)
327  result[j] += (t1 * temp4[j]);
328  }
329 
330  /* now update Pi and P(i+1) */
331  memcpy(prevprev, prev, (size_t) sizeof(double) * 2 * bw);
332  memcpy(prev, temp4, (size_t) sizeof(double) * 2 * bw);
333 
334  }
335  }
