108 void ProgAlignSignificantGPU<T>::align(const T *ref, const Dimensions &refDims,
109  const T *others, const Dimensions &otherDims,
110  unsigned device,
111  AlignmentEstimation *dest) {
112  auto hw = std::vector<HW*>();
113  for (size_t i = 0; i < 2; ++i) {
114  auto gpu = new GPU(device);
115  hw.emplace_back(gpu);
116  }
117 
118  printf("GPU %u aligning against %lu references\n", device, refDims.n());
119  auto processDims = otherDims.copyForN(std::min(otherDims.n(), m_maxBatchSize));
120 
121  auto rotEstimator = CudaRotPolarEstimator<T>();
122  initRotEstimator(rotEstimator, hw, processDims);
123  auto shiftEstimator = CudaShiftCorrEstimator<T>();
124  initShiftEstimator(shiftEstimator, hw, processDims);
125  CudaBSplineGeoTransformer<T> transformer;
126  initTransformer(transformer, hw, processDims);
127  CudaCorrelationComputer<T> mc;
128  initMeritComputer(mc, hw, processDims);
129 
130  auto aligner = IterativeAlignmentEstimator<T>(rotEstimator, shiftEstimator,
131  transformer, mc, this->getThreadPool());
132 
133  // create local copy of the reference ...
134  auto refSize = refDims.sizeSingle();
135  auto refSizeBytes = refSize * sizeof(T);
136  // ... and lock it, so that we can work asynchronously with it
137  auto refData = memoryUtils::page_aligned_alloc<T>(refSize, false);
138  hw.at(0)->lockMemory(refData, refSizeBytes);
139 
140  float reportAtPerc = 10.f;
141  float step = (refDims.n() / 100.f) * reportAtPerc;
142  float reportThreshold = step;
143  for (size_t refId = 0; refId < refDims.n(); ++refId) {
144  size_t refOffset = refId * refSize;
145  // copy reference image to page-aligned memory
146  memcpy(refData, ref + refOffset, refSizeBytes);
147 
148  if ((float)refId >= reportThreshold) { // FIXME DS remove / replace by proper progress report
149  reportThreshold += step;
150  printf("GPU %u alignment %d%% done\n", device,
151  (int)(100 * (refId / (float)refDims.n())));
152  }
153 
154  aligner.loadReference(refData);
155  for (size_t i = 0; i < otherDims.n(); i += processDims.n()) {
156  // run on a full-batch subset
157  size_t offset = std::min(i, otherDims.n() - processDims.n());
158  auto tmp = aligner.compute(others + (offset * otherDims.sizeSingle()));
159 
160  // merge results
161  dest->figuresOfMerit.insert(dest->figuresOfMerit.begin() + offset,
162  tmp.figuresOfMerit.begin(),
163  tmp.figuresOfMerit.end());
164  dest->poses.insert(dest->poses.begin() + offset,
165  tmp.poses.begin(),
166  tmp.poses.end());
167  }
168  dest++;
169  }
170 }

173 template<typename T>
174 void ProgAlignSignificantGPU<T>::updateRefs(
175  T *refs, const Dimensions &refDims,
176  const T *others, const Dimensions &otherDims,
177  const std::vector<Assignment> &assignments,
178  unsigned device,
179  size_t refOffset) {
180  const size_t elems = otherDims.sizeSingle();
181  T *workCopy = new T[m_maxBatchSize * refDims.sizeSingle()];
182  std::vector<float> workMatrices;
183  workMatrices.reserve(9 * m_maxBatchSize);
184  std::vector<float> workWeights;
185  workWeights.reserve(m_maxBatchSize);
186  T *workRef = memoryUtils::page_aligned_alloc<T>(elems, true);
187 
188  auto gpu = GPU(device);
189  gpu.pinMemory(workRef, elems * sizeof(T));
190  std::vector<HW*> hw{&gpu};
191 
192  CudaBSplineGeoTransformer<T> transformer;
193  initTransformer(transformer, hw, otherDims.copyForN(m_maxBatchSize));
194  transformer.setSrc(workCopy);
195  T norm = 0;
196 
197  std::vector<Assignment> workAssignments;
198  float reportAtPerc = 10.f;
199  float step = (refDims.n() / 100.f) * reportAtPerc;
200  float reportThreshold = step;
201  for (size_t r = 0; r < refDims.n(); ++r) {
202  if ((float)r >= reportThreshold) { // FIXME DS remove / replace by proper progress report
203  reportThreshold += step;
204  printf("GPU %u update reference %d%% done\n", device,
205  (int)(100 * (r / (float)refDims.n())));
206  }
207 
208  size_t refIndex = r + refOffset;
209  // get assignments for this reference
210  workAssignments.clear();
211  std::copy_if(assignments.begin(), assignments.end(), std::back_inserter(workAssignments),
212  [refIndex](const Assignment &a) { return a.refIndex == refIndex; });
213  // process assignments in batch
214  const size_t noOfAssignments = workAssignments.size();
215  auto finalRef = refs + (refIndex * elems);
216  if (0 == noOfAssignments) {
217  memset(finalRef, 0, elems * sizeof(T)); // clean the result
218  }
219  for (size_t offset = 0; offset < noOfAssignments; offset += m_maxBatchSize) {
220  size_t toProcess = std::min(m_maxBatchSize, noOfAssignments - offset);
221  workWeights.clear();
222  // copy image data for this batch
223  for (size_t i = 0; i < toProcess; ++i) {
224  auto &a = workAssignments.at(offset + i);
225  workWeights.emplace_back(a.weight);
226  memcpy(workCopy + (i * elems), others + (a.imgIndex * elems), elems * sizeof(T));
227  }
228  // update normalization factor
229  norm += std::accumulate(workWeights.begin(), workWeights.end(), 0.f);
230  // apply transform
231  interpolate(transformer, workCopy, workAssignments, workMatrices, offset, toProcess);
232  // sum images
233  transformer.sum(workRef, workWeights, toProcess, 1);
234  // collect intermediate result (single image per batch)
235  if (0 == offset) { // first batch -> copy data
236  memcpy(finalRef, workRef, elems * sizeof(T));
237  } else { // other batches -> sum data
238  for (size_t i = 0; i < elems; ++i) {
239  finalRef[i] += workRef[i];
240  }
241  }
242  }
243  // normalize the resulting image
244  for (size_t i = 0; i < elems; ++i) {
245  finalRef[i] /= norm;
246  }
247  // store reference
248  this->updateRefXmd(refIndex, workAssignments);
249  }
250  gpu.unpinMemory(workRef);
251  delete[] workCopy;
252  free(workRef);
253 }
