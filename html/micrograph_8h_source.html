259  template <typename T>
260  int templateScissor(const Image<T> &I,
261  const Particle_coords &P, MultidimArray<double> &result,
262  double Dmin, double Dmax, double scaleX, double scaleY,
263  bool only_check, bool fillBorders)
264  {
265  result.initZeros(Y_window_size, X_window_size);
266  int _i0 = ROUND(scaleY * P.Y) + FIRST_XMIPP_INDEX(Y_window_size);
267  int _iF = ROUND(scaleY * P.Y) + LAST_XMIPP_INDEX(Y_window_size);
268  int _j0 = ROUND(scaleX * P.X) + FIRST_XMIPP_INDEX(X_window_size);
269  int _jF = ROUND(scaleX * P.X) + LAST_XMIPP_INDEX(X_window_size);
270  int retval = 1;
271  double irange=1.0/(Dmax - Dmin);
272 
273  if (!fillBorders && (_i0 < 0 || _iF >= Ydim || _j0 < 0 || _jF >= Xdim))
274  retval = 0;
275  else
276  if (!only_check)
277  {
278 
279  for (int i = _i0, i_i0=0; i <= _iF; i++, i_i0++)
280  {
281  int ifrom=i;
282  if (fillBorders)
283  {
284  if (ifrom<0)
285  ifrom=0;
286  else if (ifrom>=Ydim)
287  ifrom=Ydim-1;
288  }
289  for (int j=_j0, j_j0=0; j<=_jF; j++, j_j0++)
290  {
291  int jfrom=j;
292  if (fillBorders)
293  {
294  if (jfrom<0)
295  jfrom=0;
296  else if (jfrom>=Xdim)
297  jfrom=Xdim-1;
298  }
299  double val=IMGPIXEL(I,ifrom,jfrom);
300  if (compute_transmitance)
301  {
302  double temp;
303  if (val < 1)
304  temp = val;
305  else
306  temp = log10(val);
307  if (compute_inverse)
308  A2D_ELEM(result,i_i0, j_j0) = (Dmax - temp) * irange;
309  else
310  A2D_ELEM(result,i_i0, j_j0) = (temp - Dmin) * irange;
311  }
312  else
313  {
314  if (compute_inverse)
315  A2D_ELEM(result, i_i0, j_j0) = (Dmax - val) * irange;
316  else
317  A2D_ELEM(result, i_i0, j_j0) = val;
318  }
319  }
320  }
321  }
322  return retval;
323  }
