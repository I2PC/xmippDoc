153 void ProgRecFourierAccel::produceSideinfo()
154 {
155  // Translate the maximum resolution to digital frequency
156  // maxResolution=sampling_rate/maxResolution;
157  maxResolutionSqr=maxResolution*maxResolution;
158 
159  // Read the input images
160  SF.read(fn_in);
161  SF.removeDisabled();
162 
163  // Ask for memory for the output volume and its Fourier transform
164  size_t objId = SF.firstObject();
165  FileName fnImg;
166  SF.getValue(MDL_IMAGE,fnImg,objId);
167  Image<double> I;
168  I.read(fnImg, HEADER);
169  int Ydim=YSIZE(I());
170  int Xdim=XSIZE(I());
171  if (Ydim!=Xdim)
172  REPORT_ERROR(ERR_MULTIDIM_SIZE,"This algorithm only works for squared images");
173  imgSize=Xdim;
174  paddedImgSize = Xdim*padding_factor_vol;
175  size_t conserveRows = (size_t) ceil((double) paddedImgSize * maxResolution * 2.0);
176  conserveRows = (size_t) ceil((double) conserveRows / 2.0);
177  maxVolumeIndexX = maxVolumeIndexYZ = 2 * conserveRows;
178 
179  // Build a table of blob values
180  Fourier_blob_table.resize(BLOB_TABLE_SIZE_SQRT);
181 
182  struct blobtype blobFourier,blobnormalized;
183  blobFourier=blob;
184  blobFourier.radius/=(padding_factor_vol*Xdim);
185  blobnormalized=blob;
186  blobnormalized.radius/=((double)padding_factor_proj/padding_factor_vol);
187  double deltaSqrt = (blob.radius*blob.radius) /(BLOB_TABLE_SIZE_SQRT-1);
188  double deltaFourier = (sqrt(3.)*Xdim/2.)/(BLOB_TABLE_SIZE_SQRT-1);
189 
190  // The interpolation kernel must integrate to 1
191  double iw0 = 1.0 / blob_Fourier_val(0.0, blobnormalized);
192  double padXdim3 = padding_factor_vol * Xdim;
193  padXdim3 = padXdim3 * padXdim3 * padXdim3;
194  double blobTableSize = blob.radius*sqrt(1./ (BLOB_TABLE_SIZE_SQRT-1));
195  for (int i = 0; i < BLOB_TABLE_SIZE_SQRT; i++)
196  {
197  //use a r*r sample instead of r
198  //DIRECT_VEC_ELEM(blob_table,i) = blob_val(delta*i, blob) *iw0;
199  blobTableSqrt[i] = blob_val(blobTableSize*sqrt((double)i), blob) *iw0;
200  //***
201  //DIRECT_VEC_ELEM(fourierBlobTableSqrt,i) =
202  // blob_Fourier_val(fourierBlobTableSize*sqrt(i), blobFourier)*padXdim3 *iw0;
203  VEC_ELEM(Fourier_blob_table,i) =
204  blob_Fourier_val(deltaFourier*i, blobFourier)*padXdim3 *iw0;
205  //#define DEBUG
206 #ifdef DEBUG
207 
208  std::cout << VEC_ELEM(Fourier_blob_table,i)
209  << " " << VEC_ELEM(fourierBlobTableSqrt,i)
210  << std::endl;
211 #endif
212  #undef DEBUG
213 
214  }
215  //iDelta = 1/delta;
216  iDeltaSqrt = 1/deltaSqrt;
217  iDeltaFourier = 1/deltaFourier;
218 
219  // Get symmetries
220  Matrix2D<double> Identity(3,3);
221  Identity.initIdentity();
222  R_repository.push_back(Identity);
223  if (fn_sym != "")
224  {
225  SymList SL;

249 Array2D<std::complex<float> >* ProgRecFourierAccel::cropAndShift(MultidimArray<std::complex<double> >& paddedFourier,
250  ProgRecFourierAccel * parent) {
251  int sizeX = parent->maxVolumeIndexX / 2; // input Fourier contains just one half of the space, second is complex conjugate
252  int sizeY = parent->maxVolumeIndexYZ;
253 
254  Array2D<std::complex<float> >* result = new Array2D<std::complex<float> >(sizeX, sizeY);
255  // convert image (shift to center and remove high frequencies)
256  std::complex<double> paddedFourierTmp;
257  int halfY = paddedFourier.ydim / 2;
258  double tempMyPadd[2];
259  for (size_t i = 0; i < paddedFourier.ydim; i++) {
260  for (size_t j = 0; j < sizeX; j++) {
261  if (i < sizeX || i >= (paddedFourier.ydim - sizeX)) {
262  // check the frequency
263  paddedFourierTmp = DIRECT_A2D_ELEM(paddedFourier, i, j);
264  FFT_IDX2DIGFREQ(j, parent->paddedImgSize, tempMyPadd[0]);
265  FFT_IDX2DIGFREQ(i, parent->paddedImgSize, tempMyPadd[1]);
266  if (tempMyPadd[0] * tempMyPadd[0] + tempMyPadd[1] * tempMyPadd[1]> parent->maxResolutionSqr) {
267  continue;
268  }
269  // do the shift
270  int myPadI = (i < halfY) ? i + sizeX : i - paddedFourier.ydim + sizeX;
271  (*result)(j, myPadI) = std::complex<float>(paddedFourierTmp.real(), paddedFourierTmp.imag());
272  }
273  }
274  }
275  return result;
276 }

278 void ProgRecFourierAccel::preloadBuffer(LoadThreadParams* threadParams,
279  ProgRecFourierAccel* parent,
280  bool hasCTF, std::vector<size_t>& objId)
281 {
282  ApplyGeoParams params;
283  MultidimArray<double> localPaddedImg;
284  FourierTransformer localTransformerImg;
285  MultidimArray< std::complex<double> > localPaddedFourier;
286  params.only_apply_shifts = true;
287  if (0 == threadParams->buffer1) {
288  threadParams->buffer1 = new ProjectionData[parent->bufferSize];
289  }
290  for (int bIndex = 0; bIndex < parent->bufferSize; bIndex++) {
291  double rot, tilt, psi;
292  Projection proj;
293  Matrix2D<double> localA(3, 3);
294  int imgIndex = threadParams->startImageIndex + bIndex;
295  ProjectionData* data = &threadParams->buffer1[bIndex];
296  if (imgIndex >= threadParams->endImageIndex) {
297  continue;
298  }
299  //Read projection from selfile, read also angles and shifts if present
300  //but only apply shifts
301  proj.readApplyGeo(*(threadParams->selFile), objId[imgIndex], params);
302  rot = proj.rot();
303  tilt = proj.tilt();
304  psi = proj.psi();
305  if (parent->do_weights && proj.weight() == 0.f) {
306  continue;
307  }
308  data->weight = (parent->do_weights) ? proj.weight() : 1.0;
309 
310  // Copy the projection to the center of the padded image
311  // and compute its Fourier transform
312  proj().setXmippOrigin();
313  localPaddedImg.initZeros(parent->paddedImgSize, parent->paddedImgSize);
314  localPaddedImg.setXmippOrigin();
315  const MultidimArray<double> &mProj = proj();
316  FOR_ALL_ELEMENTS_IN_ARRAY2D(mProj)
317  A2D_ELEM(localPaddedImg,i,j) = A2D_ELEM(mProj, i, j);
318  CenterFFT(localPaddedImg, true);
319 
320  // Fourier transformer for the images
321  localTransformerImg.setReal(localPaddedImg);
322  localTransformerImg.FourierTransform();
323  localTransformerImg.getFourierAlias(localPaddedFourier);
324 
325  // Compute the coordinate axes associated to this image
326  Euler_angles2matrix(rot, tilt, psi, localA);
327 
328  data->localAInv = localA.transpose();
329  data->img = cropAndShift(localPaddedFourier, parent);
330  data->imgIndex = imgIndex;
331  if (hasCTF) {
332  Array2D<float>* CTF = new Array2D<float>(data->img->getXSize(), data->img->getYSize());
333  Array2D<float>* modulator = new Array2D<float>(data->img->getXSize(), data->img->getYSize());
334  preloadCTF(threadParams, objId[imgIndex],parent, CTF, modulator);
335  }
336  // set data as usable
337  data->skip = false;
338  //#define DEBUG22
339 #ifdef DEBUG22
340  //CORRECTO
341 
342  if(threadParams->myThreadID%1==0)
343  {
344  proj.write((std::string) integerToString(threadParams->myThreadID) + "_" +\
345  integerToString(threadParams->imageIndex) + "proj.spi");
346 
347  ImageXmipp save44;
348  save44()=localPaddedImg;
349  save44.write((std::string) integerToString(threadParams->myThreadID) + "_" +\
350  integerToString(threadParams->imageIndex) + "local_padded_img.spi");
351 
352  FourierImage save33;
353  save33()=localPaddedFourier;
354  save33.write((std::string) integerToString(threadParams->myThreadID) + "_" +\
355  integerToString(threadParams->imageIndex) + "local_padded_fourier.spi");
356  FourierImage save22;
357  //save22()=*paddedFourier;
358  save22().alias(*(threadParams->localPaddedFourier));
359  save22.write((std::string) integerToString(threadParams->myThreadID) + "_" +\
360  integerToString(threadParams->imageIndex) + "_padded_fourier.spi");
361  }
362 
363 #endif

434  float y0 = p0.y;

573 inline void ProgRecFourierAccel::processVoxel(int x, int y, int z, const float transform[3][3], float maxDistanceSqr,
574  ProjectionData* const data) {
575  Point3D imgPos;
576  float wBlob = 1.f;
577  float wCTF = 1.f;
578  float wModulator = 1.f;
579 
580  // transform current point to center
581  imgPos.x = x - maxVolumeIndexX/2;
582  imgPos.y = y - maxVolumeIndexYZ/2;
583  imgPos.z = z - maxVolumeIndexYZ/2;
584  if (imgPos.x*imgPos.x + imgPos.y*imgPos.y + imgPos.z*imgPos.z > maxDistanceSqr) {
585  return; // discard iterations that would access pixel with too high frequency
586  }
587  // rotate around center
588  multiply(transform, imgPos);
589  // transform back and round
590  // just Y coordinate needs adjusting, since X now matches to picture and Z is irrelevant
591  int imgX = clamp((int)(imgPos.x + 0.5f), 0, data->img->getXSize() - 1);

595  wCTF = (*data->CTF)(imgX, imgY);
596  wModulator = (*data->modulator)(imgX, imgY);
597  }
598 
599  float weight = wBlob * wModulator * data->weight;
600 
601  tempVolume[z][y][x] += (*data->img)(imgX, imgY) * weight * wCTF;
602  tempWeights[z][y][x] += weight;

605 inline void ProgRecFourierAccel::processVoxelBlob(int x, int y, int z, const float transform[3][3], float maxDistanceSqr,
606  ProjectionData* const data) {
607  Point3D imgPos;
608  // transform current point to center
609  imgPos.x = x - maxVolumeIndexX/2;
610  imgPos.y = y - maxVolumeIndexYZ/2;
611  imgPos.z = z - maxVolumeIndexYZ/2;
612  if ((imgPos.x*imgPos.x + imgPos.y*imgPos.y + imgPos.z*imgPos.z) > maxDistanceSqr) {
613  return; // discard iterations that would access pixel with too high frequency
614  }
615  // rotate around center
616  multiply(transform, imgPos);
617  // transform back just Y coordinate, since X now matches to picture and Z is irrelevant
618  imgPos.y += maxVolumeIndexYZ / 2;
619 
620  // check that we don't want to collect data from far far away ...
621  float radiusSqr = blob.radius * blob.radius;
622  float zSqr = imgPos.z * imgPos.z;
623  if (zSqr > radiusSqr) return;
624 
625  // create blob bounding box
626  int minX = std::ceil(imgPos.x - blob.radius);
627  int maxX = std::floor(imgPos.x + blob.radius);
628  int minY = std::ceil(imgPos.y - blob.radius);
629  int maxY = std::floor(imgPos.y + blob.radius);
630  minX = std::max(minX, 0);
631  minY = std::max(minY, 0);
632  maxX = std::min(maxX, data->img->getXSize()-1);
633  maxY = std::min(maxY, data->img->getYSize()-1);
634  std::complex<float>* targetVolume = &tempVolume[z][y][x];
635  float* targetWeight = &tempWeights[z][y][x];
636  // ugly spaghetti code, but improves performance by app. 10%
637  if (0 != data->CTF) {
638  // check which pixel in the vicinity that should contribute
639  for (int i = minY; i <= maxY; i++) {
640  float ySqr = (imgPos.y - i) * (imgPos.y - i);
641  float yzSqr = ySqr + zSqr;
642  if (yzSqr > radiusSqr) continue;
643  for (int j = minX; j <= maxX; j++) {
644  float xD = imgPos.x - j;
645  float distanceSqr = xD*xD + yzSqr;
646  if (distanceSqr > radiusSqr) continue;
647 
648  float wCTF = (*data->CTF)(j, i);
649  float wModulator = (*data->modulator)(j, i);
650  int aux = (int) ((distanceSqr * iDeltaSqrt + 0.5f)); //Same as ROUND but avoid comparison
651  float wBlob = blobTableSqrt[aux];
652  float weight = wBlob * wModulator * data->weight;
653  *targetWeight += weight;
654  *targetVolume += (*data->img)(j, i) * weight * wCTF;
655  }
656  }
657  } else {
658  // check which pixel in the vicinity that should contribute
659  for (int i = minY; i <= maxY; i++) {
660  float ySqr = (imgPos.y - i) * (imgPos.y - i);
661  float yzSqr = ySqr + zSqr;
662  if (yzSqr > radiusSqr) continue;
663  for (int j = minX; j <= maxX; j++) {
664  float xD = imgPos.x - j;
665  float distanceSqr = xD*xD + yzSqr;
666  if (distanceSqr > radiusSqr) continue;
667 
668  int aux = (int) ((distanceSqr * iDeltaSqrt + 0.5f)); //Same as ROUND but avoid comparison
669  float wBlob = blobTableSqrt[aux];
670 
671  float weight = wBlob * data->weight;
672  *targetWeight += weight;
673  *targetVolume += (*data->img)(j, i) * weight;
674  }
675  }

688 void ProgRecFourierAccel::processProjection(
689  ProjectionData* projectionData,
690  const float transform[3][3],
691  const float transformInv[3][3])
692 {
693  int imgSizeX = projectionData->img->getXSize();
694  int imgSizeY = projectionData->img->getYSize();
695  const float maxDistanceSqr = (imgSizeX+(useFast ? 0.f : blob.radius)) * (imgSizeX+(useFast ? 0.f : blob.radius));
696  Point3D origin = {maxVolumeIndexX/2.f, maxVolumeIndexYZ/2.f, maxVolumeIndexYZ/2.f};
697  Point3D u, v, normal;
698  Point3D AABB[2];
699  Point3D cuboid[8];
700 
701  // calculate affected space
702  createProjectionCuboid(cuboid, imgSizeX, imgSizeY, useFast ? 0.f : blob.radius);
703  rotateCuboid(cuboid, transform);
704  translateCuboid(cuboid, origin);
705  computeAABB(AABB, cuboid, 0, 0, 0, maxVolumeIndexX, maxVolumeIndexYZ, maxVolumeIndexYZ);
706  getVectors(cuboid, u, v);
707  normal = getNormal(u, v);
708 
709  // prepare traversing
710  int minY, minX, minZ;
711  int maxY, maxX, maxZ;
712  minZ = floor(AABB[0].z);
713  minY = floor(AABB[0].y);
714  minX = floor(AABB[0].x);
715  maxZ = ceil(AABB[1].z);
716  maxY = ceil(AABB[1].y);
717  maxX = ceil(AABB[1].x);
718  // iterate along the longest axes, because it has the shortest projection to traverse plane
719  int nX, nY, nZ;
720  nX = std::abs(nX);
721  nY = std::abs(nY);
722  nZ = std::abs(nZ);
723 
724  if (nZ >= nX && nZ >= nY) { // iterate XY plane
725  for(int y = minY; y <= maxY; y++) {
726  for(int x = minX; x <= maxX; x++) {
727  if (useFast) {
728  float hitZ;
729  if (getZ(x, y, hitZ, u, v, *cuboid)) {
730  int z = (int)(hitZ + 0.5f); // rounding
731  processVoxel(x, y, z, transformInv, maxDistanceSqr, projectionData);
732  }
733  } else {
734  float z1, z2;
735  bool hit1 = getZ(x, y, z1, u, v, *cuboid); // lower plane
736  bool hit2 = getZ(x, y, z2, u, v, *(cuboid + 4)); // upper plane
737  if (hit1 || hit2) {
738  z1 = clamp(z1, 0, maxVolumeIndexYZ);
739  z2 = clamp(z2, 0, maxVolumeIndexYZ);
740  float lower = std::min(z1, z2);
741  float upper = std::max(z1, z2);
742  for (int z = std::floor(lower); z <= std::ceil(upper); z++) {
743  processVoxelBlob(x, y, z, transformInv, maxDistanceSqr, projectionData);
744  }
745  }
746  }
747  }
748  }
749  } else if (nY >= nX && nY >= nZ) { // iterate XZ plane
750  for(int z = minZ; z <= maxZ; z++) {
751  for(int x = minX; x <= maxX; x++) {
752  if (useFast) {
753  float hitY;
754  if (getY(x, hitY, z, u, v, *cuboid)) {
755  int y = (int)(hitY + 0.5f); // rounding
756  processVoxel(x, y, z, transformInv, maxDistanceSqr, projectionData);
757  }
758  } else {
759  float y1, y2;
760  bool hit1 = getY(x, y1, z, u, v, *cuboid); // lower plane
761  bool hit2 = getY(x, y2, z, u, v, *(cuboid + 4)); // upper plane
762  if (hit1 || hit2) {
763  y1 = clamp(y1, 0, maxVolumeIndexYZ);
764  y2 = clamp(y2, 0, maxVolumeIndexYZ);
765  float lower = std::min(y1, y2);
766  float upper = std::max(y1, y2);
767  for (int y = std::floor(lower); y <= std::ceil(upper); y++) {
768  processVoxelBlob(x, y, z, transformInv, maxDistanceSqr, projectionData);
769  }
770  }
771  }
772  }
773  }
774  } else if(nX >= nY && nX >= nZ) { // iterate YZ plane
775  for(int z = minZ; z <= maxZ; z++) {
776  for(int y = minY; y <= maxY; y++) {
777  if (useFast) {
778  float hitX;
779  if (getX(hitX, y, z, u, v, *cuboid)) {
780  int x = (int)(hitX + 0.5f); // rounding
781  processVoxel(x, y, z, transformInv, maxDistanceSqr, projectionData);
782  }
783  } else {
784  float x1, x2;
785  bool hit1 = getX(x1, y, z, u, v, *cuboid); // lower plane
786  bool hit2 = getX(x2, y, z, u, v, *(cuboid + 4)); // upper plane
787  if (hit1 || hit2) {
788  x1 = clamp(x1, 0, maxVolumeIndexX);
789  x2 = clamp(x2, 0, maxVolumeIndexX);
790  float lower = std::min(x1, x2);
791  float upper = std::max(x1, x2);
792  for (int x = std::floor(lower); x <= std::ceil(upper); x++) {
793  processVoxelBlob(x, y, z, transformInv, maxDistanceSqr, projectionData);
794  }
795  }
796  }
797  }
798  }

829 template<typename T>
830 T*** ProgRecFourierAccel::applyBlob(T***& input, float blobSize,
831  float* blobTableSqrt, float iDeltaSqrt) {
832  float blobSizeSqr = blobSize * blobSize;
833  int blob = floor(blobSize); // we are using integer coordinates, so we cannot hit anything further
834  T tmp;
835  T*** output;
836  // create new storage
837  allocate(output, maxVolumeIndexX+1, maxVolumeIndexYZ+1, maxVolumeIndexYZ+1);
838 
839  // traverse new storage
840  for (int i = 0; i <= maxVolumeIndexYZ; i++) {
841  for (int j = 0; j <= maxVolumeIndexYZ; j++) {
842  for (int k = 0; k <= maxVolumeIndexX; k++) {
843  // traverse input storage
844  tmp = (T) 0;
845  for (int z = std::max(0, i-blob); z <= std::min(maxVolumeIndexYZ, i+blob); z++) {
846  float dZSqr = (i - z) * (i - z);
847  for (int y = std::max(0, j-blob); y <= std::min(maxVolumeIndexYZ, j+blob); y++) {
848  float dYSqr = (j - y) * (j - y);
849  for (int x = std::max(0, k-blob); x <= std::min(maxVolumeIndexX, k+blob); x++) {
850  float dXSqr = (k - x) * (k - x);
851  float distanceSqr = dZSqr + dYSqr + dXSqr;
852  if (distanceSqr > blobSizeSqr) {
853  continue;
854  }
855  int aux = (int) ((distanceSqr * iDeltaSqrt + 0.5f)); //Same as ROUND but avoid comparison
856  float tmpWeight = blobTableSqrt[aux];
857  tmp += tmpWeight * input[z][y][x];
858  }
859  }
860  }
861  output[i][j][k] = tmp;
862  }
863  }
864  }
865  // free original data
866  release(input, maxVolumeIndexYZ+1, maxVolumeIndexYZ+1);
867  return output;
