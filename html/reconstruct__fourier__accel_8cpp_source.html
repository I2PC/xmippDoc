175 void ProgRecFourierAccel::produceSideinfo()
176 {
177  // Translate the maximum resolution to digital frequency
178  // maxResolution=sampling_rate/maxResolution;
179  maxResolutionSqr=maxResolution*maxResolution;
180 
181  // Read the input images
182  SF.read(fn_in);
183  SF.removeDisabled();
184 
185  // Ask for memory for the output volume and its Fourier transform
186  size_t objId = SF.firstRowId();
187  FileName fnImg;
188  SF.getValue(MDL_IMAGE,fnImg,objId);
189  Image<double> I;
190  I.read(fnImg, HEADER);
191  int Ydim=YSIZE(I());
192  int Xdim=XSIZE(I());
193  if (Ydim!=Xdim)
194  REPORT_ERROR(ERR_MULTIDIM_SIZE,"This algorithm only works for squared images");
195  imgSize=Xdim;
196  paddedImgSize = Xdim*padding_factor_vol;
197  auto conserveRows = (size_t) ceil((double) paddedImgSize * maxResolution * 2.0);
198  conserveRows = (size_t) ceil((double) conserveRows / 2.0);
199  maxVolumeIndexX = maxVolumeIndexYZ = 2 * conserveRows;
200 
201  // Build a table of blob values
202  Fourier_blob_table.resize(BLOB_TABLE_SIZE_SQRT);
203 
204  struct blobtype blobFourier,blobnormalized;
205  blobFourier=blob;
206  blobFourier.radius/=(padding_factor_vol*Xdim);
207  blobnormalized=blob;
208  blobnormalized.radius/=((double)padding_factor_proj/padding_factor_vol);
209  double deltaSqrt = (blob.radius*blob.radius) /(BLOB_TABLE_SIZE_SQRT-1);
210  double deltaFourier = (sqrt(3.)*Xdim/2.)/(BLOB_TABLE_SIZE_SQRT-1);
211 
212  // The interpolation kernel must integrate to 1
213  double iw0 = 1.0 / blob_Fourier_val(0.0, blobnormalized);
214  double padXdim3 = padding_factor_vol * Xdim;
215  padXdim3 = padXdim3 * padXdim3 * padXdim3;
216  double blobTableSize = blob.radius*sqrt(1./ (BLOB_TABLE_SIZE_SQRT-1));
217  for (int i = 0; i < BLOB_TABLE_SIZE_SQRT; i++)
218  {
219  //use a r*r sample instead of r
220  //DIRECT_VEC_ELEM(blob_table,i) = blob_val(delta*i, blob) *iw0;
221  blobTableSqrt[i] = blob_val(blobTableSize*sqrt((double)i), blob) *iw0;
222  //***
223  //DIRECT_VEC_ELEM(fourierBlobTableSqrt,i) =
224  // blob_Fourier_val(fourierBlobTableSize*sqrt(i), blobFourier)*padXdim3 *iw0;
225  VEC_ELEM(Fourier_blob_table,i) =
226  blob_Fourier_val(deltaFourier*i, blobFourier)*padXdim3 *iw0;
227  //#define DEBUG
228 #ifdef DEBUG
229 
230  std::cout << VEC_ELEM(Fourier_blob_table,i)
231  << " " << VEC_ELEM(fourierBlobTableSqrt,i)
232  << std::endl;
233 #endif
234  #undef DEBUG
235 
236  }
237  //iDelta = 1/delta;
238  iDeltaSqrt = 1/deltaSqrt;
239  iDeltaFourier = 1/deltaFourier;
240 
241  // Get symmetries
242  Matrix2D<double> Identity(3,3);
243  Identity.initIdentity();
244  R_repository.push_back(Identity);
245  if (fn_sym != "")
246  {
247  SymList SL;
248  SL.readSymmetryFile(fn_sym);
249  for (int isym = 0; isym < SL.symsNo(); isym++)
250  {
251  Matrix2D<double> L(4, 4), R(4, 4);
252  SL.getMatrices(isym, L, R);
253  R.resize(3, 3);
254  R_repository.push_back(R);
255  }
256  }

271 Array2D<std::complex<float> >* ProgRecFourierAccel::cropAndShift(MultidimArray<std::complex<double> >& paddedFourier,
272  ProgRecFourierAccel * parent) {
273  int sizeX = parent->maxVolumeIndexX / 2; // input Fourier contains just one half of the space, second is complex conjugate
274  int sizeY = parent->maxVolumeIndexYZ;
275 
276  auto* result = new Array2D<std::complex<float> >(sizeX, sizeY);
277  // convert image (shift to center and remove high frequencies)
278  std::complex<double> paddedFourierTmp;
279  int halfY = paddedFourier.ydim / 2;
280  double tempMyPadd[2];
281  for (size_t i = 0; i < paddedFourier.ydim; i++) {
282  for (size_t j = 0; j < sizeX; j++) {
283  if (i < sizeX || i >= (paddedFourier.ydim - sizeX)) {
284  // check the frequency
285  paddedFourierTmp = DIRECT_A2D_ELEM(paddedFourier, i, j);
286  FFT_IDX2DIGFREQ(j, parent->paddedImgSize, tempMyPadd[0]);
287  FFT_IDX2DIGFREQ(i, parent->paddedImgSize, tempMyPadd[1]);
288  if (tempMyPadd[0] * tempMyPadd[0] + tempMyPadd[1] * tempMyPadd[1]> parent->maxResolutionSqr) {
289  continue;
290  }
291  // do the shift
292  int myPadI = (i < halfY) ? i + sizeX : i - paddedFourier.ydim + sizeX;
293  (*result)(j, myPadI) = std::complex<float>(paddedFourierTmp.real(), paddedFourierTmp.imag());
294  }
295  }
296  }
297  return result;
298 }

300 void ProgRecFourierAccel::preloadBuffer(LoadThreadParams* threadParams,
301  ProgRecFourierAccel* parent,
302  bool hasCTF, std::vector<size_t>& objId)
303 {
304  ApplyGeoParams params;
305  MultidimArray<double> localPaddedImg;
306  FourierTransformer localTransformerImg;
307  MultidimArray< std::complex<double> > localPaddedFourier;
308  params.only_apply_shifts = true;
309  if (0 == threadParams->buffer1) {
310  threadParams->buffer1 = new ProjectionData[parent->bufferSize];
311  }
312  for (int bIndex = 0; bIndex < parent->bufferSize; bIndex++) {
313  double rot, tilt, psi;
314  Projection proj;
315  Matrix2D<double> localA(3, 3);
316  int imgIndex = threadParams->startImageIndex + bIndex;
317  ProjectionData* data = &threadParams->buffer1[bIndex];
318  if (imgIndex >= threadParams->endImageIndex) {
319  continue;
320  }
321  //Read projection from selfile, read also angles and shifts if present
322  //but only apply shifts
323  proj.readApplyGeo(*(threadParams->selFile), objId[imgIndex], params);
324  rot = proj.rot();
325  tilt = proj.tilt();
326  psi = proj.psi();
327  if (parent->do_weights && proj.weight() == 0.f) {
328  continue;
329  }
330  data->weight = (parent->do_weights) ? proj.weight() : 1.0;
331 
332  // Copy the projection to the center of the padded image
333  // and compute its Fourier transform
334  proj().setXmippOrigin();
335  localPaddedImg.initZeros(parent->paddedImgSize, parent->paddedImgSize);
336  localPaddedImg.setXmippOrigin();
337  const MultidimArray<double> &mProj = proj();
338  FOR_ALL_ELEMENTS_IN_ARRAY2D(mProj)
339  A2D_ELEM(localPaddedImg,i,j) = A2D_ELEM(mProj, i, j);
340  CenterFFT(localPaddedImg, true);
341 
342  // Fourier transformer for the images
343  localTransformerImg.setReal(localPaddedImg);
344  localTransformerImg.FourierTransform();
345  localTransformerImg.getFourierAlias(localPaddedFourier);
346 
347  // Compute the coordinate axes associated to this image
348  Euler_angles2matrix(rot, tilt, psi, localA);
349 
350  data->localAInv = localA.transpose();
351  data->img = cropAndShift(localPaddedFourier, parent);
352  data->imgIndex = imgIndex;
353  if (hasCTF) {
354  auto* CTF = new Array2D<float>(data->img->getXSize(), data->img->getYSize());
355  auto* modulator = new Array2D<float>(data->img->getXSize(), data->img->getYSize());
356  preloadCTF(threadParams, objId[imgIndex],parent, CTF, modulator);
357  }
358  // set data as usable
359  data->skip = false;
360  //#define DEBUG22
361 #ifdef DEBUG22
362  //CORRECTO
363 
364  if(threadParams->myThreadID%1==0)
365  {
366  proj.write((std::string) integerToString(threadParams->myThreadID) + "_" +\
367  integerToString(threadParams->imageIndex) + "proj.spi");
368 
369  ImageXmipp save44;
370  save44()=localPaddedImg;
371  save44.write((std::string) integerToString(threadParams->myThreadID) + "_" +\
372  integerToString(threadParams->imageIndex) + "local_padded_img.spi");
373 
374  FourierImage save33;
375  save33()=localPaddedFourier;
376  save33.write((std::string) integerToString(threadParams->myThreadID) + "_" +\
377  integerToString(threadParams->imageIndex) + "local_padded_fourier.spi");
378  FourierImage save22;
379  //save22()=*paddedFourier;
380  save22().alias(*(threadParams->localPaddedFourier));
381  save22.write((std::string) integerToString(threadParams->myThreadID) + "_" +\
382  integerToString(threadParams->imageIndex) + "_padded_fourier.spi");
383  }
384 
385 #endif
386 #undef DEBUG22
387  }

453 inline bool ProgRecFourierAccel::getZ(float x, float y, float& z, const Point3D& a, const Point3D& b, const Point3D& p0) {
454  // from parametric eq. of the plane
455  float x0 = p0.x;
456  float y0 = p0.y;
457  float z0 = p0.z;
458 
459  float u = ((y-y0)*a.x + (x0-x)*a.y) / (a.x * b.y - b.x * a.y);
460  float t = (-x0 + x - u*b.x) / (a.x);
461 
462  z = z0 + t*a.z + u*b.z;
463  return inRange(t, 0.f, 1.f) && inRange(u, 0.f, 1.f);
464 }

595 inline void ProgRecFourierAccel::processVoxel(int x, int y, int z, const float transform[3][3], float maxDistanceSqr,
596  ProjectionData* const data) {
597  Point3D imgPos;
598  float wBlob = 1.f;
599  float wCTF = 1.f;
600  float wModulator = 1.f;
601 
602  // transform current point to center
603  imgPos.x = x - maxVolumeIndexX/2;
604  imgPos.y = y - maxVolumeIndexYZ/2;
605  imgPos.z = z - maxVolumeIndexYZ/2;
606  if (imgPos.x*imgPos.x + imgPos.y*imgPos.y + imgPos.z*imgPos.z > maxDistanceSqr) {
607  return; // discard iterations that would access pixel with too high frequency
608  }
609  // rotate around center
610  multiply(transform, imgPos);
611  // transform back and round
612  // just Y coordinate needs adjusting, since X now matches to picture and Z is irrelevant
613  int imgX = clamp((int)(imgPos.x + 0.5f), 0, data->img->getXSize() - 1);

617  wCTF = (*data->CTF)(imgX, imgY);
618  wModulator = (*data->modulator)(imgX, imgY);
619  }
620 
621  float weight = wBlob * wModulator * data->weight;
622 
623  tempVolume[z][y][x] += (*data->img)(imgX, imgY) * weight * wCTF;
624  tempWeights[z][y][x] += weight;

627 inline void ProgRecFourierAccel::processVoxelBlob(int x, int y, int z, const float transform[3][3], float maxDistanceSqr,
628  ProjectionData* const data) {
629  Point3D imgPos;
630  // transform current point to center
631  imgPos.x = x - maxVolumeIndexX/2;
632  imgPos.y = y - maxVolumeIndexYZ/2;
633  imgPos.z = z - maxVolumeIndexYZ/2;
634  if ((imgPos.x*imgPos.x + imgPos.y*imgPos.y + imgPos.z*imgPos.z) > maxDistanceSqr) {
635  return; // discard iterations that would access pixel with too high frequency
636  }
637  // rotate around center
638  multiply(transform, imgPos);
639  // transform back just Y coordinate, since X now matches to picture and Z is irrelevant
640  imgPos.y += maxVolumeIndexYZ / 2;
641 
642  // check that we don't want to collect data from far far away ...
643  float radiusSqr = blob.radius * blob.radius;
644  float zSqr = imgPos.z * imgPos.z;
645  if (zSqr > radiusSqr) return;
646 
647  // create blob bounding box
648  int minX = std::ceil(imgPos.x - blob.radius);
649  int maxX = std::floor(imgPos.x + blob.radius);
650  int minY = std::ceil(imgPos.y - blob.radius);
651  int maxY = std::floor(imgPos.y + blob.radius);
652  minX = std::max(minX, 0);
653  minY = std::max(minY, 0);
654  maxX = std::min(maxX, data->img->getXSize()-1);
655  maxY = std::min(maxY, data->img->getYSize()-1);
656  std::complex<float>* targetVolume = &tempVolume[z][y][x];
657  float* targetWeight = &tempWeights[z][y][x];
658  // ugly spaghetti code, but improves performance by app. 10%
659  if (0 != data->CTF) {
660  // check which pixel in the vicinity that should contribute
661  for (int i = minY; i <= maxY; i++) {
662  float ySqr = (imgPos.y - i) * (imgPos.y - i);
663  float yzSqr = ySqr + zSqr;
664  if (yzSqr > radiusSqr) continue;
665  for (int j = minX; j <= maxX; j++) {
666  float xD = imgPos.x - j;
667  float distanceSqr = xD*xD + yzSqr;
668  if (distanceSqr > radiusSqr) continue;
669 
670  float wCTF = (*data->CTF)(j, i);
671  float wModulator = (*data->modulator)(j, i);
672  auto aux = (int) (distanceSqr * iDeltaSqrt + 0.5f); //Same as ROUND but avoid comparison
673  float wBlob = blobTableSqrt[aux];
674  float weight = wBlob * wModulator * data->weight;
675  *targetWeight += weight;
676  *targetVolume += (*data->img)(j, i) * weight * wCTF;
677  }
678  }
679  } else {
680  // check which pixel in the vicinity that should contribute
681  for (int i = minY; i <= maxY; i++) {
682  float ySqr = (imgPos.y - i) * (imgPos.y - i);
683  float yzSqr = ySqr + zSqr;
684  if (yzSqr > radiusSqr) continue;
685  for (int j = minX; j <= maxX; j++) {
686  float xD = imgPos.x - j;
687  float distanceSqr = xD*xD + yzSqr;
688  if (distanceSqr > radiusSqr) continue;
689 
690  auto aux = (int) (distanceSqr * iDeltaSqrt + 0.5f); //Same as ROUND but avoid comparison
691  float wBlob = blobTableSqrt[aux];
692 
693  float weight = wBlob * data->weight;
694  *targetWeight += weight;
695  *targetVolume += (*data->img)(j, i) * weight;
696  }
697  }
698  }

710 void ProgRecFourierAccel::processProjection(
711  ProjectionData* projectionData,
712  const float transform[3][3],
713  const float transformInv[3][3])
714 {
715  int imgSizeX = projectionData->img->getXSize();
716  int imgSizeY = projectionData->img->getYSize();
717  const float maxDistanceSqr = (imgSizeX+(useFast ? 0.f : blob.radius)) * (imgSizeX+(useFast ? 0.f : blob.radius));
718  Point3D origin = {maxVolumeIndexX/2.f, maxVolumeIndexYZ/2.f, maxVolumeIndexYZ/2.f};
719  Point3D u, v, normal;
720  Point3D AABB[2];
721  Point3D cuboid[8];
722

724  createProjectionCuboid(cuboid, imgSizeX, imgSizeY, useFast ? 0.f : blob.radius);
725  rotateCuboid(cuboid, transform);
726  translateCuboid(cuboid, origin);
727  computeAABB(AABB, cuboid, 0, 0, 0, maxVolumeIndexX, maxVolumeIndexYZ, maxVolumeIndexYZ);
728  getVectors(cuboid, u, v);
729  normal = getNormal(u, v);
730 
731  // prepare traversing
732  int minY, minZ;
733  int maxY, maxZ;
734  minZ = floor(AABB[0].z);
735  minY = floor(AABB[0].y);
736  maxZ = ceil(AABB[1].z);
737  maxY = ceil(AABB[1].y);
738 
739  for(int z = minZ; z <= maxZ; z++) {
740  for(int y = minY; y <= maxY; y++) {
741  if (useFast) {
742  float hitX;
743  if (getX(hitX, y, z, u, v, *cuboid)) {
744  auto x = (int)(hitX + 0.5f); // rounding
745  processVoxel(x, y, z, transformInv, maxDistanceSqr, projectionData);
746  }
747  } else {
748  float x1, x2;

750  bool hit2 = getX(x2, y, z, u, v, *(cuboid + 4)); // upper plane

752  x1 = clamp(x1, 0, maxVolumeIndexX);
753  x2 = clamp(x2, 0, maxVolumeIndexX);
754  float lower = std::min(x1, x2);
755  float upper = std::max(x1, x2);
756  for (int x = std::floor(lower); x <= std::ceil(upper); x++) {
757  processVoxelBlob(x, y, z, transformInv, maxDistanceSqr, projectionData);
758  }
759  }
760  }

792 template<typename T>
793 T*** ProgRecFourierAccel::applyBlob(T***& input, float blobSize,
794  float* blobTableSqrt, float iDeltaSqrt) {
795  float blobSizeSqr = blobSize * blobSize;
796  int blob = floor(blobSize); // we are using integer coordinates, so we cannot hit anything further
797  T tmp;
798  T*** output;
799  // create new storage
800  allocate(output, maxVolumeIndexX+1, maxVolumeIndexYZ+1, maxVolumeIndexYZ+1);
801 
802  // traverse new storage
803  for (int i = 0; i <= maxVolumeIndexYZ; i++) {
804  for (int j = 0; j <= maxVolumeIndexYZ; j++) {
805  for (int k = 0; k <= maxVolumeIndexX; k++) {
806  // traverse input storage
807  tmp = (T) 0;
808  for (int z = std::max(0, i-blob); z <= std::min(maxVolumeIndexYZ, i+blob); z++) {
809  float dZSqr = (i - z) * (i - z);
810  for (int y = std::max(0, j-blob); y <= std::min(maxVolumeIndexYZ, j+blob); y++) {
811  float dYSqr = (j - y) * (j - y);
812  for (int x = std::max(0, k-blob); x <= std::min(maxVolumeIndexX, k+blob); x++) {
813  float dXSqr = (k - x) * (k - x);
814  float distanceSqr = dZSqr + dYSqr + dXSqr;
815  if (distanceSqr > blobSizeSqr) {
816  continue;
817  }
818  auto aux = (int) (distanceSqr * iDeltaSqrt + 0.5f); //Same as ROUND but avoid comparison
819  float tmpWeight = blobTableSqrt[aux];
820  tmp += tmpWeight * input[z][y][x];
821  }
822  }
823  }
824  output[i][j][k] = tmp;
825  }
826  }
827  }
828  // free original data
829  release(input, maxVolumeIndexYZ+1, maxVolumeIndexYZ+1);
830  return output;
831 }

1002 void ProgRecFourierAccel::finishComputations( const FileName &out_name )
1003 {

1005  tempVolume = applyBlob(tempVolume, blob.radius, blobTableSqrt, iDeltaSqrt);
1006  tempWeights = applyBlob(tempWeights, blob.radius, blobTableSqrt, iDeltaSqrt);

1015  release(tempVolume, maxVolumeIndexYZ+1, maxVolumeIndexYZ+1);

1019  Vout().initZeros(paddedImgSize,paddedImgSize,paddedImgSize);
1020  FourierTransformer transformerVol;
1021  transformerVol.setThreadsNumber(2); // use just main and 'loading' thread
1022  transformerVol.fReal = &(Vout.data);
1023  transformerVol.setFourierAlias(VoutFourier);
1024  transformerVol.recomputePlanR2C();
1025 
1026  transformerVol.inverseFourierTransform();
1027  transformerVol.clear();
1028  CenterFFT(Vout(),false);
1029 
1030  // Correct by the Fourier transform of the blob
1031  Vout().setXmippOrigin();
1032  Vout().selfWindow(FIRST_XMIPP_INDEX(imgSize),FIRST_XMIPP_INDEX(imgSize),
1033  FIRST_XMIPP_INDEX(imgSize),LAST_XMIPP_INDEX(imgSize),
1034  LAST_XMIPP_INDEX(imgSize),LAST_XMIPP_INDEX(imgSize));
1035  double pad_relation= ((double)padding_factor_proj/padding_factor_vol);
1036  pad_relation = (pad_relation * pad_relation * pad_relation);
1037 
1038  MultidimArray<double> &mVout=Vout();
1039  double ipad_relation=1.0/pad_relation;
1040  double meanFactor2=0;
