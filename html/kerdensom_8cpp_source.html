116 void KerDenSOM::updateV(FuzzyMap* _som, const TS* _examples, const double& _reg)
117 {
118  unsigned t2 = 0; // Iteration index
119 
120  // Calculate Temporal scratch values
121  for (size_t cc = 0; cc < numNeurons; cc++)
122  {
123  double *ptrTmpMap_cc=&(tmpMap[cc][0]);
124  memset(ptrTmpMap_cc,0,dim*sizeof(double));
125  double &tmpDens_cc=tmpDens[cc];
126  if (_reg != 0)
127  tmpDens_cc = _reg * _som->getLayout().numNeig(_som, (SomPos) _som->indexToPos(cc));
128  else
129  tmpDens_cc = 0.;
130  for (size_t vv = 0; vv < numVectors; vv++)
131  {
132  double tmpU = (double) _som->memb[vv][cc];
133  tmpDens_cc += tmpU;
134  const floatFeature * ptrExample=&(_examples->theItems[vv][0]);
135  for (size_t j = 0; j < dim; j++)
136  ptrTmpMap_cc[j] += tmpU * ptrExample[j];
137  }
138  }
139 
140  // Update Code vectors using a sort of Gauss-Seidel iterative algorithm.
141  // Usually 100 iterations are enough.
142  double convergence = 1, stopError2, stopError1;
143  double *ptrTmpV=&(tmpV[0]);
144  while ((convergence > 1e-5) && (t2 < 100))
145  {
146  t2++;
147  stopError2 = 0;
148  stopError1 = 0;
149  for (size_t cc = 0; cc < numNeurons; cc++)
150  {
151  if (_reg != 0)
152  _som->localAve(_som->indexToPos(cc), tmpV);
153  double *ptrTmpMap_cc=&(tmpMap[cc][0]);
154  double iTmpDens_cc=1.0/tmpDens[cc];
155  floatFeature *ptrCodeVector_cc=&(_som->theItems[cc][0]);
156  for (size_t j = 0; j < dim; j++)
157  {
158  double tmpU = (ptrTmpMap_cc[j] + ptrTmpV[j] * _reg) * iTmpDens_cc;
159  stopError1 += fabs(ptrCodeVector_cc[j] - tmpU);
160  stopError2 += fabs(tmpU);
161  ptrCodeVector_cc[j] = (floatFeature) tmpU;
162  }
163  } // for
164  convergence = stopError1 / stopError2;
165  } // while
166 }

171 double KerDenSOM::mainIterations(FuzzyMap* _som, const TS* _examples, double& _sigma, const double& _reg)
172 {
173  int verbosity = listener->getVerbosity();
174  double stopError = 1e10, alpha, ts2;
175  size_t iter = 0;
176  if (somNSteps == 0)
177  return stopError;
178  if (verbosity == 1 || verbosity == 3)
179  listener->OnInitOperation(somNSteps);
180  do
181  {
182  iter++;
183  updateU(_som, _examples, _sigma, alpha);
184  ts2 = updateSigmaII(_som, _examples, _reg, alpha);
185  stopError = fabs(_sigma / ts2 - 1.);
186  _sigma = ts2;
187  updateV(_som, _examples, _reg);
188  if (verbosity == 1 || verbosity == 3)
189  listener->OnProgress(iter);
190  if (verbosity >= 2)
191  {
192  char s[100];
193  sprintf(s, "Iteration %d of %d. variation: %g\n", (int)iter, (int)somNSteps, stopError);
194  listener->OnReportOperation((std::string) s);
195  }
196  }
197  while ((stopError > epsilon) && (iter < somNSteps));
198  if (verbosity == 1 || verbosity == 3)
199  listener->OnProgress(somNSteps);
200  return stopError;
201 }

206 // Estimate Sigma Part I
207 double KerDenSOM::updateSigmaI(FuzzyMap* _som, const TS* _examples)
208 {
209  double t = 0;
210 
211  // Computing Sigma (Part I)
212  for (size_t vv = 0; vv < numVectors; vv++)
213  {
214  const FeatureVector &example=_examples->theItems[vv];
215  const floatFeature *ptrExample0=&example[0];
216  for (size_t cc = 0; cc < numNeurons; cc++)
217  {
218  double r = 0.0;
219  const floatFeature *ptrExample=ptrExample0;
220  const floatFeature *ptrCodeVector=&(_som->theItems[cc][0]);
221  for (size_t j = 0; j < dim; j++)
222  {
223  double diff=(double)(*ptrExample++) - (double)(*ptrCodeVector++);
224  r += diff*diff;
225  }
226  t += r * (double)(_som->memb[vv][cc]);
227  }
228  }
229  return (double)(t / (double)(numVectors*dim));
230 }

283 void KerDenSOM::updateU1(FuzzyMap* _som, const TS* _examples)
284 {
285 
286  double auxProd, auxDist, tmp;
287  size_t k, j, i;
288 
289  // Update Membership matrix
290  for (k = 0; k < numVectors; k++)
291  {
292  auxProd = 1;
293  for (j = 0; j < numNeurons; j++)
294  auxProd *= euclideanDistance(_som->theItems[j], _examples->theItems[k]);
295 
296  if (auxProd == 0.)
297  { // Apply k-means criterion (Data-CB) must be > 0
298  for (j = 0; j < numNeurons; j ++)
299  if (euclideanDistance(_som->theItems[j], _examples->theItems[k]) == 0.)
300  _som->memb[k][j] = 1.0;
301  else
302  _som->memb[k][j] = 0.0;
303  }
304  else
305  {
306  for (i = 0; i < numNeurons; i ++)
307  {
308  auxDist = 0;
309  for (j = 0; j < numNeurons; j ++)
310  {
311  tmp = euclideanDistance(_som->theItems[i], _examples->theItems[k]) /
312  euclideanDistance(_som->theItems[j], _examples->theItems[k]);
313  auxDist += pow(tmp, 2);
314  } // for j
315  _som->memb[k][i] = (floatFeature) 1.0 / auxDist;
316  } // for i
317  } // if auxProd
318  } // for k
319 }
