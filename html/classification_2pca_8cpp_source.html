481 void Running_PCA::new_sample(const Matrix1D<double> &sample)
482 {
483  n++;
484 
485  // Re-estimate sample mean
486  sum_all_samples += sample;
487  current_sample_mean = sum_all_samples / n;
488 
489  // Estimate eigenvectors
490  Matrix1D<double> un = sample;
491  un.row = false;
492  for (int j = 0; j < J; j++)
493  {
494  if (n <= j + 1)
495  {
496  // If there are not enough samples to estimate this eigenvector, then
497  // skip it
498  double norm = un.module();
499  if (norm > XMIPP_EQUAL_ACCURACY)
500  un /= norm;
501  eigenvectors.setCol(j, un);
502  }
503  else
504  {
505  // If there are enough samples
506  // Subtract the sample mean to have a zero-mean vector
507  if (j == 0)
508  un -= current_sample_mean;
509 
510  // Compute the scale of this vector as the dot product
511  // between un and the current eigenvector estimate
512  double scale = 0;
513  for (int i = 0; i < d; i++)
514  scale += un(i) * eigenvectors(i, j);
515 
516  // Re-estimate the eigenvector
517  double norm2 = 0;
518  double w1 = (double)(n - 1.0) / n;
519  double w2 = (1.0 - w1) * scale;
520  for (int i = 0; i < d; i++)
521  {
522  eigenvectors(i, j) =
523  w1 * eigenvectors(i, j) +
524  w2 * un(i);
525  norm2 += eigenvectors(i, j) *
526  eigenvectors(i, j);
527  }
528 
529  // Renormalize
530  if (norm2 > XMIPP_EQUAL_ACCURACY)
531  {
532  double norm = sqrt(norm2);
533  for (int i = 0; i < d; i++)
534  eigenvectors(i, j) /= norm;
535  }
536 
537  // Project un onto the space spanned by this eigenvector
538  double project = 0;
539  for (int i = 0; i < d; i++)
540  project += un(i) * eigenvectors(i, j);
541  for (int i = 0; i < d; i++)
542  un(i) -= project * eigenvectors(i, j);
543 
544  // Update the variance of this vector
545  sum_proj(j) += project;
546  sum_proj2(j) += project * project;
547  }
548  }
549 }
