116 void CrystalARTRecons::preProcess(GridVolume &vol_basis0, int level, int rank)
117 {
118  // Lattice vectors in BCC units
119  a = avox / artPrm.grid_relative_size;
120  b = bvox / artPrm.grid_relative_size;
121 
122  // Compute space_group
123  if (artPrm.fn_sym != "")
124  space_group = artPrm.SL.crystallographicSpaceGroup(a_mag, b_mag,
125  ang_a2b_deg);
126  else
127  space_group = sym_P1;
128 
129  // Integer lattice vectors ----------------------------------------------
130  Matrix2D<double> D;
131  computeIntegerLattice(a, b, a_mag / artPrm.grid_relative_size,
132  b_mag / artPrm.grid_relative_size,
133  ang_a2b_deg, aint, bint, D,
134  space_group);
135  // Define two more useful lattice vectors
136  ai = aint / 2;
137  bi = bint / 2;
138 
139  // Set general deformation pointer to this matrix
140  artPrm.D = new Matrix2D<double>;
141  *(artPrm.D) = D;
142  artPrm.D->resize(3, 3);
143  MAT_ELEM(*(artPrm.D), 2, 2) = 1;
144  artPrm.Dinv = new Matrix2D<double>;
145  *(artPrm.Dinv) = artPrm.D->inv();
146  artPrm.basis.setD(artPrm.D);
147 
148  // Unit cell mask within volume -----------------------------------------
149  // Compute the 4 parallelogram corners
150  Matrix1D<double> c1 = ai + bi;
151  Matrix1D<double> c2 = -ai + bi;
152  Matrix1D<double> c3 = -c1;
153  Matrix1D<double> c4 = -c2;
154  Matrix1D<double> c1c3 = c1 - c3; // These extra variables are needed because
155  Matrix1D<double> c2c4 = c2 - c4; // the compiler messes up
156 
157  // Resize unit cell mask
158  // The unit mask is a little bigger to avoid the possibility of losing
159  // any basis due to a too tight mask.
160  int mask_xdim = CEIL(XMIPP_MAX(ABS(XX(c1c3)), ABS(XX(c2c4)))) + 3;
161  int mask_ydim = CEIL(XMIPP_MAX(ABS(YY(c1c3)), ABS(YY(c2c4)))) + 3;
162  unit_cell_mask.initZeros(mask_ydim, mask_xdim);
163  unit_cell_mask.setXmippOrigin();
164 
165  // Resize the reconstructed volume
166  Matrix1D<double> r1(3), r2(3);
167  for (size_t n = 0; n < vol_basis0.VolumesNo(); n++)
168  {
169  Image<double> &V = vol_basis0(n);
170  ZZ(r1) = XMIPP_MIN(ZZ(r1), STARTINGZ(V()));
171  ZZ(r2) = XMIPP_MAX(ZZ(r2), FINISHINGZ(V()));
172  }
173  XX(r1) = STARTINGX(unit_cell_mask);
174  YY(r1) = STARTINGY(unit_cell_mask);
175  r1 *= artPrm.grid_relative_size;
176  XX(r2) = FINISHINGX(unit_cell_mask);
177  YY(r2) = FINISHINGY(unit_cell_mask);
178  r2 *= artPrm.grid_relative_size;
179  vol_basis0.resize(r1, r2);
180 
181  // Fill the unit cell mask
182  Matrix1D<double> r(2);
183  FOR_ALL_ELEMENTS_IN_ARRAY2D(unit_cell_mask)
184  {
185  // Position vector of actual BCC element been considered
186  YY(r) = i;
187  XX(r) = j;
188 
189  // Vectors from r to each corner
190  Matrix1D<double> c1r = c1 - r;
191  Matrix1D<double> c2r = c2 - r;
192  Matrix1D<double> c3r = c3 - r;
193  Matrix1D<double> c4r = c4 - r;
194 
195  // Product of each of these vectors with tha parallelogram borders
196  int sgn[4];
197  sgn[0] = SGN0_VEC_PRODUCT(c1r, -ai);
198  sgn[1] = SGN0_VEC_PRODUCT(c2r, -bi);
199  sgn[2] = SGN0_VEC_PRODUCT(c3r, ai);
200  sgn[3] = SGN0_VEC_PRODUCT(c4r, bi);
201 
202 #ifdef DEBUG_A_LOT
203 
204  std::cout << "(x,y)=(" << XX(r) << "," << YY(r) << ") " << sgn[0] << sgn[1]
205  << sgn[2] << sgn[3] << std::endl;
206 #endif
207 
208  // Now check if point is inside
209  int inside_table[4][4] = {
210  {1, 1, 1, 1},
211  {0, 1, 1, 1},
212  {1, 0, 1, 1},
213  // 1,1,0,1, Take this side out
214  // 1,1,1,0, Take this side out
215  {0, 0, 1, 1}
216  // 1,0,0,1, and their three Vertex
217  // 1,1,0,0,
218  // 0,1,1,0
219  };
220 #define INSIDE_ACCORDING_TO(n) \
221  sgn[0]==inside_table[n][0] && sgn[1]==inside_table[n][1] && \
222  sgn[2]==inside_table[n][2] && sgn[3]==inside_table[n][3]
223 
224  for (int n = 0; n < 4; n++)
225  if (INSIDE_ACCORDING_TO(n))
226  {
227  unit_cell_mask(i, j) = 1;
228 #ifdef DEBUG_A_LOT
229 
230  std::cout << " Inside\n";
231 #endif
232 
233  break;
234  }
235  }
236 
237  // Now calculate side info of basel class
238  ARTReconsBase::preProcess(vol_basis0);
239 
240  // Show all parameters --------------------------------------------------
241 #ifdef DEBUG
242  std::cout << "avox= " << avox.transpose() << std::endl;
243  std::cout << "bvox= " << bvox.transpose() << std::endl;
244  std::cout << "grid_relative_size= " << artPrm.grid_relative_size << std::endl;
245  std::cout << "a= " << a.transpose() << std::endl;
246  std::cout << "b= " << b.transpose() << std::endl;
247  std::cout << "aint= " << aint.transpose() << std::endl;
248  std::cout << "bint= " << bint.transpose() << std::endl;
249  std::cout << "ai= " << ai.transpose() << std::endl;
250  std::cout << "bi= " << bi.transpose() << std::endl;
251  std::cout << "D= " << D;
252  std::cout << "Check that a=D*aint " << (D*aint).transpose() << std::endl;
253  std::cout << "Check that b=D*bint " << (D*bint).transpose() << std::endl;
254  std::cout << "Symmetry group: " << space_group;
255  ImageXmipp I;
256  I = unit_cell_mask;
257  I.write("unit_cell_mask.xmp");
258  std::cout << "unit_cell_mask shape=";
259  unit_cell_mask.printShape();
260  std::cout << std::endl;
261 #endif
262 }

269 void CrystalARTRecons::singleStep(
270  GridVolume &vol_in, // Input Reconstructed volume
271  GridVolume *vol_out, // Output Reconstructed volume
272  Projection &theo_proj, // Projection of the reconstruction
273  // It is outside to make it visible
274  // just if it needed for any
275  // other purpose
276  Projection &read_proj, // Real projection
277  int sym_no, // Symmetry matrix index
278  Projection &diff_proj, // Difference between read and
279  // theoretical projection
280  Projection &corr_proj, // Correcting projection
281  Projection &alig_proj, // Translation alignement aux proj
282  double &mean_error, // Mean error over the pixels
283  int numIMG, // number of images in the set
284  // in SIRT the correction must
285  // be divided by this number
286  double lambda, // Lambda to be used
287  int imagen_no, // Projection number
288  const FileName &fn_ctf, // CTF to apply
289  const MultidimArray<int> *maskPtr, // Mask to apply
290  bool refine)
291 {
292  // Only works for blob volumes .............................................
293  if (artPrm.basis.type != Basis::blobs)
294  REPORT_ERROR(ERR_VALUE_INCORRECT,
295  "This function only works with blob volumes");
296 
297  // Compute lattice vectors to be used ......................................
298  Matrix1D<double> aint, bint, shift;
299  aint.resize(2);
300  bint.resize(2);
301  shift.resize(3);
302  shift.initZeros();
303 
304  symmetrizeCrystalVectors(aint, bint, shift, this->space_group, sym_no,
305  this->aint, this->bint);
306  // Project structure .......................................................
307  // The correction image is reused in this call to store the normalizing
308  // projection, ie, the projection of an all-1 volume
309 
310  project_Crystal_Volume(vol_in, artPrm.basis, theo_proj,
311  corr_proj, YSIZE(read_proj()), XSIZE(read_proj()),
312  read_proj.rot(), read_proj.tilt(), read_proj.psi(), shift,
313  aint, bint, *(artPrm.D), *(artPrm.Dinv), this->unit_cell_mask,
314  FORWARD, artPrm.eq_mode);
315  double shift_X, shift_Y;
316 
317  // #define DEBUG_SHIFT
318 #ifdef DEBUG_SHIFT
319 
320  Matrix2D<double> A(3, 3);
321  A.initIdentity();
322  dMij(A, 0, 2) = 8;
323  dMij(A, 1, 2) = -5;
324  std::cout << "A" << A;
325  //move read_proj
326  FOR_ALL_DIRECT_ELEMENTS_IN_MATRIX2D(theo_proj())
327  dMij(theo_proj(), i, j) = dMij(read_proj(), i, j);
328  applyGeometry(IMGMATRIX(read_proj), A, IMGMATRIX(theo_proj), IS_NOT_INV, WRAP);
329 #endif
330 #undef DEBUG_SHIFT
331 
332  if (artPrm.ref_trans_after != -1 &&
333  imagen_no > artPrm.ref_trans_after && imagen_no != 0)
334  {
335  CorrelationAux aux;
336  calculate_and_find_correlation_max_proj(read_proj, theo_proj,
337  alig_proj,
338  shift_X, shift_Y,
339  artPrm.ref_trans_step,
340  artPrm.ref_trans_after,
341  imagen_no, aux);
342 
343  // Apply correction
344  Matrix2D<double> Correction(3, 3);
345  alig_proj().resize(read_proj());
346  Correction.initIdentity();
347 
348  dMij(Correction, 0, 2) = - shift_X;
349  dMij(Correction, 1, 2) = - shift_Y;
350  //copy theo_proj to a temporal matrix
351  FOR_ALL_DIRECT_ELEMENTS_IN_ARRAY2D(theo_proj())
352  dAij(alig_proj(), i, j) = dAij(read_proj(), i, j);
353  applyGeometry(LINEAR, IMGMATRIX(alig_proj), IMGMATRIX(read_proj), Correction,
354  IS_NOT_INV, WRAP);
355  }
356  // Now compute differences .................................................
357  double applied_lambda = lambda / numIMG; // In ART mode, numIMG=1
358  mean_error = 0;
359  diff_proj().resize(read_proj());
360 
361  FOR_ALL_ELEMENTS_IN_ARRAY2D(IMGMATRIX(read_proj))
362  {
363  // Compute difference image and error
364  IMGPIXEL(diff_proj, i, j) = IMGPIXEL(read_proj, i, j) - IMGPIXEL(theo_proj, i, j);
365  mean_error += IMGPIXEL(diff_proj, i, j) * IMGPIXEL(diff_proj, i, j);
366 
367  // Compute the correction image
368  if (ABS(IMGPIXEL(corr_proj, i, j)) < 1)
369  IMGPIXEL(corr_proj, i, j) = SGN(IMGPIXEL(corr_proj, i, j));
370  IMGPIXEL(corr_proj, i, j) =
371  applied_lambda * IMGPIXEL(diff_proj, i, j) / IMGPIXEL(corr_proj, i, j);
372  }
373  mean_error /= XSIZE(diff_proj()) * YSIZE(diff_proj());
374 
375  // Backprojection of correction plane ......................................
376  project_Crystal_Volume(*vol_out, artPrm.basis, theo_proj,
377  corr_proj, YSIZE(read_proj()), XSIZE(read_proj()),
378  read_proj.rot(), read_proj.tilt(), read_proj.psi(), shift,
379  aint, bint, *(artPrm.D), *(artPrm.Dinv), this->unit_cell_mask,
380  BACKWARD, artPrm.eq_mode);
381 }

395 /* Compute integer lattice ================================================= */
396 void computeIntegerLattice(const Matrix1D<double> &a,
397  const Matrix1D<double> &b,
398  double a_mag_grid, double b_mag_grid,
399  double ang_a2b_deg,
400  Matrix1D<double> &aint,
401  Matrix1D<double> &bint,
402  Matrix2D<double> &D,
403  int space_group)
404 {
405 
406  // Integer lattice
407  // If we force it to be orthogonal the symmetrization is easier
408  //a_mag=original a_mag/sampling/grid_relatice_size
409  aint.resize(2);
410  bint.resize(2);
411  XX(aint) = ROUND(a_mag_grid);
412  YY(aint) = 0.;
413  XX(bint) = 0.0;
414  YY(bint) = ROUND(b_mag_grid);
415 
416  //different crystalline grids impose different restrictions
417  //on the grid, we will check them here but only if the user
418  //has provided a symmetry file
419 
420  if (space_group != sym_undefined)
421  {
422 
423  switch (space_group)
424  {
425 
426  case sym_P1:
427  break;// no check needed
428  case sym_P2_122://XX(aint) and YY(aint) should be even
429  if (XX(aint) != 2*(int)(XX(aint) / 2) ||
430  YY(bint) != 2*(int)(YY(bint) / 2))
431  {
432  std::cout << "\nLattice connstrains for P2_122 are not satisficed"
433  << "\nRound[mag_a/(sampling*grid_size)] must be even"
434  << "\nPlease modify the parmeters and try again" << std::endl;
435  exit(0);
436  }
437  break;
438  case sym_P22_12://XX(aint) and YY(aint) should be even
439  if (XX(aint) != 2*(int)(XX(aint) / 2) ||
440  YY(bint) != 2*(int)(YY(bint) / 2))
441  {
442  std::cout << "\nLattice connstrains for P22_12 are not satisficed"
443  << "\nRound[mag_a/(sampling*grid_size)] must be even"
444  << "\nPlease modify the parmeters and try again" << std::endl;
445  exit(0);
446  }
447  break;
448  case sym_P42_12://XX(aint) and YY(aint) should be even
449  if (XX(aint) != 2*(int)(XX(aint) / 2) ||
450  YY(bint) != 2*(int)(YY(bint) / 2))
451  {
452  std::cout << "\nLattice connstrains for P4212 are not satisficed"
453  << "\nRound[mag_a/(sampling*grid_size)] must be even"
454  << "\nPlease modify the parmeters and try again" << std::endl;
455  exit(0);
456  }
457  break;
458  case sym_P4:
459  case sym_P6:
460  break;// no check needed
461  default:
462  std::cerr << "\n Congratulations: you have found a bug in the\n"
463  << "routine compute_integer_lattice or\n"
464  << "you are using a non implemented symmetry group\n"
465  << std::endl;
466  exit(0);
467  break;
468  }//switch(space_group) end
469 
470  }//if (prm.fn_sym!="")
471 
472 
473  // Converting matrix
474  // a=D*aint
475  // b=D*bint
476  Matrix2D<double> L(2, 2), LI(2, 2);
477 
478  L .setCol(0, a);
479  L .setCol(1, b);
480  LI.setCol(0, aint);
481  LI.setCol(1, bint);
482  D = L * LI.inv();
483 }

527  corner1 = (double)(a0 - 1) * eprm.a + (double)(b0 - 1) * eprm.b; // CO: I'm not very satisfied with
528  corner2 = (double)(aF + 1) * eprm.a + (double)(bF + 1) * eprm.b; // the +1 and -1 but it works

538  {
539  Image<double> &V = vol(n);
540  FOR_ALL_ELEMENTS_IN_ARRAY2D(eprm.unit_cell_mask)
541  {
542  if (A2D_ELEM(eprm.unit_cell_mask, i, j))
543  {
544 #ifdef DEBUG2
