97 /************************************************************************
98 @Function: readIMAGIC
99 @Description:
100  Reading an IMAGIC image format.
101 @Algorithm:
102  A 2D file format for the IMAGIC package.
103  The header is stored in a separate file with extension ".hed" and
104  a fixed size of 1024 bytes per image.
105  The image data is stored in a single block in a file with the
106  extension ".img".
107  Byte order determination: Year and hour values
108  must be less than 256*256.
109  Data types: PACK = byte, INTG = short, REAL = float,
110  RECO,COMP = complex float.
111  Transform type: Centered (COMP data type)
112  RECO is not a transform
113  Note that the x and y dimensions are interchanged (actually a display issue).
114 @Arguments:
115  Bimage* p the image structure.
116  int select image selection in multi-image file (-1 = all images).
117 @Returns:
118  int error code (<0 means failure).
119 **************************************************************************/
123 int ImageBase::readIMAGIC(size_t select_img)
124 {
125 #undef DEBUG
126  //#define DEBUG
127 #ifdef DEBUG
128  printf("DEBUG readIMAGIC: Reading Imagic file\n");
129 #endif
130 
131  IMAGIChead* header = new IMAGIChead;
132 
133  if ( fread( header, IMAGICSIZE, 1, fhed ) < 1 )
134  REPORT_ERROR(ERR_IO_NOREAD,(String)"readIMAGIC: header file of " + filename + " cannot be read");
135 
136  // Determine byte order and swap bytes if from little-endian machine
137  if ( (swap = (( abs(header->nyear) > SWAPTRIG ) || ( header->ixlp > SWAPTRIG ))) )
138  swapPage((char *) header, IMAGICSIZE - 916, DT_Float); // IMAGICSIZE - 916 is to exclude labels from swapping
139 
140  DataType datatype=DT_Float;
141 
142  if ( strstr(header->type,"PACK") )
143  datatype = DT_UChar;
144  else if ( strstr(header->type,"INTG") )
145  datatype = DT_Short;
146  else if ( strstr(header->type,"REAL") )
147  datatype = DT_Float;
148  else if ( strstr(header->type,"RECO") )
149  {
150  datatype = DT_CFloat; // Complex data
151  transform = NoTransform;
152  }
153  else if ( strstr(header->type,"COMP") )
154  {
155  datatype = DT_CFloat; // Complex transform data
156  transform = Centered;
157  }
158 
159  // Set min-max values and other statistical values
160  if ( header->sigma == 0 && header->varian != 0 )
161  header->sigma = sqrt(header->varian);
162  if ( header->densmax == 0 && header->densmin == 0 && header->sigma != 0 )
163  {
164  header->densmin = header->avdens - header->sigma;
165  header->densmax = header->avdens + header->sigma;
166  }
167 
168  offset = 0; // separate header file
169 
170  MDMainHeader.setValue(MDL_MIN,(double)header->densmin);
171  MDMainHeader.setValue(MDL_MAX,(double)header->densmax);
172  MDMainHeader.setValue(MDL_AVG,(double)header->avdens);
173  MDMainHeader.setValue(MDL_STDDEV,(double)header->sigma);
174  MDMainHeader.setValue(MDL_SAMPLINGRATE_X,(double)1.);
175  MDMainHeader.setValue(MDL_SAMPLINGRATE_Y,(double)1.);
176  MDMainHeader.setValue(MDL_SAMPLINGRATE_Z,(double)1.);
177  MDMainHeader.setValue(MDL_DATATYPE,(int)datatype);
178 
179  int _xDim,_yDim,_zDim;
180  size_t _nDim;
181  _xDim = (int) header->iylp;
182  _yDim = (int) header->ixlp;
183  _zDim = (int) 1;
184  _nDim = (size_t) header->ifn + 1 ;
185 
186  if ( select_img > _nDim )
187  REPORT_ERROR(ERR_INDEX_OUTOFBOUNDS, formatString("readImagic: Image number %lu exceeds stack size %lu", select_img, _nDim));
188 
189  if( select_img != ALL_IMAGES )
190  _nDim = 1;
191 
192  replaceNsize = _nDim;
193  setDimensions(_xDim, _yDim, _zDim, _nDim );
194 
195  if (dataMode == HEADER || (dataMode == _HEADER_ALL && _nDim > 1)) // Stop reading if not necessary
196  {
197  delete header;
198  return 0;
199  }
200 
201  // Get the header information
202  fseek( fhed, IMG_INDEX(select_img) * IMAGICSIZE, SEEK_SET );
203 
204  MD.clear();
205  MD.resize(_nDim,MDL::emptyHeader);
206  double daux=1.;
207  for ( size_t i = 0; i < _nDim; ++i )
208  {
209  if ( fread( header, IMAGICSIZE, 1, fhed ) < 1 )
210  return(-2);
211  {
212  if ( swap )
213  swapPage((char *) header, IMAGICSIZE - 916, DT_Float);
214 
215  if (dataMode == _HEADER_ALL || dataMode == _DATA_ALL)
216  {
217  MD[i].setValue(MDL_SHIFT_X, (double)-1. * header->ixold);
218  MD[i].setValue(MDL_SHIFT_Y, (double)-1. * header->iyold);
219  MD[i].setValue(MDL_SHIFT_Z, 0.);
220  MD[i].setValue(MDL_ANGLE_ROT, (double)-1. * header->euler_alpha);
221  MD[i].setValue(MDL_ANGLE_TILT,(double)-1. * header->euler_beta);
222  MD[i].setValue(MDL_ANGLE_PSI, (double)-1. * header->euler_gamma);
223  MD[i].setValue(MDL_WEIGHT, 1.);
224  MD[i].setValue(MDL_SCALE, daux);
225  }
226  }
227  }
228  delete header;
229 
230  if (dataMode < DATA) // Don't read the individual header and the data if not necessary
231  return 0;
232 
233  size_t pad = 0;
234  readData(fimg, select_img, datatype, pad );
235 
236  return(0);
237 }
