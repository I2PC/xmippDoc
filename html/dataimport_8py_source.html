44  def importMicrographs(self):
45  """ Import a SetOfMicrographs from a given micrograph metadata.
46  (usually the result "micrographs.xmd" from Xmipp protocols)
47  If the CTF is found, a SetOfCTF will be also created.
48  """
49  self._findPathAndCtf(label=md.MDL_MICROGRAPH)
50  micSet = self.protocol._createSetOfMicrographs()
51  micSet.setObjComment('Micrographs imported from Xmipp metadata:\n'
52  '%s' % self._mdFile)
53 
54  # Update both samplingRate and acquisition with parameters
55  # selected in the protocol form
56  self.protocol.setSamplingRate(micSet)
57  micSet.setIsPhaseFlipped(self.protocol.haveDataBeenPhaseFlipped.get())
58  self.protocol.fillAcquisition(micSet.getAcquisition())
59  # Read the micrographs from the 'self._mdFile' metadata
60  # but fixing the filenames with new ones (linked or copy to extraDir)
61  readSetOfMicrographs(self._mdFile, micSet,
62  preprocessImageRow=self._preprocessMicrographRow,
63  readAcquisition=False)
64  self.protocol._defineOutputs(outputMicrographs=micSet)
65 
66  # Also create a SetOfCTF if the present
67  if self._ctfPath:
68  ctfSet = self.protocol._createSetOfCTF()
69  for mic in micSet:
70  ctf = mic.getCTF()
71  ctf.copyObjId(mic)
72  ctfSet.append(ctf)
73 
74  self.protocol._defineOutputs(outputCTF=ctfSet)
75  self.protocol._defineCtfRelation(micSet, ctfSet)

119  def _findPathAndCtf(self, label, warnings=True):
120  """ Find the relative path from which the micrographs exists
121  repect to the metadata location. Also check if it contains
122  CTF information and their relative root.
123  """
124  row = md.getFirstRow(self._mdFile)
125 
126  if row is None:
127  raise Exception("Can not import from an empty metadata: "
128  "%s" % self._mdFile)
129 
130  if not row.containsLabel(label):
131  raise Exception("Label *%s* is missing in metadata: "
132  "%s" % (md.label2Str(label), self._mdFile))
133 
134  # take only the filename part after the @
135  index, fn = xmippToLocation(row.getValue(label))
136  self._imgPath = findRootFrom(self._mdFile, fn)
137 
138  if warnings and self._imgPath is None:
139  self.protocol.warning("Binary data was not found from metadata: "
140  "%s" % self._mdFile)
141 
142  if row.containsLabel(md.MDL_CTF_MODEL):
143  self._ctfPath = findRootFrom(self._mdFile,
144  row.getValue(md.MDL_CTF_MODEL))
145  else:
146  self._ctfPath = None # means no CTF info from micrographs metadata
147 
148  if row.containsLabel(md.MDL_REF):
149  self._classFunc = self.protocol._createSetOfClasses2D
150  elif row.containsLabel(md.MDL_REF3D):
151  self._classFunc = self.protocol._createSetOfClasses3D
152  else:
153  self._classLabel = None
154  self._classFunc = None
155 
156  # Check if the MetaData contains either MDL_MICROGRAPH_ID
157  # or MDL_MICROGRAPH, this will be used when imported
158  # particles to keep track of the particle's micrograph
159  self._micIdOrName = (row.containsLabel(md.MDL_MICROGRAPH_ID) or
160  row.containsLabel(md.MDL_MICROGRAPH))
161  #init dictionary. It will be used in the preprocessing
162  self.micDict = {}
163 
164  return row
