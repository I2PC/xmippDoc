89  addUsageLine("+$ *CTF margin*: ratio between the average difference in the experimental PSD between the 1st Thon");

99  addUsageLine("+within this region. High correlations indicate good fits.");

155 void ProgPSDSort::processImage(const FileName &fnImg, const FileName &fnImgOut, const MDRow &rowIn, MDRow &rowOut)
156 {
157  CTFDescription CTF1, CTF2;
158  PSDEvaluation evaluation;
159  FileName fnMicrograph, fnPSD, fnCTF, fnCTF2;
160 
161  evaluation.ctf_envelope_ssnr = fn_in.getDir()+"envelope.xmd";
162 
163  int enabled;
164  rowIn.getValue(MDL_ENABLED,enabled);
165  rowIn.getValue(MDL_MICROGRAPH, fnMicrograph);
166  rowIn.getValue(MDL_PSD,fnPSD);
167  if (enabled==-1 || fnPSD == "NA")
168  {
169  rowOut.setValue(MDL_ENABLED,-1);
170  return;
171  }
172 
173  if (rowIn.containsLabel(MDL_CTF_MODEL))
174  {
175  rowIn.getValue(MDL_CTF_MODEL, fnCTF);
176  if (!fnCTF.exists())
177  {
178  rowOut.setValue(MDL_ENABLED,-1);
179  return;
180  }
181  CTF1.read(fnCTF);
182  }
183  else
184  CTF1.readFromMdRow(rowIn);
185 
186  FileName fnRoot = fnMicrograph.withoutExtension();
187 
188  CTF1.produceSideInfo();
189  evaluation.defocusU=CTF1.DeltafU;
190  evaluation.defocusV=CTF1.DeltafV;
191 
192 
193  if (rowIn.containsLabel(MDL_CTF_MODEL2))
194  rowIn.getValue(MDL_CTF_MODEL2,fnCTF2);
195 
196  if (!fnCTF2.empty() && fnCTF2 != "NA")
197  {
198  CTF2.read(fnCTF2);
199  CTF2.produceSideInfo();
200  }
201 
202  // Evaluate beating due to astigmatism
203  /* The argument of the cosine part of the CTF is: wu=cos(K1*deltafu*u.^2+K2*u.^4)
204  * If there is no astigmatism, then in the V direction we would have wv=cos(K1*deltafv*u.^2+K2*u.^4)
205  * and we should have (wu+wv)/2 = wu.
206  * If there is astigmatism, the sum (wu+wv)/2 will depart from the behavior of wu
207  * http://en.wikipedia.org/wiki/Beat_%28acoustics%29
208  * calling argu and argv the argument of the two waves, we have
209  *
210  * (wu+wv)/2=cos((argu+argv)/2)cos((argu-argv)/2)
211  *
212  * The term cos((argu-argv)/2) acts as an envelope which may even vanish. Let's analyze this envelope
213  * cos(0.5*(K1*deltafu*u.^2+K2*u.^4-K1*deltafv*u.^2+K2*u.^4))=cos(0.5*K1*abs(deltafu-deltafv)*u^2)
214  *
215  * When this envelope is 0, the interference between the two waves is totally destructive (we cannot apply a
216  * non-astigmatic correction to an astigmatic CTF). This happens for
217  *
218  * 0.5*K1*abs(deltafu-deltafv)*u_0^2=pi/2 ---> u_0=sqrt(PI/(K1*abs(deltafu-deltafv)))
219  *
220  * This is the expression of critBeating
221  */
222  evaluation.beating=1.0/sqrt(PI/(CTF1.K1*abs(CTF1.DeltafU-CTF1.DeltafV)));
223 
224  // Read input PSD data
225  Image<double> PSD;
226  PSD.read(fnPSD);
227 
228  // Enhance the PSD
229  ProgCTFEnhancePSD enhancePSD;
230  enhancePSD.filter_w1 = filter_w1;
231  enhancePSD.filter_w2 = filter_w2;
232  enhancePSD.decay_width = decay_width;
233  enhancePSD.mask_w1 = mask_w1;
234  enhancePSD.mask_w2 = mask_w2;
235  enhancePSD.applyFilter(PSD());
236 
237  // Evaluate the radial integral
238  PSD().setXmippOrigin();
239  Matrix1D< int > center_of_rot(2);
240  MultidimArray< double > radial_mean;
241  MultidimArray<int> radial_count;
242  radialAverage(PSD(),center_of_rot,radial_mean,radial_count);
243  radial_mean.selfABS();
244  radial_mean/=radial_mean.computeMax();
245  evaluation.PSDradialIntegral=radial_mean.sum();
246 
247  // Rotate 90 degrees and compute correlation
248  Image<double> PSDrotated;
249  rotate(LINEAR,PSDrotated(),PSD(),90);
250  evaluation.PSDcorrelation90=correlationIndex(PSD(), PSDrotated());
251 
252  // Get the fitting score and other quality criteria computed by ctf_estimate_from_micrograph
253  MetaData MDctf1;
254  MDctf1.read(fnCTF);
255  size_t objId1 = MDctf1.firstObject();
256 
257 #define GET_CTF_CRITERION(labelll,xxx) \
258  if (rowIn.containsLabel(labelll)) \
259  rowIn.getValue(labelll,xxx); \
260  else if (MDctf1.containsLabel(labelll)) \
261  MDctf1.getValue(labelll,xxx,objId1); \
262  else \
263  xxx=0;
264  GET_CTF_CRITERION(MDL_CTF_CRIT_FITTINGSCORE,evaluation.fittingScore);
265  GET_CTF_CRITERION(MDL_CTF_CRIT_FITTINGCORR13,evaluation.fittingCorr13);
266  GET_CTF_CRITERION(MDL_CTF_CRIT_PSDVARIANCE,evaluation.PSDVariance);
267  GET_CTF_CRITERION(MDL_CTF_CRIT_PSDPCA1VARIANCE,evaluation.PSDPC1Variance);
268  GET_CTF_CRITERION(MDL_CTF_CRIT_PSDPCARUNSTEST,evaluation.PSDPCRunsTest);
269 
270  // Explore the CTF
271  Matrix1D<double> u(2), freqZero1(2), freqZero2(2), freqMin1(2), pixelZero1(2), pixelMin1(2);
272  double wmax=0.5/CTF1.Tm;
273  double maxModuleZero=0, minModuleZero=1e38;
274  double N=0;
275  evaluation.maxDampingAtBorder=0;
276  evaluation.firstZeroDisagreement=-1;
277  evaluation.firstZeroAvg=0;
278  double firstZeroAvgPSD=0;
279  double firstZeroStddevPSD=0;
280  double firstMinAvgPSD=0;
281  double firstMinStddevPSD=0;
282  double firstZeroMinAvgPSD=0;
283  double firstZeroMinStddevPSD=0;
284  evaluation.maxFreq=1000;
285 
286  CTF1.precomputeValues(0.0,0.0);
287  double idamping0=1.0/CTF1.getValueDampingAt();
288  double f2pixel=CTF1.Tm*XSIZE(PSD()); // COSS *downsampling
289 
290  if (rowIn.containsLabel(MDL_CTF_DOWNSAMPLE_PERFORMED))
291  {
292  double aux;
293  rowIn.getValue(MDL_CTF_DOWNSAMPLE_PERFORMED,aux);
294  f2pixel*=aux;
295  }
296 
297  MetaData mdEnvelope;
298  Matrix1D< double > envelope(100);
299  envelope.initZeros();
300  double Nalpha = 180;
301  for (double alpha=0; alpha<=PI; alpha+=PI/Nalpha, N++)
302  {
303  VECTOR_R2(u,cos(alpha),sin(alpha));
304 
305  // Get the zero in the direction of u
306  CTF1.lookFor(1, u, freqZero1, 0);
307  double moduleZero=1.0/freqZero1.module();
308  maxModuleZero=XMIPP_MAX(maxModuleZero,moduleZero);
309  minModuleZero=XMIPP_MIN(minModuleZero,moduleZero);
310  evaluation.firstZeroAvg+=moduleZero;
311 
312  // Get the first minimum (it is at higher frequency than the zero)
313  CTF1.lookFor(1, u, freqMin1, -1);
314  pixelZero1=freqZero1*f2pixel;
315  pixelMin1=freqMin1*f2pixel;
316  double psdZero=PSD().interpolatedElement2D(XX(pixelZero1),YY(pixelZero1),0.0);
317  double psdMin=PSD().interpolatedElement2D(XX(pixelMin1),YY(pixelMin1),0.0);
318  firstMinAvgPSD+=psdMin;
319  firstMinStddevPSD+=psdMin*psdMin;
320  firstZeroAvgPSD+=psdZero;
321  firstZeroStddevPSD+=psdZero*psdZero;
322  double zeroMinDiff=psdMin-psdZero;
323  firstZeroMinAvgPSD+=zeroMinDiff;
324  firstZeroMinStddevPSD+=zeroMinDiff*zeroMinDiff;
325 
326  // Evaluate damping
327  double wx=wmax*XX(u);
328  double wy=wmax*YY(u);
329  CTF1.precomputeValues(wx,wy);
330  double damping=CTF1.getValueDampingAt();
331  damping=damping*damping;
332  evaluation.maxDampingAtBorder=XMIPP_MAX(evaluation.maxDampingAtBorder,damping);
333 
334  int idx = 0;
335  for (double w=0; w<wmax; w+=wmax/99.0)
336  {
337  wx=w*XX(u);
338  wy=w*YY(u);
339  CTF1.precomputeValues(wx,wy);
340  double normalizedDamping=fabs(CTF1.getValueDampingAt()*idamping0);
341  if (normalizedDamping>0.1)
342  evaluation.maxFreq=std::min(evaluation.maxFreq,1.0/w);
343 
344  VEC_ELEM(envelope,idx) += double(fabs(CTF1.getValueDampingAt()));
345  idx++;
346  }
347 
348  if (fnCTF2!="") {
349  CTF2.lookFor(1, u, freqZero2, 0);
350  double module2=1.0/freqZero2.module();
351  double diff=ABS(moduleZero-module2);
352  evaluation.firstZeroDisagreement=XMIPP_MAX(evaluation.firstZeroDisagreement,diff);
353  }
354  }
355 
356  size_t objId2 = mdEnvelope.firstObject();
357  int idx=0;
358  for (double w=0; w<wmax; w+=wmax/99.0)
359  {
360  mdEnvelope.setValue(MDL_RESOLUTION_FREQ,w,objId2);
361  mdEnvelope.setValue(MDL_CTF_ENVELOPE,VEC_ELEM(envelope,idx)/Nalpha,objId2);
362  objId2 = mdEnvelope.addObject();
363  idx++;
364  }
365  evaluation.firstZeroAvg/=N;
366  evaluation.firstZeroRatio=maxModuleZero/minModuleZero;
367  firstZeroAvgPSD/=N;
368  firstZeroStddevPSD=sqrt(fabs(firstZeroStddevPSD/N-firstZeroAvgPSD*firstZeroAvgPSD));
369  firstMinAvgPSD/=N;
370  firstMinStddevPSD=sqrt(fabs(firstMinStddevPSD/N-firstMinAvgPSD*firstMinAvgPSD));
371  firstZeroMinAvgPSD/=N;
372  firstZeroMinStddevPSD=sqrt(fabs(firstZeroMinStddevPSD/N-firstZeroMinAvgPSD*firstZeroMinAvgPSD));
373 
374  evaluation.firstMinimumStddev_ZeroStddev=1000;
375  evaluation.firstMinimumDiffStddev_ZeroStddev=1000;
376  if (firstZeroStddevPSD>1e-6)
377  {
378  evaluation.firstMinimumStddev_ZeroStddev=firstMinStddevPSD/firstZeroStddevPSD;
379  evaluation.firstMinimumDiffStddev_ZeroStddev=firstZeroMinAvgPSD/firstZeroStddevPSD;
380  }
381 
382  // Evaluate micrograph normality
383  ImageGeneric M;
384  M.readMapped(fnMicrograph);
385  double avg, stddev, minval, maxval;
386  M().computeStats(avg, stddev, minval, maxval);
387  Histogram1D hist;
388  compute_hist(M(), hist, minval, maxval, 400);
389  hist += 1;
390  hist /= hist.sum();
391 
392  Histogram1D histGaussian;
393  histGaussian.initZeros(hist);
394  histGaussian.hmin=hist.hmin;
395  histGaussian.hmax=hist.hmax;
396  histGaussian.step_size=hist.step_size;
397  histGaussian.istep_size=hist.istep_size;
398  FOR_ALL_ELEMENTS_IN_ARRAY1D(histGaussian) {
399  double x;
400  hist.index2val(i, x);
401  A1D_ELEM(histGaussian,i) = gaussian1D(x, stddev, avg);
402  }
403  evaluation.histogramNormality=0.5*(KLDistance(hist,histGaussian)+
404  KLDistance(histGaussian,hist));
405 
406  // Write criteria
407  rowOut.setValue(MDL_CTF_DEFOCUSU,evaluation.defocusU);
408  rowOut.setValue(MDL_CTF_DEFOCUSV,evaluation.defocusV);
409  rowOut.setValue(MDL_CTF_CRIT_FIRSTZEROAVG,evaluation.firstZeroAvg);
410  rowOut.setValue(MDL_CTF_CRIT_MAXFREQ,evaluation.maxFreq);
411  rowOut.setValue(MDL_CTF_CRIT_DAMPING,evaluation.maxDampingAtBorder);
412  if (evaluation.firstZeroDisagreement>0)
413  rowOut.setValue(MDL_CTF_CRIT_FIRSTZERODISAGREEMENT,evaluation.firstZeroDisagreement);
414  rowOut.setValue(MDL_CTF_CRIT_FIRSTZERORATIO,evaluation.firstZeroRatio);
415  rowOut.setValue(MDL_CTF_ENVELOPE_PLOT,evaluation.ctf_envelope_ssnr);
416  rowOut.setValue(MDL_CTF_CRIT_FIRSTMINIMUM_FIRSTZERO_RATIO,evaluation.firstMinimumStddev_ZeroStddev);
417  rowOut.setValue(MDL_CTF_CRIT_FIRSTMINIMUM_FIRSTZERO_DIFF_RATIO,evaluation.firstMinimumDiffStddev_ZeroStddev);
418  rowOut.setValue(MDL_CTF_CRIT_FITTINGSCORE,evaluation.fittingScore);
419  rowOut.setValue(MDL_CTF_CRIT_FITTINGCORR13,evaluation.fittingCorr13);
420  rowOut.setValue(MDL_CTF_CRIT_NONASTIGMATICVALIDITY,evaluation.beating);
421  rowOut.setValue(MDL_CTF_CRIT_PSDCORRELATION90,evaluation.PSDcorrelation90);
422  rowOut.setValue(MDL_CTF_CRIT_PSDRADIALINTEGRAL,evaluation.PSDradialIntegral);
423  rowOut.setValue(MDL_CTF_CRIT_PSDVARIANCE,evaluation.PSDVariance);
424  rowOut.setValue(MDL_CTF_CRIT_PSDPCA1VARIANCE,evaluation.PSDPC1Variance);
425  rowOut.setValue(MDL_CTF_CRIT_PSDPCARUNSTEST,evaluation.PSDPCRunsTest);
426  rowOut.setValue(MDL_CTF_CRIT_NORMALITY, evaluation.histogramNormality);
