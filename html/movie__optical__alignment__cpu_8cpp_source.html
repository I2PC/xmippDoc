258  void evaluateDisplacements(const cv::Mat *flowCurrentGroup, const cv::Mat *flowPreviousGroup, Matrix1D<double> &meanStdDev)
259  {
260  double sumX=0, sumY=0;
261  double absSumX=0, absSumY=0;
262  double sqSumX=0, sqSumY=0;
263  double valSubtract;
264  int h=flowCurrentGroup[0].rows;
265  int w=flowCurrentGroup[0].cols;
266  for(int i=0;i<h;i++)
267  for(int j=0;j<w;j++)
268  {
269  valSubtract=flowCurrentGroup[0].at<float>(i,j)-flowPreviousGroup[0].at<float>(i,j);
270  sumX+=valSubtract;
271  absSumX+=abs(valSubtract);
272  sqSumX+=valSubtract*valSubtract;
273  valSubtract=flowCurrentGroup[1].at<float>(i,j)-flowPreviousGroup[1].at<float>(i,j);
274  sumY+=valSubtract;
275  absSumY+=abs(valSubtract);
276  sqSumY+=valSubtract*valSubtract;
277  }
278  double n=h*w;
279  double avgX=sumX/n;
280  double avgY=sumY/n;
281  meanStdDev(0)=absSumX/n;
282  meanStdDev(1)=sqrt(std::max(sqSumX/n-avgX*avgX,0.0));
283  meanStdDev(2)=absSumY/n;
284  meanStdDev(3)=sqrt(std::max(sqSumY/n-avgY*avgY,0.0));
285  }

306  // If input is an stack create a metadata.

313  else
314  {
315  ImageGeneric movieStack;
316  movieStack.read(fnMovie,HEADER);
317  movieStack.getDimensions(Xdim,Ydim,Zdim,Ndim);
318  if (fnMovie.getExtension()=="mrc" and Ndim ==1)
319  Ndim = Zdim;

321  FileName fn;

323  {
324  id = movie.addObject();
325  fn.compose(i+FIRST_IMAGE,fnMovie);
326  movie.setValue(MDL_IMAGE, fn, id);
327  }
328  }
329 
330  // Get corners
331  if (yDRcorner!=-1)
332  {
333  Xdim = xDRcorner - xLTcorner +1 ;
334  Ydim = yDRcorner - yLTcorner +1 ;
335  }
336  if (Zdim!=1)
337  REPORT_ERROR(ERR_ARG_INCORRECT,"This program is meant to align 2D frames, not 3D");
338 
339  // Dark and gain images
340  Image<double> dark, igain;
341  if (fnDark!="")
342  {
343  dark.read(fnDark);
344  applyWindow(dark());
345  }
346 
347  if (fnGain!="")
348  {
349  igain.read(fnGain);
350  applyWindow(igain());
351  // Pablo with coss: Consistency with other alignment methods
352  // Do not divide by the gain, as the gain is expected to be inverted already
353  double avg = igain().computeAvg();
354  if (std::isinf(avg) || std::isnan(avg))
355  REPORT_ERROR(ERR_ARG_INCORRECT,
356  "The input gain image is incorrect, it contains infinite or nan");
357  }
358 
359  // Count the number of frames to process
360  numberOfFrames=Ndim;
361  if (nfirst<0)
362  nfirst=0;
363  if (nlast<0)
364  nlast=numberOfFrames-1;
365  numberOfFrames=nlast-nfirst+1;
366 
367  // Initialize the stack for the output movie
368  if (!fnMovieOut.isEmpty())
369  createEmptyFile(fnMovieOut, Xdim, Ydim, 1, numberOfFrames, true, WRITE_REPLACE);
370  if (!fnMovieUncOut.isEmpty())
371  createEmptyFile(fnMovieUncOut, Xdim, Ydim, 1, numberOfFrames, true, WRITE_REPLACE);
372 
373  // Prepare stack
374  if (inMemory)
375  tempStack.resizeNoCopy(numberOfFrames,1,Ydim,Xdim);
376  else
377  {
378  fnTempStack=fnmovieRoot+"tmpMovie.stk";
379  createEmptyFile(fnTempStack,Xdim,Ydim,1,numberOfFrames,true);
380  }
381 
382  int currentFrameInIdx=0,currentFrameOutIdx=0;
383  FileName fnFrame;
384  Image<double> frameImage, translatedImage;
385  MultidimArray<float> Ifloat;
386  Matrix1D<double> shift(2);
387  filterDose=new ProgMovieFilterDose(accelerationVoltage);
388  filterDose->pixel_size=sampling;
389  MultidimArray< std::complex<double> > FFTI;
390  FOR_ALL_OBJECTS_IN_METADATA(movie)
391  {
392  if (currentFrameInIdx>=nfirst && currentFrameInIdx<=nlast)
393  {
394  movie.getValue(MDL_IMAGE, fnFrame, __iter.objId);
395  frameImage.read(fnFrame);
396  applyWindow(frameImage());
397 
398  if (XSIZE(dark())>0)
399  frameImage()-=dark();
400  if (XSIZE(igain())>0)
401  frameImage()*=igain();
402 
403  if (movie.containsLabel(MDL_SHIFT_X))
404  {
405  movie.getValue(MDL_SHIFT_X, XX(shift), __iter.objId);
406  movie.getValue(MDL_SHIFT_Y, YY(shift), __iter.objId);
407  if (fabs(XX(shift))>0 || fabs(YY(shift))>0)
408  {
409 // std::cout << "Translating " << fnFrame << " by " << shift.transpose() << std::endl;
410  translate(BSPLINE3, translatedImage(), frameImage(), shift, WRAP);
411  frameImage()=translatedImage();
412  }
413  }
414 
415  if (fnMovieUncOut!="")
416  frameImage.write(fnMovieUncOut, currentFrameOutIdx+1, true, WRITE_REPLACE);
417 
418  if (compensationMode==PRECOMPENSATION)
419  {
420  transformer.FourierTransform(frameImage(), FFTI, false);
421  filterDose->applyDoseFilterToImage(YSIZE(frameImage()), XSIZE(frameImage()), FFTI,
422  dose0+currentFrameInIdx*doseStep, dose0+(currentFrameInIdx+1)*doseStep);
423  transformer.inverseFourierTransform();
424  }
425 // std::cout << "Reading " << fnFrame << " "; frameImage().printStats(); std::cout << std::endl;
426  if (inMemory)
427  {
428  typeCast(frameImage(),Ifloat);
429  memcpy(&DIRECT_NZYX_ELEM(tempStack,currentFrameOutIdx,0,0,0),&Ifloat(0,0),MULTIDIM_SIZE(Ifloat)*sizeof(float));
430  }
431  else
432  {
433  frameImage.write(fnTempStack, currentFrameOutIdx+1, true, WRITE_REPLACE);
434 // std::cout << "Writing " << currentFrameOutIdx+1 << "@" << fnTempStack << std::endl;
435  }
436  currentFrameOutIdx++;
437  }
438  currentFrameInIdx++;
439  }
