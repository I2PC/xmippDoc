444 void ProgMonoTomo::localNoise(MultidimArray<double> &noiseMap, Matrix2D<double> &noiseMatrix, int boxsize, Matrix2D<double> &thresholdMatrix)
445 {
446 // std::cout << "Analyzing local noise" << std::endl;
447 
448  int xdim = XSIZE(noiseMap);
449  int ydim = YSIZE(noiseMap);
450 
451  int Nx = xdim/boxsize;
452  int Ny = ydim/boxsize;
453 
454  noiseMatrix.initZeros(Ny, Nx);
455 
456  // For the spline regression
457  int lX=std::min(8,Nx-2), lY=std::min(8,Ny-2);
458  WeightedLeastSquaresHelper helper;
459  helper.A.initZeros(Nx*Ny,lX*lY);
460  helper.b.initZeros(Nx*Ny);
461  helper.w.initZeros(Nx*Ny);
462  helper.w.initConstant(1);
463  double hX = xdim / (double)(lX-3);
464  double hY = ydim / (double)(lY-3);
465 
466  if ( (xdim<boxsize) || (ydim<boxsize) )
467  std::cout << "Error: The tomogram in x-direction or y-direction is too small" << std::endl;
468 
469  std::vector<double> noiseVector(1);
470  std::vector<double> x,y,t;
471 
472  int xLimit, yLimit, xStart, yStart;
473 
474  long counter;
475  int idxBox=0;
476 
477  for (int X_boxIdx=0; X_boxIdx<Nx; ++X_boxIdx)
478  {
479  if (X_boxIdx==Nx-1)
480  {
481  xStart = STARTINGX(noiseMap) + X_boxIdx*boxsize;
482  xLimit = FINISHINGX(noiseMap);
483  }
484  else
485  {
486  xStart = STARTINGX(noiseMap) + X_boxIdx*boxsize;
487  xLimit = STARTINGX(noiseMap) + (X_boxIdx+1)*boxsize;
488  }
489 
490  for (int Y_boxIdx=0; Y_boxIdx<Ny; ++Y_boxIdx)
491  {
492  if (Y_boxIdx==Ny-1)
493  {
494  yStart = STARTINGY(noiseMap) + Y_boxIdx*boxsize;
495  yLimit = FINISHINGY(noiseMap);
496  }
497  else
498  {
499  yStart = STARTINGY(noiseMap) + Y_boxIdx*boxsize;
500  yLimit = STARTINGY(noiseMap) + (Y_boxIdx+1)*boxsize;
501  }
502 
503 
504 
505  counter = 0;
506  for (int i = (yStart); i<(yLimit); i++)
507  {
508  for (int j = (xStart); j<(xLimit); j++)
509  {
510  for (int k = STARTINGZ(noiseMap); k<FINISHINGZ(noiseMap); k++)
511  {
512  if (counter%257 == 0) //we take one voxel each 257 (prime number) points to reduce noise data
513  noiseVector.push_back( A3D_ELEM(noiseMap, k, i, j) );
514  ++counter;
515  }
516  }
517  }
518 
519  std::sort(noiseVector.begin(),noiseVector.end());
520  MAT_ELEM(noiseMatrix, Y_boxIdx, X_boxIdx) = noiseVector[size_t(noiseVector.size()*significance)];
521 
522  double tileCenterY=0.5*(yLimit+yStart)-STARTINGY(noiseMap); // Translated to physical coordinates
523  double tileCenterX=0.5*(xLimit+xStart)-STARTINGX(noiseMap);
524  // Construction of the spline equation system
525  long idxSpline=0;
526  for(int controlIdxY = -1; controlIdxY < (lY - 1); ++controlIdxY)
527  {
528  double tmpY = Bspline03((tileCenterY / hY) - controlIdxY);
529  VEC_ELEM(helper.b,idxBox)=MAT_ELEM(noiseMatrix, Y_boxIdx, X_boxIdx);
530  if (tmpY == 0.0)
531  {
532  idxSpline+=lX;
533  continue;
534  }
535 
536  for(int controlIdxX = -1; controlIdxX < (lX - 1); ++controlIdxX)
537  {
538  double tmpX = Bspline03((tileCenterX / hX) - controlIdxX);
539  MAT_ELEM(helper.A,idxBox,idxSpline) = tmpY * tmpX;
540  idxSpline+=1;
541  }
542  }
543  x.push_back(tileCenterX);
544  y.push_back(tileCenterY);
545  t.push_back(MAT_ELEM(noiseMatrix, Y_boxIdx, X_boxIdx));
546  noiseVector.clear();
547  idxBox+=1;
548  }
549  }
550 
551 
552  // Spline coefficients
553  Matrix1D<double> cij;
554  weightedLeastSquares(helper, cij);
555 
556  thresholdMatrix.initZeros(ydim, xdim);
557 
558  for (int i=0; i<ydim; ++i)
559  {
560  for (int j=0; j<xdim; ++j)
561  {
562  long idxSpline=0;
563 
564  for(int controlIdxY = -1; controlIdxY < (lY - 1); ++controlIdxY)
565  {
566  double tmpY = Bspline03((i / hY) - controlIdxY);
567 
568  if (tmpY == 0.0)
569  {
570  idxSpline+=lX;
571  continue;
572  }
573 
574  double xContrib=0.0;
575  for(int controlIdxX = -1; controlIdxX < (lX - 1); ++controlIdxX)
576  {
577  double tmpX = Bspline03((j / hX) - controlIdxX);
578  xContrib+=VEC_ELEM(cij,idxSpline) * tmpX;// *tmpY;
579  idxSpline+=1;
580  }
581  MAT_ELEM(thresholdMatrix,i,j)+=xContrib*tmpY;
582  }
583  }
584  }
