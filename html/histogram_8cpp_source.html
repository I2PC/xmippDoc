155 /* Percentil --------------------------------------------------------------- */
156 /* This function returns the value of the variable under which the mass% of
157  the histogram is comprised */
158 double Histogram1D::percentil(double percent_mass)
159 {
160  int i = 0;
161  double acc = 0;
162  double required_mass;
163  double percentil_i;
164  double ret_val;
165 
166  // Check it is a correct mass
167  if (percent_mass > 100)
168  REPORT_ERROR(ERR_VALUE_INCORRECT, "Asked for a percentil greater than 100");
169 
170  // Trivial cases
171  if (percent_mass == 0)
172  return (hmin);
173  if (percent_mass == 100)
174  return (hmax);
175 
176  // Any other case, find index of corresponding piece
177  required_mass = (double) no_samples * percent_mass / 100.0;
178  int N_diff_from_0 = 0;
179  while (acc < required_mass)
180  {
181  acc += A1D_ELEM(*this, i);
182  if (A1D_ELEM(*this, i) > 0)
183  N_diff_from_0++;
184  i++;
185  }
186 
187  // If the sum is just the one we want OK
188  if (acc == required_mass)
189  percentil_i = i;
190  // If there is only one sample different from 0
191  // then there is no way of setting the threshold in the middle
192  // Let's put it at the beginning of the bin
193  else if (N_diff_from_0 == 1)
194  percentil_i = i - 1;
195  // If not, then go back a step and compute which fraction of the
196  // bar is needed to finish the required mass
197  else
198  {
199  /* CO: We cannot assure that there is at least what is supposed to be
200  above this threshold. Let's move to the safe side
201  i--;
202  acc -= A1D_ELEM(*this,i);
203  percentil_i=i+(required_mass-acc)/(double) A1D_ELEM(*this,i); */
204  percentil_i = i - 1;
205  }
206 
207  // Now translate from index to range
208  index2val(percentil_i, ret_val);
209  return ret_val;
210 }

235 /* Entropy ----------------------------------------------------------------- */
236 double Histogram1D::entropy() const
237 {
238  MultidimArray<double> p;
239  p.initZeros(XSIZE(*this));
240  double pSum = 0;
241  FOR_ALL_ELEMENTS_IN_ARRAY1D(p)
242  {
243  A1D_ELEM(p,i) = A1D_ELEM(*this,i) + 1;
244  pSum += A1D_ELEM(p,i);
245  }
246  double entropy = 0;
247  double ipSum = 1.0 / pSum;
248  FOR_ALL_ELEMENTS_IN_ARRAY1D(p)
249  {
250  double pi = A1D_ELEM(p,i) * ipSum;
251  entropy -= pi * log(pi);
252  }
253  return entropy;

317 /* Detectability error ----------------------------------------------------- */
318 // Given two histograms (probability density functions) this function returns
319 // the detection error for using both, ie, the probability that given a sample
320 // I would assign it to class 1 when it belongs to class 2 and viceversa.
321 // This is computed by the calculation of the overlapping areas.
322 double detectability_error(const Histogram1D &h1, const Histogram1D &h2)
323 {
324  double hmin, hmax;
325  double step;
326  double v;
327  double error = 0;
328  int ih1, ih2; // Indexes within the histograms
329  double p1, p2; // Probability associated
330 
331  // Find global range
332  hmin = XMIPP_MAX(h1.hmin, h2.hmin);
333  hmax = XMIPP_MIN(h1.hmax, h2.hmax);
334  step = XMIPP_MIN(h1.step_size, h2.step_size) / 2;
335 
336  // Go over the range computing the errors
337  v = hmin;
338  int N = 0;
339  while (v <= hmax)
340  {
341  h1.val2index(v, ih1);
342  p1 = A1D_ELEM(h1, ih1) / h1.no_samples;
343  h2.val2index(v, ih2);
344  p2 = A1D_ELEM(h2, ih2) / h2.no_samples;
345  //#define DEBUG
346 #ifdef DEBUG
347 
348  std::cout << "Comparing at " << v << " (" << ih1 << ") p1=" << p1 << " p2= " << p2 << std::endl;
349  std::cout << " hmin " << hmin << " hmax " << hmax << " stepsize " << h1.step_size << std::endl;
350 #endif//;
351 
352  if (p1 != 0 && p2 != 0)
353  {
354  if (p1 > p2)
355  error += p2;
356  else
357  error += p1;
358  }
359  v += step;
360  N++;
361  }
362 
363  // Normalise such that the result is the area of a probability function
364  error *= step / (hmax - hmin);
365  error /= N;
366 #ifdef DEBUG
367 
368  std::cout << "Total error = " << error << std::endl;
369 #endif
370 
371  return error;
372 }
