62 void ProgTomoExtractParticleStacks::run()
63 {
64  std::cout << "Starting ... "<< std::endl;
65 
66  MetaDataVec mdts, mdcoords, mdparticlestack;
67  mdts.read(fnTs);
68  mdcoords.read(fnCoor);
69 
70  int xcoor, ycoor, zcoor, xinit, yinit, zinit;
71  size_t idx=1;
72 
73  FileName fnImg;
74 
75  Image<double> tiltImg;

83

85

87 
88  for (const auto& row : mdts)
89  {
90  row.getValue(MDL_IMAGE, fnImg);
91  row.getValue(MDL_ANGLE_TILT, tilt);
92  row.getValue(MDL_ANGLE_ROT, rot);
93 
94  tiltImg.read(fnImg);
95 
96  tsImages.push_back(ptrtiltImg);
97 
98  tsTiltAngles.push_back(tilt);
99  tsRotAngles.push_back(rot);
100  tsNames.push_back(fnImg);
101  Nimages +=1;
102  }
103 
104  size_t Xts, Yts;
105  Xts = XSIZE(ptrtiltImg);
106  Yts = YSIZE(ptrtiltImg);
107 
108 
109  int halfboxsize = floor(0.5*boxsize);
110  Image<double> finalStack;
111  auto &particlestack = finalStack();
112 
113  std::string tsid;
114  size_t elem = 0;
115 
116  for (const auto& row : mdcoords)
117  {
118  row.getValue(MDL_TSID, tsid);
119  row.getValue(MDL_XCOOR, xcoor);
120  row.getValue(MDL_YCOOR, ycoor);
121  row.getValue(MDL_ZCOOR, zcoor);
122 
123  FileName fn;
124 
125  fn = fnCoor.getBaseName() + tsid + formatString("-%i.mrc", elem);
126 
127  particlestack.initZeros(Nimages, 1, boxsize, boxsize);
128 
129  for (size_t idx = 0; idx<Nimages; idx++)
130  {
131  tsImg = tsImages[idx];
132  tilt = tsTiltAngles[idx]*PI/180;
133  rot = tsRotAngles[idx];
134 
135  std::cout << tsNames[idx] << std::endl;
136 
137  double ct = cos(tilt);
138  double st = sin(tilt);
139 
140  int x_2d, y_2d;
141 
142  x_2d = (int) (xcoor * ct + zcoor* st) + 0.5*Xts;
143  y_2d = (int) (ycoor+ 0.5*Yts);
144 
145  int xlim = x_2d + halfboxsize;
146  int ylim = y_2d + halfboxsize;
147 
148  xinit = x_2d - halfboxsize;
149  yinit = y_2d - halfboxsize;
150 
151 
152  if ((xlim>Xts) || (ylim>Yts) || (xinit<0) || (yinit<0))
153  {
154  std::cout << "skipping " << std::endl;
155  continue;
156  }
157 
158 
159 
160  if (invertContrast)
161  {
162  for (int i=yinit; i<ylim; i++)
163  {
164  int ii = i-y_2d;
165  for (int j=xinit; j<xlim; j++)
166  {
167  DIRECT_N_YX_ELEM(particlestack, idx, ii+halfboxsize, j+halfboxsize-x_2d) = -A2D_ELEM(tsImages[idx], i, j);
168  }
169  }
170  }
171  else
172  {
173  for (int i=yinit; i<ylim; i++)
174  {
175  int ii = i-y_2d;
176  for (int j=xinit; j<xlim; j++)
177  {
178  DIRECT_N_YX_ELEM(particlestack, idx, ii+halfboxsize, j+halfboxsize-x_2d) = A2D_ELEM(tsImages[idx], i, j);
179  }
180  }
181  }
182 
183 
184 
185  MDRowVec rowParticleStack;
186  rowParticleStack.setValue(MDL_TSID, tsid);
187  rowParticleStack.setValue(MDL_IMAGE, fn);
188  rowParticleStack.setValue(MDL_ANGLE_TILT, tilt);
189  rowParticleStack.setValue(MDL_ANGLE_ROT, rot);
190  rowParticleStack.setValue(MDL_XCOOR, x_2d);
191  rowParticleStack.setValue(MDL_YCOOR, y_2d);
192 
193  mdparticlestack.addRow(rowParticleStack);
194  }
195 
196 
197 
198  finalStack.write(fnOut+"/"+fn);
199 
200  elem += 1;
201  }
