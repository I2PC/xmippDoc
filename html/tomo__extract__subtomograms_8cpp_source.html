66 void ProgTomoExtractSubtomograms::run()
67 {
68  std::cout << "Starting ... "<< std::endl;
69 
70  MetaDataVec md;
71  md.read(fnCoor);
72 
73  Image<double> tomImg;
74  auto &tom = tomImg();
75  tomImg.read(fnTom);
76 
77  size_t Xtom, Ytom, Ztom;
78  Xtom = XSIZE(tom);
79  Ytom = YSIZE(tom);
80  Ztom = ZSIZE(tom);
81 
82  Image<double> subtomoImg;
83  auto &subtomo = subtomoImg();
84 
85  MultidimArray<double> subtomoExtraction;
86 
87  int xcoor;
88  int ycoor;
89  int zcoor;
90  int xinit;
91  int yinit;
92  int zinit;
93 
94  size_t idx=1;
95 
96  size_t boxSizeExtraction;
97 
98  if (downsample)
99  {
100  boxSizeExtraction = boxsize * downsampleFactor;
101  }
102  else
103  {
104  boxSizeExtraction = boxsize;
105  }
106 
107  int halfboxsize = floor(0.5*boxsize);
108  int halfBoxSizeExtraction = floor(0.5*boxSizeExtraction);
109 
110  MultidimArray<double> maskNormalize;
111  if (normalize)
112  {
113  maskNormalize.initZeros(1, boxsize, boxsize, boxsize);
114 
115  for (int k=0; k<boxsize; k++)
116  {
117  int k2 = (k-halfboxsize);
118  k2 = k2*k2;
119  for (int i=0; i<boxsize; i++)
120  {
121  int i2 = i-halfboxsize;
122  int i2k2 = i2*i2 +k2 ;
123  for (int j=0; j<boxsize; j++)
124  {
125  int j2 = (j- halfboxsize);
126  if (sqrt(i2k2 + j2*j2)>halfboxsize)
127  A3D_ELEM(maskNormalize, k, i, j) = 1;
128  }
129  }
130  }
131  }
132 
133  FourierTransformer transformer1;
134  FourierTransformer transformer2;
135  for (const auto& row : md)
136  {
137  row.getValue(MDL_XCOOR, xcoor);
138  row.getValue(MDL_YCOOR, ycoor);
139  row.getValue(MDL_ZCOOR, zcoor);
140 
141  int xlim = xcoor+halfBoxSizeExtraction;
142  int ylim = ycoor+halfBoxSizeExtraction;
143  int zlim = zcoor+halfBoxSizeExtraction;
144 
145  xinit = xcoor - halfBoxSizeExtraction;
146  yinit = ycoor - halfBoxSizeExtraction;
147  zinit = zcoor - halfBoxSizeExtraction;
148 
149  if ((xlim>Xtom) || (ylim>Ytom) || (zlim>Ztom) || (xinit<0) || (yinit<0) || (zinit<0))
150  continue;
151 
152 
153  subtomoExtraction.initZeros(1, boxSizeExtraction, boxSizeExtraction, boxSizeExtraction);
154 
155  if (invertContrast)
156  {
157  for (int k=zinit; k<zlim; k++)
158  {
159  int kk = k - zcoor;
160  for (int i=yinit; i<ylim; i++)
161  {
162  int ii = i-ycoor;
163  for (int j=xinit; j<xlim; j++)
164  {
165  A3D_ELEM(subtomoExtraction, kk+halfBoxSizeExtraction, ii+halfBoxSizeExtraction, j+halfBoxSizeExtraction-xcoor) = -A3D_ELEM(tom, k, i, j);
166  }
167  }
168  }
169  }
170  else
171  {
172  for (int k=zinit; k<zlim; k++)
173  {
174  int kk = k - zcoor;
175  for (int i=yinit; i<ylim; i++)
176  {
177  int ii = i-ycoor;
178  for (int j=xinit; j<xlim; j++)
179  {
180  A3D_ELEM(subtomoExtraction, kk+halfBoxSizeExtraction, ii+halfBoxSizeExtraction, j+halfBoxSizeExtraction-xcoor) = A3D_ELEM(tom, k, i, j);
181  }
182  }
183  }
184  }
185 
186  if (downsample)
187  {
188 
189  MultidimArray< std::complex<double> > fftSubtomoExtraction;
190  MultidimArray< std::complex<double> > fftSubtomo;
191 
192  fftSubtomo.initZeros(boxsize, boxsize, (boxsize/2)+1);
193  transformer1.FourierTransform(subtomoExtraction, fftSubtomoExtraction, true);
194 
195  #ifdef DEBUG
196  std::cout << "XSIZE(subtomoExtraction) " << XSIZE(subtomoExtraction) << std::endl;
197  std::cout << "YSIZE(subtomoExtraction) " << YSIZE(subtomoExtraction) << std::endl;
198  std::cout << "ZSIZE(subtomoExtraction) " << ZSIZE(subtomoExtraction) << std::endl;
199 
200  std::cout << "XSIZE(fftSubtomo) " << XSIZE(fftSubtomo) << std::endl;
201  std::cout << "YSIZE(fftSubtomo) " << YSIZE(fftSubtomo) << std::endl;
202  std::cout << "ZSIZE(fftSubtomo) " << ZSIZE(fftSubtomo) << std::endl;
203 
204  std::cout << "XSIZE(fftSubtomoExtraction) " << XSIZE(fftSubtomoExtraction) << std::endl;
205  std::cout << "YSIZE(fftSubtomoExtraction) " << YSIZE(fftSubtomoExtraction) << std::endl;
206  std::cout << "ZSIZE(fftSubtomoExtraction) " << ZSIZE(fftSubtomoExtraction) << std::endl;
207  #endif
208 
209  if (downsampleFactor > 1)
210  {
211  for (size_t k = 0; k < ZSIZE(fftSubtomo)/2; k++)
212  {
213  for (size_t j = 0; j < YSIZE(fftSubtomo)/2; j++)
214  {
215  for (size_t i = 0; i < XSIZE(fftSubtomo); i++)
216  {
217  // Origin cube
218  DIRECT_A3D_ELEM(fftSubtomo, k, j, i) = DIRECT_A3D_ELEM(fftSubtomoExtraction, k, j, i);
219 
220  // Oposite cube
221  DIRECT_A3D_ELEM(fftSubtomo, ZSIZE(fftSubtomo)/2 + k, YSIZE(fftSubtomo)/2 + j, i) =
222  DIRECT_A3D_ELEM(fftSubtomoExtraction, ZSIZE(fftSubtomoExtraction) - ZSIZE(fftSubtomo)/2 + k, YSIZE(fftSubtomoExtraction) - YSIZE(fftSubtomo)/2 + j, i);
223 
224  // Offset-Z cube
225  DIRECT_A3D_ELEM(fftSubtomo, ZSIZE(fftSubtomo)/2 + k, j, i) =
226  DIRECT_A3D_ELEM(fftSubtomoExtraction, ZSIZE(fftSubtomoExtraction) - ZSIZE(fftSubtomo)/2 + k, j, i);
227 
228  // Offset-Y cube
229  DIRECT_A3D_ELEM(fftSubtomo, k, YSIZE(fftSubtomo)/2 + j, i) =
230  DIRECT_A3D_ELEM(fftSubtomoExtraction, k, YSIZE(fftSubtomoExtraction) - YSIZE(fftSubtomo)/2 + j, i);
231  }
232  }
233  }
234  }
235  else // downsampleFactor < 1
236  {
237  for (size_t k = 0; k < ZSIZE(fftSubtomoExtraction)/2; k++)
238  {
239  for (size_t j = 0; j < YSIZE(fftSubtomoExtraction)/2; j++)
240  {
241  for (size_t i = 0; i < XSIZE(fftSubtomoExtraction); i++)
242  {
243  // Origin cube
244  DIRECT_A3D_ELEM(fftSubtomo, k, j, i) = DIRECT_A3D_ELEM(fftSubtomoExtraction, k, j, i);
245 
246  // Oposite cube
247  DIRECT_A3D_ELEM(fftSubtomo, ZSIZE(fftSubtomo) - ZSIZE(fftSubtomoExtraction)/2 + k, YSIZE(fftSubtomo) - YSIZE(fftSubtomoExtraction)/2 + j, i) =
248  DIRECT_A3D_ELEM(fftSubtomoExtraction, ZSIZE(fftSubtomoExtraction)/2 + k, YSIZE(fftSubtomoExtraction)/2 + j, i);
249 
250  // Offset-Z cube
251  DIRECT_A3D_ELEM(fftSubtomo, ZSIZE(fftSubtomo) - ZSIZE(fftSubtomoExtraction)/2 + k, j, i) =
252  DIRECT_A3D_ELEM(fftSubtomoExtraction, ZSIZE(fftSubtomoExtraction)/2 + k, j, i);
253 
254  // Offset-Y cube
255  DIRECT_A3D_ELEM(fftSubtomo, k, YSIZE(fftSubtomo) - YSIZE(fftSubtomoExtraction)/2 + j, i) =
256  DIRECT_A3D_ELEM(fftSubtomoExtraction, k, YSIZE(fftSubtomoExtraction)/2 + j, i);
257  }
258  }
259  }
260  }
261  subtomo.initZeros(boxsize, boxsize, boxsize);
262  transformer2.inverseFourierTransform(fftSubtomo, subtomo);
263  }
264  else
265  {
266  subtomo = subtomoExtraction; // *** reuse subtomoextraction (remove subtomo)
267  }
268 
269 
270  if (normalize)
271  {
272  double sumVal = 0, sumVal2 = 0;
273  double counter = 0;
274  FOR_ALL_DIRECT_ELEMENTS_IN_MULTIDIMARRAY(subtomo)
275  {
276  if (DIRECT_MULTIDIM_ELEM(maskNormalize, n)>0)
277  {
278  double val = DIRECT_MULTIDIM_ELEM(subtomo, n);
279  sumVal += val;
280  sumVal2 += val*val;
281  counter = counter + 1;
282  }
283  }
284 
285  double mean, sigma2;
286  mean = sumVal/counter;
287  sigma2 = sqrt(sumVal2/counter - mean*mean);
288 
289  FOR_ALL_DIRECT_ELEMENTS_IN_MULTIDIMARRAY(subtomo)
290  {
291  DIRECT_MULTIDIM_ELEM(subtomo, n) -= mean;
292  DIRECT_MULTIDIM_ELEM(subtomo, n) /= sigma2;
293  }
294  }
295 
296  FileName fn;
297  fn = fnCoor.getBaseName() + formatString("-%i.mrc", idx);
298 
299  #ifdef DEBUG
300  std::cout << fn << std::endl;
301  std::cout << fnOut << std::endl;
302  std::cout << fnOut+fn << std::endl;
303  #endif
304 
305  subtomoImg.write(fnOut+"/"+fn);
306 
307  ++idx;
308  }
309 
310  std::cout << "Subtomo substraction finished succesfully!!" << std::endl;
311 }
