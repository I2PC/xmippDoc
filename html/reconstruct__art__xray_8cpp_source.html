137  newZdim = iniZdim;

143 
144  // Rotate volume ....................................................

146  Matrix2D<double> R;
147

149

186  FOR_ALL_DIRECT_ELEMENTS_IN_ARRAY2D(mdaProjExp)
187  {
188  // Compute difference image and error

190  mean_error += DIRECT_A2D_ELEM(Idiff, i, j) * DIRECT_A2D_ELEM(Idiff, i, j);
191 
192  // Compute the correction image
193  DIRECT_A2D_ELEM(mdaProjNorm, i, j) = XMIPP_MAX(DIRECT_A2D_ELEM(mdaProjNorm, i, j), 1);
194  DIRECT_A2D_ELEM(mdaProjNorm, i, j) =
195  lambdaART * DIRECT_A2D_ELEM(Idiff, i, j) / DIRECT_A2D_ELEM(mdaProjNorm, i, j);
196  }
197  mean_error /= YXSIZE(mdaProjExp);
198 
199  projNorm.write("projNorm_2.spi");
200 
201  imTemp().alias(Idiff);
202  imTemp.write("iDiff.spi");
203 
204  projectXrayGridVolume(muVol, psf, IgeoVol, projTheo, &mdaProjNorm, BACKWARD, thMgr);
205 
206  return mean_error;
