275  """ Iterate over all the input volumes. """
276  for pointer in self.inputVolumes:
277  item = pointer.get()
278  if item is None:
279  break
280  itemId = item.getObjId()
281  if isinstance(item, Volume):
282  item.outputName = self._getExtraPath('output_vol%06d.vol' % itemId)
283  # If item is a Volume and label is empty
284  if not item.getObjLabel():
285  # Volume part of a set
286  if item.getObjParentId() is None:
287  item.setObjLabel("%s.%s" % (pointer.getObjValue(), pointer.getExtended()))
288  else:
289  item.setObjLabel('%s.%s' % (self.getMapper().getParent(item).getRunName(), item.getClassName()))
290  yield item
291  elif isinstance(item, SetOfVolumes):
292  for vol in item:
293  vol.outputName = self._getExtraPath('output_vol%06d_%03d.vol' % (itemId, vol.getObjId()))
294  # If set item label is empty
295  if not vol.getObjLabel():
296  # if set label is not empty use it
297  if item.getObjLabel():
298  vol.setObjLabel("%s - %s%s" % (item.getObjLabel(), vol.getClassName(), vol.getObjId()))
299  else:
300  vol.setObjLabel("%s - %s%s" % (self.getMapper().getParent(item).getRunName(), vol.getClassName(), vol.getObjId()))
