644 /* Produce Side Information ------------------------------------------------ */
645 void CTFDescription1D::produceSideInfo()
646 {
647  // Change units
648  double local_Cs = Cs * 1e7;
649  double local_Ca = Ca * 1e7;
650  double local_kV = kV * 1e3;
651  double local_ispr = ispr * 1e6;
652 
653  lambda=12.2643247/sqrt(local_kV*(1.+0.978466e-6*local_kV)); // See http://en.wikipedia.org/wiki/Electron_diffraction
654  //
655  // Phase shift for spherical aberration
656  // X(u)=-PI*deltaf(u)*lambda*u^2+PI/2*Cs*lambda^3*u^4
657  // ICE: X(u)=-PI/2*deltaf(u)*lambda*u^2+PI/2*Cs*lambda^3*u^4
658  // = K1*deltaf(u)*u^2 +K2*u^4
659  K1 = PI * lambda;
660  K2 = PI / 2 * local_Cs * lambda * lambda * lambda;
661 
662  // Envelope
663  // D(u)=Ed(u)*Ealpha(u)
664  // Ed(u)=exp(-1/2*PI^2*lambda^2*D^2*u^4)
665  // Ealpha(u)=exp(-PI^2*alpha^2*u^2*(Cs*lambda^2*u^2+Deltaf(u))^2)
666  // ICE: Eespr(u)=exp(-(1/4*PI*Ca*lambda*espr/kV)^2*u^4/log2)
667  // ICE: Eispr(u)=exp(-(1/2*PI*Ca*lambda*ispr)^2*u^4/log2)
668  // ICE: EdeltaF(u)=bessj0(PI*DeltaF*lambda*u^2)
669  // ICE: EdeltaR(u)=sinc(u*DeltaR)
670  // ICE: Ealpha(u)=exp(-PI^2*alpha^2*(Cs*lambda^2*u^3+Deltaf(u)*u)^2)
671  // CO: K3=pow(0.25*PI*Ca*lambda*(espr/kV,2)/log(2); Both combines in new K3
672  // CO: K4=pow(0.5*PI*Ca*lambda*ispr,2)/log(2);
673  K3 = pow(0.25 * PI * local_Ca * lambda * (espr / kV + 2 * local_ispr), 2) / log(2.0);
674  K5 = PI * DeltaF * lambda;
675  K6 = PI * PI * alpha * alpha;
676  K7 = local_Cs * lambda * lambda;
677  Ksin = sqrt(1-Q0*Q0);
678  Kcos = Q0;
679 }

699 /* Look for zeroes, maxima or minima ------------------------------------------------------------ */
700 //#define DEBUG
701 void CTFDescription1D::lookFor(int n, const Matrix1D<double> &u, Matrix1D<double> &freq, int iwhat)
702 {
703  double wmax = 1 / (2 * Tm);
704  double wstep = wmax / 300;
705  int found = 0;
706  double last_ctf = getValuePureNoPrecomputedAt(0), ctf=0.0, state=1;
707 
708  double w;
709  for (w = 0; w <= wmax; w += wstep)
710  {
711  V2_BY_CT(freq, u, w);
712  ctf = getValuePureNoPrecomputedAt(XX(freq));
713 
714  switch (iwhat)
715  {
716  case 0: // Looking for zeroes
717  if (SGN(ctf) != SGN(last_ctf))
718  found++;
719  break;
720  case 1: // Looking for maxima
721  if (w>0)
722  {
723  if (state==1) // Going up
724  {
725  if (ctf<last_ctf)
726  {
727  found++;
728  state=-1;
729  }
730  }
731  else // Going down
732  {
733  if (ctf>last_ctf)
734  state=1;
735  }
736  }
737  break;
738  case -1: // Looking for minima
739  if (w>0)
740  {
741  if (state==-1) // Going down
742  {
743  if (ctf>last_ctf)
744  {
745  found++;
746  state=1;
747  }
748  }
749  else // Going up
750  {
751  if (ctf<last_ctf)
752  state=-1;
753  }
754  }
755  break;
756  }
757  if (found == n)
758  {
759  break;
760  }
761 
762  last_ctf = ctf;
763  }
764  if (found != n)
765  {
766 
767  VECTOR_R2(freq, -1, -1);
768  }
769  else
770  {
771  // Compute more accurate zero
772 #ifdef DEBUG
773  std::cout << n << " zero: w=" << w << " (" << wmax << ") freq="
774  << (u*w).transpose()
775  << " last_ctf=" << last_ctf << " ctf=" << ctf << " ";
776 #endif
777 
778  switch (iwhat)
779  {
780  case 0:
781  w += ctf * wstep / (last_ctf - ctf);
782  break;
783  default:
784  w-=wstep;
785  }
786  V2_BY_CT(freq, u, w);
787 #ifdef DEBUG
788 
789  std::cout << " final w= " << w << " final freq=" << freq.transpose() << std::endl;
790 #endif
791  }
792 }

1424 {

1434  switch (iwhat)

1438  found++;

1441  if (w>0)
1442  {

1447  found++;

1450  }
1451  else // Going down

1455  }
1456  }

1458  case -1: // Looking for minima

1468  }
1469  else // Going up
1470  {
1471  if (ctf<last_ctf)
1472  state=-1;

1476  }
