148 bool MpiProgAngularProjectionMatching::distributeJobs(size_t * imagesToSent,
149  int node)
150 {
151  int node_index = last_chunk[node];
152  static bool reached_last_chunk = false;
153 
154  if (node_index == -1
155  || chunk_mysampling.my_exp_img_per_sampling_point[node_index].empty())
156  {
157 
158  if (!reached_last_chunk)
159  {
160  node_index = chunk_index;
161  chunk_index = (chunk_index + 1) % chunk_number;
162  reached_last_chunk = (chunk_index == 0);
163  }
164  else
165  {
166  int i = 0;
167  for (;
168  i < chunk_number
169  && chunk_mysampling.my_exp_img_per_sampling_point[chunk_index].empty();
170  ++i, chunk_index = (chunk_index + 1) % chunk_number)
171  ;
172 
173  // std::cerr << formatString("DEBUG: master: assigned %d chuck to node %d, i = %d", chunk_index, node, i);
174  if (i == chunk_number)
175  {
176  imagesToSent[0] = 0;
177  return false;
178  }
179  last_chunk[node] = node_index = chunk_index;
180  chunk_index = (chunk_index + 1) % chunk_number;
181  }
182  }
183 
184  size_t assigned_images = std::min((size_t)mpi_job_size, chunk_mysampling.my_exp_img_per_sampling_point[node_index].size());
185  imagesToSent[0] = assigned_images;
186 
187  //std::cerr << formatString("DEBUG: master: assigned %lu images to node %d\n", assigned_images, node);
188 
189  for (size_t i = 1; i <= assigned_images; ++i)
190  {
191  imagesToSent[i] =

193  // std::cerr << " " << imagesToSent[i];

320 void MpiProgAngularProjectionMatching::computeChunkAngularDistance(int symmetry,
321  int sym_order)
322 {
323  double non_reduntant_area_of_sphere =
324  chunk_mysampling.SL.nonRedundantProjectionSphere(symmetry,
325  sym_order);
326  double number_cpus = (double) node->size - 1;
327  //NEXT ONE IS SAMPLING NOT ANOTHERSAMPLING
328  double neighborhood_radius = fabs(acos(mysampling.cos_neighborhood_radius));
329  //NEXT ONE IS SAMPLING NOT ANOTHERSAMPLING
330  if (mysampling.cos_neighborhood_radius < -1.001)
331  neighborhood_radius = 0;
332  int counter = 0;
333  while (1)
334  {
335  if (counter++ > 1000)
336  {
337  chunk_angular_distance = 0.001;
338  std::cerr << "****************************************************"
339  << std::endl;
340  std::cerr << "* WARNING: The neighbourhood does not fit in memory "
341  << std::endl;
342  std::cerr << "****************************************************"
343  << std::endl;
344  break;
345  }
346  double area_chunk = non_reduntant_area_of_sphere / number_cpus;
347  //area chunk is area of spheric casket=2 PI h
348  chunk_angular_distance = acos(1 - area_chunk / (2 * PI));
349  double area_chunck_neigh = 2 * PI
350  * (1 - cos(chunk_angular_distance + neighborhood_radius));
351  //double area_chunck= 2 * PI *( 1 - cos(chunk_angular_distance));
352  //let us see how many references from the reference library fit
353  //in area_chunk, that is divide area_chunk between the voronoi
354  //region of the sampling points of the reference library
355  double areaVoronoiRegionReferenceLibrary = 2 *( 3 *( acos(
356  //NEXT ONE IS SAMPLING NOT ANOTHERSAMPLING
357  cos(mysampling.sampling_rate_rad)/(1+cos(mysampling.sampling_rate_rad)) ) ) - PI);
358  int number_of_images_that_fit_in_a_chunck_neigh =(int)
359  ceil(area_chunck_neigh / areaVoronoiRegionReferenceLibrary);
360  //#define DEBUG
361 #ifdef DEBUG
362 
363  std::cerr << "\n\ncounter " << counter << std::endl;
364  std::cerr << "area_chunk " << area_chunk << std::endl;
365  std::cerr << "2*chunk_angular_distance " << 2*chunk_angular_distance << std::endl;
366  //NEXT ONE IS SAMPLING NOT ANOTHERSAMPLING
367  std::cerr << "sampling_rate_rad " << mysampling.sampling_rate_rad
368  << " " << mysampling.sampling_rate_rad*180/PI
369  << std::endl;

371  << std::endl;
372  std::cerr << "areaVoronoiRegionReferenceLibrary " << areaVoronoiRegionReferenceLibrary << std::endl;
373  std::cerr << "number_of_images_that_fit_in_a_chunck_neigh " << number_of_images_that_fit_in_a_chunck_neigh << std::endl;

376 #endif
377 #undef DEBUG
378 
379  if (number_of_images_that_fit_in_a_chunck_neigh > max_nr_imgs_in_memory)
380  number_cpus = 1.2 * number_cpus;

382  break;
383  }
384  //chunk_angular_distance -= neighborhood_radius;
385  chunk_angular_distance *= 2.0;
386 
387  //#define DEBUG

391  << std::endl
392  << "neighborhood_radius " << neighborhood_radius
393  << std::endl;
394 #endif
395 #undef DEBUG
396  //chuck should not be bigger than a triangle in the icosahedra
