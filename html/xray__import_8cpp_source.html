326 void ProgXrayImport::getFlatfield(const FileName &fnFFinput,
327  Image<double> &Iavg)
328 {
329 
330  // Process the flatfield images
331 
332 
333  MultidimArray<double> &mdaIavg = Iavg();
334 
335  Matrix1D<double> expTimeArray, cBeamArray, slitWidthArray; // Local vectors
336 
337  DataSource ldSource = dSource;
338 
339  // Checking if fnFFinput is a single file to obtain the flatfield avg from
340  if (extFlat && isImage(fnFFinput))
341  {
342  fMD.read(fnFFinput);
343  ldSource = NONE;
344  }
345  else
346  {
347  switch (ldSource)
348  {
349  case MISTRAL:
350  if (!H5File.checkDataset(fnFFinput.getBlockName().c_str()))
351  break;
352  {
353  fMD.read(fnFFinput);
354  H5File.getDataset("NXtomo/instrument/bright_field/ExpTimes", expTimeArray, false);
355  H5File.getDataset("NXtomo/instrument/bright_field/current", cBeamArray, false);
356 
357  // If expTime is empty or only one single value in nexus file then we fill with 1
358  if (expTimeArray.size() < 2)
359  {
360  reportWarning("Input file does not contains flatfields' exposition time information.");
361  expTimeArray.initConstant(fMD.size(), 1.);
362  }
363  // If current is empty or only one single value in nexus file then we fill with 1
364  if (cBeamArray.size() < 2)
365  {
366  reportWarning("Input file does not contains flatfields' current beam information.");
367  cBeamArray.initConstant(fMD.size(), 1.);
368  }
369 
370  }
371 
372  // Since Alba does not provide slit width, we set to ones
373  slitWidthArray.initConstant(fMD.size(), 1.);
374 
375  break;
376  case BESSY:
377  {
378  size_t objId;
379 
380  for (size_t i = fIni; i <= fEnd; ++i)
381  {
382  objId = fMD.addObject();
383  fMD.setValue(MDL_IMAGE, fnFFinput + formatString("/img%d.spe", i), objId);
384  }
385  break;
386  }
387  case GENERIC:
388  {
389  // Get Darkfield
390  std::cout << "Getting darkfield from "+fnFFinput << " ..." << std::endl;
391  getDarkfield(fnFFinput, IavgDark);
392  if (darkFix)
393  IavgDark.write(fnRoot+"_"+fnFFinput.removeDirectories()+"_darkfield.xmp");
394 
395  std::vector<FileName> listDir;
396  fnFFinput.getFiles(listDir);
397  size_t objId;
398 
399  for (size_t i = 0; i < listDir.size(); ++i)
400  {
401  if (!listDir[i].hasImageExtension())
402  continue;
403  objId = fMD.addObject();
404  fMD.setValue(MDL_IMAGE, fnFFinput+"/"+listDir[i], objId);
405  }
406  }
407  break;
408  }
409  }
410 
411  if ( fMD.size() == 0 )
412  {
413  reportWarning("XrayImport::getFlatfield: No images to process");
414  return;
415  }
416 
417  ImageInfo imFFInfo;
418 
419  getImageInfo(fMD, imFFInfo);
420 
421  if ( (imFFInfo.adim.xdim != imgInfo.adim.xdim) || (imFFInfo.adim.ydim != imgInfo.adim.ydim) )
422  {
423  reportWarning(formatString("XrayImport:: Flatfield images size %dx%d different from Tomogram images size %dx%d",
424  imFFInfo.adim.xdim,imFFInfo.adim.ydim,imgInfo.adim.xdim,imgInfo.adim.ydim));
425  std::cout << "Setting crop values to fit the smallest dimensions." <<std::endl;
426 
427  // This shift in the crop sizes is exclusive of Mistral data
428  cropSizeXi = (imgInfo.adim.xdim - std::min(imgInfo.adim.xdim-cropSizeX*2, imFFInfo.adim.xdim))/2 + 1;
429  cropSizeYi = (imgInfo.adim.ydim - std::min(imgInfo.adim.ydim-cropSizeY*2, imFFInfo.adim.ydim))/2 + 1;
430  cropSizeXe = cropSizeXi - 2;
431  cropSizeYe = cropSizeYi - 2;
432  }
433 
434  int cropX = (imFFInfo.adim.xdim - (imgInfo.adim.xdim-cropSizeXi-cropSizeXe))/2;// - 1;
435  int cropY = (imFFInfo.adim.ydim - (imgInfo.adim.ydim-cropSizeYi-cropSizeYe))/2;// - 1;
436 
437 
438  int N = 0;
439  Image<double> Iaux;
440  FileName fnImg;
441 
442  FOR_ALL_OBJECTS_IN_METADATA(fMD)
443  {
444  fMD.getValue(MDL_IMAGE, fnImg, __iter.objId);
445 
446  readAndCrop(fnImg, Iaux, cropX, cropY);
447 
448 
449  if ( darkFix )
450  {
451  Iaux() -= IavgDark();
452  forcePositive(Iaux());
453  }
454 
455  double currentBeam = 1;
456  double expTime = 1;
457  double slitWidth = 1;
458 
459  switch (ldSource)
460  {
461  case MISTRAL:
462  {
463  size_t idx = fnImg.getPrefixNumber();
464  currentBeam = dMi(cBeamArray, idx-1);
465  expTime = dMi(expTimeArray, idx-1);
466  slitWidth = dMi(slitWidthArray, idx-1);
467  }
468  break;
469  case BESSY:
470  case GENERIC:
471  readCorrectionInfo(fnImg, currentBeam, expTime, slitWidth);
472  break;
473  default:
474  break;
475  }
476 
477  Iaux() *= 1.0/(currentBeam*expTime*slitWidth);
478 
479  if ( N == 0 )
480  mdaIavg = Iaux();
481  else
482  mdaIavg += Iaux();
483  N++;
484  }
485 
486  darkFix = false; // We reset just in case there is no dark field for tomo images
487 
488  if (0 == N) {
489  REPORT_ERROR(ERR_NUMERICAL, "N is zero (0), which would lead to division by zero");
490  }
491  mdaIavg/=N;
492 
493  /* Create a mask with zero valued pixels to apply boundaries median filter
494  * to avoid dividing by zero when normalizing */
495 
496  MultidimArray<char> &mdaMask = MULTIDIM_ARRAY(bpMask);
497 
498  if ( !fnBPMask.empty() && !mdaIavg.sameShape(mdaMask) )
499  REPORT_ERROR(ERR_MULTIDIM_SIZE, "XrayImport: Mask size does not match flat fields size.");
500 
501  if (BPFactor > 0)
502  {
503  double avg, std;
504  mdaIavg.computeAvgStdev(avg, std);
505  mdaMask.resize(mdaIavg, false);
506 
507  double th = avg - std*BPFactor;
508  FOR_ALL_DIRECT_ELEMENTS_IN_MULTIDIMARRAY(mdaMask)
509  dAi(mdaMask, n) = dAi(mdaIavg, n) < th;
510  }
511  else if (fnBPMask.empty())
512  mdaIavg.equal(0,mdaMask);
513 
514  MultidimArray<char> mask = mdaMask;
515  boundMedianFilter(mdaIavg,mask);
516 }

518 void runThread(ThreadArgument &thArg)
519 {
520  int thread_id = thArg.thread_id;
521  ProgXrayImport * ptrProg= (ProgXrayImport *)thArg.workClass;
522 
523  MetaData localMD;
524  Image<double> Iaux;
525  FileName fnImgIn, fnImgOut;
526  size_t first = 0, last = 0;
527  MultidimArray<char> mask;
528 
529  while (ptrProg->td->getTasks(first, last))
530  {
531  for (size_t i=first; i<=last; i++)
532  {
533  ptrProg->inMD.getValue(MDL_IMAGE, fnImgIn, ptrProg->objIds[i]);
534 
535 
536  MDRow rowGeo;
537  ptrProg->readGeoInfo(fnImgIn, rowGeo);
538 // ptrProg->readAndCrop(fnImgIn, Iaux, ptrProg->cropSizeX, ptrProg->cropSizeY);
539 
540  Iaux.read(fnImgIn);
541  Iaux().selfWindow(ptrProg->cropSizeYi,ptrProg->cropSizeXi,
542  (int)(YSIZE(Iaux())-ptrProg->cropSizeYe-1),(int)(XSIZE(Iaux())-ptrProg->cropSizeXe-1));
543 
544  Iaux().resetOrigin();
545 
546 
547 
548 
549  if (XSIZE(ptrProg->IavgDark())!=0)
550  {
551  Iaux()-=ptrProg->IavgDark();
552  forcePositive(Iaux());
553  }
554 
555 
556  double currentBeam = 1;
557  double expTime = 1;
558  double slitWidth = 1;
559 
560  if ( ptrProg->dSource == ptrProg->MISTRAL )
561  {
562  size_t idx = fnImgIn.getPrefixNumber();
563  currentBeam = dMi(ptrProg->cBeamArray, idx-1);
564  expTime = dMi(ptrProg->expTimeArray, idx-1);
565  slitWidth = dMi(ptrProg->slitWidthArray, idx-1);
566  }
567  else
568  ptrProg->readCorrectionInfo(fnImgIn, currentBeam, expTime, slitWidth);
569 
570  Iaux() *= 1.0/(currentBeam*expTime*slitWidth);
571  if (XSIZE(ptrProg->IavgFlat())!=0)
572  Iaux()/=ptrProg->IavgFlat();
573 
574  // Assign median filter to zero valued pixels to avoid -inf when applying log10
575  Iaux().equal(0,mask);
576  mask.resizeNoCopy(Iaux());
577 
578  if (XSIZE(ptrProg->bpMask()) != 0)
579  mask += ptrProg->bpMask();
580 
581  boundMedianFilter(Iaux(), mask);
582 
583  if (ptrProg->logFix)
584  {
585  Iaux().selfLog();
586  if (ptrProg->selfAttFix)
587  Iaux() *= -1.;
588  }
589 
590  fnImgOut.compose(i+1, ptrProg->fnOut);
591 
592 
593  size_t objId = localMD.addObject();
594  localMD.setValue(MDL_IMAGE,fnImgOut,objId);
595  localMD.setRow(rowGeo, objId); //
596  // localMD.setValue(MDL_ANGLE_TILT,Iaux.tilt(),objId);
597  Iaux.write(fnImgOut);
598  if (thread_id==0)
599  progress_bar(i);
600  }
601  }
602  //Lock for update the total counter
603  ptrProg->mutex.lock();
604  ptrProg->outMD.unionAll(localMD);
605  ptrProg->mutex.unlock();
606 }
