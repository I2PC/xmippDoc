177  const Dimensions &dims,
178  const T *__restrict__ data,
179  float *__restrict__ positions,
180  T *__restrict__ values) {
181  sFindUniversalChecks(dims, data, positions, values);
182 
183  if (dims.isPadded()) {

186  // locate max

245 template<typename T>
246 template<typename C>
247 void SingleExtremaFinder<T>::sFindUniversal2DAroundCenter(
248  const C &comp,
249  T startVal,
250  const CPU &cpu,
251  const Dimensions &dims,
252  const T *data,
253  float *positions, // can be nullptr
254  T * values, // can be nullptr
255  size_t maxDist) {
256  // check input
257  assert(dims.is2D());
258  assert( ! dims.isPadded());
259  assert(dims.sizeSingle() > 0);
260  assert(dims.n() > 0);
261  assert(nullptr != data);
262  assert((nullptr != positions) || (nullptr != values));
263  assert(0 < maxDist);
264  const size_t xHalf = dims.x() / 2;
265  const size_t yHalf = dims.y() / 2;
266  assert((2 * xHalf) > maxDist);
267  assert((2 * yHalf) > maxDist);
268 
269  const auto min = std::pair<size_t, size_t>(
270  std::max((size_t)0, xHalf - maxDist),
271  std::max((size_t)0, yHalf - maxDist)
272  );
273 
274  const auto max = std::pair<size_t, size_t>(
275  std::min(dims.x() - 1, xHalf + maxDist),
276  std::min(dims.y() - 1, yHalf + maxDist)
277  );
278 
279  const size_t maxDistSq = maxDist * maxDist;
280  for (size_t n = 0; n < dims.n(); ++n) {
281  size_t offsetN = n * dims.xyzPadded();
282  T extrema = startVal;
283  float pos = -1;
284  // iterate through the center
285  for (size_t y = min.second; y <= max.second; ++y) {
286  size_t offsetY = y * dims.x();
287  int logicY = (int)y - yHalf;
288  size_t ySq = logicY * logicY;
289  for (size_t x = min.first; x <= max.first; ++x) {
290  int logicX = (int)x - xHalf;
291  // continue if the Euclidean distance is too far
292  if ((ySq + (logicX * logicX)) > maxDistSq) continue;
293  // get current value and update, if necessary
294  T tmp = data[offsetN + offsetY + x];
295  if (comp(tmp, extrema)) {
296  extrema = tmp;
297  pos = offsetY + x;
298  }
299  }
300  }
301  // store results
302  if (nullptr != positions) {
303  positions[n] = pos;
304  }
305  if (nullptr != values) {
306  values[n] = extrema;
307  }
308  }
309 }
