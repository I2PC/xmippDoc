95 void ProgCorrectWiener2D::generateWienerFilter(MultidimArray<double> &Mwien, CTFDescription & ctf)
96 {
97  int paddimY = Ydim*pad;
98  int paddimX = Xdim*pad;
99  ctf.enable_CTF = true;
100  ctf.enable_CTFnoise = false;
101  ctf.produceSideInfo();
102 
103  MultidimArray<std::complex<double> > ctfComplex;
104  MultidimArray<double> ctfIm;
105 
106  Mwien.resize(paddimY,paddimX);
107 
108  ctf.Tm = sampling_rate;
109  //ctf.Tm /= pad;
110 
111  if (isIsotropic)
112  {
113  double avgdef = (ctf.DeltafU + ctf.DeltafV)/2.;
114  ctf.DeltafU = avgdef;
115  ctf.DeltafV = avgdef;
116  }
117 
118 
119  ctfIm.resize(1, 1, paddimY, paddimX,false);
120  //Esto puede estar mal. Cuidado con el sampling de la ctf!!!
121 
122  if (correct_envelope)
123  ctf.generateCTF(paddimY, paddimX, ctfComplex);
124  else
125  ctf.generateCTFWithoutDamping(paddimY, paddimX, ctfComplex);
126 
127  FOR_ALL_DIRECT_ELEMENTS_IN_ARRAY2D(ctfIm)
128  {
129  if (phase_flipped)
130  dAij(ctfIm, i, j) = fabs(dAij(ctfComplex, i, j).real());
131  else
132  dAij(ctfIm, i, j) = dAij(ctfComplex, i, j).real();
133  }
134 
135 //#define DEBUG
136 #ifdef DEBUG
137  {
138  Image<double> save;
139  save()=ctfIm;
140  save.write("PPPctf2.spi");
141  //exit(1);
142  //std::cout << "Press any key\n";
143  //char c;
144  //std::cin >> c;
145  }
146 #endif
147 #undef DEBUG
148 
149  double result;
150  FOR_ALL_DIRECT_ELEMENTS_IN_ARRAY2D(Mwien)
151  {
152  result = (DIRECT_N_YX_ELEM (ctfIm, 0, 0, i, j ));
153  dAij(Mwien,i,j) = (result *result);
154  }
155 
156 #ifdef DEBUG
157 
158 {
159  Image<double> save;
160  save()=Mwien;
161  save.write("vienerB.spi");
162 }
163 #endif
164 #undef DEBUG
165 
166 // Add Wiener constant
167 if (wiener_constant < 0.)
168 {
169 
170  // Use Grigorieff's default for Wiener filter constant: 10% of average over all Mwien terms
171  // Grigorieff JSB 157(1) (2006), pp 117-125
172  double valueW = 0.1*Mwien.computeAvg();
173  FOR_ALL_DIRECT_ELEMENTS_IN_ARRAY2D(Mwien)
174  {
175  dAij(Mwien,i,j) += valueW;
176  dAij(Mwien,i,j) = dAij(ctfIm, i, j)/dAij(Mwien, i, j);
177  }
178 }
179 else
180 {
181  FOR_ALL_DIRECT_ELEMENTS_IN_ARRAY2D(Mwien)
182  {
183  dAij(Mwien,i,j) += wiener_constant;
184  dAij(Mwien,i,j) = dAij(ctfIm, i, j)/dAij(Mwien, i, j);
185  }
186 }
187 
188 #ifdef DEBUG
189 
190 {
191  Image<double> save;
192  save()=Mwien;
193  save.write("vienerC.spi");
194 }
195 #endif
196 #undef DEBUG
197 
198 }
