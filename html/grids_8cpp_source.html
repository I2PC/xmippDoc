95 // Number of samples -------------------------------------------------------
96 int SimpleGrid::get_number_of_samples() const
97 {
98  if (R2 == -1)
99  {
100  int Zdim, Ydim, Xdim;
101  getSize(Zdim, Ydim, Xdim);
102  return Zdim*Ydim*Xdim;
103  }
104  else
105  {
106  int ZZ_lowest = (int) ZZ(lowest);
107  int YY_lowest = (int) YY(lowest);
108  int XX_lowest = (int) XX(lowest);
109  int ZZ_highest = (int) ZZ(highest);
110  int YY_highest = (int) YY(highest);
111  int XX_highest = (int) XX(highest);
112  Matrix1D<double> grid_index(3), univ_position(3);
113  int N = 0;
114  for (int k = ZZ_lowest; k <= ZZ_highest; k++)
115  for (int i = YY_lowest; i <= YY_highest; i++)
116  for (int j = XX_lowest; j <= XX_highest; j++)
117  {
118  VECTOR_R3(grid_index, j, i, k);
119  grid2universe(grid_index, univ_position);
120  if (is_interesting(univ_position)) N++;
121  }
122  return N;
123  }
124 }

144 void Grid::voxel_corners(Matrix1D<double> &Gcorner1, Matrix1D<double> &Gcorner2,
145  const Matrix2D<double> *V) const
146 {
147  Matrix1D<double> SGcorner1(3), SGcorner2(3); // Subgrid corners
148  SPEED_UP_temps012;
149 
150  // Look for the lowest and highest volume coordinate
151  Gcorner1.resize(3); // lowest and highest coord.
152  Gcorner2.resize(3);
153  for (size_t n = 0; n < GridsNo(); n++)
154  {
155  // Find box for this grid
156  bool first;
157  first = true;
158  for (int k = (int)ZZ(LG[n].lowest); k <= ZZ(LG[n].highest); k++)
159  for (int i = (int)YY(LG[n].lowest); i <= YY(LG[n].highest); i++)
160  for (int j = (int)XX(LG[n].lowest); j <= XX(LG[n].highest); j++)
161  {
162  Matrix1D<double> grid_index(3), univ_position(3);
163  VECTOR_R3(grid_index, j, i, k);
164  LG[n].grid2universe(grid_index, univ_position);
165  if (V != NULL)
166  {
167  M3x3_BY_V3x1(univ_position, *V, univ_position);
168  }
169  if (!LG[n].is_interesting(univ_position)) continue;
170  if (!first)
171  {
172  XX(SGcorner1) = XMIPP_MIN(XX(SGcorner1), XX(univ_position));
173  YY(SGcorner1) = XMIPP_MIN(YY(SGcorner1), YY(univ_position));
174  ZZ(SGcorner1) = XMIPP_MIN(ZZ(SGcorner1), ZZ(univ_position));
175 
176  XX(SGcorner2) = XMIPP_MAX(XX(SGcorner2), XX(univ_position));
177  YY(SGcorner2) = XMIPP_MAX(YY(SGcorner2), YY(univ_position));
178  ZZ(SGcorner2) = XMIPP_MAX(ZZ(SGcorner2), ZZ(univ_position));
179  }
180  else
181  {
182  SGcorner2 = SGcorner1 = univ_position;
183  first = false;
184  }
185  }
186 
187  // Compare with the rest of the grids
188  if (n != 0)
189  {
190  XX(Gcorner1) = XMIPP_MIN(XX(Gcorner1), XX(SGcorner1));
191  YY(Gcorner1) = XMIPP_MIN(YY(Gcorner1), YY(SGcorner1));
192  ZZ(Gcorner1) = XMIPP_MIN(ZZ(Gcorner1), ZZ(SGcorner1));
193 
194  XX(Gcorner2) = XMIPP_MAX(XX(Gcorner2), XX(SGcorner2));
195  YY(Gcorner2) = XMIPP_MAX(YY(Gcorner2), YY(SGcorner2));
196  ZZ(Gcorner2) = XMIPP_MAX(ZZ(Gcorner2), ZZ(SGcorner2));
197  }
198  else
199  {
200  Gcorner1 = SGcorner1;
201  Gcorner2 = SGcorner2;
202  }
203 
204 #ifdef DEBUG
205  std::cout << LG[n];
206  std::cout << "SGcorner1 " << SGcorner1.transpose() << std::endl;
207  std::cout << "SGcorner2 " << SGcorner2.transpose() << std::endl;
208  std::cout << "Gcorner1 " << Gcorner1.transpose() << std::endl;

217 /* Create CC Simple grid with a given origin ------------------------------- */
218 SimpleGrid Create_CC_grid(double relative_size, const Matrix1D<double> &corner1,
219  const Matrix1D<double> &corner2, const Matrix1D<double> &origin)
220 {
221  SimpleGrid grid;
222 
223  // The vectors of the grid are the default ones of (1,0,0), (0,1,0),
224  // and (0,0,1), and its inverse matrix is already computed
225  grid.relative_size = relative_size;
226  grid.origin = origin;
227 
228  // Compute the lowest and highest indexes inside the grid
229  grid.universe2grid(corner1, grid.lowest);
230  grid.lowest.selfFLOOR();
231  grid.universe2grid(corner2, grid.highest);
232  grid.highest.selfCEIL();
233 
234  grid.R2 = -1;
235  return grid;
236 }

397 //#define DEBUG
398 SimpleGrid Create_grid_within_sphere(double relative_size,
399  const Matrix1D<double> &origin,
400  const Matrix1D<double> &X, const Matrix1D<double> &Y,
401  const Matrix1D<double> &Z, double R2)
402 {
403  SimpleGrid grid;
404  double R = sqrt(R2);
405 
406  grid.set_X(X);
407  grid.set_Y(Y);
408  grid.set_Z(Z);
409  grid.relative_size = relative_size;
410  grid.origin = origin;
411  grid.R2 = R2;
412  grid.lowest.initZeros(3);
413  grid.highest.initZeros(3);
414  grid.prepare_grid();
415 
416  // Find grid limits
417  int iR = CEIL(R);
418  Matrix1D<double> univ_position(3), grid_position(3);
419  for (int k = -iR; k <= iR; k++)
420  for (int i = -iR; i <= iR; i++)
421  for (int j = -iR; j <= iR; j++)
422  {
423  VECTOR_R3(univ_position, j, i, k);
424  if (univ_position.module() > R) continue;
425  grid.universe2grid(univ_position, grid_position);
426  XX(grid.lowest) = XMIPP_MIN(XX(grid.lowest), FLOOR(XX(grid_position)));
427  YY(grid.lowest) = XMIPP_MIN(YY(grid.lowest), FLOOR(YY(grid_position)));
428  ZZ(grid.lowest) = XMIPP_MIN(ZZ(grid.lowest), FLOOR(ZZ(grid_position)));
429  XX(grid.highest) = XMIPP_MAX(XX(grid.highest), CEIL(XX(grid_position)));
430  YY(grid.highest) = XMIPP_MAX(YY(grid.highest), CEIL(YY(grid_position)));
431  ZZ(grid.highest) = XMIPP_MAX(ZZ(grid.highest), CEIL(ZZ(grid_position)));
432  }
433 
434 #ifdef DEBUG
435  std::cout << "Sphere radius = " << R << std::endl
436  << "relative size = " << relative_size << std::endl
437  << "X module = " << X.module() << std::endl
438  << "Y module = " << Y.module() << std::endl
439  << "Z module = " << Z.module() << std::endl
440  << grid
441  ;
442 #endif
443  return grid;
444 }
