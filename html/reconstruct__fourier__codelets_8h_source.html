36 // ============================== Used buffers ==============================

38 /* FFTs Buffer
39  * ===========
40  * float['batchSize' * ('fftSizeX' * 'fftSizeY') * 2]
41  * but only 'noOfImages' is valid (*2 since it's complex)
42  * Holds 'right side of the centered FFT', i.e. only unique values, with high frequencies in the corners
43  */
44 
45 /* Spaces Buffer
46  * RecFourierProjectionTraverseSpace['batchSize' * 'noOfSymmetries']
47  * but only 'noOfImages' is valid
48  * Traverse spaces for each FFT, 'describing' the FFTs/paddedImages
49  * NOTE: RecFourierProjectionTraverseSpace is a trivial POD struct from <reconstruction/reconstruct_fourier_projection_traverse_space.h>
50  */
51 
52 /* Image Data Buffer (only before FFTs Buffer is ready)
53  * =================
54  * double[`batchSize` * (`paddedImageSize` * `paddedImageSize`)]
55  * but only 'noOfImages' is valid
56  * Contains images to be transformed into FFTs.
57  */
58 
59 
60 /* Result Volume Buffer
61  * ====================
62  * std::complex<float>[maxVolumeIndexYZ+1][maxVolumeIndexYZ+1][maxVolumeIndexYZ+1]
63  */
64 
65 /* Result Weights Buffer
66  * =====================
67  * float[maxVolumeIndexYZ+1][maxVolumeIndexYZ+1][maxVolumeIndexYZ+1]
68  */
69 
70 /* Blob Table Squared Buffer
71  * =========================
72  * float[BLOB_TABLE_SIZE_SQRT]
73  */
