336 void MDObject::toStream(std::ostream &os, bool withFormat, bool isSql, bool escape) const
337 {
338  if (label == MDL_UNDEFINED) //if undefine label, store as a literal string
339  os << data.stringValue;
340  else
341  switch (MDL::labelType(label))
342  {
343  case LABEL_BOOL: //bools are int in sqlite3
344  os << data.boolValue;
345  break;
346  case LABEL_INT:
347  INT2STREAM(data.intValue);
348  break;
349  case LABEL_SIZET:
350  INT2STREAM(data.longintValue);
351  break;
352  case LABEL_DOUBLE:
353  DOUBLE2STREAM(data.doubleValue);
354  break;
355  case LABEL_STRING:
356  {
357  char c = _SPACE;
358  if (escape)
359  {
360  if (isSql || data.stringValue->find_first_of(_DQUOT) != String::npos)
361  c = _QUOT;
362  else if (data.stringValue->find_first_of(_QUOT) != String::npos)
363  c = _DQUOT;
364  else if (data.stringValue->find_first_of(_SPACE) != String::npos)
365  c = _QUOT;
366  else if (data.stringValue->empty())
367  c = _QUOT;
368  }
369  if (c == _SPACE)
370  os << *(data.stringValue);
371  else
372  os << c << *(data.stringValue) << c;
373  }
374  break;
375  case LABEL_VECTOR_DOUBLE:
376  {
377  std::vector<double> &vectorDouble = *(data.vectorValue);
378  if (escape)
379  os << _QUOT << " ";
380  size_t size = vectorDouble.size();
381  for (size_t i = 0; i < size; i++)
382  {
383  double v = vectorDouble[i];
384  DOUBLE2STREAM(v);
385  os << " ";
386  }
387  if (escape)
388  os << _QUOT;
389  }
390  break;
391  case LABEL_VECTOR_SIZET:
392  {
393  std::vector<size_t> &vector = *(data.vectorValueLong);
394  if (escape)
395  os << _QUOT << " ";
396  size_t size = vector.size();
397  for (size_t i = 0; i < size; i++)
398  os << vector[i] << " ";
399  if (escape)
400  os << _QUOT;
401  }
402  break;
403  case LABEL_NOTYPE:
404  if (escape) os << _QUOT;
405  os << "No type";
406  if (escape) os << _QUOT;
407  break;
408  }//close switch
409 }//close function toStream

437 bool MDObject::fromStream(std::istream &is, bool fromString)
438 {
439  if (label == MDL_UNDEFINED) //if undefine label, store as a literal string
440  {
441  String s;
442  is >> s;
443  }
444  else
445  {
446  //NOTE: int, bool and long(size_t) are read as double for compatibility with old doc files
447  double d;
448  size_t value;
449  switch (type)
450  {
451  case LABEL_BOOL: //bools are int in sqlite3
452  is >> d;
453  data.boolValue = (bool) ((int)d);
454  break;
455  case LABEL_INT:
456  is >> d;
457  data.intValue = (int) d;
458  break;
459  case LABEL_SIZET:
460  is >> d;
461  data.longintValue = (size_t) d;
462  break;
463  case LABEL_DOUBLE:
464  is >> data.doubleValue;
465  break;
466  case LABEL_STRING:
467  {
468  data.stringValue->clear();
469  String s;
470  is >> s;
471  char chr = s[0];
472  if (chr == _QUOT || chr == _DQUOT)
473  {
474  s = s.substr(1, s.size() - 1); //remove first char '
475  while (s.find_last_of(chr) == String::npos)
476  {
477  data.stringValue->append(s + " ");
478  is >> s;
479  }
480  s = s.substr(0, s.size() - 1); //remove last char '
481  }
482  data.stringValue->append(s);
483  }
484  break;
485  case LABEL_VECTOR_DOUBLE:
486  if (!fromString)
487  is.ignore(256, _QUOT);
488  //if (data.vectorValue == NULL)
489  // data.vectorValue = new std::vector<double>;
490  data.vectorValue->clear();
491  while (is >> d) //This will stop at ending "]"
492  data.vectorValue->emplace_back(d);
493  if (!fromString)
494  {
495  is.clear(); //this is for clear the fail state after found ']'
496  is.ignore(256, _QUOT); //ignore the ending ']'
497  }
498  break;
499  case LABEL_VECTOR_SIZET:
500  if (!fromString)
501  is.ignore(256, _QUOT);
502  //if (data.vectorValue == NULL)
503  // data.vectorValue = new std::vector<double>;
504  data.vectorValueLong->clear();
505  while (is >> value) //This will stop at ending "]"
506  data.vectorValueLong->emplace_back(value);
507  if (!fromString)
508  {
509  is.clear(); //this is for clear the fail state after found ']'
510  is.ignore(256, _QUOT); //ignore the ending ']'
511  }
512  break;
513  case LABEL_NOTYPE:
514  break;
515  }
516  }
517  return is.good();
518 }

538 bool MDObject::eq(const MDObject &obj, double epsilon) const {
539  // FIXME: allow to compare e.g. int & double & longint
540  if (this->label != obj.label)
541  return false;
542 
543  if (this->type != obj.type)
544  throw std::logic_error("MDObject: cannot compare == objects of different type");
545 
546  switch (this->type) {
547  case LABEL_INT:
548  return this->data.intValue == obj.data.intValue;
549 
550  case LABEL_BOOL:
551  return this->data.boolValue == obj.data.boolValue;
552 
553  case LABEL_SIZET:
554  return this->data.longintValue == obj.data.longintValue;
555 
556  case LABEL_DOUBLE:
557  return std::abs(this->data.doubleValue - obj.data.doubleValue) <= epsilon;
558 
559  case LABEL_STRING:
560  return *(this->data.stringValue) == *(obj.data.stringValue);
561 
562  case LABEL_VECTOR_DOUBLE:
563  if (this->data.vectorValue->size() != obj.data.vectorValue->size())
564  return false;
565  for (size_t i = 0; i < this->data.vectorValue->size(); i++)
566  if (std::abs((*this->data.vectorValue)[i] - (*obj.data.vectorValue)[i]) > epsilon)
567  return false;
568  return true;
569 
570  case LABEL_VECTOR_SIZET:
571  return *(this->data.vectorValueLong) == *(obj.data.vectorValueLong);
572 
573  default:
574  throw std::logic_error("MDObject: unknown data type");
575  };
576 }
