71 void getNeighbours(MetaData &mdIn, const Matrix1D<double> &projectionDir, MetaData &mdNeighbours, double maxDist)
72 {
73  Matrix1D<double> projectionDir2;
74  FileName fnImg;
75  mdNeighbours.clear();
76  MetaData mdAux;
77  size_t refno;
78  double cc;
79  FOR_ALL_OBJECTS_IN_METADATA(mdIn)
80  {
81  double rot, tilt;
82  mdIn.getValue(MDL_ANGLE_ROT,rot,__iter.objId);
83  mdIn.getValue(MDL_ANGLE_TILT,tilt,__iter.objId);
84  mdIn.getValue(MDL_IMAGE_IDX,refno,__iter.objId);
85  mdIn.getValue(MDL_MAXCC,cc,__iter.objId);
86  Euler_direction(rot,tilt,0,projectionDir2);
87 
88  double angle=acos(dotProduct(projectionDir,projectionDir2));
89  if (angle<maxDist)
90  {
91  mdIn.getValue(MDL_IMAGE,fnImg,__iter.objId);
92  size_t id=mdAux.addObject();
93  mdAux.setValue(MDL_IMAGE,fnImg,id);
94  mdAux.setValue(MDL_IMAGE_IDX,refno,id);
95  mdAux.setValue(MDL_MAXCC,cc,id);
96  }
97  }
98  mdNeighbours.removeDuplicates(mdAux,MDL_IMAGE_IDX);
99  std::vector<MDLabel> groupBy;
100  groupBy.push_back(MDL_IMAGE_IDX);
101  groupBy.push_back(MDL_IMAGE);
102  if (mdAux.size()>0)
103  mdNeighbours.aggregateGroupBy(mdAux,AGGR_MAX,groupBy,MDL_MAXCC,MDL_MAXCC);
104 }

106 void analyzeNeighbours(MetaData &mdNeighbours, const FileName &fnRef)
107 {
108  std::vector<double> cc;
109  mdNeighbours.getColumnValues(MDL_MAXCC,cc);
110  std::sort(cc.begin(),cc.end());
111 
112  double ccMedian=cc[cc.size()/2];
113  FOR_ALL_OBJECTS_IN_METADATA(mdNeighbours)
114  {
115  double cci;
116  mdNeighbours.getValue(MDL_MAXCC,cci,__iter.objId);
117  if (cci>ccMedian)
118  mdNeighbours.setValue(MDL_COST,1.0,__iter.objId);
119  else
120  mdNeighbours.setValue(MDL_COST,-1.0,__iter.objId);
121  }
122 }
