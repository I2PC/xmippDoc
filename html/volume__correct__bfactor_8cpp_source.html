143 void ProgVolumeCorrectBfactor::make_guinier_plot(MultidimArray< std::complex< double > > &FT1,
144  std::vector<fit_point2D> &guinier)
145 {
146  MultidimArray< int > radial_count(xsize);
147  MultidimArray<double> lnF(xsize);
148  Matrix1D<double> f(3);
149  fit_point2D onepoint;
150 
151  lnF.initZeros();
152  for (size_t k=0; k<ZSIZE(FT1); k++)
153  {
154  FFT_IDX2DIGFREQ(k,ZSIZE(FT1),ZZ(f));
155  double z2=ZZ(f)*ZZ(f);
156  for (size_t i=0; i<YSIZE(FT1); i++)
157  {
158  FFT_IDX2DIGFREQ(i,YSIZE(FT1),YY(f));
159  double y2z2=z2+YY(f)*YY(f);
160  for (size_t j=0; j<XSIZE(FT1); j++)
161  {
162  FFT_IDX2DIGFREQ(j,xsize,XX(f));
163  double R2=y2z2+XX(f)*XX(f);
164  if (R2>0.25)
165  continue;
166  double R=sqrt(R2);
167  int idx=ROUND(R*xsize);
168  A1D_ELEM(lnF,idx) += abs(dAkij(FT1, k, i, j));
169  ++A1D_ELEM(radial_count,idx);
170  }
171  }
172  }
173 
174  guinier.clear();
175  for (size_t i = 0; i < XSIZE(radial_count); i++)
176  {
177  double res = (xsize * sampling_rate)/(double)i;
178  if (res >= apply_maxres)
179  {
180  onepoint.x = 1. / (res * res);
181  if (lnF(i)>0.)
182  {
183  onepoint.y = log ( lnF(i) / radial_count(i) );
184  if (res <= fit_minres && res >= fit_maxres)
185  {
186  onepoint.w = 1.;
187  }
188  else
189  {
190  onepoint.w = 0.;
191  }
192  }
193  else
194  {
195  onepoint.y = 0.;
196  onepoint.w = 0.;
197  }
198  guinier.push_back(onepoint);
199  }
200  }
201 }

279 void ProgVolumeCorrectBfactor::apply_bfactor(MultidimArray< std::complex< double > > &FT1,
280  double bfactor)
281 {
282  Matrix1D<double> f(3);
283  double isampling_rate=1.0/sampling_rate;
284  FOR_ALL_DIRECT_ELEMENTS_IN_ARRAY3D(FT1)
285  {
286  FFT_IDX2DIGFREQ(j,xsize,XX(f));
287  FFT_IDX2DIGFREQ(i,YSIZE(FT1),YY(f));
288  FFT_IDX2DIGFREQ(k,ZSIZE(FT1),ZZ(f));
289  double R = f.module() * isampling_rate;
290  if (1./R >= apply_maxres)
291  {
292  dAkij(FT1, k, i, j) *= exp( -0.25* bfactor * R * R);
293  }
294  }

298  std::vector<fit_point2D> &guinier_diff)
299 {

302  {

344 void ProgVolumeCorrectBfactor::bfactor_correction(MultidimArray< double > &m1,
345  const FileName &fn_guinier)
346 {
347  MultidimArray< std::complex< double > > FT1, FT2;
348  FourierTransformer transformer;
349  double slope;
350  double intercept = 0.;
351  std::vector<fit_point2D> guinierin, guinierweighted, guinierref, guinierdiff;
352  std::vector<double> snr;
353 
354  // Transform
355  xsize=XSIZE(m1);
356  transformer.FourierTransform(m1, FT1, true);
357  make_guinier_plot(FT1,guinierin);
358 
359  // Get SNR weights and apply to guinier1
360  if (fn_fsc != "")
361  {
362  get_snr_weights(snr);
363  apply_snr_weights(FT1,snr);
364  make_guinier_plot(FT1,guinierweighted);
365  }
366  else
367  guinierweighted=guinierin;
368 
369  if (mode == BFACTOR_AUTO)
370  {
371  least_squares_line_fit(guinierweighted, slope, intercept);
372  std::cerr<<" Fitted slope= "<<slope<<" intercept= "<<intercept<<std::endl;
373  adhocB = 4. * slope;
374  }
375  else if (mode == BFACTOR_REF || mode == ALLPOINTS_REF)
376  {
377  Image<double> ref;
378  ref.read(fn_ref);
379  ref().setXmippOrigin();
380  transformer.FourierTransform(ref(), FT2, true);
381  make_guinier_plot(FT2,guinierref);
382  guinierdiff = guinierweighted;
383  for (size_t i = 0; i < guinierdiff.size(); i++)
384  {
385  (guinierdiff[i]).y -= (guinierref[i]).y;
386  }
387  if (mode == BFACTOR_REF)
388  {
389  least_squares_line_fit(guinierdiff, slope, intercept);
390  std::cerr<<" Fitted slope= "<<slope<<" intercept= "<<intercept<<std::endl;
391  adhocB = 4. * slope;
392  }
393  }
394 
395  if (mode == ALLPOINTS_REF)
396  {
397  // Now apply the allpoints correction
398  std::cerr<<"Adjust power spectrum to that of reference "<<std::endl;
399  apply_allpoints(FT1,guinierdiff);
400  }
401  else
402  {
403  // Now apply the B-factor
404  std::cerr<<"Applying B-factor of "<< adhocB << " squared Angstroms"<<std::endl;
405  apply_bfactor(FT1,adhocB);
406  }
407  // Now backtransform
408  transformer.inverseFourierTransform(FT1,m1);
409 
410  // Write Guinier-plot
411  std::vector<fit_point2D> guiniernew;
412  make_guinier_plot(FT1,guiniernew);
413  write_guinierfile(fn_guinier, guinierin, guinierweighted, guiniernew, intercept, guinierref);
414 }
