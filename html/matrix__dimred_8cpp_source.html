109 // Produce Side info ====================================================================
110 void ProgDimRed::produceSideInfo()
111 {
112  if (dimRefMethod=="PCA")
113  {
114  algorithm=&algorithmPCA;
115  } else if (dimRefMethod=="LTSA")
116  {
117  algorithm=&algorithmLTSA;
118  algorithmLTSA.setSpecificParameters(kNN);
119  } else if (dimRefMethod=="DM")
120  {
121  algorithm=&algorithmDiffusionMaps;
122  algorithmDiffusionMaps.setSpecificParameters(t,sigma);
123  } else if (dimRefMethod=="LLTSA")
124  {
125  algorithm=&algorithmLLTSA;
126  algorithmLLTSA.setSpecificParameters(kNN);
127  } else if (dimRefMethod=="LPP")
128  {
129  algorithm=&algorithmLPP;
130  algorithmLPP.setSpecificParameters(kNN,sigma);
131  } else if (dimRefMethod=="kPCA")
132  {
133  algorithm=&algorithmKernelPCA;
134  algorithmKernelPCA.setSpecificParameters(sigma);
135  } else if (dimRefMethod=="pPCA")
136  {
137  algorithm=&algorithmProbabilisticPCA;
138  algorithmProbabilisticPCA.setSpecificParameters(Niter);
139  } else if (dimRefMethod=="LE")
140  {
141  algorithm=&algorithmLaplacianEigenmap;
142  algorithmLaplacianEigenmap.setSpecificParameters(sigma,kNN);
143  } else if (dimRefMethod=="HLLE")
144  {
145  algorithm=&algorithmHessianLLE;
146  algorithmHessianLLE.setSpecificParameters(kNN);
147  } else if (dimRefMethod=="NPE")
148  {
149  algorithm=&algorithmNPE;
150  algorithmNPE.setSpecificParameters(kNN);
151  } else if (dimRefMethod=="SPE")
152  {
153  algorithm=&algorithmSPE;
154  algorithmSPE.setSpecificParameters(kNN,global);
155  }
156 
157  algorithm->setOutputDimensionality(outputDim);
158  algorithm->fnMapping=fnMapping;
159 }
