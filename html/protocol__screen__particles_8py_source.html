431 def histThresholding(valuesList, nBins=256, portion=4, takeNegatives=False):
432  """ returns the threshold to reject those values above a portionth of
433  the peak. i.e: if portion is 4, the threshold correponds to the
434  4th of the peak (in the right part).
435  """
436  if not takeNegatives:
437  # take only the positive values, negative are considered corrupted
438  valuesList = [x for x in valuesList if not x < 0]
439 
440  import numpy as np
441  while len(valuesList)*1.0/nBins < 5:
442  nBins = int(nBins/2)
443 
444  print('Thresholding with %d bins for the histogram.' % nBins)
445 
446  hist, bin_edges = np.histogram(valuesList, bins=nBins)
447 
448  histRight = hist
449  histRight[0:hist.argmax()] = 0
450 
451  idx = (np.abs(histRight-hist.max()/portion)).argmin()
452  return bin_edges[idx]

454 def rejectByVariance(inputMdFn, outputMdFn, threshold, mode):
455  """ Sets MDL_ENABLED to -1 to those items with a higher value
456  than the threshold
457  """
458  mdata = emlib.MetaData(inputMdFn)
459  for objId in mdata:
460  if mode == XmippProtScreenParticles.REJ_VARIANCE:
461  if mdata.getValue(emlib.MDL_SCORE_BY_VAR, objId) > threshold:
462  mdata.setValue(emlib.MDL_ENABLED, -1, objId)
463  elif mode == XmippProtScreenParticles.REJ_VARGINI:
464  if (mdata.getValue(emlib.MDL_SCORE_BY_VAR, objId) *
465  (1 - mdata.getValue(emlib.MDL_SCORE_BY_GINI, objId)) > threshold):
466  mdata.setValue(emlib.MDL_ENABLED, -1, objId)
467 
468  mdata.write(outputMdFn)
