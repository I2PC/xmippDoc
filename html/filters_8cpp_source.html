44  A.initZeros();
45  b.initZeros();
46  FOR_ALL_ELEMENTS_IN_ARRAY2D(I)
47  {
48  A(0, 0) += j * j;
49  A(0, 1) += j * i;
50  A(0, 2) += j;
51  A(1, 1) += i * i;
52  A(1, 2) += i;
53  A(2, 2) += 1;
54  b(0) += j * A2D_ELEM(I, i, j);

56  b(2) += A2D_ELEM(I, i, j);
57  }
58  A(1, 0) = A(0, 1);

1787  k_actual = k + kmax;
1788  for (int i = -n_max; i <= n_max; i++)
1789  {
1790  i_actual = i + imax;
1791  for (int j = -n_max; j <= n_max; j++)
