158 {
159  FileNameObject *self = (FileNameObject*) obj;
160  PyObject* pyStr1 = NULL, *pyStr2 = NULL, *pyStr3 = NULL;

162  if (self != NULL)
163  {
164  PyObject *input = NULL, *pyStr = NULL;
165  PyObject *input2 = NULL;
166  char str[1024] = "";
167  char * ext = NULL;
168  char str2[1024] = "";
169  int number = -1;
170  size_t n = PyTuple_Size(args);
171  //"kk000001.xmp"
172  if (n == 3 && PyArg_ParseTuple(args, "Ois", &input, &number, &ext))
173  {
174  pyStr = PyObject_Str(input);
175  if (pyStr != NULL)
176  strcpy(str, PyUnicode_AsUTF8(pyStr));
177  self->filename->compose(str, number, ext);
178  }
179  else if (n == 2 && PyArg_ParseTuple(args, "OO", &input, &input2))
180  {
181  if( PyUnicode_Check( input ) )
182  {
183  //"jj@kk.xmp"
184  pyStr = PyObject_Str(input);
185  pyStr2 = PyObject_Str(input2);
186  if (pyStr != NULL)
187  strcpy(str, PyUnicode_AsUTF8(pyStr));
188  if (pyStr2 != NULL)
189  strcpy(str2, PyUnicode_AsUTF8(pyStr2));
190  self->filename->compose(str, str2);
191  }
192  else if ( PyLong_Check( input ) )
193  {
194  //"1@kk.xmp"
195  number=PyLong_AsLong(input);
196  pyStr2 = PyObject_Str(input2);
197  strcpy(str2, PyUnicode_AsUTF8(pyStr2));
198  self->filename->compose(number, str2);
199  }
200  else
201  return NULL;
202  }
