151 void ProgLocSharpening::computeAvgStdev_within_binary_mask(const MultidimArray< double >&resVol,
152  const MultidimArray< double >&vol, double &stddev, bool outside )
153 {
154 
155  SPEED_UP_tempsInt;
156  double sum1 = 0;
157  double sum2 = 0;
158  int N = 0;
159  double avg=0;
160 
161  FOR_ALL_ELEMENTS_IN_COMMON_IN_ARRAY3D(resVol, vol)
162  {
163  if ((not outside && A3D_ELEM(resVol, k, i, j) > 2*sampling) ||
164  (outside && A3D_ELEM(resVol, k, i, j) < 2*sampling))
165  {
166  ++N;
167  double aux=A3D_ELEM(vol, k, i, j);
168  sum1 += aux;
169  sum2 += aux*aux;
170  }
171  }
172 
173  // average and standard deviation
174  avg = sum1 / (double) N;
175  if (N > 1)
176  stddev = sqrt(fabs(sum2 / N - avg * avg) * N / (N - 1));
177  else
178  stddev = 0;
179 }
