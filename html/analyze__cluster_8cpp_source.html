82 // Produce side info ======================================================
83 //#define DEBUG
84 void ProgAnalyzeCluster::produceSideInfo(MDLabel image_label)
85 {
86  basis = fnOutBasis!="";
87 
88  // Read input selfile and reference
89  if (SFin.size()==0)
90  {
91  SFin.read(fnSel);
92  if (SFin.size()==0)
93  return;
94  if (SFin.containsLabel(MDL_ENABLED))
95  SFin.removeObjects(MDValueEQ(MDL_ENABLED, -1));
96  MetaData SFaux;
97  SFaux.removeDuplicates(SFin,MDL_IMAGE);
98  SFin=SFaux;
99  }
100  bool subtractRef=false;
101  Image<double> Iref;
102  if (fnRef!="")
103  {
104  subtractRef=true;
105  Iref.read(fnRef);
106  Iref().setXmippOrigin();
107  }
108 
109  // Prepare mask
110  size_t Xdim,Ydim,Zdim, Ndim;
111  getImageSize(SFin,Xdim,Ydim,Zdim,Ndim,image_label);
112  mask.resize(Ydim,Xdim);
113  mask.setXmippOrigin();
114  if (dontMask)
115  mask.initConstant(1);
116  else
117  BinaryCircularMask(mask,Xdim/2, INNER_MASK);
118  int Npixels=(int)mask.sum();
119 
120  // Read all images in the class and subtract the mean
121  // once aligned
122  Image<double> I, Iaux;
123  pcaAnalyzer.clear();
124  pcaAnalyzer.reserve(SFin.size());
125  if (verbose>0)
126  {
127  std::cerr << "Processing cluster ...\n";
128  init_progress_bar(SFin.size());
129  }
130  int n=0;
131  MultidimArray<float> v(Npixels);
132  MultidimArray<double> &mIref=Iref(), Ialigned, ImirrorAligned;
133  Matrix2D<double> M;
134  AlignmentAux aux;
135  CorrelationAux aux2;
136  RotationalCorrelationAux aux3;
137  FileName fnImg;
138  FOR_ALL_OBJECTS_IN_METADATA(SFin)
139  {
140  SFin.getValue(image_label,fnImg,__iter.objId);
141  I.readApplyGeo( fnImg, SFin, __iter.objId );
142  if (XSIZE(I())!=Xdim || YSIZE(I())!=Ydim)
143  REPORT_ERROR(ERR_MULTIDIM_SIZE,"All images must be of the same size");
144  I().setXmippOrigin();
145  int idx=0;
146  if (subtractRef)
147  {
148  // Choose between this image and its mirror
149  Ialigned=I();
150  ImirrorAligned=Ialigned;
151  ImirrorAligned.selfReverseX();
152  ImirrorAligned.setXmippOrigin();
153  alignImages(mIref,Ialigned,M,WRAP,aux,aux2,aux3);
154  alignImages(mIref,ImirrorAligned,M,WRAP,aux,aux2,aux3);
155  double corr=correlationIndex(mIref,Ialigned,&mask);
156  double corrMirror=correlationIndex(mIref,ImirrorAligned,&mask);
157  if (corr>corrMirror)
158  I()=Ialigned;
159  else
160  I()=ImirrorAligned;
161 
162  FOR_ALL_ELEMENTS_IN_ARRAY2D(mask)
163  if (A2D_ELEM(mask,i,j))
164  A1D_ELEM(v,idx++)=IMGPIXEL(I,i,j)-A2D_ELEM(mIref,i,j);
165  }
166  else
167  {
168  FOR_ALL_ELEMENTS_IN_ARRAY2D(mask)
169  if (A2D_ELEM(mask,i,j))
170  A1D_ELEM(v,idx++)=IMGPIXEL(I,i,j);
171  }
172  pcaAnalyzer.addVector(v);
173  if ((++n)%10==0 && verbose>0)
174  progress_bar(n);
175  }
176  if (verbose>0)
177  progress_bar(SFin.size());
178 }

181 // Produce basis ==========================================================
182 void ProgAnalyzeCluster::produceBasis(MultidimArray<double> &basis)
183 {
184  int iimax=pcaAnalyzer.PCAbasis.size();
185  basis.initZeros(iimax,1,YSIZE(mask),XSIZE(mask));
186  for (size_t ii=0; ii<pcaAnalyzer.PCAbasis.size(); ii++)
187  {
188  int idx=0;
189  double *ptrBasis=&DIRECT_NZYX_ELEM(basis, ii, 0, 0, 0);
190  const MultidimArray<double> &PCAbasis_ii=pcaAnalyzer.PCAbasis[ii];
191  FOR_ALL_DIRECT_ELEMENTS_IN_ARRAY2D(mask)
192  if (DIRECT_A2D_ELEM(mask,i,j))
193  ptrBasis[i*XSIZE(mask)+j]=A1D_ELEM(PCAbasis_ii,idx++);
194  }
195 }

197 // Run ====================================================================
198 void ProgAnalyzeCluster::run()
199 {
200  show();
201  produceSideInfo();
202  size_t N=SFin.size();
203 
204  // Output
205  SFout=SFin;
206  if (N>Nmin)
207  {
208  pcaAnalyzer.evaluateZScore(NPCA, Niter);
209 
210  for (size_t n=0; n<N; n++)
211  {
212  int trueIdx=pcaAnalyzer.getSorted(n);
213  double zscore=pcaAnalyzer.getSortedZscore(n);
214  SFout.setValue(MDL_ZSCORE, zscore, trueIdx+1);
215  if (zscore<distThreshold || distThreshold<0)
216  SFout.setValue(MDL_ENABLED,1, trueIdx+1);
217  else
218  SFout.setValue(MDL_ENABLED,-1, trueIdx+1);
219  }
220  }
221  SFout.write(fnOut,MD_APPEND);
222  if (basis && N>Nmin)
223  {
224  fnOutBasis.deleteFile();
225  Image<double> basis;
226  produceBasis(basis());
227  basis.write(fnOutBasis);
228  }
229 }
