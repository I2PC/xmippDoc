58 void minusAdjusted(Image<double> &op1, const Image<double> &op2)

64  prm.I1=&op1();
65  prm.I2=&op2();
66 
67  Matrix1D<double> p(2), steps(2);
68  p(0)=1; // a in I'=a*I+b
69  p(1)=0; // b in I'=a*I+b
70  steps.initConstant(1);
71  double cost;
72  int iter;

76  double b=p(1);
77  FOR_ALL_DIRECT_ELEMENTS_IN_MULTIDIMARRAY(pI1)
78  DIRECT_MULTIDIM_ELEM(pI1,n)-=(a*DIRECT_MULTIDIM_ELEM(pI2,n)+b);
