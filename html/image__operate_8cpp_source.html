60 void minusAdjusted(Image<double> &op1, const Image<double> &op2)
61 {
62  MultidimArray<double> &pI1=op1();
63  const MultidimArray<double> &pI2=op2();
64 
65  MinusAdjustedPrm prm;
66  prm.I1=&op1();
67  prm.I2=&op2();
68 
69  Matrix1D<double> p(2);
70  Matrix1D<double> steps(2);
71  p(0)=1; // a in I'=a*I+b
72  p(1)=0; // b in I'=a*I+b
73  steps.initConstant(1);
74  double cost;
75  int iter;
76  powellOptimizer(p, 1, 2, &minusAdjusted_L1, &prm, 0.01, cost, iter, steps, false);
77 
78  double a=p(0);
79  double b=p(1);
80  FOR_ALL_DIRECT_ELEMENTS_IN_MULTIDIMARRAY(pI1)
81  DIRECT_MULTIDIM_ELEM(pI1,n)-=(a*DIRECT_MULTIDIM_ELEM(pI2,n)+b);
82 }
