305  inline double evaluate(double x, double y) const
306  {
307  int SplineDegree_1 = SplineDegree - 1;
308  double x_arg = (x - x0) / h_x;
309  double y_arg = (y - y0) / h_y;
310 
311  int l1 = CLIP(CEIL(x_arg - SplineDegree_1), l0, lF);
312  int l2 = CLIP(l1 + SplineDegree, l0, lF);
313  int m1 = CLIP(CEIL(y_arg - SplineDegree_1), m0, mF);
314  int m2 = CLIP(m1 + SplineDegree, m0, mF);
315  double columns = 0.0;
316  for (int m = m1; m <= m2; m++)
317  {
318  double rows = 0.0;
319  for (int l = l1; l <= l2; l++)
320  {
321  double xminusl = x_arg - (double)l;
322  double Coeff = c_ml(m, l);
323  switch (SplineDegree)
324  {
325  case 2:
326  rows += Coeff * Bspline02(xminusl);
327  break;
328  case 3:
329  rows += Coeff * Bspline03(xminusl);
330  break;
331  case 4:
332  rows += Coeff * Bspline04(xminusl);
333  break;
334  case 5:
335  rows += Coeff * Bspline05(xminusl);
336  break;
337  case 6:
338  rows += Coeff * Bspline06(xminusl);
339  break;
340  case 7:
341  rows += Coeff * Bspline07(xminusl);
342  break;
343  case 8:
344  rows += Coeff * Bspline08(xminusl);
345  break;
346  case 9:
347  rows += Coeff * Bspline09(xminusl);
348  break;
349  }
350  }
351 
352  double yminusm = y_arg - (double)m;
353  switch (SplineDegree)
354  {
355  case 2:
356  columns += rows * Bspline02(yminusm);
357  break;
358  case 3:
