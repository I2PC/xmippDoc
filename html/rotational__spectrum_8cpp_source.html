47 // Interpolate image value -------------------------------------------------
48 double Cylindrical_Wave_Decomposition::interpolate(
49  MultidimArray<double> &img, double y, double x)
50 {
51  y = y - 1; // Since Fortran starts indexing at 1
52  x = x - 1;
53  int iy = (int)y; /* Trunc the x, y coordinates to int */
54  int ix = (int)x;
55  double scale = y - iy;
56  double *ptr_yx = &DIRECT_A2D_ELEM(img, iy, ix);
57  double *ptr_y1x = ptr_yx + XSIZE(img);
58  double introw1 = *ptr_yx + scale * (*(ptr_yx + 1) - *ptr_yx);
59  double introw2 = *ptr_y1x + scale * (*(ptr_y1x + 1) - *ptr_y1x);
60  return introw1 + (x - ix)*(introw2 - introw1);
61 }

64 void Cylindrical_Wave_Decomposition::compute_cwd(MultidimArray<double> &img)
65 {
66  double coseno[1281], ampcos[5191], ampsen[5191];
67  double ac, as, bc, bs, b1, coefca, coefcb, coefsa, coefsb;
68  double d1, e1, fi, g1, h, hdpi, rh, r11, r, th, ys, x, y, zs, z, ys2;
69  int ir, ind, i1c, i1s, i2c, i2s;
70  int jr, k, kk, ntot, my, my2, my3, my4, my5, i, j, numin1, numax1;
71 
72  rh = XMIPP_MIN(r2, x0 - 4.);
73  rh = XMIPP_MIN(rh, y0 - 4.);
74  rh = XMIPP_MIN(rh, YSIZE(img) - x0 - 3.);
75  rh = XMIPP_MIN(rh, XSIZE(img) - y0 - 3.);
76 
77  ir = (int)((rh - r1) / r3 + 1);
78  ind = 0;
79  numin1 = numin + 1;
80  numax1 = numax + 1;
81 
82  for (kk = numin1;kk <= numax1;kk++)
83  {
84  k = kk - 1;
85  if (k != 0)
86  {
87  my = (int)(1 + PI * rh / 2. / k);
88  my2 = 2 * my;
89  my4 = my * k;
90  my5 = my4 - 1;
91  ntot = 4 * my4;
92  h = 2. * PI / ntot;
93  hdpi = h / PI;
94  th = k * h;
95  ys = sin(th);
96  zs = cos(th);
97  ys2 = sin(2. * th);
98  b1 = 2. / (th * th) * (1. + zs * zs - ys2 / th);
99  g1 = 4. / (th * th) * (ys / th - zs);
100  d1 = 2. * th / 45.;
101  e1 = d1 * ys * 2.;
102  d1 *= ys2;
103  coefca = (b1 + e1) * hdpi;
104  coefcb = (g1 - d1) * hdpi;
105  coefsa = (b1 - e1) * hdpi;
106  coefsb = (g1 + d1) * hdpi;
107  }
108  else
109  {
110  my = (int)(1 + PI * rh / 2.);
111  my2 = 2 * my;
112  my4 = my;
113  my5 = my4 - 1;
114  ntot = 4 * my4;
115  h = 2. * PI / ntot;
116  coefca = h / PI / 2.;
117  }
118  for (i = 1;i <= my5;i++)
119  {
120  fi = i * h;
121  coseno[i] = sin(fi);
122  }
123  coseno[my4] = 1.;
124  my3 = 2 * my4;
125  for (i = 1;i <= my5;i++)
126  {
127  coseno[my3-i] = coseno[i];
128  coseno[my3+i] = -coseno[i];
129  coseno[ntot-i] = -coseno[i];
130  coseno[ntot+i] = coseno[i];
131  }
132  coseno[my3] = 0.;
133  coseno[my3+my4] = -1.;
134  coseno[ntot] = 0.;
135  coseno[ntot+my4] = 1.;
136  r11 = r1 - r3;
137  for (jr = 1;jr <= ir;jr++)
138  {
139  ind++;
140  r = r11 + r3 * jr;
141  ac = 0.;
142  i1c = my4;
143  i1s = 0;
144  if (k != 0)
145  {
146  as = bc = bs = 0.;
147  for (i = 1;i <= k;i++)
148  {
149  i2c = my4;
150  i2s = 0;
151  for (j = 1;j <= my2;j++)
152  {
153  i1c++;
154  i1s++;
155  i2c += k;
156  i2s += k;
157  x = x0 + r * coseno[i1c];
158  y = y0 + r * coseno[i1s];
159  z = interpolate(img, y, x);
160  bc += z * coseno[i2c];
161  bs += z * coseno[i2s];
162  i1c++;
163  i1s++;
164  i2c += k;
165  i2s += k;
166  x = x0 + r * coseno[i1c];
167  y = y0 + r * coseno[i1s];
168  z = interpolate(img, y, x);
169  ac += z * coseno[i2c];
170  as += z * coseno[i2s];
171  }
172  }
173  ampcos[ind] = coefca * ac + coefcb * bc;
174  ampsen[ind] = coefsa * as + coefsb * bs;
175  }
176  else
177  for (j = 1;j <= ntot;j++)
178  {
179  i1c++;
180  i1s++;
181  x = x0 + r * coseno[i1c];
182  y = y0 + r * coseno[i1s];
183  z = interpolate(img, y, x);
184  ac += z;
185  ampcos[ind] = coefca * ac;
186  ampsen[ind] = 0.;
187  }
188  }
189  if (k == 0)
190  {
191  ac = 0.;
192  for (j = 1;j <= ir;j++)
193  {
194  r = r11 + j * r3;
195  ac += ampcos[j] * 2. * PI * r;
196  }
197  ac /= (PI * (r * r - r1 * r1));
198  for (j = 1;j <= ir;j++)
199  ampcos[j] -= ac;
200  }
201  }
202 
203  out_ampcos.initZeros((numax - numin + 1)*ir);
204  out_ampsin = out_ampcos;
205  FOR_ALL_ELEMENTS_IN_ARRAY1D(out_ampcos)
206  {
207  A1D_ELEM(out_ampcos, i) = ampcos[i+1];
208  A1D_ELEM(out_ampsin, i) = ampsen[i+1];
209  }
210 }

226 #define MAX_HARMONIC 51 /* Max. no of harmonics accepted */
227 void Rotational_Spectrum::compute_rotational_spectrum(
228  Cylindrical_Wave_Decomposition &cwd,
229  double xr1, double xr2, double xdr, double xr)
230 {
231  double *e[MAX_HARMONIC], *rv, *st, *ep[MAX_HARMONIC], *erp [MAX_HARMONIC],
232  *rp1, *rp2, *sp, *c, *s;
233  int n, m, i, j1, k, j, ir1, ir2, ndr, nr, ncol, nvez,
234  irk, k1;
235 
236  // Read the information from the Cylindrical Wave Decomposition .........
237  c = (double *) calloc(5191, sizeof(double));
238  s = (double *) calloc(5191, sizeof(double));
239  if ((NULL == c) || (NULL == s))
240  REPORT_ERROR(ERR_MEM_NOTENOUGH, "compute_rotational_spectrum::no memory");
241 
242  int numin = cwd.numin;
243  int numax = cwd.numax;
244  double rl = cwd.r1;
245  double rh = cwd.r2;
246  double dr = cwd.r3;
247  FOR_ALL_ELEMENTS_IN_ARRAY1D(cwd.out_ampcos)
248  {
249  c[i+1] = A1D_ELEM(cwd.out_ampcos, i);
250  s[i+1] = A1D_ELEM(cwd.out_ampsin, i);
251  }
252 
253  n = numax - numin + 1;
254  m = (int)((rh - rl) / dr + 1);
255  for (i = 1; i <= n; i++) {
256  e[i] = (double *) calloc(m + 1, sizeof(double));
257  if (NULL == e[i])
258  REPORT_ERROR(ERR_MEM_NOTENOUGH, "compute_rotational_spectrum::no memory");
259  }
260  rv = (double *) calloc(m + 1, sizeof(double));
261  st = (double *) calloc(m + 1, sizeof(double));
262  if ((NULL == rv) || (NULL == st))
263  REPORT_ERROR(ERR_MEM_NOTENOUGH, "compute_rotational_spectrum::no memory");
264 
265  // Computations .........................................................
266  if (numin == 0)
267  j1 = 2;
268  else
269  j1 = 1;
270  k = 0;
271  for (i = 1; i <= n; i++)
272  for (j = 1; j <= m; j++)
273  {
274  k++;
275  e[i][j] = c[k] * c[k] + s[k] * s[k];
276  }
277 
278  for (i = 1; i <= m; i++)
279  {
280  rv[i] = rl + dr * (i - 1);
281  st[i] = 0;
282  for (j = j1; j <= n; j++)
283  st[i] += e[j][i];
284  }
285 
286  ir1 = (int)((xr1 - rl) / dr + 1);
287  if (ir1 < 1)
288  ir1 = 1;
289  ir2 = (int)((XMIPP_MIN(xr2, rh) - rl) / dr + 1);
290  if (ir2 < ir1)
291  ir2 = ir1;
292  ndr = (int)(xdr / dr);
293  if (ndr < 1)
294  ndr = 1;
295  if (xr < 0)
296  nr = m;
297  else
298  nr = (int)(xr / dr + 1);
299  ir2 = XMIPP_MIN(ir2, m);
300  ncol = ir2 - nr + 1 - ir1;
301  if (ncol < 0)
302  ncol = 0;
303  else
304  ncol = 1 + ncol / ndr;
305  if (ncol == 0)
306  REPORT_ERROR(ERR_VALUE_INCORRECT, "compute_rotational_spectrum::Incorrect data");
307 
308  nvez = (ncol - 1) / 13 + 1;
309  for (i = 1; i <= n; i++) {
310  ep[i] = (double *) calloc(ncol + 1, sizeof(double));
311  erp[i] = (double *) calloc(ncol + 1, sizeof(double));
312  if ((NULL == ep[i]) || (NULL == erp[i]))
313  REPORT_ERROR(ERR_MEM_NOTENOUGH, "compute_rotational_spectrum::no memory");
314  }
315  rp1 = (double *) calloc(ncol + 1, sizeof(double));
316  rp2 = (double *) calloc(ncol + 1, sizeof(double));
317  sp = (double *) calloc(ncol + 1, sizeof(double));
318  if ((NULL == rp1) || (NULL == rp2) || (NULL == sp))
319  REPORT_ERROR(ERR_MEM_NOTENOUGH, "compute_rotational_spectrum::no memory");
320  for (k = 1; k <= ncol; k++)
321  {
322  irk = ir1 + (k - 1) * ndr - 1;
323  rp1[k] = 10 * rv[irk+1];
324  rp2[k] = 10 * rv[irk+nr];
325  sp[k] = 0;
326  for (k1 = 1; k1 <= nr; k1++)
327  sp[k] += st[irk+k1];
328  for (i = 1; i <= n; i++)
329  {
330  ep[i][k] = 0;
331  for (k1 = 1; k1 <= nr; k1 ++)
332  ep [i][k] += e[i][irk+k1];
333  erp[i][k] = 1000000. * ep[i][k] / sp[k];
334  }
335  }
336 
337  // Keep results .........................................................
338  rot_spectrum.initZeros(n - j1 + 1);
339  for (k = 1; k <= nvez; k++)
340  {
341  k1 = 13 * (k - 1) + 1;
342  for (i = j1; i <= n; i++)
343  rot_spectrum(i - j1) = erp[i][k1] / 10000;
344  }
345 
346  // Free memory
347  for (i = 1; i <= n; i++)
348  {
349  free((char *) ep[i]);
350  free((char *) erp [i]);
351  }
352  free((char *) c);
353  free((char *) s);
354  free((char *) rv);
355  free((char *) st);
356  free((char *) rp1);
357  free((char *) rp2);
358  free((char *) sp);
359 }
