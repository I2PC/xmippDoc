92 template<bool IS_REF>
93 void AProgAlignSignificant<T>::load(DataHelper &h) {
94  auto &md = h.md;
95  md.read(h.fn);
96  size_t origN = md.size();
97  md.removeDisabled();
98 
99  size_t Xdim;
100  size_t Ydim;
101  size_t Zdim;
102  size_t Ndim;
103  getImageSize(h.fn, Xdim, Ydim, Zdim, Ndim);
104  Ndim = md.size();
105 
106  if (IS_REF && (origN != Ndim)) {
107  std::cerr << h.fn << " contains disabled images. This is not expected and might lead to wrong result\n";
108  }
109 
110  auto dims = Dimensions(Xdim, Ydim, Zdim, Ndim);
111  auto dimsCropped = Dimensions((Xdim / 2) * 2, (Ydim / 2) * 2, Zdim, Ndim);
112  bool mustCrop = (dims != dimsCropped);
113  h.dims = dimsCropped;
114 
115  // FIXME DS clean up the cropping routine somehow
116  h.data = std::unique_ptr<T[]>(new T[dims.size()]);
117  auto ptr = h.data.get();
118  // routine loading the actual content of the images
119  auto routine = [&dims, ptr]
120  (int thrId, const FileName &fn, size_t storeIndex) {
121  size_t offset = storeIndex * dims.sizeSingle();
122  MultidimArray<T> wrapper(1, dims.z(), dims.y(), dims.x(), ptr + offset);
123  auto img = Image<T>(wrapper);
124  img.read(fn);
125  };
126 
127  std::vector<Image<T>> tmpImages;
128  tmpImages.reserve(m_threadPool.size());
129  if (mustCrop) {
130  std::cerr << "We need an even input (sizes must be multiple of two). Input will be cropped\n";
131  for (size_t t = 0; t < m_threadPool.size(); ++t) {
132  tmpImages.emplace_back(Xdim, Ydim);
133  }
134  }
135  // routine loading the actual content of the images
136  auto routineCrop = [&dims, &dimsCropped, ptr, &tmpImages]
137  (int thrId, const FileName &fn, size_t storeIndex) {
138  // load image
139  tmpImages.at(thrId).read(fn);
140  // copy just the part we're interested in
141  const size_t destOffsetN = storeIndex * dimsCropped.sizeSingle();
142  for (size_t y = 0; y < dimsCropped.y(); ++y) {
143  size_t srcOffsetY = y * dims.x();
144  size_t destOffsetY = y * dimsCropped.x();
145  memcpy(ptr + destOffsetN + destOffsetY,
146  tmpImages.at(thrId).data.data + srcOffsetY,
147  dimsCropped.x() * sizeof(T));
148  }
149  };
150 
151  validate(h, IS_REF);
152 
153  // load all images in parallel
154  auto futures = std::vector<std::future<void>>();
155  futures.reserve(Ndim);
156  h.rots.reserve(Ndim);
157  h.tilts.reserve(Ndim);
158  size_t i = 0;
159  FOR_ALL_OBJECTS_IN_METADATA(md) {
160  FileName fn;
161  md.getValue(MDL_IMAGE, fn, __iter.objId);
162  if (IS_REF) {
163  // if these labels are not present, we will use our default values
164  float rot = 0.f;
165  md.getValue(MDL_ANGLE_ROT, rot,__iter.objId);
166  float tilt = 0.f;
167  md.getValue(MDL_ANGLE_TILT, tilt,__iter.objId);
168  // ref label is required, as it allows for manual check of the result
169  // without it, we don't know which reference is which
170  int ref;
171  md.getValue(MDL_REF, ref,__iter.objId);
172  h.rots.emplace_back(rot);
173  h.tilts.emplace_back(tilt);
174  h.indexes.emplace_back(ref);
175  } else {
176  h.rowIds.emplace_back(__iter.objId);
177  }
178  if (mustCrop) {
179  futures.emplace_back(m_threadPool.push(routineCrop, fn, i));
180  } else {
181  futures.emplace_back(m_threadPool.push(routine, fn, i));
182  }
183  i++;
184  }
185  // wait till done
186  for (auto &f : futures) {
187  f.get();
188  }

228 template<typename T>
229 template<bool IS_ESTIMATION_TRANSPOSED>
230 void AProgAlignSignificant<T>::computeWeightsAndSave(
231  const std::vector<AlignmentEstimation> &est,
232  size_t refIndex) {
233  const size_t noOfRefs = m_referenceImages.dims.n();
234  const size_t noOfSignals = m_imagesToAlign.dims.n();
235 
236  // compute angle between two reference orientation
237  auto getAngle = [&](size_t index) {
238  return Euler_distanceBetweenAngleSets(
239  m_referenceImages.rots.at(refIndex),
240  m_referenceImages.tilts.at(refIndex),
241  0.f,
242  m_referenceImages.rots.at(index),
243  m_referenceImages.tilts.at(index),
244  0.f,
245  true);
246  };
247 
248  // find out which references are sufficiently similar
249  size_t count = 0;
250  auto mask = std::vector<bool>(noOfRefs, false);
251  for (size_t r = 0; r < noOfRefs; ++r) {
252  if ((refIndex == r)
253  || (getAngle(r) <= m_angDistance)) {
254  mask.at(r) = true;
255  count++;
256  }
257  }
258 
259  // allocate necessary memory
260  auto figsOfMerit = std::vector<WeightCompHelper>();
261  figsOfMerit.reserve(count * noOfSignals);
262 
263  // for all similar references
264  for (size_t r = 0; r < noOfRefs; ++r) {
265  if (mask.at(r)) {
266  // get figure of merit of all signals
267  for (size_t s = 0; s < noOfSignals; ++s) {
268  if (IS_ESTIMATION_TRANSPOSED) {
269  figsOfMerit.emplace_back(est.at(s).figuresOfMerit.at(r), r, s);
270  } else {
271  figsOfMerit.emplace_back(est.at(r).figuresOfMerit.at(s), r, s);
272  }
273  }
274  }
275  }
276  computeWeightsAndSave(figsOfMerit, refIndex);
277 }

279 template<typename T>
280 void AProgAlignSignificant<T>::computeWeightsAndSave(
281  std::vector<WeightCompHelper> &figsOfMerit,
282  size_t refIndex) {
283  const size_t noOfSignals = m_imagesToAlign.dims.n();
284  auto &weights = m_weights.at(refIndex);
285  weights = std::vector<float>(noOfSignals, 0); // zero weight by default
286  const size_t noOfNumbers = figsOfMerit.size();
287 
288  // sort ascending using figure of merit
289  std::sort(figsOfMerit.begin(), figsOfMerit.end(),
290  [](const WeightCompHelper &l, const WeightCompHelper &r) {
291  return l.merit < r.merit;
292  });
293  auto invMaxMerit = 1.f / figsOfMerit.back().merit;
294 
295  // set weight for all images
296  for (size_t c = 0; c < noOfNumbers; ++c) {
297  const auto &tmp = figsOfMerit.at(c);
298  if (tmp.refIndex != refIndex) {
299  continue; // current record is for different reference
300  }
301  // cumulative density function - probability of having smaller value then the rest
302  float cdf = c / (float)(noOfNumbers - 1); // <0..1> // won't work if we have just one reference
303  float merit = tmp.merit;
304  if (merit > 0.f) {
305  weights.at(tmp.imgIndex) = merit * invMaxMerit * cdf;
306  }
307  }
308 }

367 template<typename T>
368 template<bool IS_ESTIMATION_TRANSPOSED, bool USE_WEIGHT>
369 void AProgAlignSignificant<T>::computeAssignment(
370  const std::vector<AlignmentEstimation> &est) {
371  const size_t noOfRefs = m_referenceImages.dims.n();
372  auto accessor = [&](size_t image, size_t reference) {
373  if (USE_WEIGHT) {
374  return m_weights.at(reference).at(image);
375  } else if (IS_ESTIMATION_TRANSPOSED) {
376  return est.at(image).figuresOfMerit.at(reference);
377  } else {
378  return est.at(reference).figuresOfMerit.at(image);
379  }
380  };
381 
382  const size_t noOfImages = m_imagesToAlign.dims.n();
383  m_assignments.reserve(noOfImages * m_noOfBestToKeep);
384  for (size_t i = 0; i < noOfImages; ++i) {
385  // collect voting from all references
386  auto votes = std::vector<float>();
387  votes.reserve(noOfRefs);
388  for (size_t r = 0; r < noOfRefs; ++r) {
389  votes.emplace_back(accessor(i, r));
390  }
391  // for all references that we want to store, starting from the best matching one
392  for (size_t nthBest = 0; nthBest < m_noOfBestToKeep; ++nthBest) {
393  size_t refIndex;
394  float val;
395  // get the max vote
396  extractMax(votes, refIndex, val);
397  if (val <= 0) {
398  continue; // skip saving the particles which have non-positive figure of merit to the reference
399  }
400  const auto &p = IS_ESTIMATION_TRANSPOSED
401  ? (est.at(i).poses.at(refIndex).inv())
402  : (est.at(refIndex).poses.at(i));
403  m_assignments.emplace_back(refIndex, i,
404  m_weights.at(refIndex).at(i), val,
405  p);
406  }
407  }
408 }

538 template<typename T>
539 void AProgAlignSignificant<T>::run() {
540  show();
541  m_threadPool.resize(getSettings().cpuThreads);
542  // load data
543  load<false>(m_imagesToAlign);
544  load<true>(m_referenceImages);
545 
546  bool hasMoreReferences = m_allowDataSwap
547  && (m_referenceImages.dims.n() > m_imagesToAlign.dims.n());
548  if (hasMoreReferences) {
549  std::cerr << "We are swapping reference images and experimental images. "
550  "This will enhance the performance. This might lead to worse results if the experimental "
551  "images are not well centered. Use it with care!\n";
552  std::swap(m_referenceImages, m_imagesToAlign);
553  }
554 
555  // for each reference, get alignment of all images
556  updateSettings();
557  check();
558  auto alignment = align(m_referenceImages.data.get(), m_imagesToAlign.data.get());
559 
560  // process the alignment and store
561  if (hasMoreReferences) {
562  std::swap(m_referenceImages, m_imagesToAlign);
563  computeWeights<true>(alignment);
564  if (m_useWeightInsteadOfCC) {
565  computeAssignment<true, true>(alignment);
566  } else {
567  computeAssignment<true, false>(alignment);
568  }
