67 void ProgPdbNmaDeform::run()
68 {
69  PDBRichPhantom pdb;
70  MetaData modes;
71  pdb.read(fn_pdb);
72  modes.read(fn_nma);
73  modes.removeDisabled();
74  int j=0;
75  FileName fnMode;
76  MultidimArray<double> mode;
77  mode.resizeNoCopy(pdb.getNumberOfAtoms(),3);
78  FOR_ALL_OBJECTS_IN_METADATA(modes)
79  {
80  // Read mode
81  modes.getValue(MDL_NMA_MODEFILE,fnMode,__iter.objId);
82  std::ifstream fhMode;
83  fhMode.open(fnMode.c_str());
84  if (!fhMode)
85  REPORT_ERROR(ERR_IO_NOREAD,fnMode);
86  fhMode >> mode;
87  fhMode.close();
88 
89  // Apply mode
90  double lambda=A1D_ELEM(deformations,j);
91  for (size_t i=0; i<YSIZE(mode); ++i)
92  {
93  RichAtom& atom_i=pdb.atomList[i];
94  atom_i.x+=lambda*DIRECT_A2D_ELEM(mode,i,0);
95  atom_i.y+=lambda*DIRECT_A2D_ELEM(mode,i,1);
96  atom_i.z+=lambda*DIRECT_A2D_ELEM(mode,i,2);
97  }
98  j++;
99  }
100  pdb.write(fn_out);
101 }
