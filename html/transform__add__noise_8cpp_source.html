143  void processImage(const FileName &fnImg, const FileName &fnImgOut, const MDRow &rowIn, MDRow &rowOut)
144  {
145 
146  if (TYPE_POISSON == noise_type) {
147  Image<float> img;
148  img.readApplyGeo(fnImg, rowIn);
149  std::random_device rd;
150  std::mt19937 gen(rd());
151  Image<int> res(img.data.xdim, img.data.ydim, img.data.zdim, img.data.ndim);
152  const size_t count = res.data.nzyxdim;
153  const float gap = param1 - param2;
154  auto dist = std::poisson_distribution<>(0);
155  for (size_t i = 0; i < count; ++i) {
156  float mean = param1 - gap * img.data[i];
157  if (dist.mean() != mean) { // reuse distribution, if possible
158  dist = std::poisson_distribution<>(mean);
159  }
160  res.data[i] = dist(gen);
161  }
162  limit(res);
163  res.write(fnImgOut);
164  } else {
165  Image<double> img;
166  img.readApplyGeo(fnImg, rowIn);
167  img().addNoise(param1, param2, noise_type, df);
168  limit(img);
169  img.write(fnImgOut);
170  }
