72 bool checkImageFileSize(const FileName &name, const ImageInfo &imgInfo, bool error)
73 {
74  FileName ext = name.getExtension();
75  FileName dataFname;
76 
77  if ( ext.contains("hed"))
78  dataFname = name.removeLastExtension().addExtension("img");
79  else if (ext.contains("inf"))
80  dataFname = name.removeLastExtension();
81  else if (ext.contains("tif") || ext.contains("jpg") || ext.contains("hdf") || ext.contains("h5"))
82  return true;
83  else
84  dataFname = name;
85 
86  size_t explodeFactor = 1;
87 
88  if (imgInfo.datatype == DT_UHalfByte){
89  explodeFactor = 2;
90  }
91 
92 
93  size_t expectedSize = imgInfo.adim.nzyxdim*gettypesize(imgInfo.datatype) + imgInfo.offset;
94  size_t actualSize = dataFname.removeAllPrefixes().removeFileFormat().getFileSize();
95  actualSize *= explodeFactor;
96  bool result = (actualSize >= expectedSize);
97 
98  if (error && !result)
99  REPORT_ERROR(ERR_IO_SIZE, formatString("Image Extension: File %s has wrong size.\n"
100  "Expected size (at least) %u bytes. Actual size %u bytes.", name.c_str(), expectedSize, actualSize));
101 
102  return result;
103 }
