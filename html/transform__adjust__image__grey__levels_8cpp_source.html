92 void ProgTransformImageGreyLevels::preProcess()
93 {
94  // Read the reference volume
95  Image<double> V;
96  V.read(fnVol);
97  V().setXmippOrigin();
98  Xdim=XSIZE(V());
99 
100  // Construct mask
101  if (Rmax<0)
102  Rmax=Xdim/2;
103  Mask mask;
104  mask.type = BINARY_CIRCULAR_MASK;
105  mask.mode = INNER_MASK;
106  mask.R1 = Rmax;
107  mask.generate_mask(Xdim,Xdim);
108  mask2D=mask.get_binary_mask();
109  iMask2Dsum=1.0/mask2D.sum();
110 
111  // Construct projector
112  projector = new FourierProjector(V(),pad,Ts/maxResol,BSPLINE3);
113 
114  // Low pass filter
115  filter.FilterBand=LOWPASS;
116  filter.w1=Ts/maxResol;
117  filter.raised_w=0.02;
118 }

120 double transformImageGrayCost(double a, double b, ProgTransformImageGreyLevels *prm)
121 {
122  if (fabs(a-1)>prm->maxA)
123  return 1e38;
124  if (fabs(b)>prm->maxB*prm->Istddev)
125  return 1e38;
126 
127  double cost=0;
128  const MultidimArray<double> &mP=prm->P();
129  const MultidimArray<int> &mMask2D=prm->mask2D;
130  MultidimArray<double> &mIfiltered=prm->Ifiltered();
131  FOR_ALL_DIRECT_ELEMENTS_IN_MULTIDIMARRAY(mMask2D)
132  {
133 // if (DIRECT_MULTIDIM_ELEM(mMask2D,n))
134  {
135  double val=fabs((a*DIRECT_MULTIDIM_ELEM(mP,n)+b)-DIRECT_MULTIDIM_ELEM(mIfiltered,n));
136  cost+=val*val;
137  }
138  }
139  cost*=prm->iMask2Dsum;
140  return cost;
141 }

155 void ProgTransformImageGreyLevels::processImage(const FileName &fnImg, const FileName &fnImgOut, const MDRow &rowIn, MDRow &rowOut)
156 {
157  rowOut=rowIn;
158 
159  // Read input image and initial parameters
160  double rot, tilt, psi;
161  rowIn.getValue(MDL_ANGLE_ROT,rot);
162  rowIn.getValue(MDL_ANGLE_TILT,tilt);
163  rowIn.getValue(MDL_ANGLE_PSI,psi);
164 
165  double olda=1.0, oldb=0.0;
166  if (rowIn.containsLabel(MDL_CONTINUOUS_GRAY_A)){
167  rowIn.getValue(MDL_CONTINUOUS_GRAY_A,olda);
168  rowIn.getValue(MDL_CONTINUOUS_GRAY_B,oldb);
169  }
170 
171  I.read(fnImg);
172  I().setXmippOrigin();
173  Istddev=I().computeStddev();
174 
175  Ifiltered()=I();
176  filter.applyMaskSpace(Ifiltered());
177 
178  projectVolume(*projector, P, (int)XSIZE(I()), (int)XSIZE(I()), rot, tilt, psi);
179 
180 #ifdef DEBUG
181  Image<double> save;
182  save()=P();
183  std::cout << "P: "; P().printStats(); std::cout << std::endl;
184  save.write("PPPprojection.xmp");
185  save()=I();
186  std::cout << "I: "; I().printStats(); std::cout << std::endl;
187  save.write("PPPexperimental.xmp");
188  save()=P()-I();
189  std::cout << "P-I init: "; save().printStats(); std::cout << std::endl;
190  save.write("PPPdiffInit.xmp");
191 #endif
192 
193  Matrix1D<double> pAB(2), steps(2), pBA(2), p(2);
194  steps.initConstant(1);
195 
196  // Optimize
197  double costAB=-1, costBA=-1;
198  int iter;
199  try
200  {
201  costAB=1e38;
202  pAB(0)=olda; // a in I'=a*I+b
203  pAB(1)=oldb; // b in I'=a*I+b
204  powellOptimizer(pAB, 1, 2, &transformImageGrayCostAB, this, 0.01, costAB, iter, steps, verbose>=2);
205  if (costAB>1e30)
206  {
207  rowOut.setValue(MDL_ENABLED,-1);
208  pAB.initZeros();
209  pAB(0)=1; // a in I'=a*I+b
210  pAB(1)=0; // b in I'=a*I+b
211  }
212 
213  costBA=1e38;
214  pBA(0)=oldb; // a in I'=a*I+b
215  pBA(1)=olda; // b in I'=a*I+b
216  powellOptimizer(pBA, 1, 2, &transformImageGrayCostBA, this, 0.01, costBA, iter, steps, verbose>=2);
217  if (costBA>1e30)
218  {
219  rowOut.setValue(MDL_ENABLED,-1);
220  pBA.initZeros();
221  pBA(0)=0; // a in I'=a*I+b
222  pBA(1)=1; // b in I'=a*I+b
223  }
224 
225  // Decide
226  if (costAB<costBA)
227  p=pAB;
228  else
229  {
230  p(0)=pBA(1);
231  p(1)=pBA(0);
232  }
233 
234  // Apply
235  MultidimArray<double> &mI=I();
236  double ia=1.0/p(0);
237  double b=p(1);
238  FOR_ALL_DIRECT_ELEMENTS_IN_MULTIDIMARRAY(mI)
239  {
240  if (DIRECT_MULTIDIM_ELEM(mask2D,n) || true)
241  DIRECT_MULTIDIM_ELEM(mI,n)=ia*(DIRECT_MULTIDIM_ELEM(mI,n)-b);
242  else
243  DIRECT_MULTIDIM_ELEM(mI,n)=0.0;
244  }
245  rowOut.setValue(MDL_IMAGE,fnImgOut);
246  I.write(fnImgOut);
247  }
248  catch (XmippError &XE)
249  {
250  std::cerr << XE << std::endl;
251  std::cerr << "Warning: Cannot refine " << fnImg << std::endl;
252  rowOut.setValue(MDL_ENABLED,-1);
253  }
254  rowOut.setValue(MDL_CONTINUOUS_GRAY_A,p(0));
255  rowOut.setValue(MDL_CONTINUOUS_GRAY_B,p(1));
256 
257 #ifdef DEBUG
258  save()=I();
259  save.write("PPPexperimentalCorrected.xmp");
260  std::cout << "I corrected: "; I().printStats(); std::cout << std::endl;
261  save()=P()-I();
262  std::cout << "P-I final: "; save().printStats(); std::cout << std::endl;
263  save.write("PPPdiff.xmp");
264  std::cout << fnImgOut << " rewritten\n";
265  std::cout << "Press any key" << std::endl;
266  char c; std::cin >> c;
267 #endif
268 }
