702 /* Generate evenly distributed angles ====================================== */
703 void generate_even_angles(int ExtProjs, int Nrottilt, MetaData &DF,
704  const ParametersProjection &prm)
705 {
706  // We will run over the tilt angle in a deterministic way
707  // then for every tilt angle, a rot_step is computed so that
708  // it keeps the same distance in the circle generated by tilt
709  // as the sample distance at the equator (tilt=90).
710  MetaData DFaux;
711  int N = 0;
712  int limit = prm.tilt_range.samples;
713  double rot_step_at_equator = (prm.rot_range.angF - prm.rot_range.ang0) /
714  (double)(Nrot - 1);
715  for (int i = 0; i < limit; i++)
716  {
717  // Compute the corresponding deterministic tilt
718  double tilt = prm.tilt_range.ang0 +
719  (prm.tilt_range.angF - prm.tilt_range.ang0) /
720  (double)(Ntilt - 1) * i;
721  // Now compute the corresponding rotational angles
722  double rot_step;
723  if (tilt != 0 && tilt != 180)
724  rot_step = rot_step_at_equator / sin(DEG2RAD(tilt));
725  else
726  rot_step = prm.rot_range.angF - prm.rot_range.ang0 + 1;
727  for (double rot = prm.rot_range.ang0; rot <= prm.rot_range.angF; rot += rot_step)
728  {
729  // Copy this angle to those projections belonging to this group .....
730  // If there is any group
731  for (int k = 0; k < Npsi; k++)
732  {
733  // Select psi
734  double psi;
735  if (prm.psi_range.randomness == ANGLE_RANGE_DETERMINISTIC)
736  {
737  if (prm.psi_range.samples > 1)
738  {
739  psi = prm.psi_range.ang0 +
740  (prm.psi_range.angF - prm.psi_range.ang0) /
741  (double)(prm.psi_range.samples - 1) * k;
742  }
743  else
744  psi = prm.psi_range.ang0;
745  }
746  else
747  psi = prm.psi_range.ang0 +
748  (prm.psi_range.angF - prm.psi_range.ang0) /
749  (double)(Npsi - 1) * k;
750 
751  size_t iproj = ExtProjs + N + Nrottilt * k;
752  size_t idx_tmp=DFaux.firstObject(MDValueEQ(MDL_ORDER,iproj));
753  if (idx_tmp==BAD_OBJID)
754  {
755  idx_tmp=DFaux.addObject();
756  DFaux.setValue(MDL_ORDER,iproj,idx_tmp);
757  }
758  DFaux.setValue(MDL_ANGLE_ROT,rot,idx_tmp);
759  DFaux.setValue(MDL_ANGLE_TILT,tilt,idx_tmp);
760  DFaux.setValue(MDL_ANGLE_PSI,psi,idx_tmp);
761  }
762  N++;
763  }
764  }
765  DF.sort(DFaux,MDL_ORDER);
766  DF.removeLabel(MDL_ORDER);
767 }

795 /* Assign angles =========================================================== */
796 int Assign_angles(MetaData &DF, const ParametersProjection &prm,
797  const FileName &fn_sym)
798 {
799  int ExtProjs = 0, IntProjs = 0; // External and internal projections
800  int Nrottilt=0; // Number of evenly distributed
801 
802  // External generation mode
803  if (prm.fn_angle != "NULL")
804  {
805  DF.read(prm.fn_angle);
806  ExtProjs = DF.size();
807  }
808 
809  // Internal generation mode
810  if (prm.enable_angle_range)
811  {
812  randomize_random_generator();
813  if (prm.rot_range.randomness != ANGLE_EVENLY)
814  IntProjs = Nrot * Ntilt * Npsi;
815  else
816  {
817  if (fn_sym == "")
818  {
819  Nrottilt = count_even_angles(prm);
820  IntProjs = Nrottilt * Npsi;
821  }
822  else
823  IntProjs = 0;
824  }
825  if (prm.rot_range.randomness != ANGLE_EVENLY)
826  {
827  generate_angles(ExtProjs, prm.rot_range, DF, 'r', prm);
828  generate_angles(ExtProjs, prm.tilt_range, DF, 't', prm);
829  generate_angles(ExtProjs, prm.psi_range, DF, 'p', prm);
830  DF.removeLabel(MDL_ORDER);
831  }
832  else
833  {
834  if (fn_sym == "")
835  generate_even_angles(ExtProjs, Nrottilt, DF, prm);
836  else
837  {
838  SymList SL;
839  if (fn_sym != "")
840  SL.readSymmetryFile(fn_sym);
841  std::vector<double> rotList, tiltList;
842  double rot_step_at_equator = (prm.rot_range.angF - prm.rot_range.ang0) /
843  (double)(Nrot - 1);
844  make_even_distribution(rotList, tiltList, rot_step_at_equator, SL, true);
845  size_t NN=rotList.size();
846  for (size_t n=0; n<NN; ++n)
847  {
848  if (tiltList[n]>=prm.tilt_range.ang0 && tiltList[n]<=prm.tilt_range.angF &&
849  rotList[n]>=prm.rot_range.ang0 && rotList[n]<=prm.rot_range.angF)
850  {
851  size_t DFid=DF.addObject();
852  DF.setValue(MDL_ANGLE_ROT,rotList[n],DFid);
853  DF.setValue(MDL_ANGLE_TILT,tiltList[n],DFid);
854  DF.setValue(MDL_ANGLE_PSI,0.0,DFid);
855  }
856  }
857  }
858  }
859  }
860 
861  // Exit
862  return ExtProjs + IntProjs;
863 }
