132  MultidimArray<double> &mcc=cc();

136  if (Nvols!=1)

138  Matrix2D<int> coocurrence(Nimgs,Nimgs);
139  Matrix2D<double> X(Nimgs,AHCsubset);
140  MultidimArray<int> permutation;
141  AHCClassifier ahc;
142 
143  for (int iter=0; iter<AHCiter; ++iter)
144  {
145  // Take the correlation with random directions that will serve as probe functions
146  randomPermutation(Ndirs,permutation);
147  FOR_ALL_ELEMENTS_IN_MATRIX2D(X)
148  MAT_ELEM(X,i,j)=A3D_ELEM(mcc,i,0,A1D_ELEM(permutation,j));
149 
150  // Classify
151  ahc.clusterData(X,(int)Num_groups);
152 
153  // Update coocurrence
154  for (int i=0; i<Nimgs-1; ++i)
155  for (int j=i+1; j<Nimgs; ++j)
156  if (VEC_ELEM(ahc.clusterAssigned,i)==VEC_ELEM(ahc.clusterAssigned,j))
157  {
158  MAT_ELEM(coocurrence,i,j)+=1;
159  MAT_ELEM(coocurrence,j,i)=MAT_ELEM(coocurrence,i,j);

165  Matrix2D<double> D;

169

172
