462 size_t ProgPDBDictionary::canonicalOrientation2D(const MultidimArray<double> &patch, MultidimArray<double> &canonicalPatch,
463  Matrix1D<double> &patchSignature)
464 {
465  patchSignature.resizeNoCopy(4);
466  size_t nmax=rotationGroup.size();
467  double bestMoment=-1e38;
468  size_t bestIdx=0;
469  for (size_t n=0; n<nmax; ++n)
470  {
471  applyGeometry(LINEAR,auxPatch,patch,rotationGroup[n],IS_INV,DONT_WRAP);
472  // Calculate gradients
473  double momentX=0, momentY=0, momentXmY=0, momentXY=0;
474  auxPatch.setXmippOrigin();
475  FOR_ALL_ELEMENTS_IN_ARRAY2D(auxPatch)
476  {
477  double val=A2D_ELEM(auxPatch,i,j);
478  double jval=j*val;
479  double ival=i*val;
480  momentX+=jval;
481  momentY+=ival;
482  momentXmY+=jval-ival;
483  momentXY+=jval+ival;
484  }
485  double moment=momentX+momentY;
486  if (moment>bestMoment)
487  {
488  bestMoment=moment;
489  bestIdx=n;
490  canonicalPatch=auxPatch;
491  VEC_ELEM(patchSignature,0)=momentX;
492  VEC_ELEM(patchSignature,1)=momentY;
493  VEC_ELEM(patchSignature,2)=momentXmY;
494  VEC_ELEM(patchSignature,3)=momentXY;
495  }
496  }
497  patchSignature.selfNormalize();
498  return bestIdx;
499 }
