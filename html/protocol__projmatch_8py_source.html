112  # Create the library of projections
113  insertAngularProjectLibraryStep(self, iterN, refN)

119  self._insertFunctionStep('assignImagesToReferencesStep', iterN, prerequisites=projMatchSteps)

121  insertAngularClassAverageStep(self, iterN, refN)
122 
123  # Reconstruct each reference with new averages

125  # Create new class averages with images assigned

314  def calculateDeviationsStep(self, it):
315  """ Calculate both angles and shifts devitations for all iterations
316  """
317 
318  SL = emlib.SymList()
319  mdIter = emlib.MetaData()
320  #for it in self.allIters():
321  mdIter.clear()
322  SL.readSymmetryFile(self._symmetry[it])
323  md1 = emlib.MetaData(self.docFileInputAngles[it])
324  md2 = emlib.MetaData(self.docFileInputAngles[it-1])
325  #ignore disabled,
326  md1.removeDisabled()
327  md2.removeDisabled()
328 
329  #first metadata file may not have shiftx and shifty
330  if not md2.containsLabel(emlib.MDL_SHIFT_X):
331  md2.addLabel(emlib.MDL_SHIFT_X)
332  md2.addLabel(emlib.MDL_SHIFT_Y)
333  md2.fillConstant(emlib.MDL_SHIFT_X,0.)
334  md2.fillConstant(emlib.MDL_SHIFT_Y,0.)
335  oldLabels=[emlib.MDL_ANGLE_ROT,
336  emlib.MDL_ANGLE_TILT,
337  emlib.MDL_ANGLE_PSI,
338  emlib.MDL_SHIFT_X,
339  emlib.MDL_SHIFT_Y]
340  newLabels=[emlib.MDL_ANGLE_ROT2,
341  emlib.MDL_ANGLE_TILT2,
342  emlib.MDL_ANGLE_PSI2,
343  emlib.MDL_SHIFT_X2,
344  emlib.MDL_SHIFT_Y2]
345  md2.renameColumn(oldLabels,newLabels)
346  md2.addLabel(emlib.MDL_SHIFT_X_DIFF)
347  md2.addLabel(emlib.MDL_SHIFT_Y_DIFF)
348  md2.addLabel(emlib.MDL_SHIFT_DIFF)
349  mdIter.join1(md1, md2, emlib.MDL_IMAGE, emlib.INNER_JOIN)
350  SL.computeDistance(mdIter,False,False,False)
351  emlib.activateMathExtensions()
352  #operate in sqlite
353  shiftXLabel = emlib.label2Str(emlib.MDL_SHIFT_X)
354  shiftX2Label = emlib.label2Str(emlib.MDL_SHIFT_X2)
355  shiftXDiff = emlib.label2Str(emlib.MDL_SHIFT_X_DIFF)
356  shiftYLabel = emlib.label2Str(emlib.MDL_SHIFT_Y)
357  shiftY2Label = emlib.label2Str(emlib.MDL_SHIFT_Y2)
358  shiftYDiff = emlib.label2Str(emlib.MDL_SHIFT_Y_DIFF)
359  shiftDiff = emlib.label2Str(emlib.MDL_SHIFT_DIFF)
360  #timeStr = str(dtBegin)
361  operateString = shiftXDiff+"="+shiftXLabel+"-"+shiftX2Label
362  operateString += "," + shiftYDiff+"="+shiftYLabel+"-"+shiftY2Label
363  mdIter.operate(operateString)
364  operateString = shiftDiff+"=sqrt(" \
365  +shiftXDiff+"*"+shiftXDiff+"+" \
366  +shiftYDiff+"*"+shiftYDiff+");"
367  mdIter.operate(operateString)
368  iterFile = self._mdDevitationsFn(it)
369  mdIter.write(iterFile,emlib.MD_APPEND)
