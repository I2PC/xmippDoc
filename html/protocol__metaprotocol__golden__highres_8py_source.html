476  def otsu(self, ccList):
477  cc_number = len(ccList)
478  mean_weigth = 1.0 / cc_number
479  his, bins = np.histogram(ccList, int((max(ccList)-min(ccList))/0.01))
480  final_thresh = -1
481  final_value = -1
482  cc_arr = np.arange(min(ccList),max(ccList),0.01)
483  if len(cc_arr)==(len(his)+1):
484  cc_arr=cc_arr[:-1]
485  for t, j in enumerate(bins[1:-1]):
486  idx = t+1
487  pcb = np.sum(his[:idx])
488  pcf = np.sum(his[idx:])
489  Wb = pcb * mean_weigth
490  Wf = pcf * mean_weigth
491 
492  mub = np.sum(cc_arr[:idx] * his[:idx]) / float(pcb)
493  muf = np.sum(cc_arr[idx:] * his[idx:]) / float(pcf)
494  value = Wb * Wf * (mub - muf) ** 2
495 
496  if value > final_value:
497  final_thresh = bins[idx]
498  final_value = value
499 
500  return final_thresh
