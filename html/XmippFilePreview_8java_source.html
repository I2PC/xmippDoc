69  public void loadImage() {
70  if (file == null) {

75  try {
76  String filename = file.getPath();
77  if (!file.exists() || file.isDirectory()){
78  thumbnail = null;
79  return;

81  if(filename.endsWith(".pos"))
82  return;// no image on it
