84 {

96  ImageGeneric M_out;
97  if (method==SMOOTH)
98  M_out.setDatatype(DT_UChar);
99  else
100  M_out.setDatatype(DT_Float);
101  M_out.mapFile2Write(Xpdim, Ypdim, 1, fnImgOut,fnImg==fnImgOut);
102 
103  // Downsample
104  if (method==KER_RECTANGLE)
105  downsampleKernel(M_in,step,M_out);
106  else if (method==FOURIER)
107  downsampleFourier(M_in,step,M_out,nThreads);
108  else
109  downsampleSmooth(M_in,M_out);
110 
111  M_out.write(fnImgOut);
112 }

115 void downsampleKernel(const ImageGeneric &M, double step, ImageGeneric &Mp)
116 {
117  int istep=(int)step;
118  MultidimArray<double> kernel;
119  kernel.resizeNoCopy(istep,istep);
120  kernel.initConstant(1.0/MULTIDIM_SIZE(kernel));
121 
122  size_t Ydim, Xdim, Ypdim, Xpdim;
123  M().getDimensions(Xdim, Ydim);
124  Mp().getDimensions(Xpdim, Ypdim);
125 
126  // Look for input/output ranges
127  double a = 1;
128  double b = 0;
129  double scale = 1;
130  size_t ii, jj, i2, j2, i, j, y, x;
131  if (Mp.getDatatype() != DT_Float)
132  {
133  double imin, imax;
134  double omin=0., omax=0.;
135  imin=imax=M.getPixel(0,0);
136  bool ofirst = true;
137 
138  if (M.getDatatype() != DT_Float)
139  scale = (pow(2.0, Mp.getDatatypeDepth()) - 1.0) /
140  (pow(2.0, M.getDatatypeDepth()) - 1.0);
141  else if (M.getDatatype() == DT_Float)
142  scale = 1;
143  for (ii = 0, y = 0; y < Ydim && ii < Ypdim; y += istep, ++ii)
144  for (jj = 0, x = 0; x < Xdim && jj < Xpdim; x += istep, ++jj)
145  {
146  double pixval = 0;
147  for (i=0, i2=y; i<YSIZE(kernel) && i2<Ydim; ++i, ++i2)
148  for (j=0, j2=x; j<XSIZE(kernel) && j2<Xdim; ++j, ++j2)
149  {
150  double aux=M.getPixel(i2, j2);
151  imin = XMIPP_MIN(imin, aux);
152  imax = XMIPP_MAX(imax, aux);
153  pixval += A2D_ELEM(kernel,i, j) * aux;
154  }
155  pixval *= scale;
156  if (ofirst)
157  {
158  omin = omax = pixval;
159  ofirst = false;
160  }
161  else
162  {
163  omin = XMIPP_MIN(omin, pixval);
164  omax = XMIPP_MAX(omax, pixval);
165  }
166  }
167 
168  // Compute range transformation
169  double irange = imax - imin;
170  double orange = omax - omin;
171 
172  if (M.getDatatype() != DT_Float)
173  {
174  a = scale * irange / orange;
175  b = -omin;
176  }
177  else if (Mp.getDatatype() != DT_Float)
178  {
179  a = (pow(2.0, Mp.getDatatypeDepth()) - 1.0) / orange;
180  scale = 1;
181  b = -omin;
182  }
183  }
184 
185  // Really downsample
186  for (ii = 0, y = 0; y < Ydim && ii < Ypdim; y += istep, ++ii)
187  for (jj = 0, x = 0; x < Xdim && jj < Xpdim; x += istep, ++jj)
188  {
189  double pixval = 0;
190  for (i=0, i2=y; i<YSIZE(kernel) && i2<Ydim; ++i, ++i2)
191  for (j=0, j2=x; j<XSIZE(kernel)&& j2<Xdim; ++j, ++j2)
192  pixval += A2D_ELEM(kernel,i, j) * M.getPixel(i2, j2);
193  if (ii < Ypdim && jj < Xpdim)
194  {
195  if (Mp.datatype != DT_Float)
196  Mp.setPixel(ii, jj, floor(a*(pixval*scale + b)));
197  else
198  Mp.setPixel(ii, jj, pixval);
199  }
200  }
201 }

203 void downsampleFourier(const ImageGeneric &M, double step, ImageGeneric &Mp, int nThreads)
204 {
205  size_t Ydim, Xdim, Ypdim, Xpdim;
206  M().getDimensions(Xdim, Ydim);
207  Mp().getDimensions(Xpdim, Ypdim);
208 
209  // Read the micrograph in memory as doubles
210  MultidimArray<double> Mmem;
211  Mmem.setMmap(true);
212  Mmem.resizeNoCopy(Ydim,Xdim);
213  MultidimArray<std::complex<double> > MmemFourier;
214  M().getImage(Mmem);
215 
216  // Perform the Fourier transform
217  FourierTransformer transformerM;
218  transformerM.setThreadsNumber(nThreads);
219  transformerM.FourierTransform(Mmem, MmemFourier, false);
220 
221  // Create space for the downsampled image and its Fourier transform
222  MultidimArray<double> Mpmem(Ypdim,Xpdim);
223  MultidimArray<std::complex<double> > MpmemFourier;
224  FourierTransformer transformerMp;
225  transformerMp.setThreadsNumber(nThreads);
226  transformerMp.setReal(Mpmem);
227  transformerMp.getFourierAlias(MpmemFourier);
228 
229  int ihalf=YSIZE(MpmemFourier)/2+1;
230  for (int i=0; i<ihalf; i++)
231  for (size_t j=0; j<XSIZE(MpmemFourier); j++)
232  A2D_ELEM(MpmemFourier,i,j)=A2D_ELEM(MmemFourier,i,j);
233  for (size_t i=ihalf; i<YSIZE(MpmemFourier); i++)
234  {
235  size_t ip=YSIZE(MmemFourier)-YSIZE(MpmemFourier)+i;
236  for (size_t j=0; j<XSIZE(MpmemFourier); j++)
237  A2D_ELEM(MpmemFourier,i,j)=A2D_ELEM(MmemFourier,ip,j);
238  }
239 
240  // Transform data
241  transformerMp.inverseFourierTransform();
242 
243  // Find minimun and range in output data
244  double omin=0.,omax=0.;
245  Mpmem.computeDoubleMinMax(omin,omax);
246  double orange = omax - omin;
247  double a = (pow(2.0, Mp.getDatatypeDepth()) - 1.0) / orange;
248  double b = -omin;
249  double scale=1;
250  if (M.getDatatype() != DT_Float)
251  scale = (pow(2.0, Mp.getDatatypeDepth()) - 1.0) /
252  (pow(2.0, M.getDatatypeDepth()) - 1.0);
253  else if (M.getDatatype() == DT_Float)
254  scale = 1;
255 
256  // Copy back data
257  if (Mp.datatype!=DT_Float)
258  FOR_ALL_ELEMENTS_IN_ARRAY2D(Mpmem)
259  Mp.setPixel(i, j, a*(A2D_ELEM(Mpmem,i,j)*scale + b));
260  else
261  FOR_ALL_ELEMENTS_IN_ARRAY2D(Mpmem)
262  Mp.setPixel(i, j, A2D_ELEM(Mpmem,i,j));
263 }
