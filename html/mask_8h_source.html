803 template<typename T1, typename T>
804 void computeStats_within_binary_mask(const MultidimArray< T1 >& mask,
805  const MultidimArray< T >& m, double& min_val,
806  double& max_val,
807  double& avg, double& stddev)
808 {
809  SPEED_UP_tempsInt;
810  double sum1 = 0;
811  double sum2 = 0;
812  int N = 0;
813 
814  max_val = min_val = DIRECT_A3D_ELEM(m, 0, 0, 0);
815 
816  FOR_ALL_ELEMENTS_IN_COMMON_IN_ARRAY3D(mask, m)
817  {
818  if (A3D_ELEM(mask, k, i, j) > 0)
819  {
820  N++;
821 
822  double aux=A3D_ELEM(m, k, i, j);
823  // Minimum and maximum
824  if (aux < min_val)
825  min_val = aux;
826  else if (aux > max_val)
827  max_val = aux;
828 
829  // cumulative sums for average and standard deviation
830  sum1 += aux;
831  sum2 += aux*aux;
832  }
833  }
834 
835  // average and standard deviation
836  avg = sum1 / (double) N;
837  if (N > 1)
838  stddev = sqrt(fabs(sum2 / N - avg * avg) * N / (N - 1));
839  else
840  stddev = 0;
841 }

861 template<typename T>
862 void apply_binary_mask(const MultidimArray< int >& mask, const MultidimArray< T >& m_in,
863  MultidimArray< T >& m_out, T subs_val = (T) 0)
864 {
865  m_out.resize(m_in);
866  FOR_ALL_ELEMENTS_IN_ARRAY3D(m_out)
867  // If in common with the mask
868  if (k >= STARTINGZ(mask) && k <= FINISHINGZ(mask) &&
869  i >= STARTINGY(mask) && i <= FINISHINGY(mask) &&
870  j >= STARTINGX(mask) && j <= FINISHINGX(mask))
871  if (A3D_ELEM(mask, k, i, j) == 0)
872  A3D_ELEM(m_out, k, i, j) = subs_val;
873  else
874  A3D_ELEM(m_out, k, i, j) = A3D_ELEM(m_in, k, i, j);
875  // It is not in common, leave the original one
876  else
877  A3D_ELEM(m_out, k, i, j) = A3D_ELEM(m_in, k, i, j);
878 }

887  MultidimArray< T >& m_out)
