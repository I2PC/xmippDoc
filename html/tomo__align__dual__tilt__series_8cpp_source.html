216 double ProgAlignDualTiltSeries::distanceBetweenCommonLines(
217  int refi, int dualj, const Matrix2D<double> &E,
218  double X, double Y, double Z)
219 {
220  SPEED_UP_temps012;
221 
222  // Compute the direction of the common line in the
223  // universal coordinate system
224  Euler_direction(0, tiltRef(refi), 0, normali);
225  Euler_direction(0, tiltDual(dualj), 0, normalj);
226  M3x3_BY_V3x1(normalj, Et, normalj);
227 
228  vectorProduct(normali,normalj,commonline);
229  if (commonline.module()<XMIPP_EQUAL_ACCURACY)
230  return -2;
231  commonline.selfNormalize();
232 
233  // Compute the direction of the common line in each
234  // of the projection coordinate systems
235  Uproject_to_plane(commonline, 0, tiltRef(refi), 0, commonlinei);
236  M3x3_BY_V3x1(commonlinejE, E, commonline);
237  Uproject_to_plane(commonlinejE, 0, tiltDual(dualj), 0, commonlinej);
238  commonlinei.selfNormalize();
239  commonlinej.selfNormalize();
240 
241  // Compute the angle of the common line in i and j images
242  double angi=-RAD2DEG(atan2(YY(commonlinei),XX(commonlinei)));
243  double angj=-RAD2DEG(atan2(YY(commonlinej),XX(commonlinej)));
244 
245  I=imgRef[refi];
246 #ifdef DEBUG
247  debugging=true;
248  Image<double> save;
249  save()=I;
250  save.write("PPPref.xmp");
251 #endif
252 
253  selfRotate(LINEAR,I, -angi, DONT_WRAP);
254  profilei.initZeros();
255  FOR_ALL_ELEMENTS_IN_ARRAY2D(I)
256  A1D_ELEM(profilei,j)+=A2D_ELEM(I,i,j);
257 #ifdef DEBUG
258 
259  save()=I;
260  save.write("PPPrefaligned.xmp");
261 #endif
262 
263  I=imgDual[dualj];
264 #ifdef DEBUG
265 
266  save()=I;
267  save.write("PPPdual.xmp");
268 #endif
269 
270  // Modify Idual according to z
271  shiftProjectionInZ(I, dualj, Z/scaleFactor);
272 
273  // Modify Idual according to x,y,angj
274  // First translation, then rotation
275  double cj=COSD(angj);
276  double sj=SIND(angj);
277  MAT_ELEM(A,0,0)=MAT_ELEM(A,1,1)=cj;
278  MAT_ELEM(A,1,0)=sj;
279  MAT_ELEM(A,0,1)=-sj;
280  MAT_ELEM(A,0,2)=cj*X-sj*Y;
281  MAT_ELEM(A,1,2)=sj*X+cj*Y;
282  selfApplyGeometry(LINEAR, I, A, IS_NOT_INV, DONT_WRAP);
283  profilej.initZeros();
284  FOR_ALL_ELEMENTS_IN_ARRAY2D(I)
285  A1D_ELEM(profilej,j)+=A2D_ELEM(I,i,j);
286 #ifdef DEBUG
287 
288  save()=I;
289  save.write("PPPdualAligned.xmp");
290 #endif
291 
292  if (debugging)
293  {
294  std::cout << "refi=" << refi << " dualj=" << dualj << std::endl
295  << " tiltRef=" << tiltRef(refi) << " tiltDual=" << tiltDual(dualj) << std::endl
296  << " normali=" << normali.transpose() << std::endl
297  << " normalj=" << normalj.transpose() << std::endl
298  << " commonline=" << commonline.transpose() << std::endl
299  << " commonlinei=" << commonlinei.transpose() << std::endl
300  << " commonlinejE=" << commonlinejE.transpose() << std::endl
301  << " commonlinej=" << commonlinej.transpose() << std::endl
302  << " angi=" << angi << " angj=" << angj << std::endl
303  << " A\n" << A << std::endl
304  << " corr=" << correlationIndex(profilei,profilej) << std::endl
305  ;
306 
307  profilei.write("PPPi.txt");
308  profilej.write("PPPj.txt");
309 
310  std::cout << "Press any key\n";
311  char c;
312  std::cin >> c;
313  }
314  return 1-correlationIndex(profilei,profilej);
315 }

319 double ProgAlignDualTiltSeries::evaluateAlignment(const Matrix1D<double> &_alignment)
320 {
321  Euler_angles2matrix(_alignment(0),_alignment(1),_alignment(2),E);
322  Et=E.transpose();
323  double X=_alignment(3);
324  double Y=_alignment(4);
325  double Z=_alignment(5);
326 
327  int Nref=imgRef.size();
328  int Ndual=imgDual.size();
329  int Ncomparisons=0;
330  double avgDistance=0;
331  for (int nref=0; nref<Nref; nref++)
332  for (int ndual=0; ndual<Ndual; ndual++)
333  {
334  double d=distanceBetweenCommonLines(nref,ndual,E,X,Y,Z);
335  if (d>0)
336  {
337  avgDistance+=d;
338  Ncomparisons++;
339  }
340  }
341  if (Ncomparisons>0)
342  avgDistance/=Ncomparisons;
343  return avgDistance;
344 }

462 void ProgAlignDualTiltSeries::shiftProjectionInZ(MultidimArray<double> &I, int dualj, double Z) const
463 {
464  if (Z==0)
465  return;
466  FourierTransformer transformer;
467  MultidimArray< std::complex<double> > Ifft;
468  Matrix1D<double> w(3);
469  Matrix1D<int> idx(3);
470  Matrix2D<double> Edual, Edualt;
471  Euler_angles2matrix(0., tiltDual(dualj), 0., Edual);
472  Edualt=Edual.transpose();
473  transformer.FourierTransform(I,Ifft,false);
474  FOR_ALL_ELEMENTS_IN_ARRAY2D(Ifft)
475  {
476  SPEED_UP_temps012;
477  VECTOR_R2(idx,j,i);
478  FFT_idx2digfreq(I,idx,w);
479  ZZ(w)=0;
480  M3x3_BY_V3x1(w,Edualt,w);
481  double arg=ZZ(w)*Z;
482  A2D_ELEM(Ifft,i,j)*=std::complex<double>(cos(arg),sin(arg));
483  }
484  transformer.inverseFourierTransform();
485 }
