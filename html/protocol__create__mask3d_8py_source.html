207  fnVol = getImageLocation(volume)
208  Ts = volume.getSamplingRate()
209 
210  if self.volumeOperation == OPERATION_THRESHOLD:

212  "-i %s -o %s --select below %f --substitute binarize"

214  elif self.volumeOperation == OPERATION_SEGMENT:

217  args += "voxel_mass %d" % (self.nvoxels.get())
218  elif self.segmentationType == SEGMENTATION_AMINOACIDS:
219  args += "aa_mass %d %f" % (self.naminoacids.get(), Ts)
220  elif self.segmentationType == SEGMENTATION_DALTON:
221  args += "dalton_mass %d %f" % (self.dalton.get(), Ts)
222  else:
223  args += "otsu"
224  self.runJob("xmipp_volume_segment", args)
