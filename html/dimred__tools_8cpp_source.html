28 void GenerateData::generateNewDataset(const DatasetType &type, int N, double noise)
29 {
30  label.resizeNoCopy(N);
31  X.resizeNoCopy(N,3);
32  t.resizeNoCopy(N,2);
33 
34  auto g = std::mt19937();
35  if (0 == noise) {
36  g.seed(42);
37  } else {
38  std::random_device rd;
39  g.seed(rd());
40  }
41 
42  auto distUniform = std::uniform_real_distribution<>(0.0, 1.0);
43  auto distGauss = std::normal_distribution<>(0.0, 1.0);
44  switch (type) {
45  case DatasetType::SWISS:
46  {
47  for (int i=0; i<N; ++i)
48  {
49  // Generate t
50  MAT_ELEM(t,i,0)=(3 * PI / 2) * (1 + 2 * distUniform(g)); // t
51  MAT_ELEM(t,i,1) = 30 * distUniform(g); // height
52  double localT=MAT_ELEM(t,i,0);
53  double localHeight=MAT_ELEM(t,i,1);
54 
55  // Generate X
56  double s,c;
57  sincos(localT,&s,&c);
58  MAT_ELEM(X,i,0)=localT * c + noise * distGauss(g);
59  MAT_ELEM(X,i,1)=localHeight + noise * distGauss(g);
60  MAT_ELEM(X,i,2)=localT * s + noise * distGauss(g);
61 
62  // Generate label
63  VEC_ELEM(label,i)=(unsigned char)(round(localT/2)+round(localHeight/12))%2;
64  }
65  break;
66  }
67  case DatasetType::HELIX:
68  {
69  double iN=1.0/N;
70  for (int i=0; i<N; ++i)
71  {
72  // Generate t
73  MAT_ELEM(t,i,0)=2 * PI * i*iN;
74  MAT_ELEM(t,i,1)=30 * distUniform(g); // height
75  double localT=MAT_ELEM(t,i,0);
76 
77  // Generate X
78  double s,c;
79  sincos(localT,&s,&c);
80  double s8,c8;
81  sincos(8*localT,&s8,&c8);
82  MAT_ELEM(X,i,0)=(2 + c8)*c+noise*distGauss(g);
83  MAT_ELEM(X,i,1)=(2 + c8)*s+noise*distGauss(g);
84  MAT_ELEM(X,i,2)=s8+noise*distGauss(g);
85 
86  // Generate label
87  VEC_ELEM(label,i)=(unsigned char)(round(localT * 1.5))%2;
88  }
89  break;
90  }
91  case DatasetType::TWINPEAKS:
92  {
93  int actualN=round(sqrt(N));
94  X.resizeNoCopy(actualN*actualN,3);
95  t.resizeNoCopy(actualN*actualN,2);
96  label.resizeNoCopy(actualN*actualN);
97  for (int ii=0; ii<actualN; ii++)
98  {
99  for (int jj=0; jj<actualN; jj++)
100  {
101  int i=ii*actualN+jj;
102 
103  // Generate t
104  double x = 1 - 2 * distUniform(g);
105  double y = 1 - 2 * distUniform(g);
106  MAT_ELEM(t,i,0)=x;
107  MAT_ELEM(t,i,1)=y;
108 
109  // Generate X
110  MAT_ELEM(X,i,0)=x+noise*distGauss(g);
111  MAT_ELEM(X,i,1)=y+noise*distGauss(g);
112  double z=10*sin(PI * x) * tanh(3 * y);
113  MAT_ELEM(X,i,2)=z+noise*distGauss(g);
114 
115  // Generate label
116  VEC_ELEM(label,i)=(unsigned char)(round(0.1*(x+y+z-3)))%2;
117  }
118  }
119  break;
120  }
121  case DatasetType::CLUSTER3D:
122  {
123  // Create centers
124  std::vector<Matrix1D<double> > centers;
125  Matrix1D<double> center(3);
126  const int Nclusters=5;
127  for (int i=0; i<Nclusters; i++)
128  {
129  FOR_ALL_ELEMENTS_IN_MATRIX1D(center)
130  VEC_ELEM(center,i)=10*distUniform(g);
131  centers.push_back(center);
132  }
133 
134  // Measure the minimum distance between centers
135  Matrix1D<double> diff;
136  double minDistance=1e38;
137  for (int i=0; i<Nclusters-1; ++i)
138  for (int j=i+1; j<Nclusters; ++j)
139  {
140  diff=centers[i]-centers[j];
141  double distance=diff.module();
142  minDistance=std::min(minDistance,distance);
143  }
144 
145  // Create clusters
146  t.initZeros();
147  double sigma=minDistance/sqrt(12);
148  for (int n=0; n<N; ++n)
149  {
150  int i=(Nclusters*n)/N;
151  const Matrix1D<double> &center=centers[i];
152  MAT_ELEM(X,n,0)=XX(center)+(distUniform(g)-0.5)*sigma+noise*distGauss(g);
153  MAT_ELEM(X,n,1)=YY(center)+(distUniform(g)-0.5)*sigma+noise*distGauss(g);
154  MAT_ELEM(X,n,2)=ZZ(center)+(distUniform(g)-0.5)*sigma+noise*distGauss(g);
155  }
156  break;
157  }
158  case DatasetType::INTERSECT:
159  {
160  double iN=1.0/N;
161  for (int i=0; i<N; ++i)
162  {
163  // Generate t
164  MAT_ELEM(t,i,0)=2 * PI * i*iN;
165  MAT_ELEM(t,i,1)=5*distUniform(g);
166  double localT=MAT_ELEM(t,i,0);
167  double height=MAT_ELEM(t,i,1);
168 
169  // Generate X
170  double s,c;
171  sincos(localT,&s,&c);
172  MAT_ELEM(X,i,0)=c+noise*distGauss(g);
173  MAT_ELEM(X,i,1)=c*s+noise*distGauss(g);
174  MAT_ELEM(X,i,2)=height+noise*distGauss(g);
175 
176  // Generate label
177  VEC_ELEM(label,i)=(unsigned char)(round(localT *0.5)+round(height*0.5))%2;
178  }
179  break;
180  }
181  default:

183  }

208 void kNearestNeighbours(const Matrix2D<double> &X, int K, Matrix2D<int> &idx, Matrix2D<double> &distance, DimRedDistance2 f, bool computeSqrt)
209 {
210  K=std::min(K,(int)MAT_YSIZE(X)-1);
211  idx.initConstant(MAT_YSIZE(X),K,-1);
212  distance.initConstant(MAT_YSIZE(X),K,1e38);
213  for (size_t i1=0; i1<MAT_YSIZE(X)-1; ++i1)
214  for (size_t i2=i1+1; i2<MAT_YSIZE(X); ++i2)
215  {
216  // Compute the distance between i1 and i2
217  double d=0;
218  if (f==NULL)
219  for (int j=0; j<MAT_XSIZE(X); ++j)
220  {
221  double diff=MAT_ELEM(X,i1,j)-MAT_ELEM(X,i2,j);
222  d+=diff*diff;
223  }
224  else
225  d=(*f)(X,i1,i2);
226 
227  // Check if they are nearest neighbours
228  insertNeighbour(idx,distance,i1,i2,d);
229  insertNeighbour(idx,distance,i2,i1,d);
230  }
231  if (computeSqrt)
232  FOR_ALL_ELEMENTS_IN_MATRIX2D(distance)
233  MAT_ELEM(distance,i,j)=sqrt(MAT_ELEM(distance,i,j));
234 }

330 double intrinsicDimensionalityMLE(const Matrix2D<double> &X, DimRedDistance2 f)
331 {
332  int k1=5;
333  int k2=12;
334  if (k2>MAT_YSIZE(X))
335  {
336  k2=MAT_YSIZE(X)-1;
337  k1=k2/2;
338  }
339  Matrix2D<int> idx;
340  Matrix2D<double> distance;
341  kNearestNeighbours(X,k2,idx,distance,f);
342 
343  // Estimate d
344  double dsum=0;
345  std::cerr << "Estimating dimensionality ... " << std::endl;
346  init_progress_bar(MAT_YSIZE(distance));
347  for (int i=0; i<MAT_YSIZE(distance); ++i)
348  {
349  double dist=log(MAT_ELEM(distance,i,0));
350  double S=dist;
351  for (int k=1; k<MAT_XSIZE(distance); ++k)
352  {
353  dist=log(MAT_ELEM(distance,i,k));
354  S+=dist;
355  if (k>=k1)
356  {
357  double d=(k-1)/(S-dist*(k+1));
358  dsum+=d;
359  }
360  }
361  progress_bar(i);
362  }
363  progress_bar(MAT_YSIZE(distance));
364  return -dsum/((k2-k1)*MAT_YSIZE(distance));
365 }

367 // By correlation dimension
368 double intrinsicDimensionalityCorrDim(const Matrix2D<double> &X, DimRedDistance2 f)
369 {
370  int K=3;
371  Matrix2D<int> idx;
372  Matrix2D<double> distance;
373  kNearestNeighbours(X,K,idx,distance,f);
374 
375  // Compute median and maximum
376  size_t N=MAT_XSIZE(distance)*MAT_YSIZE(distance);
377  std::sort(MATRIX2D_ARRAY(distance),MATRIX2D_ARRAY(distance)+N);
378  double median=MATRIX2D_ARRAY(distance)[N/2];
379  double maxVal=MATRIX2D_ARRAY(distance)[N-1];
380  median*=median; // Because we compute dist^2 instead of dist
381  maxVal*=maxVal;
382  if (maxVal==0)
383  return 0;
384 
385  // Compute the distance of all versus all
386  double countLessMedianK=0, countLessMaxK=0;
387  std::cerr << "Estimating dimensionality ... " << std::endl;
388  init_progress_bar(MAT_YSIZE(X)-1);
389  for (size_t i1=0; i1<MAT_YSIZE(X)-1; ++i1)
390  {
391  for (size_t i2=i1+1; i2<MAT_YSIZE(X); ++i2)
392  {
393  // Compute the distance between i1 and i2
394  double d=0;
395  if (f==NULL)
396  for (int j=0; j<MAT_XSIZE(X); ++j)
397  {
398  double diff=MAT_ELEM(X,i1,j)-MAT_ELEM(X,i2,j);
399  d+=diff*diff;
400  }
401  else
402  d=(*f)(X,i1,i2);
403 
404  // Check d
405  if (d<=maxVal)
406  {
407  countLessMaxK+=1;
408  if (d<=median)
409  countLessMedianK+=1;
410  }
411  }
412  progress_bar(i1);
413  }
414  progress_bar(MAT_YSIZE(X)-1);
415  double iCombinations=2.0/(MAT_YSIZE(X)*(MAT_YSIZE(X)-1));
416  double probLessMedianK=countLessMedianK*iCombinations; // Probability of a distance being less than medianK
417  double probLessMaxK=countLessMaxK*iCombinations; // Probability of a distance being less than maxK
418  return 2*log(probLessMaxK/probLessMedianK)/log(maxVal/median);
419 }
