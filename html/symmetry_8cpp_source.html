43 int get_space_group_number(std::string spacegroup)
44 {
45  if (spacegroup == "P 21 21 2 A")
46  spacegroup = "P 21 21 2 (a)";
47  else if (spacegroup.empty())
48  throw std::runtime_error("No spacegroup, cannot continue");
49
