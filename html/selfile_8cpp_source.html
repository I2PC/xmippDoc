350 /* Merging with another selfile -------------------------------------------- */
351 void SelFile::merge(SelFile &SF)
352 {
353  std::vector<SelLine>::iterator current = SF.text_line.begin();
354  std::vector<SelLine>::iterator last = SF.text_line.end();
355  std::vector<SelLine>::iterator found;
356 
357  SelLine discrepancy;
358  discrepancy.line_type = SelLine::COMMENT;
359  discrepancy.text = "# There were discrepancy in the tags for next line, the "
360  "ACTIVE state is kept";
361 
362  while (current != last)
363  {
364  if ((*current).line_type != SelLine::DATALINE)
365  {
366  current++;
367  continue;
368  }
369  if ((found = find((*current).text)) == text_line.end())
370  {
371  // New image not found in the whole Sel File.
372  // Add it if it is not discarded
373  if ((*current).label != SelLine::DISCARDED)
374  {
375  text_line.push_back(*current);
376  no_imgs++;
377  }
378  }
379  else
380  // New image is found, check that its line is not going
381  // to be removed, if it is add it again; else, check if
382  // there is a discrepancy between them
383  if ((*found).label != (*current).label)
384  {
385  if ((*found).label < (*current).label)
386  {
387  (*found).label = SelLine::ACTIVE;
388  no_imgs++;
389  }
390  text_line.insert(found, 1, discrepancy);
391  }
392  current++;
393  }
394 }
