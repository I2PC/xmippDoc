62 void ProgMPIXrayProject::run()
63 {
64  preRun();
65 
66  // Project
67  projMD.setComment("True rot, tilt and psi; rot, tilt, psi, X and Y shifts applied");
68 
69  // // Assign mpi node information
70  // MpiNode & node = *node;
71 
72  // Calculation of data to be distributed in nodes -----------------------------------------
73  std::vector<mpiProjData> mpiData;
74  mpiProjData data;
75 
76  size_t idx = FIRST_IMAGE;
77  size_t id;
78 
79  for (double angle = projParam.tilt0; angle <= projParam.tiltF; angle += projParam.tiltStep)
80  {
81  if (projParam.singleProjection)
82  data.fn_proj = projParam.fnOut;
83  else
84  data.fn_proj.compose(idx, projParam.fnRoot + ".stk");
85 
86  // Choose Center displacement ........................................
87  double shiftX = rnd_gaus(projParam.Ncenter_avg, projParam.Ncenter_dev);
88  double shiftY = rnd_gaus(projParam.Ncenter_avg, projParam.Ncenter_dev);
89  Matrix1D<double> inPlaneShift(3);
90  VECTOR_R3(inPlaneShift,shiftX,shiftY,0);
91 
92  projParam.calculateProjectionAngles(proj,angle, 0,inPlaneShift);
93  proj.getEulerAngles(data.tRot, data.tTilt,data.tPsi);
94  proj.getShifts(data.xShift, data.yShift, data.zShift);
95 
96  // Add noise in angles and voxels ....................................
97 
98  data.rot = data.tRot + rnd_gaus(projParam.Nangle_avg, projParam.Nangle_dev);
99  data.tilt = data.tTilt + rnd_gaus(projParam.Nangle_avg, projParam.Nangle_dev);
100  data.psi = data.tPsi + rnd_gaus(projParam.Nangle_avg, projParam.Nangle_dev);
101 
102  //MPI proj.setEulerAngles(rot,tilt,psi);
103 
104  if (node->isMaster())
105  {
106  id = projMD.addObject();
107  projMD.setValue(MDL_IMAGE,data.fn_proj, id);
108 
109  projMD.setValue(MDL_ANGLE_ROT,data.tRot, id);
110  projMD.setValue(MDL_ANGLE_TILT,data.tTilt, id);
111  projMD.setValue(MDL_ANGLE_PSI,data.tPsi, id);
112  projMD.setValue(MDL_ANGLE_ROT2,data.rot, id);
113  projMD.setValue(MDL_ANGLE_TILT2,data.tilt, id);
114  projMD.setValue(MDL_ANGLE_PSI2,data.psi, id);
115  projMD.setValue(MDL_SHIFT_X,shiftX, id);
116  projMD.setValue(MDL_SHIFT_Y,shiftY, id);
117  }
118 
119  mpiData.push_back(data);
120  idx++;
121  }
122 
123  // Creation of MPI Job Handler file
124 
125  MpiTaskDistributor *jobHandler;
126  long long int nodeBlockSize = 1;
127  jobHandler = new MpiTaskDistributor(mpiData.size(), nodeBlockSize, node);
128 
129  size_t first = 0, last = 0;
130 
131  if (node->isMaster())
132  {
133  // Save metadata file with angles and shift info
134  if (!projParam.singleProjection)
135  projMD.write(projParam.fnRoot + ".sel");
136 
137  if (!(projParam.show_angles))
138  init_progress_bar(mpiData.size());
139 
140  // Creation of output file to reserve space
141  createEmptyFile(mpiData[mpiData.size()-1].fn_proj,
142  XMIPP_MIN(XSIZE(MULTIDIM_ARRAY(phantom.iniVol)),(size_t)projParam.proj_Xdim),
143  XMIPP_MIN(YSIZE(MULTIDIM_ARRAY(phantom.iniVol)),(size_t)projParam.proj_Ydim));
144 
145  std::cout << "Projecting ...\n";
146  }
147 
148  node->barrierWait();
149 
150  MPI_Barrier(MPI_COMM_WORLD);
151 
152  // Parallel node jobs
153  while (jobHandler->getTasks(first, last))
154  {
155  for (size_t k = first; k <= last; k++)
156  {
157  std::cout << "Node: " << node->rank << " - Task: " << k <<std::endl;
158  proj.setEulerAngles(mpiData[k].tRot, mpiData[k].tTilt,mpiData[k].tPsi);
159  proj.setShifts(mpiData[k].xShift, mpiData[k].yShift, mpiData[k].zShift);
160 
161  //Reset thread task distributor
162  td->clear();
163 
164  // Really project ....................................................
165  if (!projParam.only_create_angles)
166  {
167  XrayRotateAndProjectVolumeOffCentered(phantom, psf, proj, stdProj, projParam.proj_Ydim, projParam.proj_Xdim);
168  proj.write(mpiData[k].fn_proj);
169  }
170 
171  // Add noise in angles and voxels ....................................
172  // proj.setEulerAngles(mpiData[k].rot,mpiData[k].tilt,mpiData[k].psi); // Geo info is not stored in header file
173  IMGMATRIX(proj).addNoise(projParam.Npixel_avg, projParam.Npixel_dev, "gaussian");
174 
175  if (projParam.show_angles)
176  std::cout << "Node: " << node->rank << "\t" << proj.rot() << "\t"
177  << proj.tilt() << "\t" << proj.psi() << std::endl;
178 
179  if (node->isMaster() && !(projParam.show_angles))
180  progress_bar(k+1);
181  }
182  }
183  jobHandler->wait();
184 
185  delete jobHandler;
186  postRun();
187 
188  return;
189 }
