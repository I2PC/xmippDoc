482  void computeAverageAndStddev(double &avg, double &stddev,
483  int mode = FULL_CIRCLES) const
484  {
485  double aux, sum = 0., sum2=0.;
486  double twopi, w, N = 0;
487 
488  if (mode == FULL_CIRCLES)
489  twopi = 2.*PI;
490  else if (mode == HALF_CIRCLES)
491  twopi = PI;
492  else
493  REPORT_ERROR(ERR_VALUE_INCORRECT,"Incorrect mode for computeAverageAndStddev");
494 
495  size_t imax=rings.size();
496  const double *ptrRing_radius=&ring_radius[0];
497  for (size_t i = 0; i < imax; ++i, ++ptrRing_radius)
498  {
499  // take varying sampling into account
500  w = (twopi * (*ptrRing_radius)) / (double) XSIZE(rings[i]);
501  const MultidimArray<T> &rings_i=rings[i];
502  for (size_t j = 0; j < XSIZE(rings_i); j++)
503  {
504  aux = DIRECT_A1D_ELEM(rings_i,j);
505  double waux=w*aux;
506  sum += waux;
507  sum2 += waux * aux;
508  N += w;
509  }
510  }
511  if (N>0)
512  {
513  sum2 = sum2 / N;
514  avg = sum / N;
515  stddev=sqrt(fabs(sum2-avg*avg));
516  }
517  else if (N != 0.)
518  {
519  avg = sum;
520  stddev=sqrt(fabs(sum2-avg*avg));
521  }
522  else
523  stddev=avg=0;
524  }

615  void getPolarFromCartesianBSpline(const MultidimArray<T> &M1,
616  int first_ring, int last_ring, int BsplineOrder=3,
617  double xoff = 0., double yoff = 0.,
618  double oversample1 = 1., int mode1 = FULL_CIRCLES)
619  {
620  int nsam;
621  double twopi;
622  double xp, yp, minxp, maxxp, minyp, maxyp;
623 
624  MultidimArray<T> Mring;
625  rings.clear();
626  ring_radius.clear();
627  mode = mode1;
628  oversample = oversample1;
629 
630  if (mode == FULL_CIRCLES)
631  twopi = TWOPI;
632  else if (mode == HALF_CIRCLES)
633  twopi = PI;
634  else
635  REPORT_ERROR(ERR_VALUE_INCORRECT,"Incorrect mode for getPolarFromCartesian");
636 
637 
638  // Limits of the matrix (not oversized!)
639  minxp = FIRST_XMIPP_INDEX(XSIZE(M1));
640  minyp = FIRST_XMIPP_INDEX(YSIZE(M1));
641  maxxp = LAST_XMIPP_INDEX(XSIZE(M1));
642  maxyp = LAST_XMIPP_INDEX(YSIZE(M1));
643  double minxp_e=minxp-XMIPP_EQUAL_ACCURACY;
644  double minyp_e=minyp-XMIPP_EQUAL_ACCURACY;
645  double maxxp_e=maxxp+XMIPP_EQUAL_ACCURACY;
646  double maxyp_e=maxyp+XMIPP_EQUAL_ACCURACY;
647 
648  // Loop over all polar coordinates
649  for (int iring = first_ring; iring <= last_ring; iring++)
650  {
651  float radius = (float) iring;
652  // Non-constant sampling!! (always even for convenient Half2Whole of FTs)
653  nsam = 2 * (int)( 0.5 * oversample * twopi * radius );
654  nsam = XMIPP_MAX(1, nsam);
655  float dphi = twopi / (float)nsam;
656  Mring.resizeNoCopy(nsam);
657  int iphi = 0;
658 #define VEC_LEN 8
659  double axp[VEC_LEN];
660  double ayp[VEC_LEN];
661  for (iphi = 0; iphi < nsam-VEC_LEN; iphi+=VEC_LEN)
662  {
663  for (int jphi = 0; jphi < VEC_LEN; jphi++) {
664  // from polar to original cartesian coordinates
665  float phi = (iphi+jphi) * dphi;
666  axp[jphi] = sin(phi) * radius;
667  ayp[jphi] = cos(phi) * radius;
668 
669  // Origin offsets
670  axp[jphi] += xoff;
671  ayp[jphi] += yoff;
672  }
673 
674  for (int jphi = 0; jphi < VEC_LEN; jphi++) {
675  // Wrap coordinates
676  axp[jphi] = realWRAP(axp[jphi], minxp - 0.5, maxxp + 0.5);
677  ayp[jphi] = realWRAP(ayp[jphi], minyp - 0.5, maxyp + 0.5);
678  }
679 
680  // Perform the convolution interpolation
681  if (BsplineOrder==1) {
682  #pragma simd
683  for (int jphi = 0; jphi < VEC_LEN; jphi++) {
684  DIRECT_A1D_ELEM(Mring,iphi+jphi) = M1.interpolatedElement2DOutsideZero(axp[jphi],ayp[jphi]);
685  }
686  }
687  else {
688  for (int jphi = 0; jphi < VEC_LEN; jphi++) {
689  DIRECT_A1D_ELEM(Mring,iphi+jphi) = M1.interpolatedElementBSpline2D(axp[jphi],ayp[jphi],BsplineOrder);
690  }
691  }
692  }
693  for (; iphi < nsam; iphi++)
694  {
695  // from polar to original cartesian coordinates
696  float phi = iphi * dphi;
697  xp = sin(phi) * radius;
698  yp = cos(phi) * radius;
699 
700  // Origin offsets
701  xp += xoff;
702  yp += yoff;
703 
704  // Wrap coordinates
705  if (xp < minxp_e || xp > maxxp_e)
706  xp = realWRAP(xp, minxp - 0.5, maxxp + 0.5);
707  if (yp < minyp_e || yp > maxyp_e)
708  yp = realWRAP(yp, minyp - 0.5, maxyp + 0.5);
709 
710  // Perform the convolution interpolation
711  if (BsplineOrder==1)
712  DIRECT_A1D_ELEM(Mring,iphi) = M1.interpolatedElement2DOutsideZero(xp,yp);
713  else
714  DIRECT_A1D_ELEM(Mring,iphi) = M1.interpolatedElementBSpline2D(xp,yp,BsplineOrder);
715  }
716  rings.push_back(Mring);
717  ring_radius.push_back(radius);
718  }
719  }
