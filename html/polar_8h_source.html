483  void computeAverageAndStddev(double &avg, double &stddev,
484  int mode = FULL_CIRCLES) const
485  {
486  double aux, sum = 0., sum2=0.;
487  double twopi, w, N = 0;
488 
489  if (mode == FULL_CIRCLES)
490  twopi = 2.*PI;
491  else if (mode == HALF_CIRCLES)
492  twopi = PI;
493  else
494  REPORT_ERROR(ERR_VALUE_INCORRECT,"Incorrect mode for computeAverageAndStddev");
495 
496  size_t imax=rings.size();
497  const double *ptrRing_radius=&ring_radius[0];
498  for (size_t i = 0; i < imax; ++i, ++ptrRing_radius)
499  {
500  // take varying sampling into account
501  w = (twopi * (*ptrRing_radius)) / (double) XSIZE(rings[i]);
502  const MultidimArray<T> &rings_i=rings[i];
503  for (size_t j = 0; j < XSIZE(rings_i); j++)
504  {
505  aux = DIRECT_A1D_ELEM(rings_i,j);
506  double waux=w*aux;
507  sum += waux;
508  sum2 += waux * aux;
509  N += w;
510  }
511  }
512  if (N>0)
513  {
514  sum2 = sum2 / N;
515  avg = sum / N;
516  stddev=sqrt(fabs(sum2-avg*avg));
517  }
518  else if (N != 0.)
519  {
520  avg = sum;
521  stddev=sqrt(fabs(sum2-avg*avg));
522  }
523  else
524  stddev=avg=0;
525  }

550  void getCartesianCoordinates(std::vector<double> &x,
551  std::vector<double> &y,
552  std::vector<T> &data,
553  const double extra_shell = GRIDDING_K/2)
554  {
555  // Only for full circles for now!
556  if (mode != FULL_CIRCLES)
557  REPORT_ERROR(ERR_VALUE_INCORRECT,"VoronoiArea only implemented for FULL_CIRCLES mode of Polar");
558 
559  // First fill the vector with the originally sampled coordinates
560  x.clear();
561  y.clear();
562  data.clear();
563  for (int i = 0; i < rings.size(); i++)
564  {
565  int nsam = XSIZE(rings[i]);
566  float dphi = TWOPI / (float)nsam;
567  float radius = ring_radius[i];
568  for (int j = 0; j < nsam; j++)
569  {
570  float tmp = j * dphi;
571  x.push_back(radius * sin(tmp));
572  y.push_back(radius * cos(tmp));
573  data.push_back(rings[i](j));
574  }
575  }
576 
577  // Add additional points on the inside and outside of the rings
578  // Add a maximum of "extra_shell" rings
579  // Set data to zero here
580  double first_ring = ring_radius[0];
581  double last_ring = ring_radius[rings.size()-1];
582  float outer = last_ring + extra_shell;
583  float inner = XMIPP_MAX(0.,first_ring - extra_shell);
584  for (float radius = 0.; radius < outer; radius +=1.)
585  {
586  if ( (radius >= inner && radius < first_ring) ||
587  ( radius <= outer && radius > last_ring) )
588  {
589  int nsam = 2 * (int)( 0.5 * oversample * TWOPI * radius );
590  nsam = XMIPP_MAX(1, nsam);
591  float dphi = TWOPI / (float)nsam;
592  for (int j = 0; j < nsam; j++)
593  {
594  float tmp = j * dphi;
595  x.push_back(radius * sin(tmp));
596  y.push_back(radius * cos(tmp));
597  data.push_back(0.);
598  }
599  }
600  }

616  void getPolarFromCartesianBSpline(const MultidimArray<T> &M1,
617  int first_ring, int last_ring, int BsplineOrder=3,
618  double xoff = 0., double yoff = 0.,
619  double oversample1 = 1., int mode1 = FULL_CIRCLES)
620  {
621  double twopi;
622  double xp, yp, minxp, maxxp, minyp, maxyp;
623 
624  auto noOfRings = getNoOfRings(first_ring, last_ring);
625  rings.resize(noOfRings);
626  ring_radius.resize(noOfRings);
627 
628  mode = mode1;
629  oversample = oversample1;
630 
631  if (mode == FULL_CIRCLES)
632  twopi = TWOPI;
633  else if (mode == HALF_CIRCLES)
634  twopi = PI;
635  else
636  REPORT_ERROR(ERR_VALUE_INCORRECT,"Incorrect mode for getPolarFromCartesian");
637 
638 
639  // Limits of the matrix (not oversized!)
640  minxp = FIRST_XMIPP_INDEX(XSIZE(M1));
641  minyp = FIRST_XMIPP_INDEX(YSIZE(M1));
642  maxxp = LAST_XMIPP_INDEX(XSIZE(M1));
643  maxyp = LAST_XMIPP_INDEX(YSIZE(M1));
644  double minxp_e=minxp-XMIPP_EQUAL_ACCURACY;
645  double minyp_e=minyp-XMIPP_EQUAL_ACCURACY;
646  double maxxp_e=maxxp+XMIPP_EQUAL_ACCURACY;
647  double maxyp_e=maxyp+XMIPP_EQUAL_ACCURACY;
648 
649  std::lock_guard<std::mutex> lockGuard(m_mutex);
650 
651  ensureAngleCache(first_ring, last_ring);
652 
653  // Loop over all polar coordinates
654  for (int r = 0; r < noOfRings; ++r)
655  {
656  float radius = (float) r + first_ring;
657  ring_radius.at(r) = radius;
658  auto &Mring = rings.at(r);
659  // Non-constant sampling!! (always even for convenient Half2Whole of FTs)
660  int nsam = getNoOfSamples(twopi, radius);
661  Mring.resizeNoCopy(nsam);
662 
663  auto &s = m_lastSinRadius.at(r);
664  auto &c = m_lastCosRadius.at(r);
665 
666  for (int sample = 0; sample < nsam; ++sample)
667  {
668  // from polar to original cartesian coordinates
669  xp = s.at(sample);
670  yp = c.at(sample);
671 
672  // Origin offsets
673  xp += xoff;
674  yp += yoff;
675 
676  // Wrap coordinates
677  if (xp < minxp_e || xp > maxxp_e)
678  xp = realWRAP(xp, minxp - 0.5, maxxp + 0.5);
679  if (yp < minyp_e || yp > maxyp_e)
680  yp = realWRAP(yp, minyp - 0.5, maxyp + 0.5);
681 
682  // Perform the convolution interpolation
683  if (BsplineOrder==1)
684  DIRECT_A1D_ELEM(Mring,sample) = M1.interpolatedElement2DOutsideZero(xp,yp);
685  else
686  DIRECT_A1D_ELEM(Mring,sample) = M1.interpolatedElementBSpline2D(xp,yp,BsplineOrder);
687  }
688  }
689  }
