46 class XmippProtRansac(ProtInitialVolume):
47  """
48  Computes an initial 3d model from a set of projections/classes
49  using RANSAC algorithm.
50 
51  This method is based on an initial non-lineal dimensionality
52  reduction approach which allows to select representative small
53  sets of class average images capturing the most of the structural
54  information of the particle under study. These reduced sets are
55  then used to generate volumes from random orientation assignments.
56  The best volume is determined from these guesses using a random
57  sample consensus (RANSAC) approach.
58  """
59  _label = 'ransac'
