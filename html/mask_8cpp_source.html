32 void RaisedCosineMask(MultidimArray<double> &mask, double r1, double r2,
33  int mode, double x0, double y0, double z0)
34 {
35  double K = PI / (r2 - r1);
36  double r1_2=r1*r1;
37  double r2_2=r2*r2;
38  for (int k=STARTINGZ(mask); k<=FINISHINGZ(mask); ++k)
39  {
40  double k2=(k - z0) * (k - z0);
41  for (int i=STARTINGY(mask); i<=FINISHINGY(mask); ++i)
42  {
43  double i2_k2=k2+(i - y0) * (i - y0);
44  for (int j=STARTINGX(mask); j<=FINISHINGX(mask); ++j)
45  {
46  double r2=i2_k2+(j - x0) * (j - x0);
47  if (r2 <= r1_2)
48  A3D_ELEM(mask, k, i, j) = 1;
49  else if (r2 < r2_2)
50  {
51  double r=sqrt(r2);
52  A3D_ELEM(mask, k, i, j) = 0.5*(1 + cos(K * (r - r1)));
53  }
54  else

56  if (mode == OUTSIDE_MASK)

79 void KaiserMask(MultidimArray<double> &mask, double delta, double Deltaw)
80 {
81  // Convert Deltaw from a frequency normalized to 1, to a freq. normalized to PI
82  Deltaw *= PI;
83 
84  // Design Kaiser selfWindow
85  double A = -20 * log10(delta);
86  int M = CEIL((A - 8) / (2.285 * Deltaw));
87  double beta;
88  if (A > 50)
89  beta = 0.1102 * (A - 8.7);
90  else if (A >= 21)
91  beta = 0.5842 * pow(A - 21, 0.4) + 0.07886 * (A - 21);
92  else
93  beta = 0;
94 
95  // "Draw" Kaiser selfWindow
96  if (YSIZE(mask)==1 && ZSIZE(mask)==1)
97  {
98  // 1D
99  mask.resize(2*M + 1);
100  }
101  else if (ZSIZE(mask)==1)
102  {
103  // 2D
104  mask.resize(2*M + 1, 2*M + 1);
105  }
106  else

591  {
592  A3D_ELEM(mask, kpp, ipp, jpp) = 1;
593  }
594  else
