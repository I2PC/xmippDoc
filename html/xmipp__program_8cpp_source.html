607 void XmippMetadataProgram::setup(MetaData *md, const FileName &out, const FileName &oroot,
608  bool applyGeo, MDLabel image_label)
609 {
610  this->mdIn = md;
611  this->fn_out = out;
612  this->oroot = oroot;
613  this->image_label = image_label;
614  this->doRun = true;
615 
616  if (remove_disabled)
617  mdIn->removeDisabled();
618 
619  if (mdIn->isEmpty())
620  REPORT_ERROR(ERR_MD_NOOBJ, "Empty input Metadata.");
621 
622  mdInSize = mdIn->size();
623 
624  if (mdIn->isMetadataFile)
625  input_is_metadata = true;
626  else
627  {
628  if (mdInSize == 1)
629  single_image = true;
630  else
631  input_is_stack = true;
632  }
633 
634  String labelStr = MDL::label2Str(image_label);
635 
636  if (image_label == MDL_UNDEFINED)
637  REPORT_ERROR(ERR_MD_BADLABEL, formatString("Unknown image label '%s'.", labelStr.c_str()));
638 
639  if (!mdIn->containsLabel(image_label))
640  REPORT_ERROR(ERR_MD_MISSINGLABEL,
641  formatString("Image label '%s' is missing. See option --label.", labelStr.c_str()));
642 
643  /* Output is stack if, given a filename in fn_out, mdIn has multiple images.
644  * In case no output name is given, then input is overwritten and we have to
645  * check if it is stack. */
646  output_is_stack = mdInSize > 1 && oroot.empty() && (!fn_out.empty() || input_is_stack);
647 
648  /* Save metadata related to output stack only if required,
649  * and output is a stack.*/
650  //save_metadata_stack = save_metadata_stack && output_is_stack;
651 
652  // Only delete output stack in case we are not overwriting input
653  delete_output_stack = (output_is_stack && delete_output_stack) ?
654  !(fn_out.empty() && oroot.empty()) : false;
655 
656  // If the output is a stack, create empty stack file in advance to avoid concurrent access to the header
657  create_empty_stackfile = (each_image_produces_an_output && output_is_stack && !fn_out.empty());
658 
659  // if create, then we need to read the dimensions of the input stack
660  if (get_image_info || create_empty_stackfile)
661  getImageInfo(*mdIn, xdimOut, ydimOut, zdimOut, ndimOut, datatypeOut, image_label);
662 
663  apply_geo = applyGeo;
664  // if input is volume do not apply geo
665  if (zdimOut > 1)
666  apply_geo = false;
667 }//function setup

713 void XmippMetadataProgram::writeOutput()
714 {
715  if (!single_image && !mdOut.isEmpty() && !fn_out.empty())
716  {

769 void XmippMetadataProgram::run()
770 {
771  FileName fnImg, fnImgOut, fullBaseName;
772  size_t objId;
773  MDRow rowIn, rowOut;
774  mdOut.clear(); //this allows multiple runs of the same Program object
775 
776  //Perform particular preprocessing
777  preProcess();
778 
779  startProcessing();
780 
781  size_t objIndex = 0;
782 
783  if (!oroot.empty())
784  {
785  if (oext.empty())
786  oext = oroot.getFileFormat();
787  oextBaseName = oext;
788  fullBaseName = oroot.removeFileFormat();
789  baseName = fullBaseName.getBaseName();
790  pathBaseName = fullBaseName.getDir();
791  }
792 
793  //FOR_ALL_OBJECTS_IN_METADATA(mdIn)
794  while (getImageToProcess(objId, objIndex))
795  {
796  ++objIndex; //increment for composing starting at 1
797 
798  mdIn->getRow(rowIn, objId);
799  rowIn.getValue(image_label, fnImg);
800 
801  if (fnImg.empty())
802  break;
803 
804  fnImgOut = fnImg;
805 
806  if (each_image_produces_an_output)
807  {
808  if (!oroot.empty()) // Compose out name to save as independent images
809  {
810  if (oext.empty()) // If oext is still empty, then use ext of indep input images
811  {
812  if (input_is_stack)
813  oextBaseName = "spi";
814  else
815  oextBaseName = fnImg.getFileFormat();
816  }
817 
818  if (!baseName.empty() )
819  fnImgOut.compose(fullBaseName, objIndex, oextBaseName);
820  else if (fnImg.isInStack())
821  fnImgOut.compose(pathBaseName + (fnImg.withoutExtension()).getDecomposedFileName(), objIndex, oextBaseName);
822  else
823  fnImgOut = pathBaseName + fnImg.withoutExtension()+ "." + oextBaseName;
824  }
825  else if (!fn_out.empty() )
826  {
827  if (single_image)
828  fnImgOut = fn_out;
829  else
830  fnImgOut.compose(objIndex, fn_out); // Compose out name to save as stacks
831  }
832  else
833  fnImgOut = fnImg;
834  setupRowOut(fnImg, rowIn, fnImgOut, rowOut);
835  }
836  else if (produces_a_metadata)
837  setupRowOut(fnImg, rowIn, fnImgOut, rowOut);
838 
839  processImage(fnImg, fnImgOut, rowIn, rowOut);
840 
841  if (each_image_produces_an_output || produces_a_metadata)
842  mdOut.addRow(rowOut);
843 
844  checkPoint();
845  showProgress();
846  }
847  wait();
848 
849  //free iterator memory
850  delete iter;
851 
852  /* Generate name to save mdOut when output are independent images. It uses as prefix
853  * the dirBaseName in order not overwriting files when repeating same command on
854  * different directories. If baseName is set it is used, otherwise, input name is used.
855  * Then, the suffix _oext is added.*/
856  if (fn_out.empty() )
857  {
858  if (!oroot.empty())
859  {
860  if (!baseName.empty() )
861  fn_out = findAndReplace(pathBaseName,"/","_") + baseName + "_" + oextBaseName + ".xmd";
862  else
863  fn_out = findAndReplace(pathBaseName,"/","_") + fn_in.getBaseName() + "_" + oextBaseName + ".xmd";
864  }
865  else if (input_is_metadata)
866  fn_out = fn_in;
867  }
868 
869  finishProcessing();
870 
871  postProcess();
872 
873  /* Reset the default values of the program in case
874  * to be reused.*/
875  init();
876 }
