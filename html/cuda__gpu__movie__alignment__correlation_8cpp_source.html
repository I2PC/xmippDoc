35 void performFFTAndScale(T* inOutData, int noOfImgs, int inX, int inY,
36  int inBatch, int outFFTX, int outY, MultidimArray<T> &filter) {
37  assert((inX / 2 + 1) >= outFFTX);
38  assert(inY >= outY);
39  mycufftHandle handle;
40  int counter = 0;
41  std::complex<T>* h_result = (std::complex<T>*)inOutData;
42  // since memory where the input images are will be reused, we have to make
43  // sure it's big enough
44  size_t bytesImgs = inX * inY * inBatch * sizeof(T);
45  size_t bytesFFTs = outFFTX * outY * inBatch * sizeof(T) * 2; // complex
46  T* d_data = NULL;
47  T* d_filter = NULL;
48  auto h_tmpStore = new std::complex<T>[outFFTX * outY * inBatch];
49  gpuErrchk(cudaMalloc(&d_filter, filter.yxdim * sizeof(T)));
50  gpuErrchk(cudaMemcpy(d_filter, filter.data, filter.yxdim * sizeof(T), cudaMemcpyHostToDevice));
51  gpuErrchk(cudaMalloc(&d_data, std::max(bytesImgs, bytesFFTs)));
52  GpuMultidimArrayAtGpu<T> imagesGPU(inX, inY, 1, inBatch, d_data);
53  GpuMultidimArrayAtGpu<std::complex<T> > resultingFFT;
54 
55  auto loadImgs = [&](){
56  int imgToProcess = std::min(inBatch, noOfImgs - counter);
57  if (imgToProcess < 1) return (size_t)0;
58  T* h_imgLoad = inOutData + counter * inX * inY;
59  size_t bytesIn = imgToProcess * inX * inY * sizeof(T);
60  gpuErrchk(cudaMemcpy(imagesGPU.d_data, h_imgLoad, bytesIn, cudaMemcpyHostToDevice));
61  size_t bytesOut = imgToProcess * outFFTX * outY * sizeof(std::complex<T>);
62  return bytesOut;
63  };
64 
65  while (counter < noOfImgs) {
66  size_t bytesToCopyFromGPU;
67  if (0 == counter) { // first iteration
68  bytesToCopyFromGPU = loadImgs();
69  }
70  // perform FFT (out of place) and scale
71  imagesGPU.fft(resultingFFT, handle);
72  dim3 dimBlock(BLOCK_DIM_X, BLOCK_DIM_X);
73  dim3 dimGrid(ceil(outFFTX/(float)dimBlock.x), ceil(outY/(float)dimBlock.y));
74  scaleFFT2D(&dimGrid, &dimBlock,
75  (std::complex<T>*)resultingFFT.d_data,
76  (std::complex<T>*)imagesGPU.d_data,
77  inBatch, resultingFFT.Xdim, resultingFFT.Ydim,
78  outFFTX, outY, d_filter, 1.f/imagesGPU.yxdim, false);
79  gpuErrchk( cudaPeekAtLastError() );
80 
81  std::complex<T>* h_imgStore = h_result + counter * outFFTX * outY;
82  counter += inBatch;
83  // because the spatial data are consecutive, and FFT might be actually bigger than spatial data
84  // (i.e. they can overlap), we need to copy processed data to extra location
85  gpuErrchk(cudaMemcpy((void*)h_tmpStore, (void*)imagesGPU.d_data, bytesToCopyFromGPU, cudaMemcpyDeviceToHost));
86  // get new data from CPU. Now we can overwrite the location, as we have the results already
87  size_t tmp = loadImgs();
88  // copy results back to CPU, to the original memory location, consecutive
89  memcpy(h_imgStore, h_tmpStore, bytesToCopyFromGPU);
90  // and update the no of bytes we should process next iteration
91  bytesToCopyFromGPU = tmp;
92  }
93  handle.clear();
94  delete[] h_tmpStore;
95  cudaFree(d_filter);
96 }

159 void copyInRightOrder(T* d_imgs, T* h_imgs, int xDim, int yDim, bool isWithin,
160  int iStart, int iStop, int jStart, int jStop, size_t jSize,
161  size_t offset1, size_t offset2, size_t maxImgs) {
162  size_t pixelsPerImage = xDim * yDim;
163  size_t counter = 0;
164  bool ready = false;
165  for (int i = iStart; i <= iStop; i++) {
166  for (int j = isWithin ? i + 1 : 0; j < jSize; j++) {
167  if (!ready) {
168  ready = true;
169  j = jStart;
170  continue; // skip first iteration
171  }
172  if (ready) {
173  size_t actualI = offset1 + i;
174  size_t actualJ = offset2 + j;
175  size_t toCopy = jSize - j;
176  // imagine correlation in layers, correlation of 0th img with other is first layer, 1st with other is second etc
177  // compute sum of images in complete layers
178  size_t imgsInPreviousLayers = (((maxImgs - 1) + (maxImgs - actualI)) * (actualI)) / 2;
179  size_t imgsInCurrentLayer = actualJ - actualI - 1;
180  gpuErrchk(cudaMemcpy(h_imgs + (pixelsPerImage * (imgsInPreviousLayers + imgsInCurrentLayer)),
181  d_imgs + (counter * pixelsPerImage),
182  toCopy * pixelsPerImage * sizeof(T),
183  cudaMemcpyDeviceToHost));
184  counter += toCopy;
185  break; // skip to next outer iteration
186  }
187  if ((iStop == i) && (jStop == j)) {
188  return;
189  }
190  }
191  }

261 void computeCorrelations(size_t centerSize, int noOfImgs,
262  void* d_in1, size_t in1Size, void* d_in2, size_t in2Size,
263  int fftBatchSize, size_t in1Offset, size_t in2Offset,
264  GpuMultidimArrayAtGpu<std::complex<T> >& ffts,
265  GpuMultidimArrayAtGpu<T>& imgs, mycufftHandle& handler,
266  T*& result) {
267  bool isWithin = d_in1 == d_in2; // correlation is done within the same buffer
268 
269  dim3 dimBlock(BLOCK_DIM_X, BLOCK_DIM_X);
270  dim3 dimGridCorr(ceil(ffts.Xdim/(float)dimBlock.x), ceil(ffts.Ydim/(float)dimBlock.y));
271  dim3 dimGridCrop(ceil(centerSize/(float)dimBlock.x), ceil(centerSize/(float)dimBlock.y));
272 
273  size_t batchCounter = 0;
274  size_t counter = 0;
275  int origI = 0;
276  int origJ = isWithin ? 0 : -1; // kernel must skip first iteration
277  for (int i = 0; i < in1Size; i++) {
278  for (int j = isWithin ? i + 1 : 0; j < in2Size; j++) {
279  counter++;
280  bool isLastIIter = isWithin ? (i == in1Size - 2) : (i == in1Size -1);
281  if (counter == fftBatchSize || (isLastIIter && (j == in2Size -1)) ) {
282  // kernel must perform last iteration
283  // compute correlation from input buffers. Result are FFT images
284  if (std::is_same<T, float>::value) {
285  correlate<<<dimGridCorr, dimBlock>>>((float2*)d_in1, (float2*)d_in2,
286  (float2*)ffts.d_data, ffts.Xdim, ffts.Ydim,
287  isWithin, origI, i, origJ, j, in2Size);
288  } else {
289  throw std::logic_error("unsupported type");
290  }
291  // convert FFTs to space domain
292  ffts.ifft(imgs, handler);
293  // crop images in space domain, use memory for FFT to avoid realocation
294  cropSquareInCenter<<<dimGridCrop, dimBlock>>>((T*)imgs.d_data,

302  origI = i;
