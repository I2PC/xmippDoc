80 
81  if (fn == "")

88  {
89  MetaData MD;
90  MD.read(fn);

92 
93  FileName fnPSF;
94  if ( MD.getValue(MDL_IMAGE,fnPSF, id) && readVolume )
95  {
96  mode = PSF_FROM_FILE;
97  psfGen.readMapped(fn.getDir()+fnPSF);

101  Nox = aDim.xdim;
102  Noy = aDim.ydim;
103  Noz = aDim.zdim;
104 
105  psfGen().setXmippOrigin();
106 
107  // MULTIDIM_ARRAY(psfVol).setXmippOrigin();
108  }
109  else
110  {
111  std::vector<double> dimV;

114  Nox = (int)dimV[0];
115  Noy = (int)dimV[1];
116  Noz = (int)dimV[2];
117  }
118  String typeS;
119  if (!MD.getValue(MDL_CTF_XRAY_LENS_TYPE, typeS, id))
120  REPORT_ERROR(ERR_ARG_MISSING, MDL::label2Str(MDL_CTF_XRAY_LENS_TYPE) + " argument not present.");

128  lambda *= 1e-9;
129  if (!MD.getValue(MDL_CTF_XRAY_OUTER_ZONE_WIDTH,deltaR, id))
130  REPORT_ERROR(ERR_ARG_MISSING, MDL::label2Str(MDL_CTF_XRAY_OUTER_ZONE_WIDTH) + " argument not present.");

132  if (!MD.getValue(MDL_CTF_XRAY_ZONES_NUMBER,Nzp, id))
133  REPORT_ERROR(ERR_ARG_MISSING, MDL::label2Str(MDL_CTF_XRAY_ZONES_NUMBER) + " argument not present.");
134  if (!MD.getValue(MDL_MAGNIFICATION,Ms, id))
135  REPORT_ERROR(ERR_ARG_MISSING, MDL::label2Str(MDL_MAGNIFICATION) + " argument not present.");
136  if (!MD.getValue(MDL_CTF_SAMPLING_RATE,dxoPSF, id))
137  REPORT_ERROR(ERR_ARG_MISSING, MDL::label2Str(MDL_CTF_SAMPLING_RATE) + " argument not present.");
138  dxoPSF *= 1e-9;
139  if (!MD.getValue(MDL_CTF_SAMPLING_RATE_Z,dzoPSF, id))
140  REPORT_ERROR(ERR_ARG_MISSING, MDL::label2Str(MDL_CTF_SAMPLING_RATE_Z) + " argument not present.");
141  dzoPSF *= 1e-9;
142  if (!MD.getValue(MDL_CTF_LONGITUDINAL_DISPLACEMENT,DeltaZo, id))
143  REPORT_ERROR(ERR_ARG_MISSING, MDL::label2Str(MDL_CTF_LONGITUDINAL_DISPLACEMENT) + " argument not present.");
144  DeltaZo *= 1e-6;
145  }
146  else
147  {
148  FILE *fh_param;
149  if ((fh_param = fopen(fn.c_str(), "r")) == NULL)
150  REPORT_ERROR(ERR_IO_NOTOPEN,
151  (std::string)"XmippXROTF::read: There is a problem "
152  "opening the file " + fn);
153 
154  try
155  {
156  lambda = textToFloat(getParameter(fh_param, "lambda", 0, "2.43"))* 1e-9;
157  deltaR = textToFloat(getParameter(fh_param, "outer_zone_width", 0, "40")) * 1e-9;
158  Nzp = textToFloat(getParameter(fh_param, "zones_number", 0, "560"));
159  Ms = textToFloat(getParameter(fh_param, "magnification", 0, "2304"));
160  dxoPSF = textToFloat(getParameter(fh_param, "sampling_rate", 0, "1")) *1e-9;
161  if (checkParameter(fh_param, "z_sampling_rate"))
162  dzoPSF = textToFloat(getParameter(fh_param, "z_sampling_rate", 0)) *1e-9;
163  else
164  dzoPSF = dxoPSF;
165  DeltaZo = textToFloat(getParameter(fh_param, "z_axis_shift", 0, "0")) *1e-6;
166 
167  Nox = textToInteger(getParameter(fh_param, "x_dim", 0, "0"));
168 
169  }
170  catch (XmippError &XE)
171  {
172  std::cout << XE << std::endl;
173  REPORT_ERROR(ERR_IO_NOREAD, (std::string)"There is an error reading " + fn);
174  }
175  fclose(fh_param);
176  }

274 /* Produce Side Information ------------------------------------------------ */
275 void XRayPSF::calculateParams(double _dxo, double _dzo, double threshold)
276 {
277 
278  dxo = (_dxo > 0)? _dxo : dxoPSF;
279  dzo = (_dzo > 0)? _dzo : dxo;
280 
282 
283  Rlens = (4 * Nzp * deltaR)*0.5; // Eq. 9.13 from Soft X-rays and .... David Attwood
284  Flens = (2 * Rlens * deltaR)/lambda;
285 
286  Zo = (1 + 1 / Ms) * Flens;
287  Zi = Zo * Ms;
288  dxi = dxo * Ms;
289  dxiMax = lambda * Zi / (2 * Rlens);
290  DoF = 4*deltaR*deltaR/lambda;
291 
292  if(verbose > 0)
293  {
294  show();
295  verbose++; // Show only once the param adjust info
296  }
297 
298  if (mode == PSF_FROM_FILE)
299  {
300  T.initIdentity(4);
301  double scaleFactor = dxo/dxoPSF;
302  // double scaleFactorZ = dxo/dzoPSF;
303  double scaleFactorZ = 1;
304 
305  dMij(T, 0, 0) = scaleFactor;
306  dMij(T, 1, 1) = scaleFactor;
307  dMij(T, 2, 2) = scaleFactorZ;
311  MultidimArray<double> &mdaPsfVol = psfVol();
312 
313  mdaPsfVol.resize(1, (size_t)(Noz/scaleFactorZ),
314  (size_t)(Noy/scaleFactor),
315  (size_t)(Nox/scaleFactor), false);
316 
317  mdaPsfVol.setXmippOrigin();
318 
319  applyGeometry(LINEAR, mdaPsfVol, psfGen(), T,
320  IS_INV, DONT_WRAP);
321 
322  psfGen.clear(); // Free mem in case image is not mapped
323 
324  // Rescale the PSF values to keep the normalization in each X-Y plane
325  mdaPsfVol *= scaleFactor*scaleFactor;
326 
327  // Reset the sampling values so now psfVol matches input phantom's sampling
328  dxoPSF = dxo;
329  // dzoPSF = dzo;
330  T.initIdentity(4);
331 
332  if (threshold != 0.)
333  reducePSF2Slabs(threshold);
334  }
335  else // Mask is used when creating PSF on demand
336  {
337  mask = new MultidimArray<double>;
338  }
339 }

341 void XRayPSF::reducePSF2Slabs(double threshold)
342 {
343  // find max position
344  ArrayCoord maxPos;
345  MultidimArray<double> &mdaPsfVol = psfVol();
346 
347  //MULTIDIM_ARRAY_GENERIC(PSFGen).maxIndex(maxPos);
348  mdaPsfVol.maxIndex(maxPos);
349  double maxValue = A3D_ELEM(mdaPsfVol,maxPos.z,maxPos.y,maxPos.x);
350  threshold *= maxValue;
351  slabIndex.clear();
352 
353  double maxTemp = maxValue;
354  for (int k = 0; k > STARTINGZ(mdaPsfVol); --k)
355  {
356  double tempV = A3D_ELEM(mdaPsfVol,k,maxPos.y,maxPos.x);
357  if ( maxTemp - tempV > threshold)
358  {
359  slabIndex.insert(slabIndex.begin(), k);
360  maxTemp = tempV;
361  }
362  }
363  slabIndex.insert(slabIndex.begin(), STARTINGZ(mdaPsfVol));
364 
365  maxTemp = maxValue;
366  for (int k = 0; k < FINISHINGZ(mdaPsfVol); ++k)
367  {
368  double tempV = A3D_ELEM(mdaPsfVol,k,maxPos.y,maxPos.x);
369  if ( maxTemp - tempV > threshold)
370  {
371  slabIndex.push_back(k);
372  maxTemp = tempV;
373  }
374  }
375  slabIndex.push_back(FINISHINGZ(mdaPsfVol));
376 
377 } //reducePSF2Slabs

458  FourierTransformer ftGenOTF(FFTW_BACKWARD);
459  switch (mode)

462  {
463  if (type == IDEAL_FRESNEL_LENS)
464  generatePSFIdealLens(PSFi, Zpos);

466  }

520 /* Generate the Intensity PSF for a specific XR microscope configuration ------------- */
521 void XRayPSF::generatePSF()
522 {
523  calculateParams(dxoPSF, dzoPSF);
524  adjustParam();
525 
526  // PSF = new Image<double>(Nox, Noy, Noz);
527  // VOLMATRIX(*PSF).setXmippOrigin();
528 
529  MultidimArray<double> &mPsfVol = MULTIDIM_ARRAY(psfVol);
530  mPsfVol.resize(1,Noz, Niy, Nix, false);
531  mPsfVol.setXmippOrigin();
532 
533  MultidimArray<double> PSFi;
534 
535  init_progress_bar(ZSIZE(mPsfVol));
536 
537  for (int k = STARTINGZ(mPsfVol), k2 = 0; k <= FINISHINGZ(mPsfVol); k++, k2++)
538  {
539  /* We keep sign of Z, Zo and DeltaZo positives in object space for the sake of simplicity in calculations,
540  * it is, they increase opposite to Zdim in Xmipp reference system, so this is the reason to calculate
541  * the plane Z by subtracting k*dzoPSF which keeps positive in XMIPP system.
542  */
543 
544  double Z = Zo + DeltaZo - k*dzoPSF;
545 
546  switch (type)
547  {
548  case IDEAL_FRESNEL_LENS:
549  {
550  generatePSFIdealLens(PSFi, Z);
551  break;
552  }
553  case ANALYTIC_ZP:
554  {
555  REPORT_ERROR(ERR_NOT_IMPLEMENTED, "Zone plates algorithm not implemented yet.");
556  break;
557  }
558  }
559 
560  CenterFFT(PSFi, false);
561  mPsfVol.setSlice(k, PSFi);
562 
563  progress_bar(k2+1);
564  }
565 
566  /* When PSF is generated from IDEAL_FRESNEL_LENS, dimensions are adjusted to avoid subsampling,
567  * in that case it is not recommended to crop XY plane to avoid the loss of normalization in PSF plane. */
568 
569  // if (Nix != Nox || Niy != Noy)
570  // mPsfVol.selfWindow(STARTINGZ(mPsfVol),
571  // FIRST_XMIPP_INDEX(Noy),FIRST_XMIPP_INDEX(Nox),
572  // FINISHINGZ(mPsfVol),
573  // FIRST_XMIPP_INDEX(Noy)+Noy-1,FIRST_XMIPP_INDEX(Nox)+Nox-1);

666 void XRayPSF::adjustParam()

671 
672  if (mode == GENERATE_PSF)
673  {
674  switch (type)
675  {
676  case IDEAL_FRESNEL_LENS:
677  {
678  // Reset the image plane sampling value
679  dxi = dxo * Ms;
680 
681  dxl = lambda*Zi / (Nox * dxi); // Pixel X-size en the plane of lens aperture
682  dyl = lambda*Zi / (Noy * dxi); // Pixel Y-size en the plane of lens aperture
683 
684  deltaZMaxX = Zo*((Rlens*2*dxl)/(Rlens*2*dxl - Zo*lambda) - 1);
685  deltaZMaxY = Zo*((Rlens*2*dyl)/(Rlens*2*dyl - Zo*lambda) - 1);
686  deltaZMinX = Zo*((Rlens*2*dxl)/(Rlens*2*dxl + Zo*lambda) - 1);
687  deltaZMinY = Zo*((Rlens*2*dyl)/(Rlens*2*dyl + Zo*lambda) - 1);
688 
689

691  {
692  std::cout << std::endl;
693  std::cout << "----------------------" << std::endl;
694  std::cout << "XrayPSF::Param adjust:" << std::endl;

697  std::cout << "Larger volume Z plane to be calculated = " << (ABS(DeltaZo)+(Noz-1)/2*dzo)*1e6 << " um" << std::endl;
698  std::cout << "Larger allowed discrete Z plane (x,y) = (" << deltaZMaxX*1e6 << ", " << deltaZMaxY*1e6 << ") um" << std::endl;
699  }
700 
701  if (dxi>dxiMax)
702  {
703  Nix = (size_t)ceil(Nox * dxi/dxiMax);
704  Niy = (size_t)ceil(Noy * dxi/dxiMax);
705 
706  dxi *= Nox/Nix;
707 
708  AdjustType = PSFXR_INT;
709 
710  if (verbose > 1)
711  std::cout << "XrayPSF: Image plane sampling too small: increasing resolution" << std::endl;
712 
713  }
714  else
715  {
716  if (dxi < pupileSizeMin/Nox * dxiMax)
717  {
718  Nix = (size_t)ceil(pupileSizeMin * dxiMax/dxi);
719  AdjustType = PSFXR_ZPAD;
720  }
721  if (dxi < pupileSizeMin/Noy * dxiMax)
722  {
723  Niy = (size_t)ceil(pupileSizeMin * dxiMax/dxi);
724  AdjustType = PSFXR_ZPAD;
725  }
726  if (DeltaZo + Noz/2*dzo > deltaZMaxX)
727  {
728  Nix = std::max(Nix,(size_t)ceil(Zi*Rlens*2*fabs(DeltaZo+Noz/2*dzo)/(Zo*dxi*(Zo+DeltaZo+Noz/2*dzo))));
729  AdjustType = PSFXR_ZPAD;
730  }
731  if (DeltaZo - Noz/2*dzo < deltaZMinX)
732  {
733  Nix = std::max(Nix,(size_t)ceil(Zi*Rlens*2*fabs(DeltaZo-Noz/2*dzo)/(Zo*dxi*(Zo+DeltaZo-Noz/2*dzo))));
734  AdjustType = PSFXR_ZPAD;
735  }
736  if (DeltaZo + Noz/2*dzo > deltaZMaxY)
737  {
738  Niy = std::max(Niy,(size_t)ceil(Zi*Rlens*2*fabs(DeltaZo+Noz/2*dzo)/(Zo*dxi*(Zo+DeltaZo+Noz/2*dzo))));
739  AdjustType = PSFXR_ZPAD;
740  }
741  if ( DeltaZo - Noz/2*dzo < deltaZMinY)
742  {
743  Niy = std::max(Niy,(size_t)ceil(Zi*Rlens*2*fabs(DeltaZo-Noz/2*dzo)/(Zo*dxi*(Zo+DeltaZo-Noz/2*dzo))));
744  AdjustType = PSFXR_ZPAD;
745  }
746  }
747 
748  dxl = lambda*Zi / (Nix * dxi); // Pixel X-size en the plane of lens aperture
749  dyl = lambda*Zi / (Niy * dxi); // Pixel Y-size en the plane of lens aperture
750 
751  if (verbose > 1)
752  {
753  if (AdjustType!=PSFXR_STD)
754  {
755  if (AdjustType==PSFXR_ZPAD)
756  std::cout << "XrayPSF: Image plane size too small: increasing size" << std::endl;
757  std::cout << "New slice dimensions: " << "(Nix, Niy) = (" << Nix << ", " << Niy << ")" << std::endl;
758  }
759  std::cout << "(dxl, dyl) = (" << dxl*1e6 << ", " << dyl*1e6 << ") um" << std::endl;
760  std::cout << "Pupile Diameter in pixels (NpX, NpY) = (" << ceil(2*Rlens / dxl)
761  << ", " << ceil(2*Rlens / dyl) << ")" << std::endl;
762  std::cout << std::endl;
763  }
764 
765  // Calculate the mask to be applied when generating PSFIdealLens
766 
767  mask->initZeros(Niy,Nix);
768  mask->setXmippOrigin();
769 
770  double Rlens2 = Rlens*Rlens;
771  // double auxY = dyl*(1 - (int)Niy);
772  // double auxX = dxl*(1 - (int)Nix);
773 
774  for (int i = STARTINGY(*mask); i <= FINISHINGY(*mask); ++i)
775  {
776  double y = dyl * i;
777  double y2 = y * y;
778  for (int j = STARTINGX(*mask); j <= FINISHINGX(*mask); ++j)// Circular mask
779  {
781  double x = dxl * j;

784  }
785  }
786 
787 
788  break;
789  }
790  case ANALYTIC_ZP:
791  {
792  REPORT_ERROR(ERR_NOT_IMPLEMENTED, "Zone Plate algorithm not implemented yet.");
793  break;
794  }
795  }
796  }
797  else
798  {
799  // OTF.clear(); // Only for each projection
800  }
801 
802  if (verbose > 1)
803  --verbose;
804 }

807 void lensPD(MultidimArray<std::complex<double> > &Im, double Flens, double lambda, double dx, double dy)
808 {

820

822  {
823  x = dx * j;
824  phase = K * (x * x + y2);
825  A2D_ELEM(Im,i,j) = std::complex<double>(cos(phase),sin(phase));
826  }
827  }
828 }
