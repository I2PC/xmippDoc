128  from xmipp3.viewers.plotter import XmippPlotter
129  from pwem import emlib
130 
131  protML._plot_count = 0

141  def doPlot(plotName):
142  return plotName in selectedPlots
143 
144  # Remove 'mirror' from list if DoMirror is false
145  if doPlot('doShowMirror') and not protML.doMirror:
146  selectedPlots.remove('doShowMirror')
147 
148  n = len(selectedPlots)
149  if n == 0:
150  #showWarning("ML2D plots", "Nothing to plot", protML.master)
151  print("No plots")
152  return
153  elif n == 1:
154  gridsize = [1, 1]
155  elif n == 2:
156  gridsize = [2, 1]
157  else:
158  gridsize = [2, 2]
159 
160  xplotter = XmippPlotter(x=gridsize[0], y=gridsize[1])
161 
162  # Create data to plot
163  iters = range(1, lastIter+1)
164  ll = []
165  pmax = []
166  for iter in iters:
167  logs = protML._getIterClasses(it=iter, block='info')
168  md = emlib.MetaData(logs)
169  id = md.firstObject()
170  ll.append(md.getValue(emlib.MDL_LL, id))
171  pmax.append(md.getValue(emlib.MDL_PMAX, id))
172 
173  if doPlot('doShowLL'):
174  a = xplotter.createSubPlot('Log-likelihood (should increase)', 'iterations', 'LL', yformat=True)
175  a.plot(iters, ll)
176 
177  #Create plot of mirror for last iteration
178  if doPlot('doShowMirror'):
179  from numpy import arange
180  from matplotlib.ticker import FormatStrFormatter
181  md = emlib.MetaData(refs)
182  mirrors = [md.getValue(emlib.MDL_MIRRORFRAC, id) for id in md]
183  nrefs = len(mirrors)
184  ind = arange(1, nrefs + 1)
185  width = 0.85
186  a = xplotter.createSubPlot('Mirror fractions on last iteration', 'classes', 'mirror fraction')
187  a.set_xticks(ind + 0.45)
188  a.xaxis.set_major_formatter(FormatStrFormatter('%1.0f'))
189  a.bar(ind, mirrors, width, color='b')
190  a.set_ylim([0, 1.])
191  a.set_xlim([0.8, nrefs + 1])
192 
193  if doPlot('doShowPmax'):
194  a = xplotter.createSubPlot('Probabilities distribution', 'iterations', 'Pmax/Psum')
195  a.plot(iters, pmax, color='green')
196 
197  if doPlot('doShowSignalChange'):
198  md = emlib.MetaData()
199  for iter in iters:
200  fn = protML._getIterClasses(it=iter, block='classes')
201  md2 = emlib.MetaData(fn)
202  md2.fillConstant(emlib.MDL_ITER, str(iter))
203  md.unionAll(md2)
204  # 'iter(.*[1-9].*)@2D/ML2D/run_004/ml2d_iter_refs.xmd')
205  # a = plt.subplot(gs[1, 1])
206  # print("md: %s" % md)
207  md2 = emlib.MetaData()
208  md2.aggregate(md, emlib.AGGR_MAX, emlib.MDL_ITER, emlib.MDL_SIGNALCHANGE, emlib.MDL_MAX)
209  signal_change = [md2.getValue(emlib.MDL_MAX, id) for id in md2]
210  xplotter.createSubPlot('Maximum signal change', 'iterations', 'signal change')

212
