82 
83  Image<double> I;
84  if (!fnCentralSlices.empty())
85  {
86  // Choose the starting and finishing slice
87  if (Nslices==-1)
88  Nslices=ZSIZE(mV);
89  int kmiddle=ZSIZE(mV)/2;
90  int k0=kmiddle-Nslices/2;
91  int kF=k0+Nslices-1;
92 
93  // Resize the output image
94  int NslicesPerRow=maxWidth/XSIZE(mV);
95  if (NslicesPerRow==0)
96  NslicesPerRow=1;
97  else if (NslicesPerRow>Nslices)
98  NslicesPerRow=Nslices;
99  int Nrows=(int)ceil(((float)Nslices)/NslicesPerRow);
100  if (Nrows==0)
101  Nrows=1;
102  I().resizeNoCopy(Nrows*YSIZE(mV)+(Nrows-1)*sep,NslicesPerRow*XSIZE(mV)+(NslicesPerRow-1)*sep);
103  I().initConstant(mV.computeMax());
104 
105  // Copy the slices into the output image
106  int outI=0, outJ=0;
107  for (int k=k0; k<=kF; ++k)
108  {
109  int outi=outI*(YSIZE(mV)+sep);
110  int outj=outJ*(XSIZE(mV)+sep);
111  for (size_t i=0; i<YSIZE(mV); ++i, ++outi)
112  memcpy(&IMGPIXEL(I,outi,outj),&A3D_ELEM(mV,k,i,0),XSIZE(mV)*sizeof(double));
113  outJ++;
114  if (outJ==NslicesPerRow)
115  {
116  outJ=0;
117  outI++;
118  }
119  }
120  I().rangeAdjust(0,255);
121  I.write(fnCentralSlices);
122  }
123  if (fnProjections!="")
124  {
125  // Take projections
126  MultidimArray<double> pXY, pYZ, pXZ;
127  pXY.initZeros(YSIZE(mV),XSIZE(mV));
128  pXZ.initZeros(ZSIZE(mV),XSIZE(mV));
129  pYZ.initZeros(ZSIZE(mV),YSIZE(mV));
130  double maxVal=-1e38;
131  FOR_ALL_ELEMENTS_IN_ARRAY3D(mV)
132  {
133  double v=A3D_ELEM(mV,k,i,j);
134  A2D_ELEM(pXY,i,j)+=v;
135  A2D_ELEM(pXZ,k,j)+=v;
136  A2D_ELEM(pYZ,k,i)+=v;
137  maxVal=std::max(A2D_ELEM(pXY,i,j),maxVal);
138  maxVal=std::max(A2D_ELEM(pXZ,k,j),maxVal);
139  maxVal=std::max(A2D_ELEM(pYZ,k,i),maxVal);
140  }
141 
142  // Now produce output image
143  I().resizeNoCopy(XMIPP_MAX(YSIZE(mV),ZSIZE(mV)),2*(XSIZE(mV)+sep)+YSIZE(mV));
144  I().initConstant(maxVal);
145  int outi=0, outj=0;
146  for (size_t i=0; i<YSIZE(pXY); ++i, ++outi)
147  memcpy(&IMGPIXEL(I,outi,outj),&A2D_ELEM(pXY,i,0),XSIZE(pXY)*sizeof(double));
148 
149  outi=0;
