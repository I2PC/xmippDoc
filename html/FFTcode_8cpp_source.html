83 /* This is an FFT procedure which has been modified to do the evaluation
84  at K nodes given N coefficients. This is used to speed up convolution
85  by zero-striped sequences. Expects K to be larger than N */
86 /* arguments
87 
88  reals, imags - input double arrays each of size N
89  r_vals, i_vals - output double arrays each of size K
90  N is the number of coefficients, K the number of samples
91 
92  Assumes that coefficients are ordered from 0 to N-1.
93  NOTA BENE!
94  When the evaluation is complete, the data is in bit-reverse
95  order.
96  Since the purpose of this is to speed up convolution,
97  once the series is evaluated it should be left in bit reversed order.
98  In this case, brflag should be set to 0. If you set brflag to
99  1, then the results are bit-reversed
100 
101  This evaluation is modeled on polynomial division tree.
102 
103  workspace needs to be double array of size 2 * K
104 
105 */
106 
107 void FFTEval( double *reals, double *imags,
108  double *r_vals,double *i_vals,
109  int N,
110  int K,
111  double *workspace,
112  int brflag)
113 {
114  const double *rootptr;
115  double *rcptr, *icptr, *rrptr, *irptr;
116  double rtemp, itemp;
117  int rem2, j, k, l, toggle;
118 
119  double tmproot0, tmproot1;
120 
121  memcpy(workspace, reals, sizeof(double) * N);
122  memcpy(workspace+K, imags, sizeof(double) * N);
123 
124  rootptr = r4096; /* point to roots of unity in BR order */
125  rem2 = N/2; /* current size of remainders */
126  rcptr = workspace; /* points to real part of current poly coeff */
127  icptr = workspace+K; /* points to imag part of current poly coeff */
128  rrptr = r_vals; /* points to real part of current remainder coeff */
129  irptr = i_vals; /* points to imag part of current remainder coeff */
130  toggle = 0;
131 
132  for (l=0; l < ilog2fast(N); l++)
133  {
134  for (k=0; k<K; k+=rem2)
135  {
136  tmproot0 = rootptr[0]; tmproot1 = rootptr[1];
137  for (j=0; j<rem2; j++)
138  {
139  compmult(tmproot0,tmproot1,rcptr[rem2+j],icptr[rem2+j],
140  rtemp,itemp);
141  compadd(rtemp,itemp,rcptr[j],icptr[j],
142  rrptr[j],irptr[j]);
143  }
144  rrptr += rem2;
145  irptr += rem2;
146  rootptr += 2;
147  toggle++;
148  if ((toggle % 2) == 0)
149  {
150  /* check l==0 here. If l==0, then we need to
151  fake it into thinking that there are multiple
152  copies of the input coeffs */
153  if (l != 0)
154  {
155  rcptr += 2*rem2;
156  icptr += 2*rem2;
157  }
158  else
159  {
160  rcptr = workspace;
161  icptr = workspace+K;
162  }
163  }
164  } /* closes k loop - done with one level */
165 
166  if ((l % 2) == 0)
167  {
168  rcptr = r_vals;
169  icptr = i_vals;
170  rrptr = workspace;
171  irptr = workspace + K;
172  }
173  else
174  {
175  rcptr = workspace;
176  icptr = workspace + K;
177  rrptr = r_vals;
178  irptr = i_vals;
179  }
180 
181  rem2 /= 2;
182  rootptr = r4096;
183  }
184  if ((l % 2) == 0)
185  {
186  memcpy(r_vals, workspace, sizeof(double) * K);
187  memcpy(i_vals, workspace+K, sizeof(double) * K);
188  }
189 
190  if (brflag == 1) { /* bitreverse the data */
191  bitreverse(r_vals,K, workspace);
192  bitreverse(i_vals,K, workspace);
193  }
194 
195 }

222 void FFTInterp( double *reals, double *imags,
223  double *r_vals, double *i_vals,
224  int N,
225  int K,
226  double *workspace,
227  int brflag )
228 {
229  const double *rootptr;

231  double rtemp0, itemp0, rtemp1, itemp1, dn;
232  int degree, j, k, l, toggle, upperlim;
233  double rptr0, rptr1, rptr2, rptr3;
234  double tmprval, tmpival;
235 
236  dn = (double) N;
237 
238  /* normalize and permute if necessary */
239  for (j=0; j<N; j++)
240  {
241  workspace[j] = reals[j]/dn;
242  workspace[j+N] = imags[j]/dn;
243  }
244  if (brflag == 1)
245  {
246  bitreverse(workspace, N, workspace+(2*N));
247  bitreverse(workspace+N, N, workspace+(2*N));
248  }
249 
250  rootptr = r4096; /* point to roots of unity in BR order */
251  degree = 1; /* next order of polynomials - stride length */
252  rlptr = workspace; /* points to real part of lower order polys */
253  ilptr = workspace+N; /* points to imag part of lower order polys */
254  rhptr = r_vals; /* points to real part of higher order polys */
255  ihptr = i_vals; /* points to imag part of higher order polys */
256  toggle = 0;
257 
258  for (l=1; l <= ilog2fast(K); l++)
259  {
260  upperlim = N/(2*degree);
261  for (k=0; k < upperlim; k++)
262  {
263  for (j=0; j < degree; j++)
264  {
265  compadd(rlptr[j],ilptr[j],rlptr[j+degree],ilptr[j+degree],
266  rhptr[j],ihptr[j]);
267  }
268  rhptr += degree;
269  ihptr += degree;
270 
271  rptr0 = rootptr[0]; rptr1 = rootptr[1];
272  rptr2 = rootptr[2]; rptr3 = rootptr[3];
273 
274  for (j=0; j < degree; j++)
275  {
276 
277  compmult(rptr0,-rptr1,rlptr[j],ilptr[j],
278  rtemp0,itemp0);
279  compmult(rptr2,-rptr3,rlptr[j+degree],ilptr[j+degree],
280  rtemp1,itemp1);
281  compadd(rtemp0,itemp0,rtemp1,itemp1,
282  rhptr[j],ihptr[j]);
283 
284  }
285 
286  rhptr += degree;
287  ihptr += degree;
288  rlptr += (2*degree);
289  ilptr += (2*degree);
290  rootptr += 4;
291  } /* end of k loop */
292 
293  if ((l % 2) != 0)
294  {
295  rlptr = r_vals;
296  ilptr = i_vals;
297  rhptr = workspace;
298  ihptr = workspace + N;
299  }
300  else
301  {
302  rhptr = r_vals;
303  ihptr = i_vals;
304  rlptr = workspace;
305  ilptr = workspace + N;
306  }
307  degree *= 2;
308  rootptr = r4096;
309  toggle++;
310 
311  } /* end of l loop */
312 
313  if ((toggle % 2) == 0)
314  {
315  memcpy(r_vals, workspace, sizeof(double) * N);
316  memcpy(i_vals, workspace+N, sizeof(double) * N);
317  }
318 
319  for (j=0; j<K; j++)
320  {
321  tmprval = r_vals[j]; tmpival = i_vals[j];
322  for (k=K; k<N; k+=K)
323  {
324  tmprval += r_vals[j+k];
325  tmpival += i_vals[j+k];
326  }
327  r_vals[j] = tmprval; i_vals[j] = tmpival;
328  }
329 
330 }
