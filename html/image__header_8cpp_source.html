167  void processImage(const FileName &fnImg, const FileName &fnImgOut, const MDRow &rowIn, MDRow &rowOut)
168  {
169 
170  ImageGeneric img;
171 
172  switch (operation)
173  {
174  case HEADER_PRINT:
175  img.read(fnImg, _HEADER_ALL);
176  img.print();
177  break;
178  case HEADER_EXTRACT:
179  img.read(fnImg, _HEADER_ALL);
180  rowOut = img.getGeometry();
181  if (round_shifts)
182  roundShifts(rowOut);
183  rowOut.setValue(MDL_IMAGE, fnImgOut);
184  break;
185  case HEADER_ASSIGN:
186  rowOut = rowIn;
187  if (round_shifts)
188  roundShifts(rowOut);
189  img.readApplyGeo(fnImg, rowOut, params);
190  img.setDataMode(_HEADER_ALL);
191  img.write(fnImg, ALL_IMAGES, fnImg.isInStack(), WRITE_REPLACE);
192  break;
193  case HEADER_RESET:
194  img.read(fnImg, _HEADER_ALL);
195  img.initGeometry();
196  img.write(fnImg, ALL_IMAGES, fnImg.isInStack(), WRITE_REPLACE);
197  break;
198  case HEADER_SAMPLINGRATE:
199  {
200  img.read(fnImg, _HEADER_ALL);
201  if (sampling < 0)
202  {
203  double samplingRead;
204  img.image->MDMainHeader.getValue(MDL_SAMPLINGRATE_X, samplingRead);
205  std::cout << samplingRead << std::endl;
206  }
207  else
208  {
209  img.image->MDMainHeader.setValue(MDL_SAMPLINGRATE_X, sampling);
210  img.image->MDMainHeader.setValue(MDL_SAMPLINGRATE_Y, sampling);
211  img.image->MDMainHeader.setValue(MDL_SAMPLINGRATE_Z, sampling);
212  img.write(fnImg, ALL_IMAGES, fnImg.isInStack(), WRITE_REPLACE);
213  std::cout << "New sampling rate (Angstrom) = " << sampling << std::endl;
214  }
215  }
216  break;
217  case HEADER_TREE:
218  {
219  XmippH5File H5File;
220  FileName filename = fnImg.removeAllPrefixes().removeFileFormat();
221 
222  if (H5File.isHdf5(filename.c_str()))
223  {
224  H5File.openFile(fnImg.removeAllPrefixes().removeFileFormat(), H5F_ACC_RDONLY);
225  H5File.showTree();
226  }
227  else
228  REPORT_ERROR(ERR_IMG_UNKNOWN, "Unknown file format to display its data structure.");
229  }
230  break;
231  }
232  }
