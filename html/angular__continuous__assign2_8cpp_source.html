152 void ProgAngularContinuousAssign2::preProcess()
153 {
154  // Read the reference volume
155  Image<double> V;
156  if (rank==0)
157  {
158  V.read(fnVol);
159  V().setXmippOrigin();
160  Xdim=XSIZE(V());
161  }
162  else
163  {
164  size_t ydim, zdim, ndim;
165  getImageSize(fnVol, Xdim, ydim, zdim, ndim);
166  }
167 
168  Ip().initZeros(Xdim,Xdim);
169  E().initZeros(Xdim,Xdim);
170  Ifilteredp().initZeros(Xdim,Xdim);
171  Ifilteredp().setXmippOrigin();
172 
173  // Construct mask
174  if (Rmax<0)
175  Rmax=Xdim/2;
176  Mask mask;
177  mask.type = BINARY_CIRCULAR_MASK;
178  mask.mode = INNER_MASK;
179  mask.R1 = Rmax;
180  mask.generate_mask(Xdim,Xdim);
181  mask2D=mask.get_binary_mask();
182  iMask2Dsum=1.0/mask2D.sum();
183 
184  // Construct reference covariance
185  FOR_ALL_DIRECT_ELEMENTS_IN_MULTIDIMARRAY(mask2D)
186  if (DIRECT_MULTIDIM_ELEM(mask2D,n))
187  DIRECT_MULTIDIM_ELEM(E(),n)=rnd_gaus(0,1);
188  covarianceMatrix(E(),C0);
189  FOR_ALL_ELEMENTS_IN_MATRIX2D(C0)
190  {
191  double val=MAT_ELEM(C0,i,j);
192  if (val<0.5)
193  MAT_ELEM(C0,i,j)=0;
194  else
195  MAT_ELEM(C0,i,j)=1;
196  }
197 
198  // Construct projector
199  if (rank==0)
200  projector = new FourierProjector(V(),pad,Ts/maxResol,BSPLINE3);
201  else
202  projector = new FourierProjector(pad,Ts/maxResol,BSPLINE3);
203 
204  // Low pass filter
205  filter.FilterBand=LOWPASS;
206  filter.w1=Ts/maxResol;
207  filter.raised_w=0.02;
208 
209  // Transformation matrix
210  A.initIdentity(3);
211 
212  // Continuous cost
213  if (optimizeGrayValues)
214  contCost = CONTCOST_L1;
215  else
216  contCost = CONTCOST_CORR;
217 }

327 double continuous2cost(double *x, void *_prm)
328 {
329  double a=x[1];
330  double b=x[2];
331  double deltax=x[3];
332  double deltay=x[4];
333  double scalex=x[5];
334  double scaley=x[6];
335  double scaleAngle=x[7];
336  double deltaRot=x[8];
337  double deltaTilt=x[9];
338  double deltaPsi=x[10];
339  double deltaDefocusU=x[11];
340  double deltaDefocusV=x[12];
341  double deltaDefocusAngle=x[13];
342  ProgAngularContinuousAssign2 *prm=(ProgAngularContinuousAssign2 *)_prm;
343  if (prm->maxShift>0 && deltax*deltax+deltay*deltay>prm->maxShift*prm->maxShift)
344  return 1e38;
345  if (fabs(scalex)>prm->maxScale || fabs(scaley)>prm->maxScale)
346  return 1e38;
347  if (fabs(deltaRot)>prm->maxAngularChange || fabs(deltaTilt)>prm->maxAngularChange || fabs(deltaPsi)>prm->maxAngularChange)
348  return 1e38;
349  if (fabs(a-prm->old_grayA)>prm->maxA)
350  return 1e38;
351  if (fabs(b)>prm->maxB*prm->Istddev)
352  return 1e38;
353  if (fabs(deltaDefocusU)>prm->maxDefocusChange || fabs(deltaDefocusV)>prm->maxDefocusChange)
354  return 1e38;
355 // MAT_ELEM(prm->A,0,0)=1+scalex;
356 // MAT_ELEM(prm->A,1,1)=1+scaley;
357 // In Matlab
358 // syms sx sy t
359 // R=[cos(t) -sin(t); sin(t) cos(t)]
360 // S=[1+sx 0; 0 1+sy]
361 // simple(transpose(R)*S*R)
362 // [ sx - sx*sin(t)^2 + sy*sin(t)^2 + 1, -sin(2*t)*(sx/2 - sy/2)]
363 // [ -sin(2*t)*(sx/2 - sy/2), sy + sx*sin(t)^2 - sy*sin(t)^2 + 1]
364  double sin2_t=sin(scaleAngle)*sin(scaleAngle);
365  double sin_2t=sin(2*scaleAngle);
366  MAT_ELEM(prm->A,0,0)=1+scalex+(scaley-scalex)*sin2_t;
367  MAT_ELEM(prm->A,0,1)=0.5*(scaley-scalex)*sin_2t;
368  MAT_ELEM(prm->A,1,0)=MAT_ELEM(prm->A,0,1);
369  MAT_ELEM(prm->A,1,1)=1+scaley-(scaley-scalex)*sin2_t;
370  MAT_ELEM(prm->A,0,2)=prm->old_shiftX+deltax;
371  MAT_ELEM(prm->A,1,2)=prm->old_shiftY+deltay;
372  return tranformImage(prm,prm->old_rot+deltaRot, prm->old_tilt+deltaTilt, prm->old_psi+deltaPsi,
373  a, b, prm->A, deltaDefocusU, deltaDefocusV, deltaDefocusAngle, LINEAR);
374 }
