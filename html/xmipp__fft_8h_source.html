362 
363  // Execute FFT in 2 dimensions before 3D.
364  if (forward)
365  {

367  {
368  // Pointers to upper and lower half matrix.
369  T *upperHalfPtr, *lowerHalfSrcPtr, *lowerHalfDestPtr;
370

375
