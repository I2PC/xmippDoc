213  tagName[ltName] = '\0';

215  delete [] tagName;

217 
218  /*total bytes in tag/tag directory including all sub-directories
219  * (new for DM4). Actually, we don't use it*/
220  fseek(fimg, 8, SEEK_CUR);
221 
222  size_t id = header->tags.addObject();
223 
224  nodeId++;
225  header->tags.setValue(MDL_DM3_NODEID, nodeId, id);
226  header->tags.setValue(MDL_DM3_PARENTID, parentId, id);
227  header->tags.setValue(MDL_DM3_IDTAG, idTag, id);
228  header->tags.setValue(MDL_DM3_TAGNAME, stagName, id);
229 
230  /* Reading tags ===================================================================*/
231  if (idTag == 20) // Tag directory
232  {
233  unsigned char dummy;
234  int nTags;
235  xmippFREAD(&dummy,sizeof(unsigned char),1,fimg,false); // 1 = sorted (normally = 1)
236  xmippFREAD(&dummy,sizeof(unsigned char),1,fimg,false); // 0 = closed, 1 = open (normally = 0)
237 
238  xmippFREAD(&nTags,sizeof(long),1,fimg,isLE); // number of tags in tag directory
239 
240 
241  header->tags.setValue(MDL_DM3_TAGCLASS,(std::string) "Dir", id);
242  header->tags.setValue(MDL_DM3_SIZE, nTags, id);
243 
244  parentId = nodeId;
245  for (int n=1;n<=nTags;n++) // Rest of directories
246  {
247  readTagDM4(fimg, header, parentId, nodeId, isLE, swap);
248  }
249  return 0;
250  }
251  else if (idTag == 21) // Tag
252  {
253  unsigned int nnum;
254  char buf[4]; // to read %%%% symbols
255 
256  xmippFREAD(&buf,1,4,fimg,false); // To read %%%% symbols
257  xmippFREAD(&nnum,sizeof(long),1,fimg,isLE); // Size of info array
258

261  xmippFREAD(info,sizeof(long),nnum,fimg,isLE); // Reading of Info
262 
263  /* Tag classification =======================================*/
264 
265  if (nnum == 1) // Single entry tag
266  {
267  header->tags.setValue(MDL_DM3_TAGCLASS,(std::string) "Single", id);
268 
269  double tagValue = 0;
270 
271  FREADTagValueDM4(&tagValue,info[0],1,fimg,swap);
272 
273  std::vector<double> vtagValue(1);
274  vtagValue.assign(1,tagValue);
275 
276  header->tags.setValue(MDL_DM3_NUMBER_TYPE, (int) info[0], id);
277  header->tags.setValue(MDL_DM3_VALUE, vtagValue, id);
278  delete []info;
279  return tagValue;
280  }
281  else if(nnum == 3 && info[0]==20) // Tag array
282  {
283  /*nnum = 3
284  info(0) = 20
285  info(1) = number type for all values
286  info(2) = info(nnum) = size of array*/
287 
288  header->tags.setValue(MDL_DM3_TAGCLASS,(std::string) "Array", id);
289 
290  header->tags.setValue(MDL_DM3_NUMBER_TYPE, (int)info[1], id);
291  header->tags.setValue(MDL_DM3_SIZE, (int) info[nnum-1], id);
292  std::vector<double> vtagValue(1);
293  vtagValue.assign(1,(double) ftell(fimg));
294 
295  header->tags.setValue(MDL_DM3_VALUE, vtagValue, id);
296 
297  // Jump the array values
298  int k;
299  if(info[1] == 2 || info[1] == 4)
300  k = 2;
301  else if(info[1] == 3 || info[1] == 5 || info[1] == 6)
302  k = 4;
303  else if(info[1] == 10 )
304  k = 1;
305  else
306  REPORT_ERROR(ERR_ARG_INCORRECT,formatString("unsupported value info[1]=%d, read at %d",info[1],ftell(fimg)));
307 
308  fseek( fimg, ftell(fimg)+(info[nnum-1])*k , SEEK_SET );
309  delete []info;
310  return 0;
311  }
312  else if (info[0]==20 && info[1] == 15) // Tag Group array
313  {
314  /*nnum = size of array
315  info(0) = 20 (array)
316  info(1) = 15 (group)
317  info(2) = 0 (always 0)
318  info(3) = number of values in group
319  info(2*i+3) = number type for value i
320  info(nnum) = size of info array*/
321 
322  header->tags.setValue(MDL_DM3_TAGCLASS, (std::string) "GroupArray", id);
323  header->tags.setValue(MDL_DM3_SIZE, (int)info[3], id);
324  int nBytes=0, k;
325  double fieldValue;
326  for (int n=1;n<=info[3];n++)
327  {
328  fieldValue=0;
329  FREADTagValueDM4(&fieldValue,info[3+2*n],1,fimg,swap);
330 
331  if(info[3+2*n] == 2 || info[3+2*n] == 4)
332  k = 2;
333  else if(info[3+2*n] == 3 || info[3+2*n] == 5)
334  k = 4;
335  else if(info[3+2*n] == 10 )
336  k = 1;
337  nBytes+=k;
338  }
339 
340  // Jump the array values
341  fseek( fimg, ftell(fimg)+(info[nnum-1]-1)*nBytes , SEEK_SET );
342  delete []info;
343  return 0;
344  }
345  else if (info[0] == 15) // Tag Group (struct)
346  {
347  /* info[1] = length group name (always =0)
348  info[2] = number of entries in group */
349  /*nnum = size of info array
350  info(1) = 0fh
351  info(3) = number of values in group
352  info(2*i+3) = number type for value i
353  Other info entries are always zero*/
354 
355  header->tags.setValue(MDL_DM3_TAGCLASS, (std::string) "Group", id);
356  header->tags.setValue(MDL_DM3_SIZE,(int) info[2], id);
357  std::vector<double> vtagValue((int)info[2]);
358  for (int n=1;n<=info[2];n++)
359  {
360  double fieldValue=0;
361  FREADTagValueDM4(&fieldValue,info[2+2*n],1,fimg,swap);
362  vtagValue.assign(n,fieldValue);
363  }
364  header->tags.setValue(MDL_DM3_VALUE, vtagValue, id);
365  delete []info;
366  return 0;
367  }
368  delete []info;
369  }
370  return 0;
