82 /* Computes the number of non-zero entries in a table containing
83  cosine series coefficients of all the P(m,l) or G(m,l) functions
84  necessary for computing the seminaive transform for a given
85  bandwidth bw and order m. Works specifically for tables
86  generated by CosPmlTableGen()
87 */
88 
89 int TableSize(int m,
90  int bw)
91 {
92  return (((bw/2) * ((bw/2) + 1)) -
93  ((m/2)*((m/2)+1)) - ((bw/2) * (m % 2)));
94 }
95 /************************************************************************/
96 /* Spharmonic_TableSize(bw) returns an integer value for
97  the amount of space necessary to fill out an entire spharmonic
98  table. Note that in the above TableSize() formula,
99  you need to sum this formula over m as m ranges from 0 to
100  (bw-1). The critical closed form that you need is that
101 
102  \sum_{k=0}^n = \frac{(n(n+1)(2n+1)}{6}
103 
104  You also need to account for integer division.
105  From this you should derive an upper bound on the
106  amount of space.
107 
108  Some notes - because of integer division, you need to account for
109  a fudge factor - this is the additional " + bw" at the
110  end. This gaurantees that you will always have slightly more
111  space than you need, which is clearly better than underestimating!
112  Also, if bw > 512, the closed form
113  fails because of the bw*bw*bw term (at least on Sun Sparcstations)
114  so the loop computation is used instead.
115 
116  Also, the transpose is exactly the same size, obviously.
117 
118 */
119 
120 int Spharmonic_TableSize(int bw)
121 {
122  int m, sum;
123 
124  if (bw > 512)
125  {
126  sum = 0;
127 
128  for (m=0; m<bw; m++)
129  sum += TableSize(m,bw);
130 
131  return sum;
132  }
133  else
134  {
135  return (
136  (((4*(bw*bw*bw)) + (6*(bw*bw)) - (8*bw))/24)
137  + bw
138  );
139  }
140 }

500 /* RowSize returns the number of non-zero coefficients in a row of the
501  cospmltable if were really in matrix form. Helpful in transpose
502  computations. It is helpful to think of the parameter l as
503  the row of the corresponding matrix.
504 */
505 
506 int RowSize(int m,
507  int l)
508 {
509  if (l < m)
510  return 0;
511  else
512  {
513  if ((m % 2) == 0)
514  return ((l/2)+1);
515  else
516  return (((l-1)/2)+1);
517  }
518 }
519 /************************************************************************/
520 /* Transposed row size returns the number of non-zero coefficients
521  in the transposition of the matrix representing a cospmltable.
522  Used for generating arrays for inverse seminaive transform.
523  Unlike RowSize, need to know the bandwidth bw. Also, in
524  the cospml array, the first m+1 rows are empty, but in
525  the transpose, all rows have non-zero entries, and the first
526  m+1 columns are empty. So the input parameters are a bit different
527  in the you need to specify the row you want.
528 
529 */
530 
531 int Transpose_RowSize(int row,
532  int m,
533  int bw)
534 {
535 
536  if (row >= bw)
537  return 0;
538  else if ((m % 2) == 0)
539  {
540  if (row <= m)
541  return ( ((bw-m)/2) );
542  else
543  return ( ((bw-row-1)/2) + 1);
544  }
545  else
546  {
547  if (row == (bw-1))
548  return 0;
549  else if (row >= m)
550  return (Transpose_RowSize(row+1,m-1,bw));
551  else /* (row < m) */
552  return (Transpose_RowSize(row+1,m-1,bw) - (row % 2));
553  }
554 }

556 /************************************************************************/
557 /* Inverse transform is transposition of forward transform.
558  Thus, need to provide transposed version of table
559  returned by CosPmlTableGen. This function does that
560  by taking as input a cos_pml_table for a particular value
561  of bw and m, and loads the result as a
562  transposed, decimated version of it for use by an inverse
563  seminaive transform computation.
564 
565  result needs to be of size TableSize(m,bw)
566 
567 */
568 
569 void Transpose_CosPmlTableGen(int bw,
570  int m,
571  double *cos_pml_table,
572  double *result)
573 {
574  /* recall that cospml_table has had all the zeroes
575  stripped out, and that if m is odd, then it is
576  really a Gml function, which affects indexing a bit.
577  */
578 
579  double *trans_tableptr, *tableptr;
580  int i, row, rowsize, stride, offset, costable_offset;
581 
582  /* note that the number of non-zero entries is the same
583  as in the non-transposed case */
584 
585  trans_tableptr = result;
586 
587  /* now traverse the cos_pml_table , loading appropriate values
588  into the rows of transposed array */
589 
590  for (row = 0; row < bw; row++)
591  {
592  /* if m odd, no need to do last row - all zeroes */
593  if (row == (bw-1))
594  {
595  if ((m % 2) == 1)
596  return;
597  }
598 
599  /* get the rowsize for the transposed array */
600  rowsize = Transpose_RowSize(row, m, bw);
601 
602  /* compute the starting point for values in cos_pml_table */
603  if (row <= m)
604  {
605  if ((row % 2) == 0)
606  tableptr = cos_pml_table + (row/2);
607  else
608  tableptr = cos_pml_table + (m/2) + 1 + (row/2);
609  }
610  else
611  {
612  /* if row > m, then the highest degree coefficient
613  of P(m,row) should be the first coefficient loaded
614  into the transposed array, so figure out where
615  this point is.
616  */
617  offset = 0;
618  if ((m%2) == 0)
619  {
620  for (i=m; i<=row; i++)
621  offset += RowSize(m, i);
622  }
623  else
624  {
625  for (i=m;i<=row+1;i++)
626  offset += RowSize(m, i);
627  }
628  /* now we are pointing one element too far, so decrement */
629  offset--;
630 
631  tableptr = cos_pml_table + offset;
632  }
633 
634  /* stride is how far we need to jump between
635  values in cos_pml_table, i.e., to traverse the columns of the
636  cos_pml_table. Need to set initial value. Stride always
637  increases by 2 after that
638  */
639  if (row <= m)
640  stride = m + 2 - (m % 2) + (row % 2);
641  else
642  stride = row + 2;
643 
644  /* now load up this row of the transposed table */
645  costable_offset = 0;
646  for (i=0; i < rowsize; i++)
647  {
648  trans_tableptr[i] = tableptr[costable_offset];
649  costable_offset += stride;
650  stride += 2;
651 
652  } /* closes i loop */
653 
654  trans_tableptr += rowsize;
655 
656  } /* closes row loop */
657 
658 }
