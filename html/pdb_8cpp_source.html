127 /* Compute geometry -------------------------------------------------------- */
128 void computePDBgeometry(const std::string &fnPDB,
129  Matrix1D<double> &centerOfMass,
130  Matrix1D<double> &limit0, Matrix1D<double> &limitF,
131  const std::string &intensityColumn)
132 {
133  // Initialization
134  centerOfMass.initZeros(3);
135  limit0.initZeros(3);
136  limitF.initZeros(3);
137  limit0.initConstant(1e30);
138  limitF.initConstant(-1e30);
139  double total_mass = 0;
140 
141  // Open the file
142  std::ifstream fh_pdb;
143  fh_pdb.open(fnPDB.c_str());
144  if (!fh_pdb)
145  REPORT_ERROR(ERR_IO_NOTEXIST, fnPDB);
146 
147  // Process all lines of the file
148  int col=1;
149  if (intensityColumn=="Bfactor")
150  col=2;
151  while (!fh_pdb.eof())
152  {
153  // Read a ATOM line
154  std::string line;
155  getline(fh_pdb, line);
156  if (line == "")
157  continue;
158  std::string kind = line.substr(0,4);
159  if (kind != "ATOM" && kind!="HETA")
160  continue;
161 
162  // Extract atom type and position
163  // Typical line:
164  // ATOM 909 CA ALA A 161 58.775 31.984 111.803 1.00 34.78
165  std::string atom_type = line.substr(13,2);
166  double x = textToFloat(line.substr(30,8));
167  double y = textToFloat(line.substr(38,8));
168  double z = textToFloat(line.substr(46,8));
169 
170  // Update center of mass and limits
171  if (x < XX(limit0))
172  XX(limit0) = x;
173  else if (x > XX(limitF))
174  XX(limitF) = x;
175  if (y < YY(limit0))
176  YY(limit0) = y;
177  else if (y > YY(limitF))
178  YY(limitF) = y;
179  if (z < ZZ(limit0))
180  ZZ(limit0) = z;
181  else if (z > ZZ(limitF))
182  ZZ(limitF) = z;
183  double weight;
184  if (atom_type=="EN")
185  {
186  if (col==1)
187  weight=textToFloat(line.substr(54,6));
188  else if (col==2)
189  weight=textToFloat(line.substr(60,6));
190  }
191  else
192  {
193  if (kind=="HETA")
194  continue;
195  weight=(double) atomCharge(atom_type);
196  }
197  total_mass += weight;
198  XX(centerOfMass) += weight * x;
199  YY(centerOfMass) += weight * y;
200  ZZ(centerOfMass) += weight * z;
201  }
202 
203  // Finish calculations
204  centerOfMass /= total_mass;
205 
206  // Close file
207  fh_pdb.close();
208 }

589 /* Electron form factor in Fourier ----------------------------------------- */
590 double electronFormFactorFourier(double f, const Matrix1D<double> &descriptors)
591 {
592  double retval=0;
593  for (int i=1; i<=5; i++)
594  {
595  double ai=VEC_ELEM(descriptors,i);
596  double bi=VEC_ELEM(descriptors,i+5);
597  retval+=ai*exp(-bi*f*f);
598  }
599  return retval;

602 /* Electron form factor in real space -------------------------------------- */
603 /* We know the transform pair
604 
605  sqrt(pi/b)*exp(-x^2/(4*b)) <----> exp(-b*W^2)
606 
607  We also know that
608 
609  X(f)=sum_i ai exp(-bi*f^2)
610 
611  Therefore, using W=2*pi*f
612 
613  X(W)=sum_i ai exp(-bi/(2*pi)^2*W^2)
614 
615  Thus, the actual b for the inverse Fourier transform is bi/(2*pi)^2
616  And we have to divide by 2*pi to account for the Jacobian of the
617  transformation.
618 */
619 double electronFormFactorRealSpace(double r,
620  const Matrix1D<double> &descriptors)
621 {
622  double retval=0;
623  for (int i=1; i<=5; i++)
624  {
625  double ai=descriptors(i);
626  double bi=descriptors(i+5);
627  double b=bi/(4*PI*PI);
628  retval+=ai*sqrt(PI/b)*exp(-r*r/(4*b));
629  }
630  retval/=2*PI;
631  return retval;
632 }

634 /* Computation of the low pass filter -------------------------------------- */
635 // Returns the impulse response of the lowpass filter
636 void hlpf(MultidimArray<double> &f, int M, double T, const std::string &filterType,
637  MultidimArray<double> &filter, double reductionFactor=0.8,
638  double ripple=0.01, double deltaw=1.0/8.0)
639 {
640  filter.initZeros(XSIZE(f));
641  filter.setXmippOrigin();
642 
643  int Nmax=(int)CEIL(M/2.0);
644  if (filterType=="SimpleAveraging")
645  {
646  FOR_ALL_ELEMENTS_IN_ARRAY1D(filter)
647  if (ABS(i)<=Nmax)
648  filter(i)=1.0/(2*Nmax+1);
649  }
650  else if (filterType=="SincKaiser")
651  {
652  int lastIdx=FINISHINGX(filter);
653  SincKaiserMask(filter,reductionFactor*PI/M,ripple,deltaw);
654  int newLastIdx=FINISHINGX(filter);
655  if (newLastIdx>3*lastIdx)
656  filter.selfWindow(-lastIdx,lastIdx);
657  filter/=filter.sum();
658  if (FINISHINGX(f)>FINISHINGX(filter))
659  filter.selfWindow(STARTINGX(f),FINISHINGX(f));
660  else
661  f.selfWindow(STARTINGX(filter),FINISHINGX(filter));
662  }
663 }
