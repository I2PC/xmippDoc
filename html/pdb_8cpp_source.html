59 void analyzePDBAtoms(const FileName &fn_pdb, const std::string &typeOfAtom, int &numberOfAtoms, pdbInfo &at_pos)
60 {
61  //Open the pdb file
62  std::ifstream f2parse;
63  f2parse.open(fn_pdb.c_str());
64 
65  numberOfAtoms = 0;
66 
67  while (!f2parse.eof())
68  {
69  std::string line;
70  getline(f2parse, line);
71 
72  // The type of record (line) is defined in the first 6 characters of the pdb
73  std::string typeOfline = line.substr(0,4);
74 
75  if ( (typeOfline == "ATOM") || (typeOfline == "HETA"))
76  {
77  // Type of Atom
78  std::string at;
79  try
80  {
81  at = line.substr(13,2);
82  }catch (const std::out_of_range& oor)
83  {
84  std::cerr << "Out of Range error: One of the pdb lines failed selecting the atom type" << '\n';
85  }
86 
87  if (at == typeOfAtom)
88  {
89  // Atom positions
90  numberOfAtoms++;
91  double x = textToFloat(line.substr(30,8));
92  double y = textToFloat(line.substr(38,8));
93  double z = textToFloat(line.substr(46,8));
94  std::string ch = line.substr(21,1);
95 
96  // storing coordinates
97  at_pos.x.push_back(x);
98  at_pos.y.push_back(y);
99  at_pos.z.push_back(z);
100  at_pos.chain.push_back(ch);
101 
102  // Residue Number
103  auto resi = (int) textToFloat(line.substr(23,5));
104  at_pos.residue.push_back(resi);
105 
106  // Getting the bfactor = 8pi^2*u
107  double bfactorRad = textToFloat(line.substr(60,6));//sqrt(textToFloat(line.substr(60,6))/(8*PI*PI));
108  at_pos.b.push_back(bfactorRad);
109 
110  // Covalent radius of the atom
111  double rad = atomCovalentRadius(line.substr(13,2));
112  at_pos.atomCovRad.push_back(rad);
113  }
114  }
115  }

256 /* Compute geometry -------------------------------------------------------- */
257 void computePDBgeometry(const std::string &fnPDB,
258  Matrix1D<double> &centerOfMass,
259  Matrix1D<double> &limit0, Matrix1D<double> &limitF,
260  const std::string &intensityColumn)
261 {
262  // Initialization
263  centerOfMass.initZeros(3);
264  limit0.resizeNoCopy(3);
265  limitF.resizeNoCopy(3);
266  limit0.initConstant(1e30);
267  limitF.initConstant(-1e30);
268  double total_mass = 0;
269 
270  // Open the file
271  std::ifstream fh_pdb;
272  fh_pdb.open(fnPDB.c_str());
273  if (!fh_pdb)
274  REPORT_ERROR(ERR_IO_NOTEXIST, fnPDB);
275 
276  // Process all lines of the file
277  int col=1;
278  if (intensityColumn=="Bfactor")
279  col=2;
280  while (!fh_pdb.eof())
281  {
282  // Read a ATOM line
283  std::string line;
284  getline(fh_pdb, line);
285  if (line == "")
286  continue;
287  std::string kind = line.substr(0,4);
288  if (kind != "ATOM" && kind!="HETA")
289  continue;
290 
291  // Extract atom type and position
292  // Typical line:
293  // ATOM 909 CA ALA A 161 58.775 31.984 111.803 1.00 34.78
294  std::string atom_type = line.substr(13,2);
295  double x = textToFloat(line.substr(30,8));
296  double y = textToFloat(line.substr(38,8));
297  double z = textToFloat(line.substr(46,8));
298 
299  // Update center of mass and limits
300  if (x < XX(limit0))
301  XX(limit0) = x;
302  else if (x > XX(limitF))
303  XX(limitF) = x;
304  if (y < YY(limit0))
305  YY(limit0) = y;
306  else if (y > YY(limitF))
307  YY(limitF) = y;
308  if (z < ZZ(limit0))
309  ZZ(limit0) = z;
310  else if (z > ZZ(limitF))
311  ZZ(limitF) = z;
312  double weight;
313  if (atom_type=="EN")
314  {
315  if (col==1)
316  weight=textToFloat(line.substr(54,6));
317  else if (col==2)
318  weight=textToFloat(line.substr(60,6));
319  }
320  else
321  {
322  if (kind=="HETA")
323  continue;
324  weight=(double) atomCharge(atom_type);
325  }
326  total_mass += weight;
327  XX(centerOfMass) += weight * x;
328  YY(centerOfMass) += weight * y;
329  ZZ(centerOfMass) += weight * z;
330  }
331 
332  // Finish calculations
333  centerOfMass /= total_mass;
334 
335  // Close file
336  fh_pdb.close();
337 }

339 /* Apply geometry ---------------------------------------------------------- */
340 void applyGeometryToPDBFile(const std::string &fn_in, const std::string &fn_out,
341  const Matrix2D<double> &A, bool centerPDB,
342  const std::string &intensityColumn)
343 {
344  Matrix1D<double> centerOfMass;
345  Matrix1D<double> limit0;
346  Matrix1D<double> limitF;
347  if (centerPDB)
348  {
349  computePDBgeometry(fn_in, centerOfMass,limit0, limitF,
350  intensityColumn);
351  limit0 -= centerOfMass;
352  limitF -= centerOfMass;
353  }
354 
355  // Open files
356  std::ifstream fh_in;
357  fh_in.open(fn_in.c_str());
358  if (!fh_in)
359  REPORT_ERROR(ERR_IO_NOTEXIST, fn_in);
360  std::ofstream fh_out;
361  fh_out.open(fn_out.c_str());
362  if (!fh_out)
363  REPORT_ERROR(ERR_IO_NOWRITE, fn_out);
364 
365  // Process all lines of the file
366  while (!fh_in.eof())
367  {
368  // Read an ATOM line
369  std::string line;
370  getline(fh_in, line);
371  if (line == "")
372  {
373  fh_out << "\n";
374  continue;
375  }
376  std::string kind = line.substr(0,4);
377  if (kind != "ATOM" && kind != "HETA")
378  {
379  fh_out << line << std::endl;
380  continue;
381  }
382 
383  // Extract atom type and position
384  // Typical line:
385  // ATOM 909 CA ALA A 161 58.775 31.984 111.803 1.00 34.78
386  double x = textToFloat(line.substr(30,8));
387  double y = textToFloat(line.substr(38,8));
388  double z = textToFloat(line.substr(46,8));
389 
390  Matrix1D<double> v(4);
391  if (centerPDB)
392  {
393  VECTOR_R3(v,x-XX(centerOfMass),
394  y-YY(centerOfMass),z-ZZ(centerOfMass));
395  }
396  else
397  {
398  VECTOR_R3(v,x,y,z);
399  }
400  v(3)=1;
401  v=A*v;
402 
403  char aux[15];
404  sprintf(aux,"%8.3f",XX(v));
405  line.replace(30,8,aux);
406  sprintf(aux,"%8.3f",YY(v));
407  line.replace(38,8,aux);
408  sprintf(aux,"%8.3f",ZZ(v));
409  line.replace(46,8,aux);
410 
411  fh_out << line << std::endl;
412  }
413 
414  // Close files
415  fh_in.close();
416  fh_out.close();
417 }

752 /* Electron form factor in Fourier ----------------------------------------- */
753 double electronFormFactorFourier(double f, const Matrix1D<double> &descriptors)
754 {
755  double retval=0;
756  for (int i=1; i<=5; i++)
757  {
758  double ai=VEC_ELEM(descriptors,i);
759  double bi=VEC_ELEM(descriptors,i+5);
760  retval+=ai*exp(-bi*f*f);
761  }
762  return retval;
763 }

765 /* Electron form factor in real space -------------------------------------- */
766 /* We know the transform pair
767  sqrt(pi/b)*exp(-x^2/(4*b)) <----> exp(-b*W^2)
768 
769  We also know that
770 
771  X(f)=sum_i ai exp(-bi*f^2)
772 
773  Therefore, using W=2*pi*f
774 
775  X(W)=sum_i ai exp(-bi/(2*pi)^2*W^2)
776 
777  Thus, the actual b for the inverse Fourier transform is bi/(2*pi)^2
778  And we have to divide by 2*pi to account for the Jacobian of the
779  transformation.
780 */
781 double electronFormFactorRealSpace(double r,
782  const Matrix1D<double> &descriptors)
783 {
784  double retval=0;
785  for (int i=1; i<=5; i++)
786  {
787  double ai=descriptors(i);
788  double bi=descriptors(i+5);
789  double b=bi/(4*PI*PI);
790  retval+=ai*sqrt(PI/b)*exp(-r*r/(4*b));
791  }
792  retval/=2*PI;
793  return retval;
794 }
