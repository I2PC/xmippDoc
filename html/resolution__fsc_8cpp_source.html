129  void writeFiles(const FileName &fnRoot,
130  const MultidimArray<double> &freq,
131  const MultidimArray<double> &frc,
132  const MultidimArray<double> &frc_noise,
133  const MultidimArray<double> &dpr,
134  const MultidimArray<double> &error_l2,
135  double max_sam, bool do_dpr, double rFactor)
136  {
137  MetaData MD;
138 
139  FileName fn_frc;
140  if (do_o)
141  fn_frc = fn_out;
142  else
143  fn_frc = fnRoot + ".frc";
144  size_t id;
145  FOR_ALL_ELEMENTS_IN_ARRAY1D(freq)
146  {
147  if (i>0)
148  {
149  id=MD.addObject();
150  if(max_sam >0 && ((1./dAi(freq, i))<max_sam) )
151  {
152  if(do_dpr)
153  dAi(dpr, i)=0.;
154  dAi(frc, i)=0.;
155  }
156  if(min_sam >0 && ((1./dAi(freq, i))>min_sam) )
157  {
158  if(do_dpr)
159  dAi(dpr, i)=0.;
160  dAi(frc, i)=0.;
161  }
162  MD.setValue(MDL_RESOLUTION_FREQ,dAi(freq, i),id);
163  MD.setValue(MDL_RESOLUTION_FRC,dAi(frc, i),id);
164  if(do_dpr)
165  MD.setValue(MDL_RESOLUTION_DPR,dAi(dpr, i),id);
166  MD.setValue(MDL_RESOLUTION_ERRORL2,dAi(error_l2, i),id);
167  MD.setValue(MDL_RESOLUTION_FRCRANDOMNOISE,dAi(frc_noise, i),id);
168  MD.setValue(MDL_RESOLUTION_FREQREAL,1./dAi(freq, i),id);
169  }
170  }
171  MD.write(fn_frc);
172  MetaData MD2;
173  MD2.setColumnFormat(false);
174  id=MD2.addObject();
175  MD2.setValue(MDL_RESOLUTION_RFACTOR,rFactor,id);
176  fn_frc.compose("rfactor", fn_frc);
177  MD2.write(fn_frc, MD_APPEND);
178  }
