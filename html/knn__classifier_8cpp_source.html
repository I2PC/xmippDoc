42 void KNN::KNearestNeighbors(MultidimArray<double> &sample)
43 {
44  double maximum,distance;
45  int maximumIndex;
46  maxDist.resize(1,1,1,K);
47  FOR_ALL_DIRECT_ELEMENTS_IN_ARRAY1D(maxDist)
48  {
49  DIRECT_A1D_ELEM(maxDist,i)=euclideanDistance(sample,i,-1.0);
50  DIRECT_A1D_ELEM(neighborsIndex,i)=i;
51  }
52  maximumIndex=findMaxIndex(maxDist);
53  maximum=DIRECT_A1D_ELEM(maxDist,maximumIndex);
54  for (size_t i=K;i<YSIZE(__dataset);i++)
55  {
56  distance=euclideanDistance(sample,i,maximum);
57  if (distance==-1)
58  continue;
59  if (distance<maximum)
60  {
61  DIRECT_A1D_ELEM(maxDist,maximumIndex)=distance;
62  DIRECT_A1D_ELEM(neighborsIndex,maximumIndex)=i;
63  maximumIndex=findMaxIndex(maxDist);
64  maximum=DIRECT_A1D_ELEM(maxDist,maximumIndex);
65  }
66  }

120 double KNN::euclideanDistance(MultidimArray<double> &sample,int index,double maximumDist)
121 {
122  double dist=0;
123  FOR_ALL_DIRECT_ELEMENTS_IN_ARRAY1D(sample)
124  {
125  double tmp=DIRECT_A1D_ELEM(sample,i)-DIRECT_A2D_ELEM(__dataset,index,i);
126  dist+=tmp*tmp;
127  if (maximumDist>0 && sqrt(dist)>maximumDist)
128  return -1.0;
129  }
130  return sqrt(dist);
131 }
