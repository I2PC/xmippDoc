701  MDDbRowIterator(typename TypeHelpers::choose<IsConst, const MetaDataDb&, MetaDataDb&>::type &mdd, size_t _i)
702  : _mdd(mdd), _i(_i) {
703  mdd.myMDSql->selectObjects(_ids);
704  if (this->_i >= _ids.size())
705  return;

707  _mdd.initGetRow(false);
708
