685  private:
686  typename TypeHelpers::choose<IsConst, const MetaDataDb&, MetaDataDb&>::type _mdd;
687  typename TypeHelpers::choose<IsConst, MDRowSql, MDRowSql>::type _row;

690  bool _finalized = false;
691 
692  public:
693  MDDbRowIterator(typename TypeHelpers::choose<IsConst, const MetaDataDb&, MetaDataDb&>::type &mdd, size_t _i)
694  : _mdd(mdd), _i(_i) {
695  mdd.myMDSql->selectObjects(_ids);
696  if (this->_i >= _ids.size())
697  return;
698 
699  _mdd.initGetRow(false);
700 
701  _mdd.execGetRow(this->_row);
702  this->_row.set_id(this->_ids[this->_i]);
703  if (this->_i+1 == _ids.size()) {
704  _mdd.finalizeGetRow();
705  _finalized = true;
706  }
707  }
