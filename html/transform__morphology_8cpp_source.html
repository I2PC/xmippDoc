179  void processImage(const FileName &fnImg, const FileName &fnImgOut, const MDRow &rowIn, MDRow &rowOut)
180  {
181  Image<double> img, imgOut;
182  img.readApplyGeo(fnImg, rowIn);
183  imgOut().resizeNoCopy(img());
184 
185  if (binaryOperation)
186  std::cout << "Initially the image has " << img().sum()
187  << " pixels set to 1\n";
188  bool isVolume=ZSIZE(img())>1;
189  switch (operation)
190  {
191  case DILATION:
192  if (isVolume)
193  dilate3D(img(), imgOut(), neig3D, count, size);
194  else
195  dilate2D(img(), imgOut(), neig2D, count, size);
196  break;
197  case EROSION:
198  if (isVolume)
199  erode3D(img(), imgOut(), neig3D, count, size);
200  else
201  erode2D(img(), imgOut(), neig2D, count, size);
202  break;
203  case OPENING:
204  if (isVolume)
205  opening3D(img(), imgOut(), neig3D, count, size);
206  else
207  opening2D(img(), imgOut(), neig2D, count, size);
208  break;
209  case CLOSING:
210  if (isVolume)
211  closing3D(img(), imgOut(), neig3D, count, size);
212  else
213  closing2D(img(), imgOut(), neig2D, count, size);
214  break;
215  case SHARPENING:
216  if (isVolume)
217  sharpening(img(), width, strength, imgOut());
218  else
219  REPORT_ERROR(ERR_NOT_IMPLEMENTED,"Sharpening has not been implemented for images");
220  break;
221  case KEEPBIGGEST:
222  if (isVolume)
223  keepBiggestComponent(img(),0,neig3D);
224  else
225  keepBiggestComponent(img(),0,neig2D);
226  imgOut()=img();
227  break;
228  case REMOVESMALL:
229  if (isVolume)
230  removeSmallComponents(img(),smallSize,neig3D);
231  else
232  removeSmallComponents(img(),smallSize,neig2D);
233  imgOut()=img();
234  break;
235  }
236 
237  if (binaryOperation)
238  std::cout << "Finally the image has " << imgOut().sum() << " pixels set to 1\n";
239  imgOut.write(fnImgOut);
240  }
