88 static inline void multiply(const float transform[3][3], Point3D<float>& inOut) {
89  float tmp0 = transform[0][0] * inOut.x + transform[0][1] * inOut.y + transform[0][2] * inOut.z;
90  float tmp1 = transform[1][0] * inOut.x + transform[1][1] * inOut.y + transform[1][2] * inOut.z;
91  float tmp2 = transform[2][0] * inOut.x + transform[2][1] * inOut.y + transform[2][2] * inOut.z;
92  inOut.x = tmp0;
93  inOut.y = tmp1;
94  inOut.z = tmp2;

191 static void convertToExpectedSpace(T*** input, int size, MultidimArray<U>& VoutFourier) {
192  int halfSize = size / 2;
193  for (int z = 0; z <= size; z++) {
194  for (int y = 0; y <= size; y++) {
195  // shift FFT from center to corners
196  const size_t newY = (y < halfSize) ? VoutFourier.ydim - halfSize + y : y - halfSize;
197  const size_t newZ = (z < halfSize) ? VoutFourier.zdim - halfSize + z : z - halfSize;
198 
199  for (int x = 0; x <= halfSize; x++) {
200  // store to output array
201  // += in necessary as VoutFourier might be used multiple times when used with MPI
202  DIRECT_A3D_ELEM(VoutFourier, newZ, newY, x /* no need to move X */) += input[z][y][x];
203  }
204  }
205  }

216 static void computeTraverseSpace(uint32_t imgSizeX, uint32_t imgSizeY,
217  const float transform[3][3], const float transformInv[3][3], RecFourierProjectionTraverseSpace& space,
218  uint32_t maxVolumeIndexX, uint32_t maxVolumeIndexYZ, bool useFast, float blobRadius) {
219  Point3D<float> cuboid[8];
220  Point3D<float> AABB[2];
221  Point3D<float> origin = {maxVolumeIndexX/2.f, maxVolumeIndexYZ/2.f, maxVolumeIndexYZ/2.f};
222  createProjectionCuboid(cuboid, imgSizeX, imgSizeY, useFast ? 0.f : blobRadius);
223  rotateCuboid(cuboid, transform);
224  translateCuboid(cuboid, origin);
225  computeAABB(AABB, cuboid, 0, 0, 0, maxVolumeIndexX, maxVolumeIndexYZ, maxVolumeIndexYZ);
226 
227  // store data
228  space.minZ = floor(AABB[0].z);
229  space.minY = floor(AABB[0].y);
230  space.minX = floor(AABB[0].x);
231  space.maxZ = ceil(AABB[1].z);
232  space.maxY = ceil(AABB[1].y);
233  space.maxX = ceil(AABB[1].x);
234  space.topOrigin = cuboid[4];
235  space.bottomOrigin = cuboid[0];
236  space.maxDistanceSqr = (imgSizeX + (useFast ? 0.f : blobRadius))
237  * (imgSizeX + (useFast ? 0.f : blobRadius));
238  for (int i = 0; i < 3; i++) {
239  for (int j = 0; j < 3; j++) {
240  space.transformInv[i][j] = transformInv[i][j];
241  }
242  }
243 
244  // calculate best traverse direction
245  space.unitNormal.x = space.unitNormal.y = 0.f;
246  space.unitNormal.z = 1.f;
247  multiply(transform, space.unitNormal);
248  float nX = std::abs(space.unitNormal.x);
249  float nY = std::abs(space.unitNormal.y);
250  float nZ = std::abs(space.unitNormal.z);
251 
252  // biggest vector indicates ideal direction
253  if (nX >= nY && nX >= nZ) { // iterate YZ plane
254  space.dir = RecFourierProjectionTraverseSpace::Direction::YZ;
255  } else if (nY >= nX && nY >= nZ) { // iterate XZ plane
256  space.dir = RecFourierProjectionTraverseSpace::Direction::XZ;
257  } else if (nZ >= nX && nZ >= nY) { // iterate XY plane
258  space.dir = RecFourierProjectionTraverseSpace::Direction::XY;
259  }
260 }

273 template<typename T>
274 static void mirrorAndCrop(T***& input,T (*f)(T), uint32_t maxVolumeIndexX, uint32_t maxVolumeIndexYZ) {
275  T*** output;
276  // create new storage, notice that just 'right hand side - X axis' of the input will be preserved, left will be converted to its complex conjugate
277  allocate(output, maxVolumeIndexX+1, maxVolumeIndexYZ+1, maxVolumeIndexYZ+1);
278  // traverse old storage
279  for (int z = 0; z <= maxVolumeIndexYZ; z++) {
280  for (int y = 0; y <= maxVolumeIndexYZ; y++) {
281  for (int x = 0; x <= maxVolumeIndexYZ; x++) {
282  if (x < maxVolumeIndexX) {
283  int newPos[3];
284  // mirror against center of the volume, e.g. [0,0,0]->[size,size,size]. It will fit as the input space is one voxel bigger
285  newPos[0] = maxVolumeIndexYZ - x;
286  newPos[1] = maxVolumeIndexYZ - y;
287  newPos[2] = maxVolumeIndexYZ - z;
288  output[newPos[2]][newPos[1]][newPos[0]-maxVolumeIndexX] += f(input[z][y][x]);
289  } else {
290  // copy with X shifted by (-halfSize)
291  output[z][y][x-maxVolumeIndexX] += input[z][y][x];
292  }
293  }
294  }
295  }
296  // free original data
297  release(input, maxVolumeIndexYZ+1, maxVolumeIndexYZ+1);
298  // set new data
299  input = output;
300 }

318 static void forceHermitianSymmetry(std::complex<float>***& tempVolume, float***& tempWeights, int maxVolumeIndexYZ) {
319  const int x = 0;
320  for (int z = 0; z <= maxVolumeIndexYZ; z++) {
321  for (int y = 0; y <= maxVolumeIndexYZ/2; y++) {
322  // mirror against center of the volume, e.g. [0,0,0]->[size,size,size]. It will fit as the input space is one voxel biger
323  const int newX = x;
324  const int newY = maxVolumeIndexYZ - y;
325  const int newZ = maxVolumeIndexYZ - z;
326 
327  const std::complex<float> averageVol = 0.5f * (tempVolume[newZ][newY][newX] + conj(tempVolume[z][y][x]));
328  const float averageWeight = 0.5f * (tempWeights[newZ][newY][newX] + tempWeights[z][y][x]);
329 
330  tempVolume[newZ][newY][newX] = averageVol;
331  tempVolume[z][y][x] = conj(averageVol);
332  tempWeights[newZ][newY][newX] = tempWeights[z][y][x] = averageWeight;
333  }
334  }

342 static void processWeights(std::complex<float>***& tempVolume, float***& tempWeights, int maxVolumeIndexX, int maxVolumeIndexYZ, double paddingFactorProj, double paddingFactorVol, int imgSize) {
343  // Get a first approximation of the reconstruction
344  float corr2D_3D = static_cast<float>(pow(paddingFactorProj, 2.0) / (imgSize * pow(paddingFactorVol, 3.0)));
345  for (int z = 0; z <= maxVolumeIndexYZ; z++) {
346  for (int y = 0; y <= maxVolumeIndexYZ; y++) {
347  for (int x = 0; x <= maxVolumeIndexX; x++) {
348  const float weight = tempWeights[z][y][x];
349 
350  if (weight > ACCURACY)
351  tempVolume[z][y][x] *= corr2D_3D / weight;
352  else
353  tempVolume[z][y][x] = 0;
354  }
355  }
356  }

376 template<typename T>
377 static T*** applyBlob(T***& input, float blobSize,
378  float* blobTableSqrt, float iDeltaSqrt,
379  const int maxVolumeIndexX, const int maxVolumeIndexYZ) {
380  float blobSizeSqr = blobSize * blobSize;
381  int blob = static_cast<int>(floor(blobSize)); // we are using integer coordinates, so we cannot hit anything further
382  T*** output;
383  // create new storage
384  allocate(output, maxVolumeIndexX+1, maxVolumeIndexYZ+1, maxVolumeIndexYZ+1);
385 
386  // traverse new storage
387  for (int i = 0; i <= maxVolumeIndexYZ; i++) {
388  for (int j = 0; j <= maxVolumeIndexYZ; j++) {
389  for (int k = 0; k <= maxVolumeIndexX; k++) {
390  // traverse input storage
391  T tmp = (T) 0;
392  for (int z = std::max(0, i-blob); z <= std::min(maxVolumeIndexYZ, i+blob); z++) {
393  float dZSqr = (i - z) * (i - z);
394  for (int y = std::max(0, j-blob); y <= std::min(maxVolumeIndexYZ, j+blob); y++) {
395  float dYSqr = (j - y) * (j - y);
396  for (int x = std::max(0, k-blob); x <= std::min(maxVolumeIndexX, k+blob); x++) {
397  float dXSqr = (k - x) * (k - x);
398  float distanceSqr = dZSqr + dYSqr + dXSqr;
399  if (distanceSqr > blobSizeSqr) {
400  continue;
401  }
402  int aux = (int) ((distanceSqr * iDeltaSqrt + 0.5)); //Same as ROUND but avoid comparison
403  float tmpWeight = blobTableSqrt[aux];
404  tmp += tmpWeight * input[z][y][x];
405  }
406  }
407  }
408  output[i][j][k] = tmp;
409  }
410  }
411  }
412  // free original data
413  release(input, maxVolumeIndexYZ+1, maxVolumeIndexYZ+1);
414  return output;
415 }
