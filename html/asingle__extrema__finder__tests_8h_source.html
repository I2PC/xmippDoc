62  static void runPerSignal(F f, const Dimensions &dims) {
63  int threads = std::min((size_t)CPU::findCores(), dims.n());
64  auto workers = std::vector<std::thread>();
65  int workloadPerWorker = std::ceil(dims.n() / (float)threads);
66 
67  auto workload = [&](int id){
68  const size_t first = id * workloadPerWorker;
69  const size_t last = std::min(first + workloadPerWorker, dims.n());
70  for (size_t n = first; n < last; ++n) {
71  f(n, id);
72  }
73  };
74  for (size_t w = 0; w < threads; ++w) {
75  workers.emplace_back(workload, w);
76  }
77  for (auto &w : workers) {
78  w.join();
79  }
