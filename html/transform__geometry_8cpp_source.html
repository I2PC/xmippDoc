125 void ProgTransformGeometry::calculateRotationMatrix()
126 {
127  const char * rotateType = getParam("--rotate_volume");
128  Matrix1D<double> xyz(3);
129  if (!STR_EQUAL(rotateType, "icosahedral"))
130  { // params are char for icosahedral option
131  XX(xyz) = getDoubleParam("--rotate_volume", 1); //rot
132  YY(xyz) = getDoubleParam("--rotate_volume", 2); //tilt
133  ZZ(xyz) = getDoubleParam("--rotate_volume", 3); //psi
134  }
135 
136  if (STR_EQUAL(rotateType, "euler"))
137  Euler_angles2matrix(XX(xyz), YY(xyz), ZZ(xyz), R, true);
138  else if (STR_EQUAL(rotateType, "matrix"))
139  {
140  R.initZeros(4,4);
141  MAT_ELEM(R, 3, 3) = 1;
142  for (int i = 0; i < 9; ++i)
143  {
144  int r = i / 3;
145  int c = i % 3;
146  MAT_ELEM(R, r, c) = getDoubleParam("--rotate_volume", i+1);
147  }
148  }
149  else if (STR_EQUAL(rotateType, "alignZ"))
150  alignWithZ(xyz, R);
151  else if (STR_EQUAL(rotateType, "icosahedral"))
152  {
153  const char * icoFrom = getParam("--rotate_volume", 1);
154  const char * icoTo = getParam("--rotate_volume", 2);
155  rotation3DMatrixFromIcoOrientations(icoFrom, icoTo, R);
156  }
157  else
158  {
159  double ang = getDoubleParam("--rotate_volume", 1);
160  XX(xyz) = getDoubleParam("--rotate_volume", 2); //axis x
161  YY(xyz) = getDoubleParam("--rotate_volume", 3); //y
162  ZZ(xyz) = getDoubleParam("--rotate_volume", 4);//z
163  rotation3DMatrix(ang, xyz, R, true);
164  }
165 }

167 void ProgTransformGeometry::preProcess()
168 {
169  //If zdimOut greater than 1, is a volume and should apply transform
170  dim = (isVol = (zdimOut > 1)) ? 3 : 2;
171 
172  applyTransform = applyTransform || isVol || mdInSize == 1;
173  R.initIdentity(dim + 1);
174  A.initIdentity(dim + 1);
175 
176  MDRow rowGeo;
177  rowGeo.setValue(MDL_SHIFT_X, getDoubleParam("--shift", 0));
178  rowGeo.setValue(MDL_SHIFT_Y, getDoubleParam("--shift", 1));
179  rowGeo.setValue(MDL_SCALE, getDoubleParam("--scale"));
180 
181  if (isVol)
182  {
183  if (checkParam("--rotate_volume"))
184  calculateRotationMatrix();
185  else
186  mdVol = true;
187  rowGeo.setValue(MDL_SHIFT_Z, getDoubleParam("--shift", 2));
188 
189  }
190  else
191  rowGeo.setValue(MDL_ANGLE_PSI, getDoubleParam("--rotate"));
192 
193  geo2TransformationMatrix(rowGeo, A);
194 
195  A = A * R;
196 
197  if (flip)
198  {
199  MAT_ELEM(A, 0, 0) *= -1.;
200  MAT_ELEM(A, 0, 1) *= -1.;
201  if (dim == 3)
202  MAT_ELEM(A, 0, 2) *= -1.;
203  }
204  if (inverse)
205  A = A.inv();
206 }
