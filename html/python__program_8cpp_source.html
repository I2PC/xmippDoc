306  auto *self = (ProgramObject*) obj;

309  PyObject *list = nullptr;

311  {

314  size_t size = PyList_Size(list);
315  PyObject * item = nullptr;
316  PyObject *pyStr1 = nullptr;
317  PyObject *str_exc_type = nullptr;
318  auto ** argv = new char*[size];
319  std::vector<double> vValue(size);
320  for (size_t i = 0; i < size; ++i)
321  {
322  item = PyList_GetItem(list, i);
323  if (!PyUnicode_Check(item))
324  {
325  PyErr_SetString(PyExc_TypeError,
326  "Program arguments should be of type string");
327  delete[] argv;
328  return nullptr;
329  }
330 
331  str_exc_type = PyObject_Str(item); //Now a unicode object
332  argv[i] = (char*)PyUnicode_AsUTF8(str_exc_type);
333 
334  if (i == 0)
335  {
336  FileName temp(argv[i]);
337  temp = temp.removeDirectories();
338  argv[i] = strdup(temp.c_str());
339  }
340  }
341  self->program->read((int)size, (const char **)argv);
342  if (self->program->doRun)
343  Py_RETURN_TRUE;
