65  void readParams()
66  {
67  fn_in = getParam("-i");
68  fn_out = getParam("-o");
69  if (fn_out=="")
70  fn_out=fn_in;
71  PDBmode=fn_in.contains(".pdb");
72  String operation;
73  shift.initZeros(3);
74  scale.resize(3);
75  scale.initConstant(1);
76  A3D.initIdentity(4);
77  Euler_mode = Align_mode = Axis_mode = false;
78  centerPDB = checkParam("--center_pdb");
79  operation=getParam("--operation");
80  if (operation=="shift")
81  {
82  double x=getDoubleParam("--operation",1);
83  double y=getDoubleParam("--operation",2);
84  double z=getDoubleParam("--operation",3);
85  shift=vectorR3(x,y,z);
86  }
87  else if (operation=="scale")
88  {
89  double x=getDoubleParam("--operation",1);
90  double y=getDoubleParam("--operation",2);
91  double z=getDoubleParam("--operation",3);
92  scale=vectorR3(x,y,z);
93  }
94  else if (operation=="rotate_euler")
95  {
96  Euler_mode = true;
97  rot = getDoubleParam("--operation",1);
98  tilt = getDoubleParam("--operation",2);
99  psi = getDoubleParam("--operation",3);
100  Euler_angles2matrix(rot, tilt, psi, A3D, true);
101  }
102  else if (operation=="rotate_align_with_z")
103  {
104  Align_mode = true;
105  double x=getDoubleParam("--operation",1);
106  double y=getDoubleParam("--operation",2);
107  double z=getDoubleParam("--operation",3);
108  axis=vectorR3(x,y,z);
109  alignWithZ(axis, A3D);
110  }
111  else if (operation=="rotate_axis")
112  {
113  Axis_mode = true;
114  double x=getDoubleParam("--operation",1);
115  double y=getDoubleParam("--operation",2);
116  double z=getDoubleParam("--operation",3);
117  ang=getDoubleParam("--operation",4);
118  axis=vectorR3(x,y,z);
119  rotation3DMatrix(ang, axis, A3D);
120  }
121 
122  // Apply shift
123  A3D(0, 3) = XX(shift);
124  A3D(1, 3) = YY(shift);
125  A3D(2, 3) = ZZ(shift);

127  // Apply scale

136  A3D(2, 2) *= ZZ(scale);
