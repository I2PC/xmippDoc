55 void applyTransformation(const MultidimArray<double> &V2,
56  MultidimArray<double> &Vaux,
57  double *p)
58 {
59  Matrix1D<double> r(3);
60  Matrix2D<double> A, Aaux;
61 
62  double greyScale = p[0];
63  double greyShift = p[1];
64  double rot = p[2];
65  double tilt = p[3];
66  double psi = p[4];
67  double scale = p[5];
68  ZZ(r) = p[6];
69  YY(r) = p[7];
70  XX(r) = p[8];
71 
72  Euler_angles2matrix(rot, tilt, psi, A, true);
73  translation3DMatrix(r,Aaux);
74  A = A * Aaux;
75  scale3DMatrix(vectorR3(scale, scale, scale),Aaux);
76  A = A * Aaux;
77 
78  applyGeometry(LINEAR, Vaux, V2, A, IS_NOT_INV, WRAP);
79  if (greyScale!=1 || greyShift!=0)
80  FOR_ALL_DIRECT_ELEMENTS_IN_MULTIDIMARRAY(Vaux)
81  DIRECT_MULTIDIM_ELEM(Vaux,n)=DIRECT_MULTIDIM_ELEM(Vaux,n)*greyScale+greyShift;
82 }

87 {

90  // Correlate

99  break;
100  }
