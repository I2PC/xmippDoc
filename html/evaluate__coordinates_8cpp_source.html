64 void ProgEvaluateCoordinates::run()
65 {
66  MDRow row;
67  int evalXCoor, evalYCoor, gtXCoor, gtYCoor;
68  int truePosivites = 0, totalEval = 0, totalGT = 0;
69 
70  for (int m = 1; m <= numMic; m++)
71  {
72  // Here you need to change the identifiers based on datasets
73  // TODO: loading names and counts of mics automatically?
74  FileName micGT = formatString("%s_%04d@%s", rootName.c_str(), m, fnGt.c_str());
75  FileName micEval = formatString("%s_%04d@%s", rootName.c_str(), m, fnEval.c_str());
76 
77  GT.read(micGT);
78  Eval.read(micEval);
79 
80  int iterE = 1;
81  FOR_ALL_OBJECTS_IN_METADATA(Eval)
82  {
83  Eval.getRow(row, iterE);
84  row.getValue(MDL_XCOOR, evalXCoor);
85  row.getValue(MDL_YCOOR, evalYCoor);
86 
87  int iterG = 1;
88  FOR_ALL_OBJECTS_IN_METADATA(GT)
89  {
90  GT.getRow(row, iterG);
91  row.getValue(MDL_XCOOR, gtXCoor);
92  row.getValue(MDL_YCOOR, gtYCoor);
93 
94  if ((evalXCoor > (gtXCoor-errMargin-1)) &&
95  (evalXCoor < (gtXCoor+errMargin+1)) &&
96  (evalYCoor > (gtYCoor-errMargin-1)) &&
97  (evalYCoor < (gtYCoor+errMargin+1)))
98  truePosivites++;
99 
100  iterG++;
101  }
102  iterE++;
103  }
104  totalEval += Eval.size();
105  totalGT += GT.size();
106  }
107 
108  std::cout << std::endl;
109  std::cout << "True positives (correctly picked particles): ";
110  std::cout << truePosivites << std::endl;
111  std::cout << "False positives (incorrectly picked particles): ";
112  std::cout << totalEval - truePosivites << std::endl;
113  std::cout << "False negatives (missed particles): ";
114  std::cout << totalGT - truePosivites << std::endl << std::endl;
115 
116  std::cout << "How many relevant particles are picked:" << std::endl;
117  std::cout << "True positive rate: ";
118  double truePositiveRate = (0 == totalGT) ? 0 : (double) truePosivites / totalGT;
119  std::cout << truePositiveRate << std::endl << std::endl;
120 
121  std::cout << "Ratio of wrongly picked particles and all picked particles:";
122  std::cout << std::endl;
123  std::cout << "False positive rate: ";
124  std::cout << (double) (totalEval - truePosivites) / totalEval;
125  std::cout << std::endl << std::endl;
126 }
