90  imgsFn = self._getFileName('imgsFn')
91  fnVol = self._getFileName('fnVol')
92 
93  inputParticles = self.inputParticles.get()
94  writeSetOfParticles(inputParticles, imgsFn)
95  Xdim = inputParticles.getXDim()
96  Ts = inputParticles.getSamplingRate()
97  newTs = self.targetResolution.get() * 1.0 /3.0
98  newTs = max(Ts, newTs)
99  self.newXdim = int(Xdim * Ts / newTs)
100  writeInfoField(self._getExtraPath(), "sampling", md.MDL_SAMPLINGRATE, newTs)
101  writeInfoField(self._getExtraPath(), "size", md.MDL_XSIZE, self.newXdim)

104  "-i %s -o %s --save_metadata_stack %s --fourier %d" %

135  def createOutputStep(self):
136  from sklearn.manifold import TSNE
137  fnOut = self._getFileName('fnOut')
138  mdOut = md.MetaData(fnOut)
139  i = 0
140  for row in md.iterRows(mdOut):
141  coeffs = mdOut.getValue(md.MDL_SPH_COEFFICIENTS, row.getObjId())
142  if i==0:
143  coeffMatrix = coeffs
144  else:
145  coeffMatrix = np.vstack((coeffMatrix, coeffs))
146  i+=1
147  X_tsne_1d = TSNE(n_components=1).fit_transform(coeffMatrix)
148  X_tsne_2d = TSNE(n_components=2).fit_transform(coeffMatrix)
149 
150  newMdOut = md.MetaData()
151  i=0
152  for row in md.iterRows(mdOut):
153  newRow = row
154  newRow.setValue(md.MDL_SPH_TSNE_COEFF1D, float(X_tsne_1d[i,0]))
155  newRow.setValue(md.MDL_SPH_TSNE_COEFF2D, [float(X_tsne_2d[i, 0]), float(X_tsne_2d[i, 1])])
156  newRow.addToMd(newMdOut)
157  i+=1
158  newMdOut.write(fnOut)
