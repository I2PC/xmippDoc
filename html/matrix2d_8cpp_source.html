61 void solveLinearSystem(PseudoInverseHelper &h, Matrix1D<double> &result)
62 {
63  Matrix2D<double> &A=h.A;
64  Matrix1D<double> &b=h.b;
65  Matrix2D<double> &AtA=h.AtA;
66  Matrix2D<double> &AtAinv=h.AtAinv;
67  Matrix1D<double> &Atb=h.Atb;
68 
69  // Compute AtA and Atb
70  int I=MAT_YSIZE(A);
71  int J=MAT_XSIZE(A);
72  AtA.initZeros(J,J);
73  Atb.initZeros(J);
74  for (int i=0; i<J; ++i)
75  {
76  for (int j=0; j<J; ++j)
77  {
78  double AtA_ij=0;
79  for (int k=0; k<I; ++k)
80  AtA_ij+=MAT_ELEM(A,k,i)*MAT_ELEM(A,k,j);
81  MAT_ELEM(AtA,i,j)=AtA_ij;
82  }
83  double Atb_i=0;
84  for (int k=0; k<I; ++k)
85  Atb_i+=MAT_ELEM(A,k,i)*VEC_ELEM(b,k);
86  VEC_ELEM(Atb,i)=Atb_i;
87  }
88 
89  // Compute the inverse of AtA
90  AtA.inv(AtAinv);
91 
92  // Now multiply by Atb
93  result.initZeros(J);
94  FOR_ALL_ELEMENTS_IN_MATRIX2D(AtAinv)
95  VEC_ELEM(result,i)+=MAT_ELEM(AtAinv,i,j)*VEC_ELEM(Atb,j);
96 }

99 void weightedLeastSquares(WeightedLeastSquaresHelper &h, Matrix1D<double> &result)
100 {
101  Matrix2D<double> &A=h.A;
102  Matrix1D<double> &b=h.b;
103  Matrix1D<double> &w=h.w;
104 
105  // See http://en.wikipedia.org/wiki/Least_squares#Weighted_least_squares
106  FOR_ALL_ELEMENTS_IN_MATRIX1D(w)
107  {
108  double wii=sqrt(VEC_ELEM(w,i));
109  VEC_ELEM(b,i)*=wii;

111  MAT_ELEM(A,i,j)*=wii;
112  }
113  solveLinearSystem(h,result);

117 //#define DEBUG
118 //#define DEBUG_MORE
119 double ransacWeightedLeastSquaresBasic(WeightedLeastSquaresHelper &h, Matrix1D<double> &result,
120  double tol, int Niter, double outlierFraction)
121 {
122  int N=MAT_YSIZE(h.A); // Number of equations
123  int M=MAT_XSIZE(h.A); // Number of unknowns
124 
125  // Initialize a vector with all equation indexes
126  std::vector<int> eqIdx;
127  eqIdx.reserve(N);
128  for (int n=0; n<N; ++n)
129  eqIdx.push_back(n);
130  int *eqIdxPtr=&eqIdx[0];
131 
132 #ifdef DEBUG_MORE
133  // Show all equations
134  for (int n=0; n<N; n++)
135  {
136  std::cout << "Eq. " << n << " w=" << VEC_ELEM(h.w,n) << " b=" << VEC_ELEM(h.b,n) << " a=";
137  for (int j=0; j<M; j++)
138  std::cout << MAT_ELEM(h.A,n,j) << " ";
139  std::cout << std::endl;
140  }
141 #endif
142 
143  // Resize a WLS helper for solving the MxM equation systems
144  PseudoInverseHelper haux;
145  haux.A.resizeNoCopy(M,M);
146  haux.b.resizeNoCopy(M);
147  Matrix2D<double> &A=haux.A;
148  Matrix1D<double> &b=haux.b;
149 
150  // Solve Niter randomly chosen equation systems
151  double bestError=1e38;
152  const int Mdouble=M*sizeof(double);
153  int minNewM=(int)((1.0-outlierFraction)*N-M);
154  if (minNewM<0)
155  minNewM=0;
156 
157  Matrix1D<double> resultAux;
158  Matrix1D<int> idxIn(N);
159  WeightedLeastSquaresHelper haux2;
160  for (int it=0; it<Niter; ++it)
161  {
162 #ifdef DEBUG_MORE
163  std::cout << "Iter: " << it << std::endl;
164 #endif
165  idxIn.initZeros();
166 
167  // Randomly select M equations
168  std::random_device rd;
169  std::mt19937 g(rd());
170  std::shuffle(eqIdx.begin(), eqIdx.end(), g);
171 
172  // Select the equation system
173  for (int i=0; i<M; ++i)
174  {
175  int idx=eqIdxPtr[i];
176  memcpy(&MAT_ELEM(A,i,0),&MAT_ELEM(h.A,idx,0),Mdouble);
177  VEC_ELEM(b,i)=VEC_ELEM(h.b,idx);
178  VEC_ELEM(idxIn,idx)=1;
179 #ifdef DEBUG_MORE
180  std::cout << " Using Eq.: " << idx << " for first solution" << std::endl;
181 #endif
182  }
183 
184  // Solve the equation system
185  // We use LS because the weight of some of the equations might be too low
186  // and then the system is ill conditioned
187  solveLinearSystem(haux, resultAux);
188 
189  // Study the residuals of the rest
190  int newM=0;
191  for (int i=M+1; i<N; ++i)
192  {
193  int idx=eqIdxPtr[i];
194  double bp=0;
195  for (int j=0; j<M; ++j)
196  bp+=MAT_ELEM(h.A,idx,j)*VEC_ELEM(resultAux,j);
197  if (fabs(bp-VEC_ELEM(h.b,idx))<tol)
198  {
199  VEC_ELEM(idxIn,idx)=1;
200  ++newM;
201  }
202 #ifdef DEBUG_MORE
203  std::cout << " Checking Eq.: " << idx << " err=" << bp-VEC_ELEM(h.b,idx) << std::endl;
204 #endif
205  }
206 
207  // If the model represent more points
208  if (newM>minNewM)
209  {
210  Matrix2D<double> &A2=haux2.A;
211  Matrix1D<double> &b2=haux2.b;
212  Matrix1D<double> &w2=haux2.w;
213  A2.resizeNoCopy(M+newM,M);
214  b2.resizeNoCopy(M+newM);
215  w2.resizeNoCopy(M+newM);
216 
217  // Select the equation system
218  int targeti=0;
219  for (int i=0; i<N; ++i)
220  if (VEC_ELEM(idxIn,i))
221  {
222  memcpy(&MAT_ELEM(A2,targeti,0),&MAT_ELEM(h.A,i,0),Mdouble);
223  VEC_ELEM(b2,targeti)=VEC_ELEM(h.b,i);
224  VEC_ELEM(w2,targeti)=VEC_ELEM(h.w,i);
225  ++targeti;
226  }
227 
228  // Solve it with WLS
229  weightedLeastSquares(haux2, resultAux);
230 
231  // Compute the mean error
232  double err=0;
233  for (int i=0; i<M+newM; ++i)
234  {
235  double bp=0;
236  for (int j=0; j<M; ++j)
237  bp+=MAT_ELEM(A2,i,j)*VEC_ELEM(resultAux,j);
238  err+=fabs(VEC_ELEM(b2,i)-bp)*VEC_ELEM(w2,i);
239  }
240  err/=(M+newM);
241  if (err<bestError)
242  {
243  bestError=err;
244  result=resultAux;
245 #ifdef DEBUG
246  std::cout << "Best solution iter: " << it << " Error=" << err << " frac=" << (float)(M+newM)/VEC_XSIZE(h.b) << std::endl;
247 #ifdef DEBUG_MORE
248  std::cout << "Result:" << result << std::endl;
249  for (int i=0; i<M+newM; ++i)
250  {
251  double bp=0;
252  for (int j=0; j<M; ++j)
253  bp+=MAT_ELEM(A2,i,j)*VEC_ELEM(resultAux,j);
254  std::cout << "Eq. " << i << " w=" << VEC_ELEM(w2,i) << " b2=" << VEC_ELEM(b2,i) << " bp=" << bp << std::endl;
255  err+=fabs(VEC_ELEM(b2,i)-bp)*VEC_ELEM(w2,i);
256  }
257 #endif
258 #endif
259  }
260  }
261  }
262  return bestError;
263 }

287 void ransacWeightedLeastSquares(WeightedLeastSquaresHelper &h, Matrix1D<double> &result,
288  double tol, int Niter, double outlierFraction, int Nthreads)
289 {
290  // Read and preprocess the images
291  pthread_t * th_ids = new pthread_t[Nthreads];
292  ThreadRansacArgs * th_args = new ThreadRansacArgs[Nthreads];
293  for (int nt = 0; nt < Nthreads; nt++) {
294  // Passing parameters to each thread
295  th_args[nt].myThreadID = nt;
296  th_args[nt].h = &h;
297  th_args[nt].tol = tol;
298  th_args[nt].Niter = Niter/Nthreads;
299  th_args[nt].outlierFraction = outlierFraction;
300  pthread_create((th_ids + nt), NULL, threadRansacWeightedLeastSquares,
301  (void *) (th_args + nt));
302  }
303 
304  // Waiting for threads to finish
305  double err=1e38;
306  for (int nt = 0; nt < Nthreads; nt++)
307  {
308  pthread_join(*(th_ids + nt), NULL);
309  if (th_args[nt].error<err)
310  {
311  err=th_args[nt].error;
312  result=th_args[nt].result;
313  }
314  }
315 
316  // Threads structures are not needed any more
317  delete []th_ids;
318  delete []th_args;

321 void normalizeColumns(Matrix2D<double> &A)
322 {
323  if (MAT_YSIZE(A)<=1)
324  return;
325 
326  // Compute the mean and standard deviation of each column
327  Matrix1D<double> avg, stddev;
328  avg.initZeros(MAT_XSIZE(A));
329  stddev.initZeros(MAT_XSIZE(A));
330 
331  FOR_ALL_ELEMENTS_IN_MATRIX2D(A)
332  {
333  double x=MAT_ELEM(A,i,j);
334  VEC_ELEM(avg,j)+=x;
335  VEC_ELEM(stddev,j)+=x*x;
336  }
337 
338  double iN=1.0/MAT_YSIZE(A);
339  FOR_ALL_ELEMENTS_IN_MATRIX1D(avg)
340  {
341  VEC_ELEM(avg,i)*=iN;
342  VEC_ELEM(stddev,i)=sqrt(fabs(VEC_ELEM(stddev,i)*iN - VEC_ELEM(avg,i)*VEC_ELEM(avg,i)));
343  if (VEC_ELEM(stddev,i)>XMIPP_EQUAL_ACCURACY)
344  VEC_ELEM(stddev,i)=1.0/VEC_ELEM(stddev,i);
345  else
346  VEC_ELEM(stddev,i)=0.0;
347  }
348 
349  // Now normalize
350  FOR_ALL_ELEMENTS_IN_MATRIX2D(A)
351  MAT_ELEM(A,i,j)=(MAT_ELEM(A,i,j)-VEC_ELEM(avg,j))*VEC_ELEM(stddev,j);
352 }

494 {

496  component.resizeNoCopy(N);
497  component.initConstant(-1);
498 
499  int nextComponent=0;
500  bool workDone=false;
501  std::queue<size_t> toExplore;
502  do
503  {
504  workDone=false;
505  // Find next unvisited element
506  bool found=false;
507  size_t seed=0;
508  FOR_ALL_ELEMENTS_IN_MATRIX1D(component)
509  if (VEC_ELEM(component,i)<0)

511  seed=i;
512  found=true;

518  {

531  if (VEC_ELEM(component,j)<0)
532  {
533  VEC_ELEM(component,j)=currentComponent;
534  toExplore.push(j);
535  }

537  }
538  }
539  workDone=true;
540  }
541  } while (workDone);
