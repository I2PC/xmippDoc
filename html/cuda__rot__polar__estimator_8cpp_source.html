35 void CudaRotPolarEstimator<T>::init2D() {
36  release();
37  auto s = this->getSettings();
38  if (2 != s.hw.size()) {
39  REPORT_ERROR(ERR_ARG_INCORRECT, "Two GPU streams are needed");
40  }
41  try {
42  m_mainStream = dynamic_cast<GPU*>(s.hw.at(0));
43  m_backgroundStream = dynamic_cast<GPU*>(s.hw.at(1));
44  } catch (std::bad_cast&) {
45  REPORT_ERROR(ERR_ARG_INCORRECT, "Instance of GPU expected");
46  }
47  m_mainStream->set();
48  m_backgroundStream->set();
49 
50  // all rings have the same number of samples, to make FT easier
51  m_samples = std::max(1, 2 * (int)(M_PI * s.lastRing)); // keep this even
52  auto batchPolar = FFTSettingsNew<T>(m_samples, 1, 1, // x, y, z
53  s.otherDims.n() * s.getNoOfRings(), // each signal has N rings
54  s.batch * s.getNoOfRings()); // so we have to multiply by that
55  if (s.allowTuningOfNumberOfSamples) {
56  // try to tune number of samples. We don't mind using more samples (higher precision)
57  // if we can also do it faster!
58  auto proposal = CudaFFT<T>::findOptimal(*m_mainStream,
59  // test small batch, as we want the results as soon as possible (should be around 1s)
60  batchPolar.createSubset(std::min((size_t)1000, batchPolar.sDim().n())),
61  0, false, 10, false, false);
62  if (proposal.has_value()) {
63  batchPolar = FFTSettingsNew<T>(proposal.value().sDim().x(), 1, 1, // x, y, z
64  s.otherDims.n() * s.getNoOfRings(), // each signal has N rings
65  s.batch * s.getNoOfRings()); // so we have to multiply by that
66  m_samples = batchPolar.sDim().x();
67  }
68  }
69 
70  auto singlePolar = FFTSettingsNew<T>(m_samples, 1, 1, // x, y, z
71  s.getNoOfRings(), // each signal has N rings
72  s.getNoOfRings()); // process all at once
73 
74  // allocate host memory
75  gpuErrchk(cudaMalloc(&m_d_ref, singlePolar.fBytes())); // FT of the samples
76  if (s.allowDataOverwrite) { // we need space just for reference(s)
77  gpuErrchk(cudaMalloc(&m_d_batch, s.otherDims.sizeSingle() * s.refDims.n() * sizeof(T))); // Cartesian reference(s)
78  } else { // we need space for the whole batch (which should be bigger than no of references)
79  gpuErrchk(cudaMalloc(&m_d_batch, s.otherDims.sizeSingle() * s.batch * sizeof(T))); // Cartesian batch
80  }
81  gpuErrchk(cudaMalloc(&m_d_batchPolarFD, batchPolar.fBytesBatch())); // FT of the polar samples
82  gpuErrchk(cudaMalloc(&m_d_batchPolarOrCorr, batchPolar.sBytesBatch())); // IFT of the polar samples
83  size_t sumsBytes = m_samples * s.batch * sizeof(T);
84  gpuErrchk(cudaMalloc(&m_d_sumsOrMaxPos, sumsBytes));
85  gpuErrchk(cudaMalloc(&m_d_sumsSqr, sumsBytes));
86 
87  // FT plans
88  m_singleToFD = CudaFFT<T>::createPlan(*m_mainStream, singlePolar);
89  m_batchToFD = CudaFFT<T>::createPlan(*m_mainStream, batchPolar);
90  auto inversePolar = FFTSettingsNew<T>(m_samples, 1, 1, // x, y, z
91  s.batch, s.batch, // while computing correlation, we also sum the rings,
92  false,false); // i.e. we end up with batch * samples elements
93  m_batchToSD = CudaFFT<T>::createPlan(*m_mainStream, inversePolar);
94  gpuErrchk(cudaMalloc(&m_d_batchCorrSumFD, inversePolar.fBytesBatch()));
95 
96  // allocate device memory
97  m_h_batchMaxPositions = new float[m_samples * s.batch];
98 
99  // synch primitives
100  m_mutex = new std::mutex();
101  m_cv = new std::condition_variable();
102 }

246 void CudaRotPolarEstimator<T>::loadThreadRoutine(T *others) {
247  m_backgroundStream->set();
248  auto lStream = *(cudaStream_t*)m_backgroundStream->stream();
249  auto s = this->getSettings();
250  for (size_t offset = 0; offset < s.otherDims.n(); offset += s.batch) {
251  std::unique_lock<std::mutex> lk(*m_mutex);
252  // wait till the data is processed
253  m_cv->wait(lk, [&]{return !m_isDataReady;});
254  // how many signals to process
255  size_t toProcess = std::min(s.batch, s.otherDims.n() - offset);
256 
257  T *src = others + offset * s.otherDims.sizeSingle();
258  size_t bytes = toProcess * s.otherDims.sizeSingle() * sizeof(T);
259 
260  auto kind = m_backgroundStream->isGpuPointer(src)
261  ? cudaMemcpyDeviceToDevice
262  : cudaMemcpyHostToDevice;
263  // copy memory
264  gpuErrchk(cudaMemcpyAsync(
265  m_d_batch,
266  src,
267  bytes,
268  kind, lStream));
269  // block until data is loaded
270  m_backgroundStream->synch();
271 
272  // notify processing stream that it can work
273  m_isDataReady = true;
274  m_cv->notify_one();
275  }

301 template<typename T>
302 template<bool FULL_CIRCLE>
303 void CudaRotPolarEstimator<T>::sNormalize(
304  const GPU &gpu,
305  const Dimensions &dim,
306  T * __restrict__ d_in,
307  T * __restrict__ d_sums,
308  T * __restrict__ d_sumsSqr,
309  int posOfFirstRing) {
310  dim3 dimBlock(64);
311  dim3 dimGrid(
312  ceil((dim.x() * dim.n()) / (float)dimBlock.x));
313 
314  auto stream = *(cudaStream_t*)gpu.stream();
315  // clear the arrays
316  size_t bytes = dim.n() * sizeof(T);
317  gpuErrchk(cudaMemset(d_sums, 0, bytes));
318  gpuErrchk(cudaMemset(d_sumsSqr, 0, bytes));
319  computeSumSumSqr<T, FULL_CIRCLE>
320  <<<dimGrid, dimBlock, 0, stream>>> (
321  d_in, dim.x(), dim.y(), dim.n(),
322  d_sums, d_sumsSqr, posOfFirstRing);
323 
324  const T piConst = FULL_CIRCLE ? (2 * M_PI) : M_PI;
325  // sum of the first n terms of an arithmetic sequence
326  // a1 = first radius (posOfFirstRing)
327  // an = last radius (dim.y() - 1 + posOfFirstRing)
328  // s = n * (a1 + an) / 2
329  const T radiiSum = (dim.y() * (2 * posOfFirstRing + dim.y() - 1)) / (T)2;
330  T norm = piConst * radiiSum;
331 
332  normalize<T>
333  <<<dimGrid, dimBlock, 0, stream>>> (
334  d_in, dim.x(), dim.y(), dim.n(),
335  norm,
336  d_sums, d_sumsSqr);
337 }

371 template<typename T>
372 template<bool FULL_CIRCLE>
373 void CudaRotPolarEstimator<T>::computeRotation2DOneToN(T *others) {
374  const auto &s = this->getSettings();
375  bool isReady = this->isInitialized() && this->isRefLoaded();
376  if ( ! isReady) {
377  REPORT_ERROR(ERR_LOGIC_ERROR, "Not ready to execute. Call init() and load reference");
378  }
379  if ( ! GPU::isMemoryPinned(others)) {
380  REPORT_ERROR(ERR_LOGIC_ERROR, "Input memory has to be pinned (page-locked)");
381  }
382  if (s.allowDataOverwrite && ( ! m_mainStream->isGpuPointer(others))) {
383  REPORT_ERROR(ERR_LOGIC_ERROR, "Incompatible parameters: allowDataOverwrite && 'others' data on host");
384  }
385 
386  m_mainStream->set();
387  // make sure that all work (e.g. loading and processing of the reference) is done
388  m_mainStream->synch();
389  // start loading data at the background
390  m_isDataReady = false;
391  const bool createLocalCopy = ( ! s.allowDataOverwrite);
392  std::thread loadingThread;
393  if (createLocalCopy) {
394  loadingThread = std::thread(&CudaRotPolarEstimator<T>::loadThreadRoutine, this, others);
395  }
396 
397  // process signals in batches
398  for (size_t offset = 0; offset < s.otherDims.n(); offset += s.batch) {
399  // how many signals to process
400  size_t toProcess = std::min(s.batch, s.otherDims.n() - offset);
401  auto inCart = s.otherDims.copyForN(toProcess);
402  auto outPolar = Dimensions(m_samples, s.getNoOfRings(), 1, toProcess);
403  auto outPolarFourier = Dimensions(m_samples / 2 + 1, s.getNoOfRings(), 1, toProcess);
404  auto resSize = Dimensions(m_samples, 1, 1, toProcess);
405 
406  // get proper data
407  if (createLocalCopy) {
408  waitAndConvert<FULL_CIRCLE>(inCart, outPolar, s.firstRing);
409  } else {
410  T *in = others + (offset * s.otherDims.sizeSingle());
411  sComputePolarTransform<FULL_CIRCLE>(*m_mainStream,
412  inCart, in,
413  outPolar, m_d_batchPolarOrCorr,
414  s.firstRing);
415  }
416 
417  // It seems that the normalization is not necessary (does not change the result
418  // of the synthetic tests. It can however prevent e.g. overflow, hence it's kept here)
419 // sNormalize<FULL_CIRCLE>(*m_mainStream,
420 // outPolar, m_d_batchPolarOrCorr,
421 // m_d_sumsOrMaxPos, m_d_sumsSqr,
422 // s.firstRing);
423 
424  CudaFFT<T>::fft(*m_batchToFD, m_d_batchPolarOrCorr, m_d_batchPolarFD);
425 
426  sComputeCorrelationsOneToN(*m_mainStream, m_d_batchPolarFD, m_d_batchCorrSumFD, m_d_ref, outPolarFourier, s.firstRing);
427 
428  CudaFFT<T>::ifft(*m_batchToSD, m_d_batchCorrSumFD, m_d_batchPolarOrCorr);
429 
430  // locate maxima for each signal (results are always floats)
431  auto d_positions = (float*)m_d_sumsOrMaxPos;
432  ExtremaFinder::CudaExtremaFinder<T>::sFindMax(
433  *m_mainStream, resSize, m_d_batchPolarOrCorr, d_positions, nullptr);
434 
435  // copy data back (they are always float)
436  auto stream = *(cudaStream_t*)m_mainStream->stream();
437  gpuErrchk(cudaMemcpyAsync(
438  m_h_batchMaxPositions,
439  d_positions,
440  resSize.n() * sizeof(float), // one position per signal
441  cudaMemcpyDeviceToHost, stream));
442  m_mainStream->synch();
443 
444  // convert positions of the maxima to angles
445  for (int n = 0; n < resSize.n(); ++n) {
446  auto angle = m_h_batchMaxPositions[n] * (FULL_CIRCLE ? (T)360 : (T)180) / resSize.x();
447  this->getRotations2D().emplace_back(angle);
448  }
449  }
450  if (createLocalCopy) {
451  loadingThread.join();
452  }
453 }

455 template<typename T>
456 void CudaRotPolarEstimator<T>::check() {
457  auto s = this->getSettings();
458  if (s.refDims.x() != s.refDims.y()) {
459  // because of the rings
460  REPORT_ERROR(ERR_ARG_INCORRECT, "This estimator can work only with square signal");
461  }
462  if (s.refDims.x() < 6) {
463  // we need some edge around the biggest ring, to avoid invalid memory access
464  REPORT_ERROR(ERR_ARG_INCORRECT, "The input signal is too small.");
465  }
466  if (s.refDims.isPadded()) {
467  REPORT_ERROR(ERR_ARG_INCORRECT, "Padded signal is not supported");
468  }
469  if ((s.lastRing + 1) * 2 >= s.refDims.x()) {
470  // bilinear interpolation we use does not check boundaries
471  REPORT_ERROR(ERR_ARG_INCORRECT, "Last ring is too big, this would cause invalid memory access");
472  }
473 }
