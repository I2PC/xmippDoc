126 /* Compute protein geometry ------------------------------------------------ */
127 void ProgPdbValueToVol::computeProteinGeometry()
128 {
129  std::ifstream fh_pdb;
130  std::ofstream fh_out(fn_out);
131  fh_pdb.open(fn_pdb.c_str());
132 
133  MetaData mdmean;
134  size_t objId;
135  objId = mdmean.addObject();
136 
137  if (!fh_pdb)
138  REPORT_ERROR(ERR_IO_NOTEXIST, fn_pdb);
139 
140  double suma=0, sumaP=0;
141  int numA=0;
142 
143 
144  while (!fh_pdb.eof())
145  {
146  // Read an ATOM line
147  std::string line;
148  getline(fh_pdb, line);
149  if (line == "")
150  {
151  fh_out << line << " \n";
152  continue;
153  }
154  std::string kind = line.substr(0,4);
155  if (kind != "ATOM" && kind !="HETA")
156  {
157  fh_out << line << " \n";
158  continue;
159  }
160 
161  // Extract atom type and position
162  // Typical line:
163  // ATOM 909 CA ALA A 161 58.775 31.984 111.803 1.00 34.78
164  std::string atom_type = line.substr(13,2);
165  double x = textToFloat(line.substr(30,8));
166  double y = textToFloat(line.substr(38,8));
167  double z = textToFloat(line.substr(46,8));
168 
169  // Correct position
170  Matrix1D<double> r(3);
171  VECTOR_R3(r, x, y, z);
172  r *= 1/Ts;
173 
174  // Characterize atom
175  double radius2=radius*radius;
176 
177  // Find the part of the volume that must be updated
178  int k0 = XMIPP_MAX(FLOOR(ZZ(r) - radius), STARTINGZ(inputVol));
179  int kF = XMIPP_MIN(CEIL(ZZ(r) + radius), FINISHINGZ(inputVol));
180  int i0 = XMIPP_MAX(FLOOR(YY(r) - radius), STARTINGY(inputVol));
181  int iF = XMIPP_MIN(CEIL(YY(r) + radius), FINISHINGY(inputVol));
182  int j0 = XMIPP_MAX(FLOOR(XX(r) - radius), STARTINGX(inputVol));
183  int jF = XMIPP_MIN(CEIL(XX(r) + radius), FINISHINGX(inputVol));
184 
185  int ka = XMIPP_MAX(FLOOR(ZZ(r)), STARTINGZ(inputVol));
186  int ia = XMIPP_MAX(FLOOR(YY(r)), STARTINGZ(inputVol));
187  int ja = XMIPP_MAX(FLOOR(XX(r)), STARTINGZ(inputVol));
188 
189 
190  float atomS=0.0;
191  float atomP=0.0;
192  float atomN=0.0;
193  float value=0.0;
194  int cont=0;
195  for (int k = k0; k <= kF; k++)
196  {
197  double zdiff=ZZ(r) - k;
198  double zdiff2=zdiff*zdiff;
199  for (int i = i0; i <= iF; i++)
200  {
201  double ydiff=YY(r) - i;
202  double zydiff2=zdiff2+ydiff*ydiff;
203  for (int j = j0; j <= jF; j++)
204  {
205  double xdiff=XX(r) - j;
206  double rdiffModule2=zydiff2+xdiff*xdiff;
207  if (withMask)
208  {
209  if ( (rdiffModule2<radius2 || (k==ka && i==ia && j==ja)) && (inputMask(k, i , j)>0.00001) )
210  {
211  atomS += A3D_ELEM(inputVol,k, i, j);
212  ++cont;
213 
214  //Positivity
215  if (A3D_ELEM(inputVol,k, i, j) < 0)
216  {
217  value = -A3D_ELEM(inputVol,k, i, j);
218  atomP += atomP;
219  }
220  else
221  atomP += A3D_ELEM(inputVol,k, i, j);
222 
223  //Negativity
224  if (A3D_ELEM(inputVol,k, i, j) > 0)
225  {
226  value = -A3D_ELEM(inputVol,k, i, j);
227  atomN += atomN;
228  }
229  else
230  atomN += A3D_ELEM(inputVol,k, i, j);
231 
232  }
233  }
234  else
235  {
236  if ( (rdiffModule2<radius2) || (k==ka && i==ia && j==ja))
237  {
238  atomS+=A3D_ELEM(inputVol,k, i, j);
239  ++cont;
240 
241  //Positivity
242  if (A3D_ELEM(inputVol,k, i, j) < 0)
243  {
244  value = -A3D_ELEM(inputVol,k, i, j);
245  atomP += value;
246  }
247  else
248  atomP += A3D_ELEM(inputVol,k, i, j);
249 
250 
251  //Negativity
252  if (A3D_ELEM(inputVol,k, i, j) > 0)
253  {
254  value = -A3D_ELEM(inputVol,k, i, j);
255  atomN += atomN;
256  }
257  else
258  atomN += A3D_ELEM(inputVol,k, i, j);
259 
260  }
261  }
262 
263  }
264  }
265  }
266 
267  if (atomS>=0)
268  atomS = atomP;
269  else
270  atomS = atomN;
271 
272  if (atomS != 0)
273  atomS=atomS/cont;
274  else
275  atomS=0.00;
276 
277  if (atomP != 0)
278  atomP=atomP/cont;
279  else
280  atomP=0.00;
281 
282 
283  ++numA;
284  suma+=atomS;
285  sumaP+=atomP;
286 
287  std::stringstream stream;
288  stream << std::fixed << std::setprecision(2) << atomS;
289  std::string s = stream.str();
290 
291  if (atomS<0)
292  line.replace(55, 5, s, 0, 5);
293  else
294  line.replace(56, 4, s, 0, 4);
295 // std::cout << line << std::endl;
296 
297  fh_out << line << " \n";
298  }
299 
300  double mean = suma/numA;
301  double meanA = sumaP/numA;
302  std::cout << "mean value: = " << mean << std::endl;
303  std::cout << "absolute mean value: = " << meanA << std::endl;
304 
305  mdmean.setValue(MDL_VOLUME_SCORE1, mean, objId);
306  mdmean.setValue(MDL_VOLUME_SCORE2, meanA, objId);
307  mdmean.write(fnMD);
308 
309  // Close file
310  fh_pdb.close();
311  fh_out.close();
312 
313 }
