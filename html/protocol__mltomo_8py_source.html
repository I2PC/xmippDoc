293  def createInputMd(self, vols):
294  fnVols = self._getExtraPath('input_volumes.xmd')
295  if self.copyAlignment:
296  alignType = vols.getAlignment()
297  else:
298  alignType = ALIGN_NONE
299 
300  writeSetOfVolumes(vols, fnVols,
301  postprocessImageRow=self._postprocessVolumeRow,
302  alignType=alignType)
303  if not vols.hasAlignment() or not self.copyAlignment:
304  mdFn = md.MetaData(fnVols)
305  mdFn.fillConstant(md.MDL_ANGLE_ROT, 0.)
306  mdFn.fillConstant(md.MDL_ANGLE_TILT, 0.)
307  mdFn.fillConstant(md.MDL_ANGLE_PSI, 0.)
308  mdFn.fillConstant(md.MDL_SHIFT_X, 0.)
309  mdFn.fillConstant(md.MDL_SHIFT_Y, 0.)
310  mdFn.fillConstant(md.MDL_SHIFT_Z, 0.)
311  mdFn.write(fnVols, md.MD_OVERWRITE)
312 
313  # set missing angles
314  missType = ['wedge_y', 'wedge_x', 'pyramid', 'cone']
315  missNum = self.missingDataType.get()
316  missAng = self.missingAng.get()
317  missDataFn = self._getExtraPath('wedges.xmd')
318  missAngValues = str(missAng).strip().split()
319  thetaY0, thetaYF, thetaX0, thetaXF = 0, 0, 0, 0
320  mdFn = md.MetaData()
321 
322  if missNum == MISSING_WEDGE_X:
323  thetaX0, thetaXF = missAngValues
324  elif missNum == MISSING_WEDGE_Y:
325  thetaY0, thetaYF = missAngValues
326  elif missNum == MISSING_PYRAMID:
327  thetaY0, thetaYF, thetaX0, thetaXF = missAngValues
328  else: # MISSING_CONE
329  thetaY0 = missAngValues[0]
330 
331  for i in range(1, vols.getSize() + 1):
332  row = md.Row()
333  row.setValue(md.MDL_MISSINGREGION_NR, missNum + 1)
334  row.setValue(md.MDL_MISSINGREGION_TYPE, missType[missNum])
335  row.setValue(md.MDL_MISSINGREGION_THX0, float(thetaX0))
336  row.setValue(md.MDL_MISSINGREGION_THXF, float(thetaXF))
337  row.setValue(md.MDL_MISSINGREGION_THY0, float(thetaY0))
338  row.setValue(md.MDL_MISSINGREGION_THYF, float(thetaYF))
339  row.addToMd(mdFn)
340 
341  mdFn.write(missDataFn, md.MD_APPEND)
