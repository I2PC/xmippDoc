42 bool MpiTaskDistributor::distributeMaster()
43 {
44  int size = node->size;
45  size_t workBuffer[3];
46  MPI_Status status;
47  int finalizedWorkers = 0;
48 
49  while (finalizedWorkers < size - 1)
50  {
51  //wait for request form workers
52  MPI_Recv(0, 0, MPI_INT, MPI_ANY_SOURCE, TAG_WORK_REQUEST, MPI_COMM_WORLD, &status);
53 
54  workBuffer[0] = ThreadTaskDistributor::distribute(workBuffer[1], workBuffer[2]) ? 1 : 0;
55 
56  if (workBuffer[0] == 0) //no more jobs, count finalized workers
57  finalizedWorkers++;
58  //send response (either task or finish answer)
59  MPI_Send(workBuffer, 3, MPI_LONG_LONG_INT, status.MPI_SOURCE, TAG_WORK_RESPONSE, MPI_COMM_WORLD);
60  }
61  return false;
62 }

64 bool MpiTaskDistributor::distributeSlaves(size_t &first, size_t &last)
65 {
66  // Worker nodes should ask for task to master
67  // Result of workBuffer:
68  // workBuffer[0] = 0 if no more jobs, 1 otherwise
69  // workBuffer[1] = first
70  // workBuffer[2] = last
71  size_t workBuffer[3];
72  MPI_Status status;
73  //any message from the master, is tag is TAG_STOP then stop
74  MPI_Send(0, 0, MPI_INT, 0, TAG_WORK_REQUEST, MPI_COMM_WORLD);
75  MPI_Recv(workBuffer, 3, MPI_LONG_LONG_INT, 0, TAG_WORK_RESPONSE, MPI_COMM_WORLD, &status);
76 
77  first = workBuffer[1];
78  last = workBuffer[2];
79 
80  return (workBuffer[0] == 1);
81 }
