242 // Compute fitness =========================================================
243 double ObjFunc_nma_alignment_vol::eval(Vector X, int *nerror) {
244  int dim = global_nma_vol_prog->numberOfModes;
245 
246  for (int i = 0; i < dim; i++) {
247  global_nma_vol_prog->trial(i) = X[i];
248  }
249 
250  FileName fnRandom = global_nma_vol_prog->createDeformedPDB();
251  const char * randStr = fnRandom.c_str();
252  double retval=std::numeric_limits<double>::max();
253 
254  FileName fnShiftsAngles = fnRandom + "_angles_shifts.txt";
255  const char * shifts_angles = fnShiftsAngles.c_str();
256 
257  String fnVolume1 = global_nma_vol_prog->currentVolName;
258  String fnVolume2 = fnRandom + "_deformedPDB.vol";
259 
260  if (global_nma_vol_prog->tilt0!=-90 || global_nma_vol_prog->tiltF!=90 ){
261  global_nma_vol_prog->flip = true;
262  runSystem("xmipp_transform_geometry",formatString("-i %s -o %s_currentvolume.vol --rotate_volume euler 0 90 0 -v 0",fnVolume1.c_str(),randStr));
263  fnVolume1 = fnRandom+"_currentvolume.vol";
264  }
265 
266  const char * Volume1 = fnVolume1.c_str();
267  const char * Volume2 = fnVolume2.c_str();
268 
269  if (global_nma_vol_prog->alignVolumes){
270  runSystem("xmipp_volume_align",formatString("--i1 %s --i2 %s --frm %f %d %d %d --store %s -v 0 ",
271  Volume1,Volume2,global_nma_vol_prog->frm_freq, global_nma_vol_prog->frm_shift, global_nma_vol_prog->tilt0, global_nma_vol_prog->tiltF, shifts_angles));
272  //first just see what is the score
273  global_nma_vol_prog->AnglesShiftsAndScore = fopen(shifts_angles, "r");
274  // fit_value is the last element in shifts_angles. To get it without looping on all the file, we seek the end_of_file -10 (10 always work because fit_value is always < 1).
275  fseek(global_nma_vol_prog->AnglesShiftsAndScore, -10, SEEK_END);
276  fscanf(global_nma_vol_prog->AnglesShiftsAndScore, "%f,", &global_nma_vol_prog->fit_value);
277  fclose(global_nma_vol_prog->AnglesShiftsAndScore);
278  retval = 1 + global_nma_vol_prog->fit_value;
279  }
280 
281  else{
282  global_nma_vol_prog->Vdeformed.read(formatString("%s_deformedPDB.vol",fnRandom.c_str()));
283  auto mask = (0 == XSIZE(global_nma_vol_prog->mask)) ? nullptr : &global_nma_vol_prog->mask;
284  retval = 1 - correlationIndex(global_nma_vol_prog->V(), global_nma_vol_prog->Vdeformed(), mask);
285  //global_nma_vol_prog->V().printStats();
286  //global_nma_vol_prog->Vdeformed().printStats();
287  //std::cout << correlationIndex(global_nma_vol_prog->V(),global_nma_vol_prog->Vdeformed()) << std::endl;
288  }
289 
290  if(global_nma_vol_prog->updateBestFit(retval, dim) && global_nma_vol_prog->alignVolumes){
291  global_nma_vol_prog->AnglesShiftsAndScore = fopen(shifts_angles, "r");
292  for (int i = 0; i < 6; i++){
293  fscanf(global_nma_vol_prog->AnglesShiftsAndScore, "%f,", &global_nma_vol_prog->Best_Angles_Shifts[i]);
294  }
295  fclose(global_nma_vol_prog->AnglesShiftsAndScore);
296  }
297 
298  runSystem("rm", formatString("-rf %s* &", randStr));
299  //std::cout << global_nma_vol_prog->trial << " -> " << retval << std::endl;
300  return retval;
301 }

306 void ProgNmaAlignmentVol::processImage(const FileName &fnImg,
307  const FileName &fnImgOut, const MDRow &rowIn, MDRow &rowOut) {
308  static size_t imageCounter = 0;
309  ++imageCounter;
310 
311  ObjectiveFunction *of;
312 
313  int dim = numberOfModes;
314 
315  parameters.initZeros(dim);
316  V.read(fnImg);
317  currentVolName = fnImg;
318  sprintf(nameTemplate, "_node%d_img%lu_XXXXXX", rangen, (long unsigned int)imageCounter);
319 
320  trial.initZeros(dim);
321  trial_best.initZeros(dim);
322 
323  fitness_min=1000000.0;
324 
325  of = new ObjFunc_nma_alignment_vol(1, dim);
326 
327  of->xStart.setSize(dim);
328  for (int i = 0; i < dim; i++)
329  of->xStart[i] = 0.;
330 
331  double rhoStart=trustradius_scale*rhoStartBase;
332  double rhoEnd=trustradius_scale*rhoEndBase;
333 
334  CONDOR(rhoStart, rhoEnd, niter, of);
335 
336  trial = parameters = trial_best;
337 
338  parameters.resize(VEC_XSIZE(parameters) + 1);
339  parameters(VEC_XSIZE(parameters) - 1) = fitness_min;
340 
341  writeVolumeParameters(fnImg);
342  if (fnOutPDB!="")
343  deformPDB(fnPDB,fnOutPDB,fnModeList,trial_best);
344  delete of;
345 }

347 void ProgNmaAlignmentVol::writeVolumeParameters(const FileName &fnImg) {
348  MetaData md;
349  size_t objId = md.addObject();
350  md.setValue(MDL_IMAGE, fnImg, objId);
351  md.setValue(MDL_ENABLED, 1, objId);
352 
353  std::vector<double> vectortemp;
354  double energy=0;
355  for (int j = 0; j < numberOfModes; j++)
356  {
357  double lambdaj=parameters(j);

359  energy+=lambdaj*lambdaj;
360  }
361 
362 
363  energy/=numberOfModes;
364  md.setValue(MDL_ANGLE_ROT, (double)global_nma_vol_prog->Best_Angles_Shifts[0], objId);
365  md.setValue(MDL_ANGLE_TILT, (double)global_nma_vol_prog->Best_Angles_Shifts[1], objId);

368  md.setValue(MDL_SHIFT_Y, (double)global_nma_vol_prog->Best_Angles_Shifts[4], objId);
369  md.setValue(MDL_SHIFT_Z, (double)global_nma_vol_prog->Best_Angles_Shifts[5], objId);
