130  firstImage = self.inputParticles.get().getFirstItem()
131  fnClasses = self._params['kclasses']
132  mdClasses = "classes@%s" % fnClasses
133  fnClassStack = self._params['classes']

136  md = emlib.MetaData(mdClasses)
137  image = ImageHandler().createImage()
138 
139  counter = 1
140 
141  for objId in md:
142  imageName = "%06d@%s" % (counter, fnClassStack)
143  averageName = "%06d@%s" % (counter, fnAverageStack)
144 
145  if md.getValue(emlib.MDL_CLASS_COUNT, objId) > 0:
146  # compute the average of images assigned to this class
147  classPrefix = 'class%06d' % counter
148  classMd = '%s_images@%s' % (classPrefix, fnClasses)
149  classRoot = self._getTmpPath(classPrefix)
150  self.runJob('xmipp_image_statistics',
151  '-i %s --save_image_stats %s -v 0' % (classMd, classRoot))
152  image.read(classRoot + 'average.xmp')
153  else:
154  # Create emtpy image as average
155  image.read(firstImage.getLocation()) # just to take the right dimensions and datatype
156  image.initConstant(0.)
157 
158  image.write(averageName)
159  md.setValue(emlib.MDL_IMAGE, imageName, objId)
160  md.setValue(emlib.MDL_IMAGE2, averageName, objId)
161 
162  counter += 1
