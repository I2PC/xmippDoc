36  double rot2p, tilt2p, psi2p, psi2 = 0.;
37  double diff_rot, diff_tilt;
38 
39  int isymmax=SL.symsNo();
40  for (int isym = 0; isym <= isymmax; isym++)
41  {
42  if (isym == 0)
43  {
44  rot2p = rot2;
45  tilt2p = tilt2;
46  psi2p = psi2;
47  }
48  else
49  {
50  SL.getMatrices(isym - 1, Laux, Raux,false);
51  Euler_apply_transf(Laux, Raux, rot2, tilt2, psi2, rot2p, tilt2p, psi2p);
52  }
53 
54  double aux=rot - rot2p;
55  diff_rot = fabs(realWRAP(aux, -180, 180));
56  aux=tilt - tilt2p;
57  diff_tilt = fabs(realWRAP(aux, -180, 180));
58  if ((rot_limit - diff_rot) > 1e-3 && (tilt_limit - diff_tilt) > 1e-3)
59  are_unique = false;
60  Euler_another_set(rot2p, tilt2p, psi2p, rot2p, tilt2p, psi2p);
61  aux=rot - rot2p;
62  diff_rot = fabs(realWRAP(aux, -180, 180));
63  aux=tilt - tilt2p;
64  diff_tilt = fabs(realWRAP(aux, -180, 180));
65  if ((rot_limit - diff_rot) > 1e-3 && (tilt_limit - diff_tilt) > 1e-3)
66  are_unique = false;
67  if (!include_mirrors)
68  {
69  Euler_up_down(rot2p, tilt2p, psi2p, rot2p, tilt2p, psi2p);
70  aux=rot - rot2p;
71  diff_rot = fabs(realWRAP(aux, -180, 180));
72  aux=tilt - tilt2p;
73  diff_tilt = fabs(realWRAP(aux, -180, 180));

131 void make_even_distribution(std::vector<double> &rotList, std::vector<double> &tiltList,
132  double sampling, SymList &SL, bool include_mirror)
133 {
134  int rot_nstep, tilt_nstep = ROUND(180. / sampling) + 1;
135  double rot_sam, tilt, tilt_sam;
136  bool append;
137  tilt_sam = (180. / tilt_nstep);
138 
139  // Create evenly distributed angles
140  rotList.clear();
141  tiltList.clear();
142  rotList.reserve(20000); // Normally there are many less directions than 20000
143  tiltList.reserve(20000); // Set to 20000 to avoid resizing
144  Matrix2D<double> L(3,3),R(3,3);
145  for (int tilt_step = 0; tilt_step < tilt_nstep; tilt_step++)
146  {
147  tilt = ((double)tilt_step / (tilt_nstep - 1)) * 180.;
148  if (tilt > 0)
149  rot_nstep = CEIL(360. * sin(DEG2RAD(tilt)) / sampling);
150  else
151  rot_nstep = 1;
152  rot_sam = 360. / (double)rot_nstep;
153  for (double rot = 0.; rot < 360.; rot += rot_sam)
154  {
155  // Check whether by symmetry or mirror the angle has been included already
156  append = true;
157  size_t imax=rotList.size();
158  double *ptrRot=NULL;
159  double *ptrTilt=NULL;
160  if (imax>0)
161  {
162  ptrRot=&rotList[0];
163  ptrTilt=&tiltList[0];
164  }
165  for (size_t i=0; i<imax; ++i, ++ptrRot, ++ptrTilt)
166  {
167  if (!directions_are_unique(rot, tilt, *ptrRot, *ptrTilt, rot_sam, tilt_sam, SL,
168  include_mirror, L, R))
169  {
170  append = false;
171  break;
172  }
173  }
174  if (append)
175  {
176  rotList.push_back(rot);
177  tiltList.push_back(tilt);
178  }
179  }
180  }
