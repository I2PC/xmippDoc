476  def readPartsFromMics(self, micList, outputParts):
477  """ Read the particles extract for the given list of micrographs
478  and update the outputParts set with new items.
479  """
480  p = Particle()
481  for mic in micList:
482  # We need to make this dict because there is no ID in the .xmd file
483  coordDict = {}
484  for coord in self.coordDict[mic.getObjId()]:
485  pos = self._getPos(coord)
486  if pos in coordDict:
487  print("WARNING: Ignoring duplicated coordinate: %s, id=%s" %
488  (coord.getObjId(), pos))
489  coordDict[pos] = coord
490 
491  added = set() # Keep track of added coords to avoid duplicates
492  fnMicXmd = self._getMicXmd(mic)
493  if exists(fnMicXmd):
494  for row in md.iterRows(fnMicXmd):
495  pos = (row.getValue(md.MDL_XCOOR), row.getValue(md.MDL_YCOOR))
496  coord = coordDict.get(pos, None)
497  if coord is not None and coord.getObjId() not in added:
498  # scale the coordinates according to particles dimension.
499  coord.scale(self.getBoxScale())
500  p.copyObjId(coord)
501  p.setLocation(xmippToLocation(row.getValue(md.MDL_IMAGE)))
502  p.setCoordinate(coord)
503  p.setMicId(mic.getObjId())
504  p.setCTF(mic.getCTF())
505  # adding the variance and Gini coeff. value of the mic zone
506  setXmippAttributes(p, row, md.MDL_SCORE_BY_VAR)
507  setXmippAttributes(p, row, md.MDL_SCORE_BY_GINI)
508  if row.containsLabel(md.MDL_ZSCORE_DEEPLEARNING1):
509  setXmippAttributes(p, row, md.MDL_ZSCORE_DEEPLEARNING1)
510 
511  # disabled particles (in metadata) should not add to the
512  # final set
513  if row.getValue(md.MDL_ENABLED) > 0:
514  outputParts.append(p)
515  added.add(coord.getObjId())
516 
517  # Release the list of coordinates for this micrograph since it
518  # will not be longer needed
519  del self.coordDict[mic.getObjId()]
