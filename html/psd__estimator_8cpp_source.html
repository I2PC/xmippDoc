29 std::vector<Rectangle<Point2D<size_t>>> PSDEstimator<T>::getPatchesLocation(
30  const std::pair<size_t, size_t> &borders,
31  const Dimensions &micrograph,
32  const Dimensions &patch,
33  float overlap) {
34  assert(micrograph.x() >= patch.x());
35  assert(micrograph.y() >= patch.y());
36  assert((2 * borders.first + patch.x()) <= micrograph.x());
37  assert((2 * borders.second + patch.y()) <= micrograph.y());
38  assert(overlap >= 0.f);
39  assert(overlap < 1.f);
40 
41  size_t stepX = std::max((1 - overlap) * patch.x(), 1.f);
42  size_t stepY = std::max((1 - overlap) * patch.y(), 1.f);
43 
44  size_t maxX = micrograph.x() - borders.first - patch.x();
45  size_t maxY = micrograph.y() - borders.second - patch.y();
46 
47  auto result = std::vector<Rectangle<Point2D<size_t>>>();
48 
49  size_t y = borders.second;
50  while (y < (maxY + stepY)) {
51  size_t ys = std::min(y, maxY);
52  size_t ye = ys + patch.y() - 1;
53  size_t x = borders.first;
54  while (x < (maxX + stepX)) {
55  size_t xs = std::min(x, maxX);
56  size_t xe = xs + patch.x() - 1;
57  auto tl = Point2D<size_t>(xs, ys);
58  auto br = Point2D<size_t>(xe, ye);
59  result.emplace_back(tl, br);
60  x += stepX;
61  }
62  y += stepY;
63  }
64  return result;
65 }

69  float overlap, const Dimensions &patchDim, MultidimArray<T> &psd,
70  unsigned fftThreads) {

76  overlap);

78  auto settings = FFTSettingsNew<T>(patchDim);
79 
80  // prepare data for FT - set proper sizes and allocate aligned dat for faster execution
81  // XXX HACK
82  MultidimArray<T> patchData;
83  patchData.destroyData = false;
84  patchData.setDimensions(patchDim.x(), patchDim.y(), 1, 1);
85  patchData.nzyxdimAlloc = patchData.nzyxdim;
86  patchData.data = (T*)transformer::allocateAligned(settings.sBytesSingle());
87 
88  MultidimArray<T> smoother;

90  smoother.resetOrigin();

93  auto magnitudes = new T[settings.fElemsBatch()](); // initialize to zero
94 
95  auto hw = CPU(fftThreads);
96  auto plan = transformer::createPlan(hw, settings, true);
97 
98  for (auto &p : patches) {
99  // get patch data
100  window2D(micrograph, patchData,
101  p.tl.y, p.tl.x, p.br.y, p.br.x);
102  // normalize, otherwise we would get 'white cross'
103  patchData.statisticsAdjust((T)0, (T)1);
104  patchData.resetOrigin();
105  // apply edge attenuation
106  patchData *= smoother;
107  // perform FFT
108  transformer::fft(plan, patchData.data, patchFS);
109  // get average of amplitudes
110  for (size_t n = 0; n < settings.fElemsBatch(); ++n) {
111  auto v = patchFS[n]; // / (T)settings.sDim().xyz();
112  auto mag = sqrt((v.real() * v.real()) + (v.imag() * v.imag()));
113  magnitudes[n] += mag;
114  }
115  }

119  half2whole(magnitudes, psd.data, settings, [&](bool mirror, T val){return val;});

123  transformer::release(plan);
124  transformer::release(patchData.data);
125 }
