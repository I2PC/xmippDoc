489 double MultidimArray<double>::interpolatedElement2D(double x, double y, double outside_value) const
490 {
491  double dx0 = floor(x);
492  int x0=(int)dx0;
493  double fx = x - dx0;
494  int x1 = x0 + 1;
495  double dy0 = floor(y);
496  int y0=(int)dy0;
497  double fy = y - dy0;
498  int y1 = y0 + 1;
499 
500  int i0=STARTINGY(*this);
501  int j0=STARTINGX(*this);
502  int iF=FINISHINGY(*this);
503  int jF=FINISHINGX(*this);
504 
505  /* Next code avoids checking two times some variables that are doubles.
506  * The code before was:
507  ASSIGNVAL2D(d00,y0,x0);
508  ASSIGNVAL2D(d01,y0,x1);
509  ASSIGNVAL2D(d10,y1,x0);
510  ASSIGNVAL2D(d11,y1,x1);
511  */
512  double d00, d10, d11, d01, *ref;
513  if ((x0 >= j0) && (x1 <= jF) && (y0 >= i0) && (y1 <= iF))
514  {
515  ref = &A2D_ELEM(*this, y0, x0);
516  d00 = (*ref++);
517  d01 = (*ref);
518  ref = &A2D_ELEM(*this, y1, x0);
519  d10 = (*ref++);
520  d11 = (*ref);
521  }
522  else
523  {
524  bool outX0,outX1,outY0,outY1;
525  outX0 = (x0 < j0) || (x0 > jF);
526  outX1 = (x1 < j0) || (x1 > jF);
527  outY0 = (y0 < i0) || (y0 > iF);
528  outY1 = (y1 < i0) || (y1 > iF);
529 
530  if (outY0)
531  {
532  d00 = outside_value;
533  d01 = outside_value;
534  d10 = outside_value;
535  d11 = outside_value;
536  }
537  else if(outY1)
538  {
539  d10 = outside_value;
540  d11 = outside_value;
541 
542  if (outX0)
543  {
544  d00 = outside_value;
545  d01 = outside_value;
546  }
547  else
548  {
549  d00 = A2D_ELEM(*this, y0, x0);
550  if (outX1)
551  {
552  d01 = outside_value;
553  }
554  else

557  }
558  }
559  }
560  else

563  {
564  d00 = outside_value;
565  d10 = outside_value;
566  d01 = outside_value;
567  d11 = outside_value;
568  }
569  else

575  }
576  else
