29 void ExtractSubset::createSubset(const Settings &s) {
30  s.check();
31  auto it = MDIterator(s.md);
32  // get to proper position
33  for (size_t i = 0; i < s. first; ++i) {
34  it.moveNext();
35  }
36  MDRow row;
37  auto destMD = MetaData();
38  auto img = Image<float>();
39  // iterate through all items
40  for (size_t i = 0; i < s.count; it.moveNext(), i++) {
41  // orig name
42  FileName origName;
43  s.md.getValue(MDL_IMAGE, origName, it.objId);
44  // new name
45  FileName newName;
46  newName.compose(i + 1, s.outStk); // within stk file, index images from one (1)
47  // copy row, replace name
48  s.md.getRow(row, it.objId);
49  row.setValue(MDL_IMAGE, newName);
50  row.setValue(MDL_ENABLED, 1);
51  destMD.addRow(row);
52  // copy image
53  // FIXME DS maybe we can do this more efficiently
54  img.read(origName);
55  img.write(newName, i, true, WRITE_APPEND);
56  }
57  // store metadata
58  destMD.write(s.outXmd);
59 }

61 void ExtractSubset::Settings::check() const {
62  // target directories must exist, so that we can write there
63  if ( ! outXmd.getDir().exists()) {
64  REPORT_ERROR(ERR_IO_NOTEXIST, "Directory " + outXmd.getDir() + " does not exist");
65  }
66  if ( ! outStk.getDir().exists()) {
67  REPORT_ERROR(ERR_IO_NOTEXIST, "Directory " + outStk.getDir() + " does not exist");
68  }
69  // target files must not exist, otherwise we will append to them
70  if (outXmd.exists()) {
71  REPORT_ERROR(ERR_IO_NOTEXIST, "File " + outXmd + " already exists.");
72  }
73  if (outStk.exists()) {
74  REPORT_ERROR(ERR_IO_NOTEXIST, "File " + outStk + " already exists");
75  }
76  const size_t n = md.size();
77  if (first >= n) {
78  REPORT_ERROR(ERR_LOGIC_ERROR, "Zero-based index of the first item ("
79  + std::to_string(first) + ") is bigger than size of the metadata ("
80  + std::to_string(n) + ")");
81  }
82  if (first + count > n) {
83  REPORT_ERROR(ERR_LOGIC_ERROR, "Out of range ("
84  + std::to_string(first) + " + "
85  + std::to_string(count) + " >= "
86  + std::to_string(n) + ")");
87  }
88 }
