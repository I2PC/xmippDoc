220 template<typename T>
221 void solve(const Matrix2D<T>& A, const Matrix1D<T>& b, Matrix1D<T>& result)
222 {
223  if (MAT_XSIZE(A) == 0)
224  REPORT_ERROR(ERR_MATRIX_EMPTY, "Solve: Matrix is empty");
225 
226  if (MAT_XSIZE(A) != MAT_YSIZE(A))
227  REPORT_ERROR(ERR_MATRIX_SIZE, "Solve: Matrix is not squared");
228 
229  if (MAT_XSIZE(A) != b.size())
230  REPORT_ERROR(ERR_MATRIX_SIZE, "Solve: Different sizes of Matrix and Vector");
231 
232  if (b.isRow())
233  REPORT_ERROR(ERR_MATRIX_DIM, "Solve: Not correct vector shape");
234 
235  // Perform LU decomposition and then solve
236  Matrix1D< int > indx;
237  T d;
238  Matrix2D<T> LU;
239  ludcmp(A, LU, indx, d);
240  result = b;
241  lubksb(LU, indx, result);
242 }

247 template<typename T>
248 void solveBySVD(const Matrix2D< T >& A, const Matrix1D< T >& b,
249  Matrix1D< double >& result, double tolerance)
250 {
251  if (A.Xdim() == 0)
252  REPORT_ERROR(ERR_MATRIX_EMPTY, "Solve: Matrix is empty");
253 
254  if (A.Xdim() != A.Ydim())
255  REPORT_ERROR(ERR_MATRIX_SIZE, "Solve: Matrix is not squared");
256 
257  if (A.Xdim() != b.size())
258  REPORT_ERROR(ERR_MATRIX_SIZE, "Solve: Different sizes of Matrix and Vector");
259 
260  if (b.isRow())
261  REPORT_ERROR(ERR_MATRIX_DIM, "Solve: Not correct vector shape");
262 
263  // First perform de single value decomposition
264  // Xmipp interface that calls to svdcmp of numerical recipes
265  Matrix2D< double > u, v;
266  Matrix1D< double > w;
267  svdcmp(A, u, w, v);
268 
269  // Here is checked if eigenvalues of the svd decomposition are acceptable
270  // If a value is lower than tolerance, the it's zeroed, as this increases
271  // the precision of the routine.
272  for (int i = 0; i < w.size(); i++)
273  if (w(i) < tolerance)
274  w(i) = 0;
275 
276  // Set size of matrices
277  result.resize(b.size());
278 
279  // Xmipp interface that calls to svdksb of numerical recipes
280  Matrix1D< double > bd;
281  typeCast(b, bd);
282  svbksb(u, w, v, bd, result);
283 }
