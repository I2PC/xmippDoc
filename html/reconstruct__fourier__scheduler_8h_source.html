77  /*
78  * Debug tracking of the task state.
79  * This makes sure that the assertions on the task lifecycle hold.
80  *
81  * There are two assumed paths:
82  * 1. Regular
83  * SUBMITTED -> QUEUED (-> stealing happens here) -> POPPED -> EXECUTING -> DONE
84  * 2. Outside of the scheduler
85  * SUBMITTED -> QUEUED_OUTSIDE -> EXECUTING -> DONE
86  *
87  * Since the StarPU documentation does not really specify this, these methods assert that it holds.
88  */
89 
90  void stateOnPush() {
91  switchState(SUBMITTED, QUEUED);
92  }
93  void stateOnPushNotify() {
94  switchState(SUBMITTED, QUEUED_OUTSIDE);
95  }
96  void stateOnPop() {
97  switchState(QUEUED, POPPED);
98  }
99  void stateOnPreExec() {
100  switchState(POPPED, QUEUED_OUTSIDE, EXECUTING);
101  }
102  void stateOnPostExec() {
103  switchState(EXECUTING, DONE);
104  }
