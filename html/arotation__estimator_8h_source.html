69  void check() const {
70  if (0 == hw.size()) {
71  REPORT_ERROR(ERR_VALUE_INCORRECT, "HW contains zero (0) devices");
72  }
73  if ( ! refDims.isValid()) {
74  REPORT_ERROR(ERR_LOGIC_ERROR, "'Reference' dimensions are invalid (contain 0)");
75  }
76  if ( ! otherDims.isValid()) {
77  REPORT_ERROR(ERR_LOGIC_ERROR, "'Other' dimensions are invalid (contain 0)");
78  }
79  if ( ! refDims.equalExceptNPadded(otherDims)) {
80  REPORT_ERROR(ERR_LOGIC_ERROR, "Dimensions of the reference and other signals differ");
81  }
82  if (AlignType::None == type) {
83  REPORT_ERROR(ERR_LOGIC_ERROR, "'None' alignment type is set. This is invalid value");
84  }
85  if ((AlignType::OneToN == type)
86  && (1 != refDims.n())) {
87  REPORT_ERROR(ERR_LOGIC_ERROR, "More than one reference specified for alignment type 1:N");
88  }
89  if ((AlignType::MToN == type)
90  && (1 == refDims.n())) {
91  REPORT_ERROR(ERR_LOGIC_ERROR, "Single reference specified for alignment type M:N");
92  }
93  if (batch > otherDims.n()) {
94  REPORT_ERROR(ERR_LOGIC_ERROR, "Batch is bigger than number of signals");
95  }
96  if (0 == batch) {
97  REPORT_ERROR(ERR_LOGIC_ERROR, "Batch is zero (0)");
98  }
99  if (0 == maxRotDeg) {
100  REPORT_ERROR(ERR_VALUE_INCORRECT, "Max rotation is zero (0)");
101  }
102  if (0 == lastRing) {
103  REPORT_ERROR(ERR_VALUE_INCORRECT, "Last ring is zero (0)");
104  }
105  if (0 == firstRing) {
106  REPORT_ERROR(ERR_VALUE_INCORRECT, "First ring is zero (0)");
107  }
108  if (lastRing <= firstRing) {
109  REPORT_ERROR(ERR_VALUE_INCORRECT, "Last ring is bigger (or equal) than first ring");
110  }
111  if (lastRing >= refDims.x()) {
112  REPORT_ERROR(ERR_VALUE_INCORRECT, "Last ring is too big");
113  }
114  if (firstRing >= refDims.x()) {
115  REPORT_ERROR(ERR_VALUE_INCORRECT, "First ring is too big");
116  }
117  }
