530 void MDObject::toStream(std::ostream &os, bool withFormat, bool isSql, bool escape) const
531 {
532  if (label == MDL_UNDEFINED) //if undefine label, store as a literal string
533  os << data.stringValue;
534  else
535  switch (MDL::labelType(label))
536  {
537  case LABEL_BOOL: //bools are int in sqlite3
538  os << data.boolValue;
539  break;
540  case LABEL_INT:
541  INT2STREAM(data.intValue);
542  break;
543  case LABEL_SIZET:
544  INT2STREAM(data.longintValue);
545  break;
546  case LABEL_DOUBLE:
547  DOUBLE2STREAM(data.doubleValue);
548  break;
549  case LABEL_STRING:
550  {
551  char c = _SPACE;
552  if (escape)
553  {
554  if (isSql || data.stringValue->find_first_of(_DQUOT) != String::npos)
555  c = _QUOT;
556  else if (data.stringValue->find_first_of(_QUOT) != String::npos)
557  c = _DQUOT;
558  else if (data.stringValue->find_first_of(_SPACE) != String::npos)
559  c = _QUOT;
560  else if (data.stringValue->empty())
561  c = _QUOT;
562  }
563  if (c == _SPACE)
564  os << *(data.stringValue);
565  else
566  os << c << *(data.stringValue) << c;
567  }
568  break;
569  case LABEL_VECTOR_DOUBLE:
570  {
571  std::vector<double> &vectorDouble = *(data.vectorValue);
572  if (escape)
573  os << _QUOT << " ";
574  size_t size = vectorDouble.size();
575  for (size_t i = 0; i < size; i++)
576  {
577  double v = vectorDouble[i];
578  DOUBLE2STREAM(v);
579  os << " ";
580  }
581  if (escape)
582  os << _QUOT;
583  }
584  break;
585  case LABEL_VECTOR_SIZET:
586  {
587  std::vector<size_t> &vector = *(data.vectorValueLong);
588  if (escape)
589  os << _QUOT << " ";
590  size_t size = vector.size();
591  for (size_t i = 0; i < size; i++)
592  os << vector[i] << " ";
593  if (escape)
594  os << _QUOT;
595  }
596  break;
597  case LABEL_NOTYPE:
598  if (escape) os << _QUOT;
599  os << "No type";
600  if (escape) os << _QUOT;
601  break;
602  }//close switch
603 }//close function toStream

631 bool MDObject::fromStream(std::istream &is, bool fromString)
632 {
633  if (label == MDL_UNDEFINED) //if undefine label, store as a literal string
634  {
635  String s;
636  is >> s;
637  }
638  else
639  {
640  //NOTE: int, bool and long(size_t) are read as double for compatibility with old doc files
641  double d;
642  size_t value;
643  switch (type)
644  {
645  case LABEL_BOOL: //bools are int in sqlite3
646  is >> d;
647  data.boolValue = (bool) ((int)d);
648  break;
649  case LABEL_INT:
650  is >> d;
651  data.intValue = (int) d;
652  break;
653  case LABEL_SIZET:
654  is >> d;
655  data.longintValue = (size_t) d;
656  break;
657  case LABEL_DOUBLE:
658  is >> data.doubleValue;
659  break;
660  case LABEL_STRING:
661  {
662  data.stringValue->clear();
663  String s;
664  is >> s;
665  char chr = s[0];
666  if (chr == _QUOT || chr == _DQUOT)
667  {
668  s = s.substr(1, s.size() - 1); //remove first char '
669  while (s.find_last_of(chr) == String::npos)
670  {
671  data.stringValue->append(s + " ");
672  is >> s;
673  }
674  s = s.substr(0, s.size() - 1); //remove last char '
675  }
676  data.stringValue->append(s);
677  }
678  break;
679  case LABEL_VECTOR_DOUBLE:
680  if (!fromString)
681  is.ignore(256, _QUOT);
682  //if (data.vectorValue == NULL)
683  // data.vectorValue = new std::vector<double>;
684  data.vectorValue->clear();
685  while (is >> d) //This will stop at ending "]"
686  data.vectorValue->push_back(d);
687  if (!fromString)
688  {
689  is.clear(); //this is for clear the fail state after found ']'
690  is.ignore(256, _QUOT); //ignore the ending ']'
691  }
692  break;
693  case LABEL_VECTOR_SIZET:
694  if (!fromString)
695  is.ignore(256, _QUOT);
696  //if (data.vectorValue == NULL)
697  // data.vectorValue = new std::vector<double>;
698  data.vectorValueLong->clear();
699  while (is >> value) //This will stop at ending "]"
700  data.vectorValueLong->push_back(value);
701  if (!fromString)
702  {
703  is.clear(); //this is for clear the fail state after found ']'
704  is.ignore(256, _QUOT); //ignore the ending ']'
705  }
706  break;
707  case LABEL_NOTYPE:
708  break;
709  }
710  }
711  return is.good();
712 }
