156  def alignParticlesStep(self):
157 
158  fhInputTranMat = self._getExtraPath('transformation-matrix.txt')
159  outParticlesFn = self._getExtraPath('outputParticles.xmd')
160  transMatFromFile = np.loadtxt(fhInputTranMat)
161  transformationMat = np.reshape(transMatFromFile,(4,4))
162  transform = Transform()
163  transform.setMatrix(transformationMat)
164 
165  resultMat = Transform()
166  outputParts = md.MetaData()
167  mdToAlign = md.MetaData(self.imgsInputFn)
168  for row in md.iterRows(mdToAlign):
169  inMat = rowToAlignment(row, ALIGN_PROJ)
170  partTransformMat = inMat.getMatrix()
171  partTransformMatrix = np.matrix(partTransformMat)
172  newTransformMatrix = np.matmul(transformationMat, partTransformMatrix)
173  resultMat.setMatrix(newTransformMatrix)
174  rowOut = md.Row()
175  rowOut.copyFromRow(row)
176  alignmentToRow(resultMat, rowOut, ALIGN_PROJ)
177  rowOut.addToMd(outputParts)
178  outputParts.write(outParticlesFn)
