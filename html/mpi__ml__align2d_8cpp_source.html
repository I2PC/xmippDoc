49 void MpiML2DBase<T>::sendDocfile(const MultidimArray<double> &docfiledata)
50 {
51 
52  // Write intermediate files
53  if (!node->isMaster())
54  {
55  // All slaves send docfile data to the master
56  int s_size = MULTIDIM_SIZE(docfiledata);
57  MPI_Send(&s_size, 1, MPI_INT, 0, TAG_DOCFILESIZE,
58  MPI_COMM_WORLD);
59  MPI_Send(MULTIDIM_ARRAY(docfiledata), s_size, MPI_DOUBLE,
60  0, TAG_DOCFILE, MPI_COMM_WORLD);
61  }
62  else
63  {
64  // Master fills docfile
65  // Master's own contribution
66  auto * ml2d = (ML2DBaseProgram*)program;
67  ml2d->addPartialDocfileData(docfiledata, ml2d->myFirstImg, ml2d->myLastImg);
68  int s_size;
69  size_t first_img, last_img;
70  MPI_Status status;
71 
72  for (size_t docCounter = 1; docCounter < node->size; ++docCounter)
73  {
74  // receive in order
75  MPI_Recv(&s_size, 1, MPI_INT, docCounter, TAG_DOCFILESIZE,
76  MPI_COMM_WORLD, &status);
77  MPI_Recv(MULTIDIM_ARRAY(docfiledata), s_size,
78  MPI_DOUBLE, docCounter, TAG_DOCFILE,
79  MPI_COMM_WORLD, &status);
80  divide_equally(ml2d->nr_images_global, node->size, docCounter, first_img, last_img);
81  ml2d->addPartialDocfileData(docfiledata, first_img, last_img);
82  }
83  }
84 }

297 void MpiProgMLF2D::expectation()
298 {
299  MultidimArray<double> Maux, Vaux;
300  double aux;
301  Maux.resize(dim, dim);
302  Maux.setXmippOrigin();
303 
304  ProgMLF2D::expectation();
305  MPI_Allreduce(&LL, &aux, 1, MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
306  LL = aux;
307  MPI_Allreduce(&sumcorr, &aux, 1, MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
308  sumcorr = aux;
309  MPI_Allreduce(&wsum_sigma_offset, &aux, 1, MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
310  wsum_sigma_offset = aux;
311  if (do_kstest)
312  {
313  Vaux.resize(sumhist);
314  MPI_Allreduce(MULTIDIM_ARRAY(sumhist), MULTIDIM_ARRAY(Vaux),
315  MULTIDIM_SIZE(sumhist), MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
316  FOR_ALL_DIRECT_ELEMENTS_IN_MULTIDIMARRAY(Vaux)
317  DIRECT_MULTIDIM_ELEM(sumhist, n) = DIRECT_MULTIDIM_ELEM(Vaux, n);
318  for (size_t ires = 0; ires < hdim; ires++)
319  {
320  Vaux.resize(sumhist);
321  MPI_Allreduce(MULTIDIM_ARRAY(resolhist[ires]), MULTIDIM_ARRAY(Vaux),
322  MULTIDIM_SIZE(resolhist[ires]), MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
323  FOR_ALL_DIRECT_ELEMENTS_IN_MULTIDIMARRAY(Vaux)
324  DIRECT_MULTIDIM_ELEM(resolhist[ires], n) = DIRECT_MULTIDIM_ELEM(Vaux, n);
325  }
326  }
327  for (int refno = 0;refno < model.n_ref; refno++)
328  {
329  MPI_Allreduce(MULTIDIM_ARRAY(wsum_Mref[refno]), MULTIDIM_ARRAY(Maux),
330  MULTIDIM_SIZE(wsum_Mref[refno]), MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
331  wsum_Mref[refno] = Maux;
332  if (do_ctf_correction)
333  {
334  MPI_Allreduce(MULTIDIM_ARRAY(wsum_ctfMref[refno]), MULTIDIM_ARRAY(Maux),
335  MULTIDIM_SIZE(wsum_ctfMref[refno]), MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
336  wsum_ctfMref[refno] = Maux;
337  }
338  MPI_Allreduce(&sumw[refno], &aux, 1, MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
339  sumw[refno] = aux;
340  MPI_Allreduce(&sumw2[refno], &aux, 1, MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
341  sumw2[refno] = aux;
342  MPI_Allreduce(&sumwsc2[refno], &aux, 1, MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
343  sumwsc2[refno] = aux;
344  MPI_Allreduce(&sumwsc[refno], &aux, 1, MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
345  sumwsc[refno] = aux;
346  MPI_Allreduce(&sumw_mirror[refno], &aux, 1, MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
347  sumw_mirror[refno] = aux;
348  }
349  for (size_t ifocus = 0;ifocus < nr_focus;ifocus++)
350  {
351  for (size_t ii = 0; ii < Mwsum_sigma2[ifocus].size(); ii++)
352  {
353  MPI_Allreduce(&Mwsum_sigma2[ifocus][ii], &aux, 1, MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
354  Mwsum_sigma2[ifocus][ii] = aux;
355  }
356  if (do_student)
357  {
358  MPI_Allreduce(&sumw_defocus[ifocus], &aux, 1, MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
359  sumw_defocus[ifocus] = aux;
360  }
361  }
362 }//end of expectation
