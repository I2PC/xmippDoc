36 class XmippProtDenoiseParticles(ProtProcessParticles):
37  """ Remove particles noise by filtering them.
38  This filtering process is based on a projection over a basis created
39  from some averages (extracted from classes). This filtering is not
40  intended for processing particles. The huge filtering they will be
41  passed through is known to remove part of the signal with the noise.
42  However this is a good method for clearly see which particle are we
43  going to process before it's done.
44  """
45  _label = 'denoise particles'
