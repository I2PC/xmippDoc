1191  double &rot2, double &tilt2, double &psi2,
1192  bool projdir_mode, bool check_mirrors,
1193  bool object_rotation)

1202

1217  else
1218  Euler_apply_transf(L, R, rot2, tilt2, psi2, rot2p, tilt2p, psi2p);
1219  }
1220 
1221  double ang_dist = Euler_distanceBetweenAngleSets_fast(E1,rot2p, tilt2p, psi2p,
1222  projdir_mode, E2);
1223

1225  {
1226  best_rot2 = rot2p;

1229  best_ang_dist = ang_dist;
1230  }
1231

1233  {
1234  Euler_mirrorY(rot2p, tilt2p, psi2p, rot2p, tilt2p, psi2p);
1235  double ang_dist_mirror = Euler_distanceBetweenAngleSets_fast(E1,
1236  rot2p, tilt2p, psi2p,projdir_mode, E2);
1237

1239  {
1240  best_rot2 = rot2p;
1241  best_tilt2 = tilt2p;
1242  best_psi2 = psi2p;
1243  best_ang_dist = ang_dist_mirror;
1244  }
1245 
1246  }
1247  }
