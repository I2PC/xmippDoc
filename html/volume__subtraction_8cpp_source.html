35 void ProgVolumeSubtraction::defineParams() {
36  // Usage
37  addUsageLine("This program modifies a volume as much as possible in order "
38  "to assimilate it to another one, "
39  "without loosing the important information in it ('adjustment "
40  "process'). Then, the subtraction of "
41  "the two volumes can be optionally calculated. Sharpening: "
42  "reference volume must be an atomic "
43  "structure previously converted into a density map of the "
44  "same specimen than in input volume 2.");
45  // Parameters
46  addParamsLine("--i1 <volume> : Reference volume");
47  addParamsLine("--i2 <volume> : Volume to modify");
48  addParamsLine("[-o <structure=\"\">]\t: Volume 2 modified or "
49  "volume difference");
50  addParamsLine("\t: If no name is given, "
51  "then output_volume.mrc");
52  addParamsLine("[--sub]\t: Perform the "
53  "subtraction of the volumes. Output will be the difference");
54  addParamsLine("[--sigma <s=3>]\t: Decay of the filter "
55  "(sigma) to smooth the mask transition");
56  addParamsLine(
57  "[--iter <n=5>]\t: Number of iterations for the adjustment process");
58  addParamsLine("[--mask1 <mask=\"\">] : Mask for volume 1");
59  addParamsLine("[--mask2 <mask=\"\">] : Mask for volume 2");
60  addParamsLine(
61  "[--maskSub <mask=\"\">]\t: Mask for subtraction region");
62  addParamsLine(
63  "[--cutFreq <f=0>]\t: Filter both volumes with a filter which "
64  "specified cutoff frequency (i.e. resolution inverse, <0.5)");
65  addParamsLine(
66  "[--lambda <l=1>]\t: Relaxation factor for Fourier Amplitude POCS, "
67  "i.e. 'how much modification of volume Fourier amplitudes', between 1 "
68  "(full modification, recommended) and 0 (no modification)");
69  addParamsLine("[--radavg]\t: Match the radially averaged Fourier "
70  "amplitudes when adjusting the amplitudes instead of taking "
71  "directly them from the reference volume");
72  addParamsLine(
73  "[--computeEnergy]\t: Compute the energy difference between each step "
74  "(energy difference gives information about the convergence of the "
75  "adjustment process, while it can slightly slow the performance)");
76  addParamsLine(
77  "[--saveV1 <structure=\"\"> ]\t: Save subtraction intermediate file "
78  "(vol1 filtered) just when option --sub is passed, if not passed the "
79  "input reference volume is not modified");

81  "[--saveV2 <structure=\"\"> ]\t: Save subtraction intermediate file "

138 void POCSFourierAmplitude(const MultidimArray<double> &V1FourierMag,
139  MultidimArray<std::complex<double>> &V2Fourier, double l) {
140  FOR_ALL_DIRECT_ELEMENTS_IN_MULTIDIMARRAY(V1FourierMag) {
141  double mod = std::abs(DIRECT_MULTIDIM_ELEM(V2Fourier, n));
142  if (mod > 1e-10) // Condition to avoid divide by zero, values smaller than
143  // this threshold are considered zero
144  DIRECT_MULTIDIM_ELEM(V2Fourier, n) *=
145  ((1 - l) + l * DIRECT_MULTIDIM_ELEM(V1FourierMag, n)) / mod;
146  }
147 }

149 void POCSFourierAmplitudeRadAvg(MultidimArray<std::complex<double>> &V,
150  double l, const MultidimArray<double> &rQ,
151  int V1size_x, int V1size_y, int V1size_z) {
152  int V1size2_x = V1size_x/2;
153  double V1sizei_x = 1.0/V1size_x;
154  int V1size2_y = V1size_y/2;
155  double V1sizei_y = 1.0/V1size_y;
156  int V1size2_z = V1size_z/2;
157  double V1sizei_z = 1.0/V1size_z;
158  double wx;
159  double wy;
160  double wz;
161  for (int k=0; k<ZSIZE(V); ++k)
162  {
163  FFT_IDX2DIGFREQ_FAST(k,V1size_z,V1size2_z,V1sizei_z,wz)
164  double wz2 = wz*wz;
165  for (int i=0; i<YSIZE(V); ++i)
166  {
167  FFT_IDX2DIGFREQ_FAST(i,V1size_y,V1size2_y,V1sizei_y,wy)
168  double wy2_wz2 = wy*wy + wz2;
169  for (int j=0; j<XSIZE(V); ++j)
170  {
171  FFT_IDX2DIGFREQ_FAST(j,V1size_x,V1size2_x,V1sizei_x,wx)
172  double w = sqrt(wx*wx + wy2_wz2);
173  auto iw = std::min((int)floor(w*V1size_x), (int)XSIZE(rQ) -1);
174  DIRECT_A3D_ELEM(V,k,i,j)*=(1-l)+l*DIRECT_MULTIDIM_ELEM(rQ,iw);
175  }
176  }
177  }
178 }

225 void radialAverage(const MultidimArray<double> &VolFourierMag,
226  const MultidimArray<double> &V, MultidimArray<double> &radial_mean) {
227  MultidimArray<double> radial_count;
228  int Vsize2_x = int(XSIZE(V))/2;
229  double Vsizei_x = 1.0/int(XSIZE(V));
230  int Vsize2_y = int(YSIZE(V))/2;
231  double Vsizei_y = 1.0/int(YSIZE(V));
232  int Vsize2_z = int(ZSIZE(V))/2;
233  double Vsizei_z = 1.0/int(ZSIZE(V));
234  double wx;
235  double wy;
236  double wz;
237  auto maxrad = int(floor(sqrt(Vsize2_x*Vsize2_x + Vsize2_y*Vsize2_y + Vsize2_z*Vsize2_z)));
238  radial_count.initZeros(maxrad);
239  radial_mean.initZeros(maxrad);
240  for (int k=0; k<Vsize2_z; ++k)
241  {
242  FFT_IDX2DIGFREQ_FAST(k,ZSIZE(V),Vsize2_z,Vsizei_z,wz)
243  double wz2 = wz*wz;
244  for (int i=0; i<Vsize2_y; ++i)
245  {
246  FFT_IDX2DIGFREQ_FAST(i,YSIZE(V),Vsize2_y,Vsizei_y,wy)
247  double wy2_wz2 = wy*wy + wz2;
248  for (int j=0; j<Vsize2_x; ++j)
249  {
250  FFT_IDX2DIGFREQ_FAST(j,XSIZE(V),Vsize2_x,Vsizei_x,wx)
251  double w = sqrt(wx*wx + wy2_wz2);
252  auto iw = (int)round(w*int(XSIZE(V)));
253  DIRECT_A1D_ELEM(radial_mean,iw)+=DIRECT_A3D_ELEM(VolFourierMag,k,i,j);
254  DIRECT_A1D_ELEM(radial_count,iw)+=1.0;
255  }
256  }
257  }
258  radial_mean/= radial_count;
259 }

261 MultidimArray<double> computeRadQuotient(const MultidimArray<double> &v1Mag,
262  const MultidimArray<double> &vMag, const MultidimArray<double> &V1,
263  const MultidimArray<double> &V) {
264  // Compute the quotient of the radial mean of the volumes to use it in POCS amplitude
265  MultidimArray<double> radial_meanV1;
266  radialAverage(v1Mag, V1, radial_meanV1);
267  MultidimArray<double> radial_meanV;
268  radialAverage(vMag, V, radial_meanV);
269  MultidimArray<double> radQuotient = radial_meanV1 / radial_meanV;
270  FOR_ALL_ELEMENTS_IN_ARRAY1D(radQuotient)
271  radQuotient(i) = std::min(radQuotient(i), 1.0);
272  return radQuotient;
273 }

282 Image<double> subtraction(Image<double> V1, Image<double> &V,
283  const MultidimArray<double> &mask, const FileName &fnVol1F,
284  const FileName &fnVol2A, FourierFilter &filter2, double cutFreq) {
285  Image<double> V1Filtered;
286  V1Filtered() = V1();
287  if (cutFreq != 0){
288  filter2.applyMaskSpace(V1Filtered());
289  }
290  if (!fnVol1F.isEmpty()) {
291  V1Filtered.write(fnVol1F);
292  }
293  if (!fnVol2A.isEmpty()) {
294  V.write(fnVol2A);
295  }
296  FOR_ALL_DIRECT_ELEMENTS_IN_MULTIDIMARRAY(V1())
297  DIRECT_MULTIDIM_ELEM(V1(), n) =
298  DIRECT_MULTIDIM_ELEM(V1(), n) * (1 - DIRECT_MULTIDIM_ELEM(mask, n)) +
299  (DIRECT_MULTIDIM_ELEM(V1Filtered(), n) -
300  std::min(DIRECT_MULTIDIM_ELEM(V(), n),
301  DIRECT_MULTIDIM_ELEM(V1Filtered(), n))) *
302  DIRECT_MULTIDIM_ELEM(mask, n);
303  return V1;
304 }

357 /* Core of the program: processing needed to adjust input
358  * volume V2 to reference volume V1. Several iteration of
359  * this processing should be run. */
360 void ProgVolumeSubtraction::runIteration(Image<double> &V,Image<double> &V1,const MultidimArray<double> &radQuotient,
361  const MultidimArray<double> &V1FourierMag,const MultidimArray<std::complex<double>> &V2FourierPhase,
362  const MultidimArray<double> &mask, FourierFilter &filter2) {
363  if (computeE)
364  std::cout << "---Iter " << n << std::endl;
365 
366  transformer2.FourierTransform(V(), V2Fourier, false);
367  if (radavg) {
368  auto V1size_x = (int)XSIZE(V1());
369  auto V1size_y = (int)YSIZE(V1());
370  auto V1size_z = (int)ZSIZE(V1());
371  POCSFourierAmplitudeRadAvg(V2Fourier, lambda, radQuotient, V1size_x, V1size_y, V1size_z);
372  }
373  else {
374  POCSFourierAmplitude(V1FourierMag, V2Fourier, lambda);
375  }
376  transformer2.inverseFourierTransform();
377  if (computeE) {
378  computeEnergy(Vdiff(), V());
379  Vdiff = V;
380  }
381  POCSMinMax(V(), v1min, v1max);
382  if (computeE) {
383  computeEnergy(Vdiff(), V());
384  Vdiff = V;
385  }
386  POCSmask(mask, V());
387  if (computeE) {
388  computeEnergy(Vdiff(), V());
389  Vdiff = V;
390  }
391  transformer2.FourierTransform();
392  POCSFourierPhase(V2FourierPhase, V2Fourier);
393  transformer2.inverseFourierTransform();
394  if (computeE) {
395  computeEnergy(Vdiff(), V());
396  Vdiff = V;
397  }
398  POCSnonnegative(V());
399  if (computeE) {
400  computeEnergy(Vdiff(), V());
401  Vdiff = V;
402  }
403  double std2 = V().computeStddev();
404  V() *= std1 / std2;
405  if (computeE) {
406  computeEnergy(Vdiff(), V());
407  Vdiff = V;
408  }
409 
410  if (cutFreq != 0) {
411  filter2.generateMask(V());
412  filter2.do_generate_3dmask = true;
413  filter2.applyMaskSpace(V());
414  if (computeE) {
415  computeEnergy(Vdiff(), V());
416  Vdiff = V;
417  }
418  }

421 void ProgVolumeSubtraction::run() {
422  show();
423  Image<double> V1;
424  V1.read(fnVol1);
425  auto mask = createMask(V1, fnMask1, fnMask2);
426  POCSmask(mask, V1());
427  POCSnonnegative(V1());
428  V1().computeDoubleMinMax(v1min, v1max);
429  Image<double> V;
430  V.read(fnVol2);
431  POCSmask(mask, V());
432  POCSnonnegative(V());
433  std1 = V1().computeStddev();
434  Vdiff = V;
435  auto V2FourierPhase = computePhase(V());
436  auto V1FourierMag = computeMagnitude(V1());
437  auto V2FourierMag = computeMagnitude(V());
438  auto radQuotient = computeRadQuotient(V1FourierMag, V2FourierMag, V1(), V());
439  FourierFilter filter2;
440  createFilter(filter2, cutFreq);
441  for (n = 0; n < iter; ++n) {
442  runIteration(V, V1, radQuotient, V1FourierMag, V2FourierPhase, mask, filter2);
443  }
444  if (performSubtraction) {
445  auto masksub = getSubtractionMask(fnMaskSub, mask);
446  V1.read(fnVol1);
447  V = subtraction(V1, V, masksub, fnVol1F, fnVol2A, filter2, cutFreq);
448  }
449  /* The output of this program is either a modified
450  * version of V (V') or the subtraction between
451  * V1 and V' if performSubtraction flag is activated' */
452  V.write(fnOutVol);
453 }
