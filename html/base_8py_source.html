202 def findRow(md, label, value):
203  """ Query the metadata for a row with label=value.
204  Params:
205  md: metadata to query.
206  label: label to check value
207  value: value for equal condition
208  Returns:
209  Row object of the row found.
210  None if no row is found with label=value
211  """
212  mdQuery = emlib.MetaData() # store result
213  mdQuery.importObjects(md, emlib.MDValueEQ(label, value))
214  n = mdQuery.size()
215 
216  if n == 0:
217  row = None
218  elif n == 1:
219  row = emlib.metadata.Row()
220  row.readFromMd(mdQuery, mdQuery.firstObject())
221  else:
222  raise Exception("findRow: more than one row found matching the query "
223  "%s = %s" % (emlib.label2Str(label), value))
224 
225  return row

376  def produceAlignedImagesStep(self, volumeIsCTFCorrected, fn, images):
377 
378  from numpy import array, dot
379  fnOut = 'classes_aligned@' + fn
380  MDin = emlib.MetaData(images)
381  MDout = emlib.MetaData()
382  n = 1
383  hasCTF = MDin.containsLabel(emlib.MDL_CTF_MODEL)
384  for i in MDin:
385  fnImg = MDin.getValue(emlib.MDL_IMAGE,i)
386  fnImgRef = MDin.getValue(emlib.MDL_IMAGE_REF,i)
387  maxCC = MDin.getValue(emlib.MDL_MAXCC,i)
388  rot = MDin.getValue(emlib.MDL_ANGLE_ROT,i)
389  tilt = MDin.getValue(emlib.MDL_ANGLE_TILT,i)
390  psi =-1.*MDin.getValue(emlib.MDL_ANGLE_PSI,i)
391  flip = MDin.getValue(emlib.MDL_FLIP,i)
392  if flip:
393  psi = -psi
394  eulerMatrix = emlib.Euler_angles2matrix(0., 0., psi)
395  x = MDin.getValue(emlib.MDL_SHIFT_X,i)
396  y = MDin.getValue(emlib.MDL_SHIFT_Y,i)
397  shift = array([x, y, 0])
398  shiftOut = dot(eulerMatrix, shift)
399  [x,y,z]= shiftOut
400  if flip:
401  x = -x
402  id = MDout.addObject()
403  MDout.setValue(emlib.MDL_IMAGE, fnImg, id)
404  MDout.setValue(emlib.MDL_IMAGE_REF, fnImgRef, id)
405  MDout.setValue(emlib.MDL_IMAGE1, "%05d@%s"%(n, self._getExtraPath("diff.stk")), id)
406  if hasCTF:
407  fnCTF = MDin.getValue(emlib.MDL_CTF_MODEL,i)
408  MDout.setValue(emlib.MDL_CTF_MODEL,fnCTF,id)
409  MDout.setValue(emlib.MDL_MAXCC, maxCC, id)
410  MDout.setValue(emlib.MDL_ANGLE_ROT, rot, id)
411  MDout.setValue(emlib.MDL_ANGLE_TILT, tilt, id)
412  MDout.setValue(emlib.MDL_ANGLE_PSI, psi, id)
413  MDout.setValue(emlib.MDL_SHIFT_X, x,id)
414  MDout.setValue(emlib.MDL_SHIFT_Y, y,id)
415  MDout.setValue(emlib.MDL_FLIP,flip,id)
416  MDout.setValue(emlib.MDL_ENABLED,1,id)
417  n+=1
418  MDout.write(fnOut,emlib.MD_APPEND)
419 
420  # Actually create the differences
421  img = emlib.Image()
422  imgRef = emlib.Image()
423  if hasCTF and volumeIsCTFCorrected:
424  Ts = MDin.getValue(emlib.MDL_SAMPLINGRATE, MDin.firstObject())
425 
426  for i in MDout:
427  img.readApplyGeo(MDout,i)
428  imgRef.read(MDout.getValue(emlib.MDL_IMAGE_REF,i))
429  if hasCTF and volumeIsCTFCorrected:
430  fnCTF = MDout.getValue(emlib.MDL_CTF_MODEL,i)
431  emlib.applyCTF(imgRef, fnCTF, Ts)
432  img.convert2DataType(emlib.DT_DOUBLE)
433  imgDiff = img-imgRef
434  imgDiff.write(MDout.getValue(emlib.MDL_IMAGE1,i))
