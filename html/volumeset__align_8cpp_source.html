112 // Compute fitness =========================================================
113 void ProgVolumeSetAlign::computeFitness(){
114  FileName fnRandom;
115  fnRandom.initUniqueName(nameTemplate,fnOutDir);
116  const char * randStr = fnRandom.c_str();
117 
118  FileName fnShiftsAngles = fnRandom + "_angles_shifts.txt";
119  const char * shifts_angles = fnShiftsAngles.c_str();
120 
121  String fnVolume1 = this->currentVolName;
122  String fnVolume2 = this->fnREF;
123 
124  if (this->tilt0!=-90 || this->tiltF!=90 ){
125  this->flipped = true;
126  runSystem("xmipp_transform_geometry",formatString("-i %s -o %s_currentvolume.vol --rotate_volume euler 0 90 0 -v 0",fnVolume1.c_str(),randStr));
127  fnVolume1 = fnRandom+"_currentvolume.vol";
128  }
129 
130  const char * Volume1 = fnVolume1.c_str();
131  const char * Volume2 = fnVolume2.c_str();
132 
133  int err;
134 
135  runSystem("xmipp_volume_align",formatString("--i1 %s --i2 %s --frm %f %d %d %d --store %s -v 0 ",
136  Volume1,Volume2,this->frm_freq, this->frm_shift, this->tilt0, this->tiltF, shifts_angles));
137  //The first 6 parameters are angles and shifts, and the 7th is the fitness value
138  fnAnglesAndShifts = fopen(shifts_angles, "r");
139  for (int i = 0; i < 6; i++){
140  err = fscanf(fnAnglesAndShifts, "%f,", &this->Matrix_Angles_Shifts[i]);
141  if (1!=err)
142  REPORT_ERROR(ERR_IO, "reading the angles and shifts was not successful");
143  }
144  err = fscanf(fnAnglesAndShifts, "%f,", &this->fitness);
145  if(1!=err)
146  REPORT_ERROR(ERR_IO, "reading the fitness value was not successful");
147 
148  fclose(fnAnglesAndShifts);
149 
150  runSystem("rm", formatString("-rf %s* &", randStr));
151 }

164 void ProgVolumeSetAlign::writeVolumeParameters(const FileName &fnImg) {
165  MetaData md;
166  size_t objId = md.addObject();
167  md.setValue(MDL_IMAGE, fnImg, objId);
168  md.setValue(MDL_ENABLED, 1, objId);
169  md.setValue(MDL_ANGLE_ROT, (double)this->Matrix_Angles_Shifts[0], objId);
170  md.setValue(MDL_ANGLE_TILT, (double)this->Matrix_Angles_Shifts[1], objId);
171  md.setValue(MDL_ANGLE_PSI, (double)this->Matrix_Angles_Shifts[2], objId);
172  md.setValue(MDL_SHIFT_X, (double)this->Matrix_Angles_Shifts[3], objId);
173  md.setValue(MDL_SHIFT_Y, (double)this->Matrix_Angles_Shifts[4], objId);
174  md.setValue(MDL_SHIFT_Z, (double)this->Matrix_Angles_Shifts[5], objId);
175  md.setValue(MDL_MAXCC, -(double)this->fitness, objId);
176  if (this->flipped){
177  md.setValue(MDL_ANGLE_Y, 90.0 , objId);
178  }
179  else{
180  md.setValue(MDL_ANGLE_Y, 0.0 , objId);
181  }
182  md.append(fnOutDir+"/AlignedSoFar.xmd");
183 }
