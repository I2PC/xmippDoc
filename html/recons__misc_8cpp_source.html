138 void sortPerpendicular(int numIMG, ReconsInfo *IMG_Inf,
139  MultidimArray<int> &ordered_list, int N)
140 {
141  int i, j;
142  MultidimArray<short> chosen(numIMG); // 1 if that image has been already
143  // chosen
144  double min_prod;
145  int min_prod_proj=0;
146  Matrix2D<double> v(numIMG, 3);
147  Matrix2D<double> euler;
148  MultidimArray<double> product(numIMG);
149 
150  // Initialization
151  ordered_list.resize(numIMG);
152  for (i = 0; i < numIMG; i++)
153  {
154  Matrix1D<double> z;
155  // Initially no image is chosen
156  A1D_ELEM(chosen, i) = 0;
157 
158  // Compute the Euler matrix for each image and keep only
159  // the third row of each one
160  Euler_angles2matrix(IMG_Inf[i].rot, IMG_Inf[i].tilt, 0., euler);
161  euler.getRow(2, z);
162  v.setRow(i, z);
163  }
164 
165  // Pick first projection as the first one to be presented
166  i = 0;
167  A1D_ELEM(chosen, i) = 1;
168  A1D_ELEM(ordered_list, 0) = i;
169 
170  // Choose the rest of projections
171  std::cout << "Sorting projections ...\n";
172  init_progress_bar(numIMG - 1);
173  Matrix1D<double> rowj, rowi_1, rowi_N_1;
174  for (i = 1; i < numIMG; i++)
175  {
176  // Compute the product of not already chosen vectors with the just
177  // chosen one, and select that which has minimum product
178  min_prod = MAXFLOAT;
179  v.getRow(A1D_ELEM(ordered_list, i - 1),rowi_1);
180  if (N != -1 && i > N)
181  v.getRow(A1D_ELEM(ordered_list, i - N - 1),rowi_N_1);
182  for (j = 0; j < numIMG; j++)
183  {
184  if (!A1D_ELEM(chosen, j))
185  {
186  v.getRow(j,rowj);
187  A1D_ELEM(product, j) += ABS(dotProduct(rowi_1,rowj));
188  if (N != -1 && i > N)
189  A1D_ELEM(product, j) -= ABS(dotProduct(rowi_N_1,rowj));
190  if (A1D_ELEM(product, j) < min_prod)
191  {
192  min_prod = A1D_ELEM(product, j);
193  min_prod_proj = j;
194  }
195  }
196  }
197 
198  // Store the chosen vector and mark it as chosen
199  A1D_ELEM(ordered_list, i) = min_prod_proj;
200  A1D_ELEM(chosen, min_prod_proj) = 1;
201 
202  // The progress bar is updated only every 10 images
203  if (i % 10 == 0)
204  progress_bar(i);
205  }
206 
207  // A final call to progress bar to finish a possible small piece
208  progress_bar(numIMG - 1);
209  std::cout << std::endl;
210 }
