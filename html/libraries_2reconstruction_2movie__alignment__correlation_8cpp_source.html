70 template<typename T>
71 void ProgMovieAlignmentCorrelation<T>::loadData(const MetaData& movie,
72  const Image<T>& dark, const Image<T>& igain) {
73  sizeFactor = this->getScaleFactor();
74  MultidimArray<T> filter;
75  FourierTransformer transformer;
76  bool firstImage = true;
77  int n = -1;
78  FileName fnFrame;
79  Image<T> croppedFrame, reducedFrame;
80 
81  if (this->verbose) {
82  std::cout << "Computing Fourier transform of frames ..." << std::endl;
83  init_progress_bar(movie.size());
84  }
85 
86  FOR_ALL_OBJECTS_IN_METADATA(movie)
87  {
88  ++n;
89  if (n >= this->nfirst && n <= this->nlast) {
90  this->loadFrame(movie, dark, igain, __iter.objId, croppedFrame);
91 
92  if (firstImage) {
93  firstImage = false;
94  newXdim = croppedFrame().xdim * sizeFactor;
95  newYdim = croppedFrame().ydim * sizeFactor;
96  filter = this->createLPF(this->getPixelResolution(sizeFactor), Dimensions(newXdim,
97  newYdim));
98  }
99 
100  // Reduce the size of the input frame
101  scaleToSizeFourier(1, newYdim, newXdim, croppedFrame(),
102  reducedFrame());
103 
104  // Now do the Fourier transform and filter
105  MultidimArray<std::complex<T> > *reducedFrameFourier =
106  new MultidimArray<std::complex<T> >;
107  transformer.FourierTransform(reducedFrame(), *reducedFrameFourier,
108  true);
109  for (size_t nn = 0; nn < filter.nzyxdim; ++nn) {
110  T wlpf = DIRECT_MULTIDIM_ELEM(filter, nn);
111  DIRECT_MULTIDIM_ELEM(*reducedFrameFourier,nn) *= wlpf;
112  }
113  frameFourier.push_back(reducedFrameFourier);
114  }
115  if (this->verbose)
116  progress_bar(n);
117  }
118  if (this->verbose)
119  progress_bar(movie.size());
120 }

122 template<typename T>
123 void ProgMovieAlignmentCorrelation<T>::computeShifts(size_t N,
124  const Matrix1D<T>& bX, const Matrix1D<T>& bY, const Matrix2D<T>& A) {
125  int idx = 0;
126  assert(frameFourier.size() > 0);
127  MultidimArray<T> Mcorr;
128  Mcorr.resizeNoCopy(newYdim, newXdim);
129  Mcorr.setXmippOrigin();
130  CorrelationAux aux;
131  for (size_t i = 0; i < N - 1; ++i) {
132  for (size_t j = i + 1; j < N; ++j) {
133  bestShift(*frameFourier[i], *frameFourier[j], Mcorr, bX(idx),
134  bY(idx), aux, NULL, this->maxShift * sizeFactor);
135  bX(idx) /= sizeFactor; // scale to expected size
136  bY(idx) /= sizeFactor;
137  if (this->verbose)
138  std::cerr << "Frame " << i + this->nfirst << " to Frame "
139  << j + this->nfirst << " -> ("
140  << bX(idx) << ","
141  << bY(idx) << ")\n";
142  for (int ij = i; ij < j; ij++)
143  A(idx, ij) = 1;
144 
145  idx++;
146  }
147  }
148 }

150 template<typename T>
151 void ProgMovieAlignmentCorrelation<T>::applyShiftsComputeAverage(
152  const MetaData& movie, const Image<T>& dark, const Image<T>& igain,
153  Image<T>& initialMic, size_t& Ninitial, Image<T>& averageMicrograph,
154  size_t& N, const AlignmentResult<T> &globAlignment) {
155  // Apply shifts and compute average
156  Image<T> frame, croppedFrame, reducedFrame, shiftedFrame;
157  Matrix1D<T> shift(2);
158  FileName fnFrame;
159  int frameIndex = -1;
160  Ninitial = N = 0;
161  const T binning = this->getOutputBinning();
162  FOR_ALL_OBJECTS_IN_METADATA(movie)
163  {
164  frameIndex++;
165  if ((frameIndex >= this->nfirstSum) && (frameIndex <= this->nlastSum)) {
166  // user might want to align frames 3..10, but sum only 4..6
167  // by deducting the first frame that was aligned, we get proper offset to the stored memory
168  int frameOffset = frameIndex - this->nfirst;
169  // get shifts
170  XX(shift) = -globAlignment.shifts.at(frameOffset).x; // we want to compensate
171  YY(shift) = -globAlignment.shifts.at(frameOffset).y; // the shift
172  bool isZeroShift = (XX(shift) == YY(shift)) // if shift is the same in both dimensions
173  && (XX(shift) == (T)0); // and it's zero
174 
175  // load frame
176  this->loadFrame(movie, dark, igain, __iter.objId, croppedFrame);
177  if (binning > 0) {
178  scaleToSizeFourier(1, floor(YSIZE(croppedFrame()) / binning),
179  floor(XSIZE(croppedFrame()) / binning),
180  croppedFrame(), reducedFrame());
181  croppedFrame() = reducedFrame();
182  }
183 
184  if ( ! this->fnInitialAvg.isEmpty()) {
185  if (frameIndex == this->nfirstSum)
186  initialMic() = croppedFrame();
187  else
188  initialMic() += croppedFrame();
189  Ninitial++;
190  }
191 
192  if (this->fnAligned != "" || this->fnAvg != "") {
193  // if there's no shift
194  if (isZeroShift) {
195  if (nullptr == shiftedFrame().data) {
196  shiftedFrame().resizeNoCopy(croppedFrame());
197  }
198  std::swap(shiftedFrame().data, croppedFrame().data);
199  } else {
200  if (this->outsideMode == OUTSIDE_WRAP)
201  translate(this->BsplineOrder, shiftedFrame(),
202  croppedFrame(), shift, WRAP);
203  else if (this->outsideMode == OUTSIDE_VALUE)
204  translate(this->BsplineOrder, shiftedFrame(),
205  croppedFrame(), shift, DONT_WRAP,
206  this->outsideValue);
207  else
208  translate(this->BsplineOrder, shiftedFrame(),
209  croppedFrame(), shift, DONT_WRAP,
210  croppedFrame().computeAvg());
211  }
212  if (this->fnAligned != "")
213  shiftedFrame.write(this->fnAligned, frameOffset + 1, true,
214  WRITE_REPLACE);
215  if (this->fnAvg != "") {
216  if (frameIndex == this->nfirstSum)
217  averageMicrograph() = shiftedFrame();
218  else
219  averageMicrograph() += shiftedFrame();
220  N++;
221  }
222  if (isZeroShift) {
223  std::swap(shiftedFrame().data, croppedFrame().data);
224  }
225  }
226  if (this->verbose > 1) {
227  std::cout << "Frame " << std::to_string(frameIndex) << " processed." << std::endl;
228  }
229  }
230  }
231 }
