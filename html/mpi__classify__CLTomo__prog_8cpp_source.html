724  std::cout << "Looking for node ..." << std::endl;
725 #endif
726

728  {
729  // Check if q is neighbour of the oldnode

734  {

753  proceed = true;

755  }

761  // Try this image

763  P[q]->fitBasic(Iaux, assignment);
764 
765  VEC_ELEM(corrList,q) = assignment.score;

770  prmCL3Dprog->classifyAllImages)

772  bestq = q;

774  bestAssignment = assignment;
775  }
776  }
777  }

782 #ifdef DEBUG

784 #endif
785

787  if (newnode != -1)
788  {
789  P[newnode]->updateProjection(I, bestAssignment);
790  }

809 
810  std::vector<int> oldAssignment, newAssignment;

812  int iter = 1;
813  bool goOn = true;
814  MetaData MDChanges;
815  Image<double> I;
816  int progressStep = XMIPP_MAX(1,Nimgs/60);
817  CL3DAssignment assignment;
818  while (goOn)
819  {
820  if (prmCL3Dprog->node->rank == 0)
821  {
822  std::cout << "Iteration " << iter << " ...\n";
823  std::cerr << "Iteration " << iter << " ...\n";

825  }
826  LOG(formatString("Iteration %d",iter));
827 
828  int K = XMIPP_MIN(prmCL3Dprog->Nneighbours+1,Q);
829  if (K == 0)
830  K = Q;
831  for (int q = 0; q < Q; q++)
832  P[q]->lookForNeighbours(P, K);
833 
834  int node;
835  double corrSum = 0;
836  SF->getColumnValues(MDL_REF, oldAssignment);
837  int *ptrOld = &(oldAssignment[0]);
838  for (size_t n = 0; n < Nimgs; ++n, ++ptrOld)
839  *ptrOld -= 1;
840  SF->fillConstant(MDL_REF, "-1");

851  LOG(formatString("Analyzing %s oldAssignment=%d newAssignment=%d",I.name().c_str(),oldAssignment[imgCounter], node));
852  SF->setValue(MDL_REF, node + 1, objId);
853  if (assignment.score>0)
854  corrSum += assignment.score;
855  if (prmCL3Dprog->node->rank == 0 && imgCounter % progressStep == 0)
856  progress_bar(imgCounter);

866  {

869  }

873  MPI_Allreduce(MPI_IN_PLACE, &corrSum, 1, MPI_DOUBLE, MPI_SUM,
874  MPI_COMM_WORLD);
875  shareAssignments(true, true);
876 
877  // Some report
878  size_t idMdChanges=0;
879  if (prmCL3Dprog->node->rank == 0)
880  {
881  progress_bar(Nimgs);

883  std::cout << "\nAverage correlation with input vectors="
884  << avgSimilarity << std::endl;

888  }
889

891  SF->getColumnValues(MDL_REF, newAssignment);
892  int *ptrNew = &(newAssignment[0]);
893  for (size_t n = 0; n < Nimgs; ++n, ++ptrNew)
894  *ptrNew -= 1;
895  int Nchanges = 0;
896  if (iter > 1)
897  {
898  int *ptrNew = &(newAssignment[0]);
899  ptrOld = &(oldAssignment[0]);
900  for (size_t n = 0; n < Nimgs; ++n, ++ptrNew, ++ptrOld)
901  if (*ptrNew != *ptrOld)
902  ++Nchanges;
903  }
904  if (prmCL3Dprog->node->rank == 0)
905  {
906  std::cout << "Number of assignment changes=" << Nchanges

909  MDChanges.write(
910  formatString("info@%s_classes_level_%02d.xmd",fnOut.c_str(), level));
911  }
912 
913  // Check if there are empty nodes
914  if (Q > 1)
915  {
916  bool smallNodes;

918  {
919  smallNodes = false;
920  int largestNode = -1, sizeLargestNode = -1, smallNode = -1;

923  {

925  {
926  smallNode = q;
927  sizeSmallestNode = P[q]->currentListImg.size();
928  }

930  {

933  }
934  }

936  REPORT_ERROR(ERR_UNCLASSIFIED,"All classes are of size 0: normally this happens when images are too noisy");
937  if (largestNode == -1 || smallNode == -1)
938  break;
939  if (sizeSmallestNode < prmCL3Dprog->PminSize * Nimgs / Q * 0.01 && sizeSmallestNode<0.25*sizeLargestNode)
940  {
941  if (prmCL3Dprog->node->rank == 0 && prmCL3Dprog->verbose)
942  std::cout << "Splitting node " << largestNode << " (" << sizeLargestNode
943  << ") by overwriting " << smallNode << " (" << sizeSmallestNode << " )" << std::endl;
944  smallNodes = true;

947  std::vector<CL3DAssignment> &currentListImgSmall =
948  P[smallNode]->currentListImg;
949  size_t iimax = currentListImgSmall.size();
950  for (size_t ii = 0; ii < iimax; ii++)
951  SF->setValue(MDL_REF, -1, currentListImgSmall[ii].objId);
952  delete P[smallNode];
953 
954  // Clear the old assignment of the images in the large node
955  std::vector<CL3DAssignment> &currentListImgLargest =
956  P[largestNode]->currentListImg;
957  iimax = currentListImgLargest.size();
958  for (size_t ii = 0; ii < iimax; ii++)
959  SF->setValue(MDL_REF, -1,
960  currentListImgLargest[ii].objId);
961

963  CL3DClass *node1 = new CL3DClass();
964  CL3DClass *node2 = new CL3DClass();
965  std::vector<size_t> splitAssignment;
966  splitNode(P[largestNode], node1, node2, splitAssignment);
967  delete P[largestNode];
968 
969  // Keep the results of the split
970  P[largestNode] = node1;
971  P[smallNode] = node2;

973  for (size_t ii = 0; ii < iimax; ii += 2)
974  {
975  if (splitAssignment[ii + 1] == 1)
976  SF->setValue(MDL_REF, largestNode + 1,
977  splitAssignment[ii]);
978  else
979  SF->setValue(MDL_REF, smallNode + 1,
980  splitAssignment[ii]);
981  }
982  }
983  }
984  while (smallNodes);
985  }
986

988  write(fnOut,level);
989 
990  if ((iter > 1 && Nchanges < 0.005 * Nimgs && Q > 1) || iter >= prmCL3Dprog->Niter)
991  goOn = false;
992  iter++;
993  }
994 
995  std::sort(P.begin(), P.end(), SDescendingClusterSort());
996 }

1032  bool success = true;
1033  do
1034  {
1035  // Sort the currentListImg to make sure that all nodes have the same order
1036  std::sort(node->currentListImg.begin(),node->currentListImg.end(),CL3DAssignmentComparator);
