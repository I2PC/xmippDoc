291  size_t maxPow5 = std::ceil(std::log(max) * log_5);
292  size_t maxPow7 = std::ceil(std::log(max) * log_7);
293 
294  for (size_t a = 1; a <= maxPow2; a++) { // we want at least one multiple of two
295  for (size_t b = 0; b <= maxPow3; b++) {
296  for (size_t c = 0; c <= maxPow5; c++) {
297  for (size_t d = 0; d <= maxPow7; d++) {
298  size_t value = std::pow(2, a) * std::pow(3, b)
299  * std::pow(5, c) * std::pow(7, d);
300  bool incCond = !crop && ((value >= num) && (value <= max));
301  bool decrCond = crop && (value <= num);
302  if (incCond || decrCond) {
303  Polynom p;
304  p.value = value;
305  p.exponent2 = a;
306  p.exponent3 = b;
307  p.exponent5 = c;
308  p.exponent7 = d;
309  p.invocations = getInvocations(p, isFloat);
310  p.noOfPrimes = getNoOfPrimes(p);
311  result->push_back(p);
312  }
313  }
314  }

316  }

321  *SizeOptimizer::filterOptimal(std::vector<SizeOptimizer::Polynom> *input,
322  bool crop) {
323

325  new std::set<Polynom, valueComparator>(!crop);
326 
327  size_t size = input->size();
328  if (0 == size) {
329  result->insert(UNIT);
330  return result;
331  }
332 
333  // add the most near polynom
334  Polynom &minInv = input->at(0);
335  Polynom &closest = minInv;
336  for (size_t i = 1; i < size; i++) {
337  Polynom &tmp = input->at(i);
338  if (tmp.invocations < minInv.invocations) {
339  // update min kernel invocations needed
340  minInv = tmp;
341  }
342  if (closest.value > tmp.value) {
343  closest = tmp;
344  }
345  }
346  result->insert(closest);
347 
348  // add all polynoms which have a minimal number of kernel invocations
349  for (size_t i = 0; i < size; i++) {
350  Polynom &tmp = input->at(i);
351  if ((tmp.invocations <= (minInv.invocations + 2)) &&
352  (tmp.noOfPrimes <= 4)) {
353  result->insert(tmp);
354  }
355  }
