75 void ProgCtfCorrectIdr::processImage(const FileName &fnImg, const FileName &fnImgOut, const MDRow &rowIn, MDRow &rowOut)
76 {
77  FileName fn_ctf;
78  rowIn.getValue(MDL_CTF_MODEL,fn_ctf);
79 
80  // Read current input image
81  Ireal.readApplyGeo(fnImg, rowIn);
82  int Ydim = YSIZE(Ireal());
83  int Xdim = XSIZE(Ireal());
84 
85  // Project the volume in the same direction
86  projectVolume(V(), Itheo, Ydim, Xdim, Ireal.rot(), Ireal.tilt(), Ireal.psi());
87 
88  // Copy to theo_CTF and resize
89  Itheo_CTF = Itheo();
90  Itheo_CTF.setXmippOrigin();
91  Itheo_CTF.selfWindow(FIRST_XMIPP_INDEX(2*Ydim), FIRST_XMIPP_INDEX(2*Xdim),
92  LAST_XMIPP_INDEX(2*Ydim), LAST_XMIPP_INDEX(2*Xdim));
93 
94  // Read CTF file
95  if (last_fn_ctf!=fn_ctf)
96  {
97  ctf.FilterBand = CTF;
98  ctf.ctf.read(fn_ctf);
99  ctf.ctf.enable_CTFnoise = false;
100  ctf.ctf.produceSideInfo();
101  ctf.generateMask(Itheo_CTF);
102  ctf.correctPhase();
103  last_fn_ctf=fn_ctf;
104  }
105 
106  // Apply CTF
107  ctf.applyMaskSpace(Itheo_CTF);
108  Itheo_CTF.selfWindow(FIRST_XMIPP_INDEX(Ydim), FIRST_XMIPP_INDEX(Xdim),
109  LAST_XMIPP_INDEX(Ydim), LAST_XMIPP_INDEX(Xdim));
110 
111 #ifdef DEBUG
112 
113  Itheo.write("PPPtheo.xmp");
114  Ireal.write("PPPreal.xmp");
115 #endif
116 
117  // Apply IDR process
118  MultidimArray<double> &mItheo=Itheo();
119  const MultidimArray<double> &mIreal=Ireal();
120  FOR_ALL_DIRECT_ELEMENTS_IN_MULTIDIMARRAY(mItheo)
121  DIRECT_MULTIDIM_ELEM(mItheo, n) = mu * DIRECT_MULTIDIM_ELEM(mIreal, n) +
122  (DIRECT_MULTIDIM_ELEM(mItheo, n) -
123  mu * DIRECT_MULTIDIM_ELEM(Itheo_CTF, n));
124 
125  // Save output image
126  Itheo.write(fnImgOut);
127 
128 #ifdef DEBUG
129 
130  Image<double> save;
131  save()=Itheo_CTF;
132  save.write("PPPtheo_CTF.xmp");
133  save() = Itheo() - mu * Itheo_CTF;
134  save.write("PPPdiff.xmp");
135  Itheo.write("PPPidr.xmp");
136  std::cout << "Press any key to continue\n";
137  char c;
138  std::cin >> c;
139 #endif
140 }
