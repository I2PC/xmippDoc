89 double ProgVolDeformSph::distance(double *pclnm)
90 {
91  if (applyTransformation)
92  {
93  VO().initZeros(VR());
94  VO().setXmippOrigin();
95  }
96  int l,n,m;
97  size_t idxY0=VEC_XSIZE(clnm)/4;
98  size_t idxZ0=2*idxY0;
99  size_t idxR=3*idxY0;
100 // double Ncount=0.0;
101  double totalVal=0.0;
102  double diff2=0.0;
103  const MultidimArray<double> &mVR=VR();
104  const MultidimArray<double> &mVI=VI();
105  FOR_ALL_ELEMENTS_IN_MATRIX1D(clnm)
106  VEC_ELEM(clnm,i)=pclnm[i+1];
107 #ifdef DEBUG
108  std::cout << "Starting to evaluate\n" << clnm << std::endl;
109 #endif
110  double modg=0.0;
111  double voxelR, absVoxelR, voxelI, diff;
112  for (int k=STARTINGZ(mVR); k<=FINISHINGZ(mVR); k++)
113  {
114  for (int i=STARTINGY(mVR); i<=FINISHINGY(mVR); i++)
115  {
116  for (int j=STARTINGX(mVR); j<=FINISHINGX(mVR); j++)
117  {
118  double gx=0.0, gy=0.0, gz=0.0;
119  for (size_t idx=0; idx<idxY0; idx++)
120  {
121  double Rmax=VEC_ELEM(clnm,idx+idxR);
122  double Rmax2=Rmax*Rmax;
123  double iRmax=1.0/Rmax;
124  double k2=k*k;
125  double kr=k*iRmax;
126  double k2i2=k2+i*i;
127  double ir=i*iRmax;
128  double r2=k2i2+j*j;
129  double jr=j*iRmax;
130  double rr=std::sqrt(r2)*iRmax;
131  double zsph=0.0;
132  if (r2<Rmax2)
133  {
134  spherical_index2lnm(idx,l,n,m);
135  zsph=ZernikeSphericalHarmonics(l,n,m,jr,ir,kr,rr);
136  }
137 
138 #ifdef NEVERDEFINED
139  if (ir!=0&jr!=0&rr!=0)
140  {
141  x = zsph*(ir/std::sqrt(ir*ir+jr*jr))*(kr/rr);
142  y = zsph*(ir/rr);
143  z = zsph*(jr/std::sqrt(ir*ir+jr*jr));
144  gx += VEC_ELEM(clnm,idx) *x;
145  gy += VEC_ELEM(clnm,idx+idxY0)*y;
146  gz += VEC_ELEM(clnm,idx+idxZ0)*z;
147  }
148  else
149  {
150  gx += VEC_ELEM(clnm,idx) *zsph;
151  gy += VEC_ELEM(clnm,idx+idxY0)*zsph;
152  gz += VEC_ELEM(clnm,idx+idxZ0)*zsph;
153  }
154 #endif
155  if (rr>0 || l==0)
156  {
157  gx += VEC_ELEM(clnm,idx) *(zsph);
158  gy += VEC_ELEM(clnm,idx+idxY0) *(zsph);
159  gz += VEC_ELEM(clnm,idx+idxZ0) *(zsph);
160  }
161  }
162 
163  for (int idv=0; idv<volumesR.size(); idv++)
164  {
165  voxelR=A3D_ELEM(volumesR[idv](),k,i,j);
166  absVoxelR=fabs(voxelR);
167  voxelI=volumesI[idv]().interpolatedElement3D(j+gx,i+gy,k+gz);
168  if (applyTransformation && idv == 0)
169  VO(k,i,j)=voxelI;
170  diff=voxelR-voxelI;
171  diff2+=absVoxelR*diff*diff;
172  modg+=absVoxelR*(gx*gx+gy*gy+gz*gz);
173  // Ncount++;
174  totalVal += absVoxelR;
175  }
176 
177  if (saveDeformation)
178  {
179  Gx(k,i,j)=gx;
180  Gy(k,i,j)=gy;
181  Gz(k,i,j)=gz;
182  }
183  }
184  }
185  }
186 
187  deformation=std::sqrt(modg/(totalVal));
188 
189 #ifdef DEBUG
190  //save.write("PPPIdeformed.vol");
191  //save()-=VR();
192  //save.write("PPPdiff.vol");
193  //save()=VR();
194  //save.write("PPPR.vol");
195  std::cout << "Error=" << deformation << " " << std::sqrt(diff2/totalVal) << std::endl;
196  //std::cout << "Press any key\n";
197  //char c; std::cin >> c;
198 #endif
199  if (applyTransformation)
200  VO.write(fnVolOut);
201  return std::sqrt(diff2/totalVal);
202 }
203 #undef DEBUG

212 void ProgVolDeformSph::run() {
213  Matrix1D<int> nh;
214  nh.resize(depth+2);
215  nh.initConstant(0);
216  nh(1)=1;
217 
218  saveDeformation=false;
219  Numsph(nh);
220 
221  VI.read(fnVolI);
222  VR.read(fnVolR);
223  if (Rmax<0)
224  Rmax=XSIZE(VI())/2;
225 
226  VI().setXmippOrigin();
227  VR().setXmippOrigin();
228 
229  // Filter input and reference volumes according to the values of sigma
230  FourierFilter filter;
231  filter.FilterShape = REALGAUSSIAN;
232  filter.FilterBand = LOWPASS;
233  filter.generateMask(VI());
234 
235  // We need also to normalized the filtered volumes to compare them appropiately
236  MultidimArray<int> bg_mask;
237  Image<double> auxI = VI;
238  Image<double> auxR = VR;
239  bg_mask.resizeNoCopy(VI().zdim, VI().ydim, VI().xdim);
240  bg_mask.setXmippOrigin();
241  BinaryCircularMask(bg_mask, Rmax, OUTSIDE_MASK);
242 
243  normalize_Robust(auxI(), bg_mask, true);
244  normalize_Robust(auxR(), bg_mask, true);
245 
246  volumesI.push_back(auxI());
247  volumesR.push_back(auxR());
248 
249  for (int ids=0; ids<sigma.size(); ids++)
250  {
251  Image<double> auxI = VI;
252  Image<double> auxR = VR;
253  filter.w1 = sigma[ids];
254 
255  // Filer input vol
256  filter.do_generate_3dmask = true;
257  filter.applyMaskSpace(auxI());
258  normalize_Robust(auxI(), bg_mask, true);
259  volumesI.push_back(auxI);
260 
261  // Filter ref vol
262  filter.applyMaskSpace(auxR());
263  normalize_Robust(auxR(), bg_mask, true);
264  volumesR.push_back(auxR);
265  }
266 
267  Matrix1D<double> steps, x, prevsteps;
268  for (int h=0;h<VEC_XSIZE(nh)-1;h++)
269  {
270  L = nh(h+1);
271  prevL = nh(h);
272  prevsteps=steps;
273  steps.clear();
274  std::cout<<std::endl;
275  std::cout<<"-------------------------- Spherical harmonic depth: "<<h<<" --------------------------"<<std::endl;
276  steps.initConstant(4*L,1);
277  if (h==0)
278  {
279  clnm.initZeros(VEC_XSIZE(steps));
280  x.initZeros(VEC_XSIZE(steps));
281 
282  }
283  else
284  {
285  x.resize(VEC_XSIZE(steps),false);
286  copyvectors(clnm,x);
287  clnm=x;
288  }
289 
290  for(int d=VEC_XSIZE(x)-L+prevL;d<VEC_XSIZE(x);d++)
291  {
292  x(d)=Rmax;
293  clnm(d)=Rmax;
294  }
295 #ifdef NEVERDEFINED
296  for (int d=0;d<VEC_XSIZE(x);d++)
297  {
298  if (x(d)==0)
299  {
300  steps(d) = 1;
301  }
302  else if (d>=VEC_XSIZE(steps)+nh(h)-nh(h+1)&d<VEC_XSIZE(steps)+nh(h+1))
303  {
304  steps(d) = 1;
305  }
306  else
307  {
308  steps(d) = 0;
309  }
310  //std::cout<<steps(d)<<" ";
311  }
312  //std::cout<<std::endl;
313 #endif
314  if (h!=0)
315  {
316  minimizepos(steps,prevsteps);
317  }
318  else
319  {
320  steps(VEC_XSIZE(steps)-1)=0;
321  }
322  int iter;
323  double fitness;
324  powellOptimizer(x, 1, VEC_XSIZE(steps), &volDeformSphGoal, this,
325  0.01, fitness, iter, steps, true);
326 
327  std::cout<<std::endl;
328  std::cout << "Deformation " << deformation << std::endl;
329  std::ofstream deformFile;
330  deformFile.open ("./deformation.txt");
331  deformFile << deformation;
332  deformFile.close();
333 
334  }
335  applyTransformation=true;
336  if (analyzeStrain)
337  {
338  saveDeformation=true;
339  Gx().initZeros(VR());
340  Gy().initZeros(VR());
341  Gz().initZeros(VR());
342  Gx().setXmippOrigin();
343  Gy().setXmippOrigin();
344  Gz().setXmippOrigin();
345  }
346 
347  distance(x.adaptForNumericalRecipes()); // To save the output volume
348 
349  if (analyzeStrain)
350  computeStrain();
351 }

408 void ProgVolDeformSph::computeStrain()
409 {
410  Image<double> LS, LR;
411  LS().initZeros(Gx());
412  LR().initZeros(Gx());
413 
414  // Gaussian filter of the derivatives
415  FourierFilter f;
416  f.FilterBand=LOWPASS;
417  f.FilterShape=REALGAUSSIAN;
418  f.w1=2;
419  f.applyMaskSpace(Gx());
420  f.applyMaskSpace(Gy());
421  f.applyMaskSpace(Gz());
422 
423  Gx.write("PPPGx.vol");
424  Gy.write("PPPGy.vol");
425  Gz.write("PPPGz.vol");
426 
427  MultidimArray<double> &mLS=LS();
428  MultidimArray<double> &mLR=LR();
429  MultidimArray<double> &mGx=Gx();
430  MultidimArray<double> &mGy=Gy();
431  MultidimArray<double> &mGz=Gz();
432  Matrix2D<double> U(3,3), D(3,3), H(3,3);
433  std::vector< std::complex<double> > eigs;
434  H.initZeros();
435  for (int k=STARTINGZ(mLS)+2; k<=FINISHINGZ(mLS)-2; ++k)
436  {
437  int km1=k-1;
438  int kp1=k+1;
439  int km2=k-2;
440  int kp2=k+2;
441  for (int i=STARTINGY(mLS)+2; i<=FINISHINGY(mLS)-2; ++i)
442  {
443  int im1=i-1;
444  int ip1=i+1;
445  int im2=i-2;
446  int ip2=i+2;
447  for (int j=STARTINGX(mLS)+2; j<=FINISHINGX(mLS)-2; ++j)
448  {
449  int jm1=j-1;
450  int jp1=j+1;
451  int jm2=j-2;
452  int jp2=j+2;
453  MAT_ELEM(U,0,0)=Dx(mGx); MAT_ELEM(U,0,1)=Dy(mGx); MAT_ELEM(U,0,2)=Dz(mGx);
454  MAT_ELEM(U,1,0)=Dx(mGy); MAT_ELEM(U,1,1)=Dy(mGy); MAT_ELEM(U,1,2)=Dz(mGy);
455  MAT_ELEM(U,2,0)=Dx(mGz); MAT_ELEM(U,2,1)=Dy(mGz); MAT_ELEM(U,2,2)=Dz(mGz);
456 
457  MAT_ELEM(D,0,0) = MAT_ELEM(U,0,0);
458  MAT_ELEM(D,0,1) = MAT_ELEM(D,1,0) = 0.5*(MAT_ELEM(U,0,1)+MAT_ELEM(U,1,0));
459  MAT_ELEM(D,0,2) = MAT_ELEM(D,2,0) = 0.5*(MAT_ELEM(U,0,2)+MAT_ELEM(U,2,0));
460  MAT_ELEM(D,1,1) = MAT_ELEM(U,1,1);
461  MAT_ELEM(D,1,2) = MAT_ELEM(D,2,1) = 0.5*(MAT_ELEM(U,1,2)+MAT_ELEM(U,2,1));
462  MAT_ELEM(D,2,2) = MAT_ELEM(U,2,2);
463 
464  MAT_ELEM(H,0,1) = 0.5*(MAT_ELEM(U,0,1)-MAT_ELEM(U,1,0));
465  MAT_ELEM(H,0,2) = 0.5*(MAT_ELEM(U,0,2)-MAT_ELEM(U,2,0));
466  MAT_ELEM(H,1,2) = 0.5*(MAT_ELEM(U,1,2)-MAT_ELEM(U,2,1));
467  MAT_ELEM(H,1,0) = -MAT_ELEM(H,0,1);
468  MAT_ELEM(H,2,0) = -MAT_ELEM(H,0,2);
469  MAT_ELEM(H,2,1) = -MAT_ELEM(H,1,2);
470 
471  A3D_ELEM(mLS,k,i,j)=fabs(D.det());
472  allEigs(H,eigs);
473  for (size_t n=0; n < eigs.size(); n++)
474  {
475  double imagabs=fabs(eigs[n].imag());
476  if (imagabs>1e-6)
477  {
478  A3D_ELEM(mLR,k,i,j)=imagabs*180/PI;
479  break;
480  }
481  }
482  }
483  }
484  LS.write(fnVolOut.withoutExtension()+"_strain.mrc");
485  LR.write(fnVolOut.withoutExtension()+"_rotation.mrc");
486  }
