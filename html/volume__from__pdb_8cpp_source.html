239 /* Compute protein geometry ------------------------------------------------ */
240 void ProgPdbConverter::computeProteinGeometry()
241 {
242  Matrix1D<double> limit0(3), limitF(3);
243  computePDBgeometry(fn_pdb, centerOfMass, limit0, limitF, intensityColumn);
244  if (doCenter)
245  {
246  limit0-=centerOfMass;
247  limitF-=centerOfMass;
248  }
249  limit.resize(3);
250  XX(limit) = XMIPP_MAX(ABS(XX(limit0)), ABS(XX(limitF)));
251  YY(limit) = XMIPP_MAX(ABS(YY(limit0)), ABS(YY(limitF)));
252  ZZ(limit) = XMIPP_MAX(ABS(ZZ(limit0)), ABS(ZZ(limitF)));
253 
254  // Update output size if necessary
255  if (output_dim == -1)
256  {
257  int max_dim = XMIPP_MAX(CEIL(ZZ(limit) * 2 / Ts) + 5, CEIL(YY(limit) * 2 / Ts) + 5);
258  max_dim = XMIPP_MAX(max_dim, CEIL(XX(limit) * 2 / Ts) + 5);
259  if (useBlobs)
260  output_dim = (int)NEXT_POWER_OF_2(max_dim);
261  else
262  output_dim = max_dim+10;
263  }

266 /* Create protein at a high sampling rate ---------------------------------- */
267 void ProgPdbConverter::createProteinAtHighSamplingRate()
268 {
269  // Create an empty volume to hold the protein
270  int finalDim;
271  if (highTs!=Ts)
272  finalDim=(int)NEXT_POWER_OF_2(output_dim / (highTs/Ts));
273  else
274  finalDim=output_dim;
275  Vhigh().initZeros(finalDim,finalDim,finalDim);
276  Vhigh().setXmippOrigin();
277  if (verbose)
278  std::cout << "The highly sampled volume is of size " << XSIZE(Vhigh())
279  << std::endl;
280 
281  // Fill the volume with the different atoms
282  std::ifstream fh_pdb;
283  fh_pdb.open(fn_pdb.c_str());
284  if (!fh_pdb)
285  REPORT_ERROR(ERR_IO_NOTEXIST, fn_pdb);
286 
287  // Process all lines of the file
288  int col=1;
289  if (intensityColumn=="Bfactor")
290  col=2;
291  while (!fh_pdb.eof())
292  {
293  // Read an ATOM line
294  std::string line;
295  getline(fh_pdb, line);
296  if (line == "")
297  continue;
298  std::string kind = line.substr(0,4);
299  if (kind != "ATOM" && kind !="HETA")
300  continue;
301 
302  // Extract atom type and position
303  // Typical line:
304  // ATOM 909 CA ALA A 161 58.775 31.984 111.803 1.00 34.78
305  std::string atom_type = line.substr(13,2);
306  double x = textToFloat(line.substr(30,8));
307  double y = textToFloat(line.substr(38,8));
308  double z = textToFloat(line.substr(46,8));
309 
310  // Correct position
311  Matrix1D<double> r(3);
312  VECTOR_R3(r, x, y, z);
313  if (doCenter)
314  r -= centerOfMass;
315  r /= highTs;
316 
317  // Characterize atom
318  double weight, radius;
319  if (!useFixedGaussian)
320  {
321  if (noHet && kind=="HETA")
322  continue;
323  atomBlobDescription(atom_type, weight, radius);
324  }
325  else
326  {
327  radius=4.5*sigmaGaussian;
328  if (col==1)
329  weight=textToFloat(line.substr(54,6));
330  else
331  weight=textToFloat(line.substr(60,6));
332  }
333  blob.radius = radius;
334  if (usePoorGaussian)
335  radius=XMIPP_MAX(radius/Ts,4.5);
336  double GaussianSigma2=(radius/(3*sqrt(2.0)));
337  if (useFixedGaussian)
338  GaussianSigma2=sigmaGaussian;
339  GaussianSigma2*=GaussianSigma2;
340  double GaussianNormalization = 1.0/pow(2*PI*GaussianSigma2,1.5);
341 
342  // Find the part of the volume that must be updated
343  int k0 = XMIPP_MAX(FLOOR(ZZ(r) - radius), STARTINGZ(Vhigh()));
344  int kF = XMIPP_MIN(CEIL(ZZ(r) + radius), FINISHINGZ(Vhigh()));
345  int i0 = XMIPP_MAX(FLOOR(YY(r) - radius), STARTINGY(Vhigh()));
346  int iF = XMIPP_MIN(CEIL(YY(r) + radius), FINISHINGY(Vhigh()));
347  int j0 = XMIPP_MAX(FLOOR(XX(r) - radius), STARTINGX(Vhigh()));
348  int jF = XMIPP_MIN(CEIL(XX(r) + radius), FINISHINGX(Vhigh()));
349 
350  // Fill the volume with this atom
351  Matrix1D<double> rdiff(3);
352  for (int k = k0; k <= kF; k++)
353  for (int i = i0; i <= iF; i++)
354  for (int j = j0; j <= jF; j++)
355  {
356  VECTOR_R3(rdiff, XX(r) - j, YY(r) - i, ZZ(r) - k);
357  rdiff*=highTs;
358  if (useBlobs)
359  Vhigh(k, i, j) += weight * blob_val(rdiff.module(), blob);
360  else if (usePoorGaussian || useFixedGaussian)
361  Vhigh(k, i, j) += weight *
362  exp(-rdiff.module()*rdiff.module()/(2*GaussianSigma2))*
363  GaussianNormalization;
364  }
365  }
366 
367  // Close file
368  fh_pdb.close();
369 }

371 /* Create protein at a low sampling rate ----------------------------------- */
372 void ProgPdbConverter::createProteinAtLowSamplingRate()
373 {
374  // Compute the integer downsapling factor
375  int M = FLOOR(Ts / highTs);
376  double current_Ts = highTs;
377 
378  // Use Bsplines pyramid if possible
379  int levels = FLOOR(log10((double)M) / log10(2.0) + XMIPP_EQUAL_ACCURACY);
380  pyramidReduce(BSPLINE3, Vlow(), Vhigh(), levels);
381  current_Ts *= pow(2.0, levels);
382  Vhigh.clear();
383 
384  // Now scale using Bsplines
385  int new_output_dim = CEIL(XSIZE(Vlow()) * current_Ts / Ts);
386  scaleToSize(BSPLINE3, Vhigh(), Vlow(),
387  new_output_dim, new_output_dim, new_output_dim);
388  Vlow() = Vhigh();
389  Vlow().setXmippOrigin();
390 
391  // Return to the desired size
392  Vlow().selfWindow(FIRST_XMIPP_INDEX(output_dim), FIRST_XMIPP_INDEX(output_dim),
393  FIRST_XMIPP_INDEX(output_dim), LAST_XMIPP_INDEX(output_dim),
394  LAST_XMIPP_INDEX(output_dim), LAST_XMIPP_INDEX(output_dim));
395 }
