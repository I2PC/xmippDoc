157 size_t MDRowVec::newCol(const MDLabel label) {
158  size_t i;
159  if (_no_columns != nullptr)
160  i = (*_no_columns)++;
161  else
162  i = _row->size();
163 
164  if (_col_to_label != nullptr) {
165  while (_row->size() < i) {
166  size_t j = _row->size();
167  _row->emplace_back(MDObject((*_col_to_label).at(j)));
168  }
169  if (i < (*_col_to_label).size()) {
170  (*_col_to_label).at(i) = label;
171  } else {
172  assert(i == (*_col_to_label).size());
173  (*_col_to_label).push_back(label);
174  }
175  }
176 
177  assert(_row->size() == i);
178  (*_label_to_col).at(label) = i;
179  _row->emplace_back(MDObject(label));
180  return i;
181 }
