45 def getXmippLabels():
46  ''' Parse the labels definition from the 'libraries/data/metadata_label.h' file '''
47  labelHeader = getXmippPath('libraries', 'data', 'metadata_label.h')
48  f = open(labelHeader)
49  labels = []
50  comments = {}
51  labelsPrefixes = ['MDL_', 'RLN_', 'BSOFT_']
52 
53  for line in f:
54  line = line.strip()
55  for prefix in labelsPrefixes:
56  if line.startswith(prefix) and '///<' in line:
57  parts = line.split('///<')
58  mdl = parts[0].strip()[:-1] # remove last comma
59  comm = parts[1].strip()
60  comments[mdl] = comm
61  if line.startswith('MDL::addLabel(' + prefix):
62  l = line.find('(')
63  r = line.find(')')
64  parts = line[l + 1:r].split(',')
65  label = {}
66  label['name'] = parts[2].replace('"', '').strip()
67  label['type'] = parts[1].strip()
68  label['enum'] = parts[0].strip()
69  label['comment'] = comments.get(label['enum'], "")
70  labels.append(label)
71  return labels
