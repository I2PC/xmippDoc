95 /* Make 3D CTF ------------------------------------------------------------- */
96 void ProgCtfCorrectAmplitude3D::generateCTF1D(const FileName &fnCTF, size_t nr_steps,
97  MultidimArray<double> &CTF1D)
98 {
99  // Read the CTF
100  ctf.FilterBand = CTF;
101  ctf.ctf.enable_CTFnoise = false;
102  ctf.ctf.read(fnCTF);
103  ctf.ctf.produceSideInfo();
104 
105  double maxres = ( 0.5 * sqrt(3.) ) / ctf.ctf.Tm;
106  double stepsize = maxres / (double)nr_steps;
107  CTF1D.resizeNoCopy(nr_steps);
108  double freq = 0.;
109 
110  freq=0.;
111  for (size_t step=0; step < nr_steps; step++)
112  {
113  if ( (minFreq < 0) || (1./freq < minFreq) )
114  {
115  ctf.ctf.precomputeValues(freq, 0.0);
116  A1D_ELEM(CTF1D,step)=ctf.ctf.getValueAt();
117  if (isFlipped)
118  A1D_ELEM(CTF1D,step)=fabs(CTF1D(step));
119  }
120  else
121  A1D_ELEM(CTF1D,step)=1.;
122  freq+=stepsize;
123  }
124 }

126 /* Make Wiener filters ------------------------------------------------------------- */
127 void ProgCtfCorrectAmplitude3D::generateWienerFilters()
128 {
129  MultidimArray<double> CTF1D, sumterm;
130  std::ofstream fh;
131  double res;
132  double tot_nr_imgs = 0;
133  // Oversample the 1D CTF and Wiener filter vectors OVERSAMPLE times
134  // Use 0.55*sqrt(3) to make sure all pixels fit in...
135  size_t nr_steps= (size_t)ceil(OVERSAMPLE * 0.55 * sqrt((double)(Zdim*Zdim + Ydim*Ydim + Xdim*Xdim)));
136 
137  Vctfs1D.clear();
138  Vwien1D.clear();
139  int ii = 0;
140  FileName fnVol, fnCTF;
141  FOR_ALL_OBJECTS_IN_METADATA(ctfdat)
142  {
143  // Calculate 1D CTF
144  ctfdat.getValue(MDL_IMAGE,fnVol,__iter.objId);
145  ctfdat.getValue(MDL_CTF_MODEL,fnCTF,__iter.objId);
146  generateCTF1D(fnCTF,nr_steps,CTF1D);
147  Vctfs1D.push_back(CTF1D);
148 
149  // Get the number of images contributing to this group
150  size_t NumberOfImages;
151  ctfdat.getValue(MDL_CLASS_COUNT,NumberOfImages,__iter.objId);
152  tot_nr_imgs += NumberOfImages;
153 
154  // Calculate denominator of the Wiener filter
155  if (ii==0)
156  sumterm.resizeNoCopy(CTF1D);
157  FOR_ALL_ELEMENTS_IN_ARRAY1D(sumterm)
158  {
159  double CTF1Di=A1D_ELEM(CTF1D,i);
160  A1D_ELEM(sumterm,i) += NumberOfImages * CTF1Di * CTF1Di;
161  }
162  ii++;
163  }
164 
165  // Add (normalized) Wiener filter constant
166  sumterm += tot_nr_imgs*wienerConstant;
167 
168  double maxwien=0, minwien=1e38;
169  ii=0;
170  FOR_ALL_OBJECTS_IN_METADATA(ctfdat)
171  {
172  size_t NumberOfImages;
173  ctfdat.getValue(MDL_CLASS_COUNT,NumberOfImages,__iter.objId);
174  CTF1D=Vctfs1D[ii];
175  CTF1D*=NumberOfImages;
176  CTF1D/=sumterm;
177  Vwien1D.push_back(CTF1D);
178 
179  // Write CTF and Wiener filter curves to disc
180  if (verbose>2)
181  {
182  FileName fn_tmp;
183  fn_tmp = fnRoot + "_wien";
184  fn_tmp.compose(fn_tmp, ii+1, "txt");
185  fh.open((fn_tmp).c_str(), std::ios::out);
186  if (!fh)
187  REPORT_ERROR(ERR_IO_NOWRITE, fn_tmp);
188  for (size_t step = 0; step < nr_steps; step++)
189  {
190  res = (step * sqrt(3.) ) /
191  (OVERSAMPLE * sqrt( (double) (Zdim*Zdim + Ydim*Ydim + Xdim*Xdim) ) );
192  if (res<=0.5)
193  fh << res << " " << Vwien1D[ii](step) << " " << Vctfs1D[ii](step) << "\n";
194  }
195  fh.close();
196  }
197  ii++;
198  }
199 
200  // Some output to screen
201  std::cerr <<" ---------------------------------------------------"<<std::endl;
202  std::cerr <<" + Number of defocus groups = "<<ctfdat.size()<<std::endl;
203  std::cerr <<" + Total number of images = "<<tot_nr_imgs<<std::endl;
204  std::cerr <<" + Normalized Wiener constant = "<<tot_nr_imgs*wienerConstant<<std::endl;
205  std::cerr <<" + Minimum Wiener filter value = "<<minwien<<std::endl;
206  std::cerr <<" + Maximum Wiener filter value = "<<maxwien<<std::endl;
207  std::cerr <<" ---------------------------------------------------"<<std::endl;
208 }

211 void ProgCtfCorrectAmplitude3D::generateVolumes()
212 {
213  Image<double> V;
214  FileName fnVol, fnCTF;
215  MultidimArray<std::complex<double> > fft,fft_out;
216  Matrix1D<int> idx(3);
217  Matrix1D<double> freq(3);
218 
219  int ii = 0;
220  double Xdim2=Xdim*Xdim;
221  double Ydim2=Ydim*Ydim;
222  double Zdim2=Zdim*Zdim;
223  FOR_ALL_OBJECTS_IN_METADATA(ctfdat)
224  {
225  ctfdat.getValue(MDL_IMAGE,fnVol,__iter.objId);
226  ctfdat.getValue(MDL_CTF_MODEL,fnCTF,__iter.objId);
227  V.read(fnVol);
228  FourierTransform(V(),fft);
229  if (ii == 0)
230  fft_out.initZeros(fft);
231  MultidimArray<double>& Vwien1D_ii=Vwien1D[ii];
232  FOR_ALL_ELEMENTS_IN_ARRAY3D(fft)
233  {
234  XX(idx) = j;
235  YY(idx) = i;
236  ZZ(idx) = k;
237  FFT_idx2digfreq(fft, idx, freq);
238  int ires= (int)round(OVERSAMPLE*sqrt(XX(freq)*XX(freq)*Xdim2+
239  YY(freq)*YY(freq)*Ydim2+
240  ZZ(freq)*ZZ(freq)*Zdim2));
241  A3D_ELEM(fft_out,k,i,j)+=A1D_ELEM(Vwien1D_ii,ires)*A3D_ELEM(fft,k,i,j);
242  }
243  ii++;
244  }
245 
246  // Inverse Fourier Transform
247  InverseFourierTransform(fft_out,V());
248  fnVol=fnRoot+"_deconvolved.vol";
249  V.write(fnVol);
250 
251  // Calculate CTF-affected volumes
252  ii = 0;
253  FOR_ALL_OBJECTS_IN_METADATA(ctfdat)
254  {
255  ctfdat.getValue(MDL_IMAGE,fnVol,__iter.objId);
256  ctfdat.getValue(MDL_CTF_MODEL,fnCTF,__iter.objId);
257  MultidimArray<double>& Vwien1D_ii=Vwien1D[ii];
258  FOR_ALL_ELEMENTS_IN_ARRAY3D(fft)
259  {
260  XX(idx) = j;
261  YY(idx) = i;
262  ZZ(idx) = k;
263  FFT_idx2digfreq(fft, idx, freq);
264  int ires= (int)round(OVERSAMPLE*sqrt(XX(freq)*XX(freq)*Xdim2+
265  YY(freq)*YY(freq)*Ydim2+
266  ZZ(freq)*ZZ(freq)*Zdim2));
267  A3D_ELEM(fft,k,i,j)+=A1D_ELEM(Vwien1D_ii,ires)*A3D_ELEM(fft_out,k,i,j);
268  }
269  ii++;
270 
271  fnVol=fnRoot+"_ctffiltered_group";
272  fnVol.compose(fnVol,ii,"vol");
273  InverseFourierTransform(fft,V());
274  V.write(fnVol);
275  }
276 }
