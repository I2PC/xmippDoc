225 // Perform complete search =================================================
226 void ProgNmaAlignment::performCompleteSearch(const FileName &fnRandom,
227  int pyramidLevel) const {
228  String program;
229  String arguments;
230  const char * randStr = fnRandom.c_str();
231 
232  // Reduce the image
233  FileName fnDown = formatString("%s_downimg.xmp", fnRandom.c_str());
234  if (pyramidLevel != 0) {
235  Image<double> I;
236  I.read(currentImgName);
237  selfPyramidReduce(BSPLINE3, I(), pyramidLevel);
238  I.write(fnDown);
239  } else
240  copyImage(currentImgName.c_str(), fnDown.c_str());
241 
242  mkdir((fnRandom+"_ref").c_str(), S_IRWXU);
243 
244  double angSampling=2*RAD2DEG(atan(1.0/((double) imgSize / pow(2.0, (double) pyramidLevel+1))));
245  angSampling=std::max(angSampling,discrAngStep);
246  program = "xmipp_angular_project_library";
247  arguments = formatString(
248  "-i %s_deformedPDB.vol -o %s_ref/ref.stk --sampling_rate %f -v 0",
249  randStr, randStr, angSampling);
250  if (projMatch)
251  arguments +=formatString(
252  " --compute_neighbors --angular_distance -1 --experimental_images %s_downimg.xmp", randStr);
253 
254  runSystem(program, arguments, false);
255 
256  String refSelStr = formatString("%s_ref/ref.doc", randStr);
257 
258  if (fnmask != "") {
259  program = "xmipp_transform_mask";
260  arguments = formatString("-i %s --mask binary_file %s", refSelStr.c_str(),
261  fnmask.c_str());
262  runSystem(program, arguments, false);
263  }
264 
265  // Perform alignment
266  String fnOut=formatString("%s_angledisc.xmd",randStr);
267  if (!projMatch) {
268  program = "xmipp_angular_discrete_assign";
269  arguments = formatString(
270  "-i %s_downimg.xmp --ref %s -o %s --psi_step 5 --max_shift_change %d --search5D -v 0",
271  randStr, refSelStr.c_str(), fnOut.c_str(), (int)round((double) imgSize / (10.0 * pow(2.0, (double) pyramidLevel))));
272  } else {
273  String refStkStr = formatString("%s_ref/ref.stk", randStr);
274  program = "xmipp_angular_projection_matching";
275  arguments = formatString(
276  "-i %s_downimg.xmp --ref %s -o %s --search5d_step 1 --max_shift %d -v 0",
277  randStr, refStkStr.c_str(), fnOut.c_str(), (int)round((double) imgSize / (10.0 * pow(2.0, (double) pyramidLevel))));
278  }
279  runSystem(program, arguments, false);
280  if (projMatch)
281  {
282  MetaData MD;
283  MD.read(fnOut);
284  bool flip;
285  size_t id=MD.firstObject();
286  MD.getValue(MDL_FLIP,flip,id);
287  if (flip)
288  {
289  // This is because continuous assignment does not understand flips
290 
291  double shiftX, rot, tilt, psi, newrot, newtilt, newpsi;
292  // Change sign in shiftX
293  MD.getValue(MDL_SHIFT_X,shiftX,id);
294  MD.setValue(MDL_SHIFT_X,-shiftX,id);
295 
296  // Change Euler angles
297  MD.getValue(MDL_ANGLE_ROT,rot,id);
298  MD.getValue(MDL_ANGLE_TILT,tilt,id);
299  MD.getValue(MDL_ANGLE_PSI,psi,id);
300  Euler_mirrorY(rot,tilt,psi,newrot,newtilt,newpsi);
301  MD.setValue(MDL_ANGLE_ROT,newrot,id);
302  MD.setValue(MDL_ANGLE_TILT,newtilt,id);
303  MD.setValue(MDL_ANGLE_PSI,newpsi,id);
304  MD.write(fnOut);
305  }
306  }
307 }

309 // Continuous assignment ===================================================
310 double ProgNmaAlignment::performContinuousAssignment(const FileName &fnRandom,
311  int pyramidLevel) const {
312  // Perform alignment
313  const char * randStr = fnRandom.c_str();
314  String fnResults=formatString("%s_anglecont.xmd", randStr);
315  bool costSource=true;
316  String program = "xmipp_angular_continuous_assign";
317  String arguments =
318  formatString(
319  "-i %s_angledisc.xmd --ref %s_deformedPDB.vol -o %s --gaussian_Fourier %f --gaussian_Real %f --zerofreq_weight %f -v 0",
320  randStr, randStr, fnResults.c_str(), gaussian_DFT_sigma,
321  gaussian_Real_sigma, weight_zero_freq);
322  runSystem(program, arguments, false);
323 
324  // Pick up results
325  MetaData DF(fnResults);
326  MDRow row;
327  DF.getRow(row, DF.firstObject());
328  row.getValue(MDL_ANGLE_ROT, trial(VEC_XSIZE(trial) - 5));
329  row.getValue(MDL_ANGLE_TILT, trial(VEC_XSIZE(trial) - 4));
330  row.getValue(MDL_ANGLE_PSI, trial(VEC_XSIZE(trial) - 3));
331  row.getValue(MDL_SHIFT_X, trial(VEC_XSIZE(trial) - 2));
332  trial(VEC_XSIZE(trial) - 2) *= pow(2.0, (double) pyramidLevel);
333  row.getValue(MDL_SHIFT_Y, trial(VEC_XSIZE(trial) - 1));
334  trial(VEC_XSIZE(trial) - 1) *= pow(2.0, (double) pyramidLevel);
335  double tempvar;
336  if (!costSource) {
337  row.getValue(MDL_MAXCC, tempvar);
338  tempvar = -tempvar;
339  } else
340  row.getValue(MDL_COST, tempvar);
341  return tempvar;
342 }
