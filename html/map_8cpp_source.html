543 void RECTLayout::localAve(const FuzzyMap* _som, const SomPos& _center, std::vector<double>& _aveVector) const
544 {
545  int j;
546  int dim = _som->itemAt(0).size();
547  double *ptrAveVector=&(_aveVector[0]);
548  memset(ptrAveVector,0,dim*sizeof(double));
549  int tmpi = _center.first;
550  int tmpj = _center.second;
551  int kk = 0;
552  if ((tmpi - 1) >= 0)
553  {
554  kk++;
555  const floatFeature *codevector=&(_som->itemAt(_som->PosToIndex(SomPos(tmpi - 1, tmpj)))[0]);
556  for (j = 0; j < dim; j++)
557  ptrAveVector[j] += codevector[j];
558  }
559  if ((tmpi + 1) < (int)_som->width())
560  {
561  kk++;
562  const floatFeature *codevector=&(_som->itemAt(_som->PosToIndex(SomPos(tmpi + 1, tmpj)))[0]);
563  for (j = 0; j < dim; j++)
564  ptrAveVector[j] += codevector[j];
565  }
566  if ((tmpj - 1) >= 0)
567  {
568  kk++;
569  const floatFeature *codevector=&(_som->itemAt(_som->PosToIndex(SomPos(tmpi, tmpj-1)))[0]);
570  for (j = 0; j < dim; j++)
571  ptrAveVector[j] += codevector[j];
572  }
573  if ((tmpj + 1) < (int)_som->height())
574  {
575  kk++;
576  const floatFeature *codevector=&(_som->itemAt(_som->PosToIndex(SomPos(tmpi, tmpj+1)))[0]);
577  for (j = 0; j < dim; j++)
578  ptrAveVector[j] += codevector[j];
579  }
580  if (_som->height() == 1 || _som->width() == 1)
581  for (j = 0; j < dim; j++)
582  ptrAveVector[j] *= 1.0/2.0;
583  else
584  for (j = 0; j < dim; j++)
585  ptrAveVector[j] *= 1.0/4.0;
586 }

626 double HEXALayout::dist(const SomPos& _center, const SomPos& _v) const
627 {
628  double ret, diff;
629  diff = _center.first - _v.first;
630 
631  if (((_center.second - _v.second) % 2) != 0)
632  {
633  if ((_center.second % 2) == 0)
634  {
635  diff -= 0.5;
636  }
637  else
638  {
639  diff += 0.5;
640  }
641  }
642  ret = diff * diff;
643  diff = _center.second - _v.second;
644  ret += 0.75 * diff * diff;
645  ret = (double) sqrt(ret);
646  return(ret);
647 }
