330  @staticmethod
331  def colorMapToColorList(steps, colorMap):
332  """ Returns a list of pairs resolution, hexColor to be used in chimera
333  scripts for coloring the volume and the colorKey """
334 
335  # Get the map used by monoRes
336  colors = ()
337  ratio = 255.0/(len(steps)-1)
338  for index, step in enumerate(steps):
339  colorPosition = int(round(index*ratio))
340  rgb = colorMap(colorPosition)[:3]
341  colors += step,
342  rgbColor = mcolors.rgb2hex(rgb)
343  colors += rgbColor,
344 
345  return colors
