169 void PCAMahalanobisAnalyzer::learnPCABasis(size_t NPCA, size_t Niter)
170 {
171  // Take the first vectors for the PCA basis
172  MultidimArray<double> vPCA;
173  NPCA=XMIPP_MIN(NPCA,v.size());
174  std::vector<size_t> used;
175  PCAbasis.clear();
176  for (size_t n=0; n<NPCA; n++)
177  {
178  size_t nRandom;
179  do {
180  nRandom=(size_t)round((v.size()-1)*rnd_unif(0,1));
181  } while (std::find(used.begin(),used.end(),nRandom)!=used.end());
182  typeCast(v[nRandom],vPCA);
183  PCAbasis.push_back(vPCA);
184  used.push_back(nRandom);
185  }
186 
187  size_t N=v.size();
188  for (size_t n=0; n<Niter; n++)
189  {
190  // E-step ..........................................................
191  // Compute C^t*C
192  Matrix2D<double> CtC(NPCA,NPCA);
193  for (size_t ii=0; ii<NPCA; ii++)
194  {
195  const MultidimArray<double> &Iii=PCAbasis[ii];
196  for (size_t jj=ii; jj<NPCA; jj++)
197  {
198  const MultidimArray<double> &Ijj=PCAbasis[jj];
199  CtC(ii,jj)=0;
200  FOR_ALL_DIRECT_ELEMENTS_IN_ARRAY1D(Ijj)
201  MAT_ELEM(CtC,ii,jj)+=
202  DIRECT_A1D_ELEM(Iii,i)*DIRECT_A1D_ELEM(Ijj,i);
203 
204  if (ii!=jj)
205  CtC(jj,ii)=CtC(ii,jj);
206 
207  }
208  }
209 
210  Matrix2D<double> CtY, X;
211  projectOnPCABasis(CtY);
212  X=CtC.inv()*CtY;
213 
214  // M-step ..........................................................
215  Matrix2D<double> XtXXtinv=X.transpose()*(X*X.transpose()).inv();
216  for (size_t ii=0;ii<NPCA;ii++)
217  {
218  MultidimArray<double> &Ipca=PCAbasis[ii];
219  const size_t unroll=4;
220  size_t nmax=unroll*(MULTIDIM_SIZE(Ipca)/unroll);
221  Ipca.initZeros();
222  for (size_t jj=0; jj<N; jj++)
223  {
224  const MultidimArray<float> &I=v[jj];
225  const float *ptrI=MULTIDIM_ARRAY(I);
226  double *ptrPCA=MULTIDIM_ARRAY(Ipca);
227  double val=MAT_ELEM(XtXXtinv,jj,ii);
228  size_t n;
229  for (n=0; n<nmax; n+=unroll, ptrPCA+=unroll, ptrI+=unroll)
230  {
231  *ptrPCA += *ptrI * val;
232  *(ptrPCA+1) += *(ptrI+1) * val;
233  *(ptrPCA+2) += *(ptrI+2) * val;
234  *(ptrPCA+3) += *(ptrI+3) * val;
235  }
236  for (n=nmax, ptrI=MULTIDIM_ARRAY(I)+nmax, ptrPCA=MULTIDIM_ARRAY(Ipca)+nmax;
237  n<MULTIDIM_SIZE(I); ++n, ++ptrI, ++ptrPCA)
238  *ptrPCA += *ptrI * val;
239  }
240  }
241  }
242 
243  //Obtain the Orthonormal vectors for the C (According to paper)
244  gramSchmidt();
245 
246  //Generate a Matrix for true PCA and compute the average C'*data
247  Matrix2D<double> data;
248  double *refData;
249  MultidimArray<double> average;
250  data.initZeros(NPCA,v.size());
251  average.initZeros(NPCA);
252  refData = &MAT_ELEM(data,0,0);
253  for (size_t ii=0;ii<NPCA;ii++)
254  {
255  MultidimArray<double> &C=PCAbasis[ii];
256  for (size_t jj=0;jj<v.size();jj++)
257  {
258  MultidimArray<float> &D=v[jj];
259  FOR_ALL_DIRECT_ELEMENTS_IN_ARRAY1D(C)
260  (*refData) += DIRECT_A1D_ELEM(C,i)*DIRECT_A1D_ELEM(D,i);
261  DIRECT_A1D_ELEM(average,ii)+=(*refData)++;
262  }
263  }
264  average/=v.size();
265  FOR_ALL_ELEMENTS_IN_MATRIX2D(data)
266  MAT_ELEM(data,i,j)-=DIRECT_A1D_ELEM(average,i);
267 
268  Matrix2D<double> covarMatrix;
269  Matrix2D<double> u;
270  Matrix2D<double> v;
271  covarMatrix=data*data.transpose();
272  svdcmp(covarMatrix,u,w,v);
273 
275  size_t xsize=XSIZE(PCAbasis[0]);
276  MultidimArray<double> temp;
277  for (size_t i=0;i<xsize;i++)
278  {
279  temp.initZeros(1,NPCA);
280  for (size_t j=0;j<NPCA;j++)
281  for (size_t k=0;k<NPCA;k++)
282  {
283  const MultidimArray<double> &Iii=PCAbasis[k];
284  DIRECT_A1D_ELEM(temp,j)+=DIRECT_A1D_ELEM(Iii,i)* MAT_ELEM(v,k,j);
285  }

288  const MultidimArray<double> &Iii=PCAbasis[k];

290  }
291  }
292

312 void PCAMahalanobisAnalyzer::gramSchmidt()
313 {
314  MultidimArray<double> v,orthonormalBasis;
315  for (size_t ii=0;ii<PCAbasis.size();ii++)
316  {
318  MultidimArray<double> &Iii=PCAbasis[ii];
319  if (ii==0)
320  {
321  double inorm=1.0/sqrt(Iii.sum2());
322  for (size_t i=0;i<XSIZE(Iii);i++)
323  DIRECT_A1D_ELEM(Iii,i)=DIRECT_A1D_ELEM(Iii,i)*inorm;
324  continue;
325  }
327  v.initZeros(XSIZE(Iii));
328  orthonormalBasis.initZeros(XSIZE(Iii));
329  for (size_t jj=0;jj<ii;jj++)
330  {
331  const MultidimArray<double> &Ijj=PCAbasis[jj];
332  double dotProduct=Iii.dotProduct(Ijj);
333  FOR_ALL_DIRECT_ELEMENTS_IN_ARRAY1D(v)
334  DIRECT_A1D_ELEM(v,i)-=dotProduct*DIRECT_A1D_ELEM(Ijj,i);
335  }
336  FOR_ALL_DIRECT_ELEMENTS_IN_ARRAY1D(orthonormalBasis)
337  DIRECT_A1D_ELEM(orthonormalBasis,i)= DIRECT_A1D_ELEM(Iii,i)+ DIRECT_A1D_ELEM(v,i);
338  double inorm=1.0/sqrt(orthonormalBasis.sum2());
339  for (size_t i=0;i<XSIZE(Iii);i++)
340  DIRECT_A1D_ELEM(Iii,i)=DIRECT_A1D_ELEM(orthonormalBasis,i)*inorm;
341  }
342 }

345 void PCAMahalanobisAnalyzer::computeStatistics(MultidimArray<double> & avg,
