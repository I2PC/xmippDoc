128 void ClassicTrainingVectors::readSelf(std::istream& _is)
129 {
130 #ifndef _NO_EXCEPTION
131  try
132  {
133 #endif
134  clear();
135  std::string line;
136 
137  // Determines the number of rows and columns in the training set
138 
139  long dim, size;
140  _is >> dim;
141  _is >> line;
142  if (!sscanf(line.c_str(), "%ld", &size))
143  {
144  int x, y;
145  _is >> x;
146  _is >> y;
147  size = x * y;
148  }
149  getline(_is, line);
150  theItems.resize(size);
151  theTargets.resize(size);
152 
153  for (int i = 0; i < size; i++)
154  {
155  std::vector<floatFeature> v;
156  v.resize(dim);
157  for (int j = 0; j < dim; j++)
158  {
159  floatFeature var;
160  _is >> var;
161  v[j] = var;
162  }
163  getline(_is, line);
164  theItems[i] = v;
165  theTargets[i] = removeSpaces(line);
166  }
167 
168 #ifndef _NO_EXCEPTION
169  }
170  catch (std::exception& e)
171  {
172  std::ostringstream msg;
173  msg << e.what() << std::endl << "Error reading the training set";
174  throw std::runtime_error(msg.str());

179 void ClassicTrainingVectors::read(const FileName& fnIn)
180 {
181  clear();
182 
183  // Read header and content
184  MetaData vectorHeader(formatString("vectorHeader@%s",fnIn.c_str()));
185  MetaData vectorContent(formatString("vectorContent@%s",fnIn.c_str()));
186  size_t Nvectors;
187  size_t vectorSize;
188  size_t id=vectorHeader.firstObject();
189  vectorHeader.getValue(MDL_CLASSIFICATION_DATA_SIZE,vectorSize,id);
190  vectorHeader.getValue(MDL_COUNT,Nvectors,id);
191  theItems.reserve(Nvectors);
192  theTargets.reserve(Nvectors);
193 
194  // Read the data
195  FileName fnInRaw=formatString("%s.vec",fnIn.withoutExtension().c_str());
196  std::ifstream fhInRaw(fnInRaw.c_str(),std::ios::binary);
197  if (!fhInRaw)
198  REPORT_ERROR(ERR_IO_NOTEXIST,fnInRaw);
199  std::vector<floatFeature> v;
200  v.resize(vectorSize);
201  float *buffer=new float[vectorSize];
202  String fnImg;
203  size_t order;
204  FOR_ALL_OBJECTS_IN_METADATA(vectorContent)
205  {
206  vectorContent.getValue(MDL_IMAGE,fnImg,__iter.objId);
207  vectorContent.getValue(MDL_ORDER,order,__iter.objId);
208 
209  // Read raw values
210  fhInRaw.seekg(order*vectorSize*sizeof(float));
211  fhInRaw.read((char*)buffer,vectorSize*sizeof(float));
212  if (!fhInRaw)
213  REPORT_ERROR(ERR_IO_NOREAD,
214  formatString("Could not read image %lu from %s",
215  order,fnInRaw.c_str()));
216  for (size_t i=0; i<vectorSize; ++i)
217  v[i]=buffer[i];
218  theTargets.push_back(fnImg);
219  theItems.push_back(v);
220  }
221  delete []buffer;
222 }

358 void ClassicTrainingVectors::normalizeFeature(unsigned _i)
359 {
360  using namespace std;
361 
362  // Do some validation
363 
364  if (_i > itemAt(0).size())
365  {
366  std::ostringstream msg;
367  msg << "Out of range. No variable at position " << _i;
368  throw std::out_of_range(msg.str());
369  }
370 
371  // first calculates the mean
372  floatFeature mean = 0;
373  int nn = 0;
374  for (size_t it = 0; it < size(); it++)
375  {
376  if (!isnan(itemAt(it)[_i]))
377  {
378  mean += itemAt(it)[_i];
379  nn++;
380  }
381 
382  }
383  mean /= (floatFeature) nn;
384 
385  // Then calculates SD
386  floatFeature sd = 0;
387  for (size_t it = 0; it < size(); it++)
388  {
389  if (!isnan(itemAt(it)[_i]))
390  sd += (itemAt(it)[_i] - mean) * (itemAt(it)[_i] - mean);
391  }
392  sd = sqrt(sd / (floatFeature)(nn - 1));
393 
394  // Now normalize the variable
395  if (sd != 0)
396  {
397  for (size_t it = 0; it < size(); it++)
398  {
399  if (!isnan(itemAt(it)[_i]))
400  itemAt(it)[_i] = (itemAt(it)[_i] - mean) / sd;
401  }
402  }
403 
404  varStats[_i].mean = mean;
405  varStats[_i].sd = sd;
406 }
