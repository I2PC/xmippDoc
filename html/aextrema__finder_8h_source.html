54 class ExtremaFinderSettings {
55 public:
56  std::vector<HW*> hw;

63  Point3D<size_t> getCenter() const {
64  return Point3D<size_t>(dims.x() / 2, dims.y() / 2, dims.z() / 2);

67  void check() const {
68  if (0 == hw.size()) {
69  REPORT_ERROR(ERR_VALUE_INCORRECT, "HW contains zero (0) devices");
70  }
71  if ( ! dims.isValid()) {
72  REPORT_ERROR(ERR_LOGIC_ERROR, "Dimensions are invalid (contain 0)");
73  }
74  if (0 == batch) {
75  REPORT_ERROR(ERR_LOGIC_ERROR, "Batch is zero (0)");
76  }
77  if ((SearchType::MaxAroundCenter == searchType)
78  || (SearchType::LowestAroundCenter == searchType)) {
79  const auto center = getCenter();
80  if (0 == maxDistFromCenter) {
81  REPORT_ERROR(ERR_LOGIC_ERROR, "'maxDistFromCenter' is set to zero (0)");
82  }
83  if (dims.is1D()) {
84  if (maxDistFromCenter >= center.x) {
85  REPORT_ERROR(ERR_LOGIC_ERROR, "'maxDistFromCenter' is bigger than half of the signal's X dimension");
86  }
87  } else if (dims.is2D()) {
88  if ((maxDistFromCenter >= center.x)
89  || (maxDistFromCenter >= center.y)) {
90  REPORT_ERROR(ERR_LOGIC_ERROR, "'maxDistFromCenter' is bigger than half of the signal's X or Y dimensions");
91  }
92  } else {
93  if ((maxDistFromCenter >= center.x)
94  || (maxDistFromCenter >= center.y)
95  || (maxDistFromCenter >= center.z)) {
96  REPORT_ERROR(ERR_LOGIC_ERROR, "'maxDistFromCenter' is bigger than half of the signal's X, Y or Z dimensions");
97  }
98  }
99  }
100  }
101 };
