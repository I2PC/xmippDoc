171 bool ArgLexer::nextToken()
172 {
173  //make it visible by default
174  pToken->visibility = 0;
175  pToken->starred = false;
176 
177  if (line == input.size())
178  {
179  setupToken(TOK_END);
180  return false;
181  }
182 
183  char c = input[line][pos];
184 
185  //Skip all white spaces
186  while (line < input.size() && pos < input[line].length() && isspace(c
187  = input[line][pos]))
188  {
189  ++pos;
190  if (c == '\0' || c == '\n' || pos == input[line].length()) //Next line
191  {
192  nextLine();
193  }
194  }
195 
196  if (line == input.size())
197  setupToken(TOK_END);
198  else if (isalpha(c) || c == '_')
199  {
200  offset = 0;
201  while (isalnum(c) || c == '_')
202  {
203  ++offset;
204  c = input[line][pos + offset];
205  }
206  setupToken(TOK_ID);
207  }
208  else if (isalnum(c) || (c == '-' && isdigit(input[line][pos + 1])))
209  {
210  offset = 1;
211  ArgTokenType t = TOK_INT;
212  while (isdigit(input[line][pos + offset]))
213  ++offset;
214  if (input[line][pos + offset] == '.')
215  {
216  ++offset;
217  while (isdigit(input[line][pos + offset]))
218  ++offset;
219  t = TOK_FLOAT;
220  }
221  if (input[line][pos + offset] == 'e')
222  {
223  ++offset;
224  if (input[line][pos + offset] != '+' && input[line][pos + offset]
225  != '-')
226  {
227  std::cerr << "ERROR: expected '+' or '-' " << std::endl
228  << "at line: " << line + 1 << " pos: " << offset + 1
229  << std::endl;
230  exit(1);
231  }
232  ++offset;
233  while (isdigit(input[line][pos + offset]))
234  ++offset;
235  t = TOK_FLOAT;
236 
237  }
238  setupToken(t);
239  }
240  else
241  {
242  offset = 1;
243  bool empty;
244  switch (c)
245  {
246  case '<':
247  setupToken(TOK_LAN);
248  break;
249  case '>':
250  setupToken(TOK_RAN);
251  break;
252  case '=':
253  if (input[line][pos + offset] == '=')
254  {
255  pos = input[line].find_first_not_of('=', pos + 1);
256  checkVisibility();
257  offset = input[line].find_first_of('=', pos + 1);
258  offset -= pos;
259  setupToken(TOK_SECTION);
260  nextLine();
261  }
262  else
263  //simple equal sign '='
264  setupToken(TOK_EQ);
265  break;
266  case '[':
267  setupToken(TOK_LBRA);
268  break;
269  case ']':
270  setupToken(TOK_RBRA);
271  break;
272  case ';':
273  setupToken(TOK_SEMI);
274  break;
275  case ',':
276  setupToken(TOK_COMMA);
277  break;
278  case '+':
279  setupToken(TOK_PLUS);
280  break;
281  case '.':
282  if (input[line][pos + offset] == '.' &&
283  input[line][pos + offset + 1] == '.')
284  {
285  offset += 2;
286  setupToken(TOK_ETC);
287  }
288  break;
289  case ':':
290  ++pos;
291 
292  checkVisibility();
293 
294  offset = input[line].find_first_of("\n\r", pos);
295  offset -= pos;
296  setupToken(TOK_COMM);
297  nextLine();
298  break;
299  case '-':
300  if (input[line][pos + offset] == '-')
301  ++offset;
302  empty = true;
303  c = input[line][pos + offset];
304  if (isalpha(c) || c == '_') //should start with letter or _
305  {
306  while (isalnum(c) || c == '_')
307  {
308  empty = false;
309  ++offset;
310  c = input[line][pos + offset];
311  }
312  }
313  if (empty)
314  {
315  std::cerr << "ERROR: Params should be of the form -ID or --ID"
316  << std::endl;
317  exit(1);
318  }
319  setupToken(TOK_OPT);
320  checkVisibility();
321  checkIndependent();
322  break;
323  case '"':
324  offset = input[line].find_first_of('"', pos + 1);
325  ++pos;
326  offset -= pos;
327  setupToken(TOK_STR);
328  break;
329  default:
330  std::cerr << "ERROR: Unexpected character '" << c << "'"
331  << std::endl << "at line: " << line + 1 << " pos: " << pos
332  + 1 << std::endl;
333  std::cerr << "WRONG LINE: " << input[line] << std::endl;
334  exit(1);
335  }
336  }
337 
338  //ConsolePrinter * cp = new ConsolePrinter();
339  //cp->printToken(pToken);
340 
341  return true;
342 }

465 bool ArgumentDef::parse()
466 {
467  // A -> < ID DEF > | <...>
468  // DEF -> = VALUE | e
469  // VALUE -> INT | FLOAT | STRING
470  consume(TOK_LAN);
471  if (lookahead(TOK_ID))
472  {
473  consume(TOK_ID);
474  name = token.lexeme;
475  if (lookahead(TOK_EQ))
476  {
477  consume(TOK_EQ);
478  //Consume a value, that can be int, float or string
479  ArgTokenType t = lookahead();
480 
481  if (t == TOK_INT || t == TOK_FLOAT || t == TOK_STR || t == TOK_ID)
482  consume(t);
483  else
484  unexpectedToken(" expecting INT, FLOAT, STRING or ID.");
485 
486  hasDefault = true;
487  argDefault = token.lexeme;
488  }
489  }
490  else
491  {
492  consume(TOK_ETC);
493  name = token.lexeme;
494  isList = true;
495  }
496  consume(TOK_RAN);
497  return true;
498 }

589 bool ParamDef::parse()
590 {
591  ProgramDef * prog = (ProgramDef*) parent->parent;
592  notOptional = true;
593  orBefore = false;
594  independent = false;
595  counter = 0;
596 
597  //Param Definition(OD)
598  //OD -> OH CL
599  //Param Header(OH)
600  //OH -> O | [O] | OR O
601  //Param(O)
602  //O -> param AL
603  if (lookahead(TOK_LBRA))
604  {
605  consume(TOK_LBRA);
606  notOptional = false;
607  }
608  else if (lookahead(TOK_OR))
609  {
610  consume(TOK_OR);
611  orBefore = true;
612  }
613  prog->addParamExclusiveGroup(this);
614 
615  consume(TOK_OPT);
616  name = token.lexeme;
617  visible = token.visibility;
618  independent = token.starred;
619 
620  prog->addParamName(name, this);
621 
622  //Parse argument list
623  parseArgumentList();
624 
625  if (notOptional == false)
626  consume(TOK_RBRA);
627 
628  //Parse comment list
629  parseCommentList(comments);
630 
631  //WHERE section
632  while (lookahead(TOK_WHERE))
633  {
634  consume(TOK_WHERE);
635  while (lookahead(TOK_LAN))
636  {
637  consume(TOK_LAN);
638  consume(TOK_ID);
639  ArgumentDef * pArg = findArgument(token.lexeme);
640  if (pArg == NULL)
641  {
642  std::cerr << "ERROR; on WHERE definition.\n Param '" << name
643  << "' not contains argument '" << token.lexeme << "'"
644  << std::endl;
645  exit(1);
646  }
647  consume(TOK_RAN);
648 
649  while (lookahead(TOK_ID))
650  {
651  ParamDef * pOpt = new ParamDef(pLexer, this);
652  pOpt->consume(TOK_ID);
653 
654  pOpt->name = pOpt->token.lexeme;
655  pOpt->parseArgumentList();
656  pOpt->parseCommentList(pOpt->comments);
657  pOpt->parseParamList(TOK_REQUIRES, prog, pOpt->requires, false);
658  pArg->subParams.push_back(pOpt);
659  }
660 
661  }
662  }
663 
664  //ALIAS section
665  parseParamList(TOK_ALIAS, prog, aliases, true);
666 
667  //REQUIRES section
668  parseParamList(TOK_REQUIRES, prog, requires, false);
669 
670  return true;
671 }
