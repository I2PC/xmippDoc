198 void ProgRecFourierGPU::produceSideinfo()
199 {
200  // Translate the maximum resolution to digital frequency
201  // maxResolution=sampling_rate/maxResolution;
202  maxResolutionSqr=maxResolution*maxResolution;
203 
204  // Read the input images
205  SF.read(fn_in);
206  SF.removeDisabled();
207  SF.getDatabase()->activateThreadMuting();
208 
209  // Ask for memory for the output volume and its Fourier transform
210  size_t objId = SF.firstObject();
211  FileName fnImg;
212  SF.getValue(MDL_IMAGE,fnImg,objId);
213  Image<double> I;
214  I.read(fnImg, HEADER);
215  int Ydim=YSIZE(I());
216  int Xdim=XSIZE(I());
217  if (Ydim!=Xdim)
218  REPORT_ERROR(ERR_MULTIDIM_SIZE,"This algorithm only works for squared images");
219  imgSize=Xdim;
220  paddedImgSize = Xdim*padding_factor_vol;
221  size_t conserveRows = (size_t) ceil((double) paddedImgSize * maxResolution * 2.0);
222  conserveRows = (size_t) ceil((double) conserveRows / 2.0);
223  maxVolumeIndexX = maxVolumeIndexYZ = 2 * conserveRows;
224 
225  // Build a table of blob values
226  Fourier_blob_table.resize(BLOB_TABLE_SIZE_SQRT);
227 
228  struct blobtype blobFourier,blobnormalized;
229  blobFourier=blob;
230  blobFourier.radius/=(padding_factor_vol*Xdim);
231  blobnormalized=blob;
232  blobnormalized.radius/=((double)padding_factor_proj/padding_factor_vol);
233  double deltaSqrt = (blob.radius*blob.radius) /(BLOB_TABLE_SIZE_SQRT-1);
234  double deltaFourier = (sqrt(3.)*Xdim/2.)/(BLOB_TABLE_SIZE_SQRT-1);
235 
236  // The interpolation kernel must integrate to 1
237  iw0 = 1.0 / blob_Fourier_val(0.0, blobnormalized);
238  double padXdim3 = padding_factor_vol * Xdim;
239  padXdim3 = padXdim3 * padXdim3 * padXdim3;
240  double blobTableSize = blob.radius*sqrt(1./ (BLOB_TABLE_SIZE_SQRT-1));
241  for (int i = 0; i < BLOB_TABLE_SIZE_SQRT; i++)
242  {
243  //use a r*r sample instead of r
244  //DIRECT_VEC_ELEM(blob_table,i) = blob_val(delta*i, blob) *iw0;
245  blobTableSqrt[i] = blob_val(blobTableSize*sqrt((double)i), blob) *iw0;
246  //***
247  //DIRECT_VEC_ELEM(fourierBlobTableSqrt,i) =
248  // blob_Fourier_val(fourierBlobTableSize*sqrt(i), blobFourier)*padXdim3 *iw0;
249  VEC_ELEM(Fourier_blob_table,i) =
250  blob_Fourier_val(deltaFourier*i, blobFourier)*padXdim3 *iw0;
251  //#define DEBUG
252 #ifdef DEBUG
253 
254  std::cout << VEC_ELEM(Fourier_blob_table,i)
255  << " " << VEC_ELEM(fourierBlobTableSqrt,i)
256  << std::endl;
257 #endif
258  #undef DEBUG
259 
260  }
261  //iDelta = 1/delta;
262  iDeltaSqrt = 1/deltaSqrt;
263  iDeltaFourier = 1/deltaFourier;
264 
265  // Get symmetries
266  Matrix2D<double> Identity(3,3);
267  Identity.initIdentity();
268  R_repository.push_back(Identity);
269  if (fn_sym != "")
270  {
271  SymList SL;
272  SL.readSymmetryFile(fn_sym);
273  for (int isym = 0; isym < SL.symsNo(); isym++)
274  {
275  Matrix2D<double> L(4, 4), R(4, 4);
276  SL.getMatrices(isym, L, R);
277  R.resize(3, 3);
278  R_repository.push_back(R);
279  }
280  }

283 void ProgRecFourierGPU::cropAndShift(
284  MultidimArray<std::complex<double> >& paddedFourier,
285  ProgRecFourierGPU* parent,
286  RecFourierBufferData* buffer,
287  float* dest) {
288  int sizeX = buffer->fftSizeX;
289 
290  // convert image (shift to center and remove high frequencies)
291  std::complex<double> paddedFourierTmp;
292  int halfY = paddedFourier.ydim / 2;
293  double tempMyPadd[2];
294  for (size_t i = 0; i < paddedFourier.ydim; i++) {
295  for (size_t j = 0; j < sizeX; j++) {
296  if (i < sizeX || i >= (paddedFourier.ydim - sizeX)) {
297  // check the frequency
298  paddedFourierTmp = DIRECT_A2D_ELEM(paddedFourier, i, j);
299  FFT_IDX2DIGFREQ(j, parent->paddedImgSize, tempMyPadd[0]);
300  FFT_IDX2DIGFREQ(i, parent->paddedImgSize, tempMyPadd[1]);
301  if (tempMyPadd[0] * tempMyPadd[0] + tempMyPadd[1] * tempMyPadd[1]> parent->maxResolutionSqr) {
302  paddedFourierTmp = std::complex<double>(0.0, 0.0);
303  }
304  // do the shift
305  int myPadI = (i < halfY) ? i + sizeX : i - paddedFourier.ydim + sizeX;
306  int index = myPadI * sizeX + j;
307  dest[index * 2] = paddedFourierTmp.real();
308  dest[index * 2 + 1] = paddedFourierTmp.imag();
309  }
310  }
311  }

314 void ProgRecFourierGPU::prepareBuffer(RecFourierWorkThread* threadParams,
315  ProgRecFourierGPU* parent,
316  bool hasCTF, std::vector<size_t>& objId)
317 {
318  ApplyGeoParams params;
319  params.only_apply_shifts = true;
320  MultidimArray<float> localPaddedImgFloat;
321  MultidimArray<double> localPaddedImgDouble;
322  FourierTransformer localTransformerImg;
323  // FIXME cannot be complex float due to build errors
324  MultidimArray< std::complex<double> > localPaddedFourier;
325 
326  RecFourierBufferData* buffer = threadParams->buffer;
327  buffer->noOfImages = 0; // 'clean' buffer
328  for (int projIndex = 0; projIndex < parent->bufferSize; projIndex++) {
329  double rot, tilt, psi;
330  Projection proj;
331  Matrix2D<double> Ainv(3, 3);
332  int imgIndex = threadParams->startImageIndex + projIndex;
333 
334  if (imgIndex >= threadParams->endImageIndex) {
335  continue;
336  }
337  //Read projection from selfile, read also angles and shifts if present
338  //but only apply shifts (set above)
339  // FIXME following line is a current bottleneck, as it calls BSpline interpolation
340  proj.readApplyGeo(*(threadParams->selFile), objId[imgIndex], params);
341  if (parent->do_weights && proj.weight() == 0.f) {
342  continue;
343  }
344 
345  // Compute the coordinate axes associated to this projection
346  rot = proj.rot();
347  tilt = proj.tilt();
348  psi = proj.psi();
349  Euler_angles2matrix(rot, tilt, psi, Ainv);
350 
351  Ainv = Ainv.transpose();
352  float transf[3][3];
353  float transfInv[3][3];
354 
355  // prepare transforms for all symmetries
356  int travSpaceOffset = buffer->noOfImages * buffer->noOfSymmetries;
357  for (int j = 0; j < parent->R_repository.size(); j++) {
358  RecFourierProjectionTraverseSpace* space = &buffer->spaces[travSpaceOffset + j];
359 
360  Matrix2D<double> A_SL=parent->R_repository[j]*(Ainv);
361  Matrix2D<double> A_SLInv=A_SL.inv();
362  A_SL.convertTo(transf);
363  A_SLInv.convertTo(transfInv);
364 
365  parent->computeTraverseSpace(buffer->fftSizeX, buffer->fftSizeY, projIndex,
366  transf, transfInv, space);
367  space->weight = (parent->do_weights) ? proj.weight() : 1.0;
368  }
369 
370  // Copy the projection to the center of the padded image
371  // and compute its Fourier transform, if requested
372  proj().setXmippOrigin();
373  const MultidimArray<double> &mProj = proj();
374  if (buffer->hasFFTs) {
375  localPaddedImgDouble.initZeros(buffer->paddedImgSize, buffer->paddedImgSize);
376  localPaddedImgDouble.setXmippOrigin();
377  FOR_ALL_ELEMENTS_IN_ARRAY2D(mProj)
378  A2D_ELEM(localPaddedImgDouble,i,j) = A2D_ELEM(mProj, i, j);
379  CenterFFT(localPaddedImgDouble, true);
380 
381  // Fourier transformer for the images
382  localTransformerImg.setReal(localPaddedImgDouble);
383  localTransformerImg.FourierTransform();
384  localTransformerImg.getFourierAlias(localPaddedFourier);
385  cropAndShift(localPaddedFourier, parent,
386  buffer, buffer->getNthItem(buffer->FFTs, projIndex));
387  } else {
388  localPaddedImgFloat.initZeros(parent->paddedImgSize, parent->paddedImgSize);
389  localPaddedImgFloat.setXmippOrigin();
390  FOR_ALL_ELEMENTS_IN_ARRAY2D(mProj)
391  A2D_ELEM(localPaddedImgFloat,i,j) = A2D_ELEM(mProj, i, j);
392  CenterFFT(localPaddedImgFloat, true);
393 
394  // add image at the end of the stack (that is already long enough)
395  memcpy(buffer->getNthItem(buffer->paddedImages, projIndex),
396  localPaddedImgFloat.data,
397  buffer->getPaddedImgByteSize());
398  }
399 
400  if (hasCTF) {
401  computeCTFCorrection(threadParams, objId[imgIndex],parent, buffer, projIndex);
402  }
403 
404  buffer->noOfImages++; // new image added to buffer
405  }

409  XMIPP_TRY // in case some method throws a xmipp exception
410 
411  RecFourierWorkThread* threadParams = (RecFourierWorkThread *) threadArgs;
412  ProgRecFourierGPU* parent = threadParams->parent;

421

427  parent->maxVolumeIndexX / 2, parent->maxVolumeIndexYZ, parent->paddedImgSize,
428  parent->bufferSize, (int)parent->R_repository.size());
429  pinMemory(threadParams->buffer);
430  // allocate GPU
431  allocateWrapper(threadParams->buffer, threadParams->gpuStream);
432 
433  // main work routine
434  int firstImageIndex = threadParams->startImageIndex;
435  int lastImageIndex = threadParams->endImageIndex;
436  for(int startLoadIndex = firstImageIndex;
437  startLoadIndex <= lastImageIndex;
438  startLoadIndex += parent->bufferSize) {
439  // load data
440  threadParams->startImageIndex = startLoadIndex;
441  threadParams->endImageIndex = std::min(lastImageIndex+1, startLoadIndex+parent->bufferSize);
442  prepareBuffer(threadParams, parent, hasCTF, objId);
443 
444  // send them for processing
445  if (threadParams->buffer->noOfImages > 0) { // it can happen that all images are skipped
446  processBufferGPU(parent->tempVolumeGPU, parent->tempWeightsGPU,
447  threadParams->buffer,
448  parent->blob.radius, parent->maxVolumeIndexYZ, parent->useFast,
449  parent->maxResolutionSqr,
450  threadParams->gpuStream,
451  parent->blob.order, parent->blob.alpha);
452  parent->logProgress(threadParams->buffer->noOfImages);
453  }
454  // once the processing finished, buffer can be reused
455  }
456 
457  // clean after itself
458  releaseWrapper(threadParams->gpuStream);
459  unpinMemory(threadParams->buffer);
460  threadParams->buffer->~RecFourierBufferData(); // <- explicit destructor call
461  free(rawMem);
462  threadParams->buffer = NULL;
463  threadParams->selFile = NULL;
464  barrier_wait( &parent->barrier );// notify that thread finished
465 
466  XMIPP_CATCH // catch possible exception
467  return NULL;
468 }

617 template<typename T>
618 T*** ProgRecFourierGPU::applyBlob(T***& input, float blobSize,
619  float* blobTableSqrt, float iDeltaSqrt) {
620  float blobSizeSqr = blobSize * blobSize;
621  int blob = floor(blobSize); // we are using integer coordinates, so we cannot hit anything further
622  T tmp;
623  T*** output;
624  // create new storage
625  allocate(output, maxVolumeIndexX+1, maxVolumeIndexYZ+1, maxVolumeIndexYZ+1);
626 
627  // traverse new storage
628  for (int i = 0; i <= maxVolumeIndexYZ; i++) {
629  for (int j = 0; j <= maxVolumeIndexYZ; j++) {
630  for (int k = 0; k <= maxVolumeIndexX; k++) {
631  // traverse input storage
632  tmp = (T) 0;
633  for (int z = std::max(0, i-blob); z <= std::min(maxVolumeIndexYZ, i+blob); z++) {
634  float dZSqr = (i - z) * (i - z);
635  for (int y = std::max(0, j-blob); y <= std::min(maxVolumeIndexYZ, j+blob); y++) {
636  float dYSqr = (j - y) * (j - y);
637  for (int x = std::max(0, k-blob); x <= std::min(maxVolumeIndexX, k+blob); x++) {
638  float dXSqr = (k - x) * (k - x);
639  float distanceSqr = dZSqr + dYSqr + dXSqr;
640  if (distanceSqr > blobSizeSqr) {
641  continue;
642  }
643  int aux = (int) ((distanceSqr * iDeltaSqrt + 0.5)); //Same as ROUND but avoid comparison
644  float tmpWeight = blobTableSqrt[aux];
645  tmp += tmpWeight * input[z][y][x];
646  }
647  }
648  }
649  output[i][j][k] = tmp;
650  }
651  }
652  }
653  // free original data
654  release(input, maxVolumeIndexYZ+1, maxVolumeIndexYZ+1);
655  return output;
