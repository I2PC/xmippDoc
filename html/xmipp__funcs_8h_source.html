331 void computeStats(const std::vector<T> &V, double& avg, double& stddev,
332  T& minval, T& maxval)
333 {
334  if (V.size()<= 0)
335  return;
336 
337  avg = 0;
338  stddev = 0;
339 
340  minval = maxval = V[0];
341 
342  size_t nmax=V.size();
343  const T* ptr=&V[0];
344  for(size_t n=0; n<nmax; ++n, ++ptr)
345  {
346  double val=*ptr;
347  avg += val;
348  stddev += val * val;
349 
350  if (val > maxval)
351  maxval = val;
352  else if (val < minval)
353  minval = val;
354  }
355  avg /= nmax;
356 
357  if (nmax > 1)
358  {
359  stddev = stddev / nmax - avg * avg;
360  stddev *= nmax / (nmax - 1);
361 
362  // Foreseeing numerical instabilities
363  stddev = sqrt(static_cast< double >(ABS(stddev)));
364  }
365  else
366  stddev = 0;
367 }
