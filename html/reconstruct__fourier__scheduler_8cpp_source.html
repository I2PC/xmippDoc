48 void task_data::recompute_metrics(const std::bitset<STARPU_NMAXWORKERS> &available_workers, unsigned sched_ctx_id, bool includeTransferTime) {
49  for (int workerid = 0; workerid < STARPU_NMAXWORKERS; ++workerid) {
50  best_implementation_by_worker[workerid] = -1;
51  best_implementation_time_by_worker[workerid] = MAX_TIME;
52 
53  if (!available_workers[workerid]) {
54  continue;
55  }
56 
57  unsigned mask;
58  if (!starpu_worker_can_execute_task_impl(workerid, task, &mask)) {
59  continue;
60  }
61 
62  unsigned pickedImpl = 0;
63  uint64_t pickedImplTime = std::numeric_limits<uint64_t>::max();
64  const auto perfArchetype = starpu_worker_get_perf_archtype(workerid, sched_ctx_id);
65 
66  for (unsigned impl = 0; impl < sizeof(unsigned) && mask != 0; ++impl) {
67  unsigned bit = 1u << impl;
68  if (mask & bit) {
69  mask &= ~bit;
70 
71  uint64_t implTime = sanitize_time(starpu_task_expected_length(task, perfArchetype, impl));
72  if (includeTransferTime) {
73  implTime += sanitize_time(starpu_task_expected_data_transfer_time_for(task, workerid));
74  }
75 
76  if (implTime < pickedImplTime) {
77  pickedImpl = impl;
78  pickedImplTime = implTime;
79  }
80  }
81  }
82 
83  best_implementation_by_worker[workerid] = pickedImpl;
84  best_implementation_time_by_worker[workerid] = pickedImplTime;
85  }
86 }

91 int task_data::find_best_worker(const std::bitset<STARPU_NMAXWORKERS> &available_workers) {
92  int pickedWorkerIds[STARPU_NMAXWORKERS];
93  unsigned pickedWorkerIdsCount = 0;
94  double pickedWorkerIdFinishTime = INFINITY;
95 
96  for (int workerId = 0; workerId < STARPU_NMAXWORKERS; ++workerId) {
97  if (!available_workers[workerId] || best_implementation_by_worker[workerId] == -1) {
98  continue;
99  }
100 
101  double finishTime = best_implementation_time_by_worker[workerId];
102  if (finishTime < pickedWorkerIdFinishTime) {
103  pickedWorkerIds[0] = workerId;
104  pickedWorkerIdsCount = 1;
105  pickedWorkerIdFinishTime = finishTime;
106  } else if (finishTime == pickedWorkerIdFinishTime) {
107  pickedWorkerIds[pickedWorkerIdsCount++] = workerId;
108  }
109  }
110 
111  assert(pickedWorkerIdsCount > 0 && "No worker to schedule task to");
112 
113  // It is not very important what this value is, it just should be somewhat unique for each run, so that most tasks
114  // are uniformly distributed across available workers.
115  static std::atomic<unsigned> roundRobinNext { 0 };
116  unsigned pickedWorkerIndex = roundRobinNext++ % pickedWorkerIdsCount;
117 
118  return pickedWorkerIds[pickedWorkerIndex];
119 }

150 static void rfs_add_workers(unsigned sched_ctx_id, int *workerids, unsigned nworkers) {
151  auto* data = (rfs_data*)starpu_sched_ctx_get_policy_data(sched_ctx_id);
152 
153  for (unsigned i = 0; i < nworkers; ++i) {
154  const int worker = workerids[i];
155  if (starpu_worker_is_combined_worker(worker)) {
156  continue;// Just in case
157  }
158  data->workers_mask.set(worker);
159  data->workers[worker].deviceKey = timing_device_key(starpu_worker_get_type(worker), starpu_worker_get_devid(worker));
160  data->workers[worker].busy = true;
161  RFS_LOG("%d", worker);
162  }
163 
164  // Recreate next-worker cycles
165  for (int worker = 0; worker < STARPU_NMAXWORKERS; ++worker) {
166  if (!data->workers_mask[worker])
167  continue;
168 
169  // (reset captains for next step)
170  data->workers[worker].combinedWorkerCaptain = -1;
171 
172  data->workers[worker].nextSimilarWorker = worker;
173  for (int nextWorkerOffset = 1; nextWorkerOffset < STARPU_NMAXWORKERS; ++nextWorkerOffset) {
174  int nextWorker = (worker + nextWorkerOffset) % STARPU_NMAXWORKERS;
175  if (!data->workers_mask[nextWorker] || data->workers[worker].deviceKey != data->workers[nextWorker].deviceKey)
176  continue;
177  data->workers[worker].nextSimilarWorker = nextWorker;
178  break;
179  }
180  }
181 
182  // Find combined workers
183  const unsigned workerCount = starpu_worker_get_count();
184  const unsigned combinedWorkerCount = starpu_combined_worker_get_count();
185  for (int combinedWorker = (int) workerCount; combinedWorker < workerCount + combinedWorkerCount; ++combinedWorker) {
186  int size = 0;
187  int* workers = nullptr;
188  starpu_combined_worker_get_description(combinedWorker, &size, &workers);
189  if (size <= 1) {
190  continue;
191  }
192 
193  bool valid = true;
194  starpu_worker_archtype combinedType = starpu_worker_get_type(workers[0]);
195  for (int i = 0; i < size; ++i) {
196  if (!data->workers_mask[workers[i]]) {
197  valid = false;
198  break;
199  }
200  auto type = starpu_worker_get_type(workers[i]);
201  if (type != combinedType) {
202  RFS_LOG("Mixed type worker not supported %d", combinedWorker);
203  valid = false;
204  break;
205  }
206  }
207 
208  if (!valid) {
209  continue;
210  }
211 
212  // We can use this combined worker!
213  bool used = false;
214  for (int i = 0; i < size; ++i) {
215  auto& wData = data->workers[workers[i]];
216  if (wData.combinedWorkerCaptain == -1) {
217  wData.combinedWorkerCaptain = combinedWorker;
218  used = true;
219  break;
220  }
221  }
222 
223  if (!used) {
224  RFS_LOG("Can't use combined worker %d, no available captain", combinedWorker);
225  continue;
226  }
227 
228  RFS_LOG("Registered combined worker %d (%d workers)", combinedWorker, size);
229  }
230 }

318 static int rfs_push_task(starpu_task * task) {
319  const unsigned sched_ctx_id = task->sched_ctx;
320  auto* data = (rfs_data*)starpu_sched_ctx_get_policy_data(sched_ctx_id);
321 
322  assert(task->sched_data);
323  auto* taskData = (task_data*)(task->sched_data);
324  taskData->stateOnPush();
325 
326  // Find best worker
327  const int32_t previousPickedWorkerId = taskData->picked_worker;
328  const uint32_t previousPickedWorkerTime = taskData->best_implementation_by_worker[previousPickedWorkerId];
329  const int32_t pickedWorkerId = taskData->best_worker(data->workers_mask, sched_ctx_id, true);
330  assert(pickedWorkerId >= 0);
331 
332  // If the preferred worker has changed after including the transfer time, remove the load estimate from the previous
333  // worker and add it to the new worker (done later to prevent duplicate mutex locking)
334  if (previousPickedWorkerId != pickedWorkerId) {
335  taskData->picked_worker = pickedWorkerId;
336 
337  if (previousPickedWorkerId >= 0) {
338  worker_data &prevPickedWorkerData = data->workers[previousPickedWorkerId];
339  prevPickedWorkerData.workerMutex.lock();
340  prevPickedWorkerData.total_load -= previousPickedWorkerTime;
341  prevPickedWorkerData.workerMutex.unlock();
342  }
343  }
344 
345  starpu_task_set_implementation(task, taskData->best_implementation_by_worker[pickedWorkerId]);
346 
347  starpu_worker_lock(pickedWorkerId);
348  worker_data& workerData = data->workers[pickedWorkerId];
349 
350  workerData.workerMutex.lock();
351  if (previousPickedWorkerId != pickedWorkerId) {
352  workerData.total_load += taskData->best_implementation_time_by_worker[pickedWorkerId];
353  }
354  workerData.queue.addLast(taskData);
355  auto queuedTaskCount = workerData.queue.getSize();
356  uint64_t totalLoad = workerData.total_load;
357  workerData.workerMutex.unlock();
358  starpu_push_task_end(task);
359 
360  if (queuedTaskCount <= PREFETCH_TASK_COUNT) {
361  starpu_idle_prefetch_task_input_for(task, pickedWorkerId);
362  }
363  starpu_wake_worker_locked(pickedWorkerId);
364  starpu_worker_unlock(pickedWorkerId);
365 
366 
367  RFS_LOG("Scheduled onto %d (%" PRIu64 " us)", pickedWorkerId, totalLoad);
368  return 0;
369 }

379 static starpu_task* steal_single_parallel_job(unsigned thiefWorkerId, unsigned combinedWorkerThiefId, unsigned thiefWorkerGroupSize,
380  unsigned victimWorkerId, worker_data &victimWorkerData,
381  uint32_t &workerJobLength) {
382  starpu_task* stolenTask = nullptr;
383  rfs::removeMatching(victimWorkerData.queue, [thiefWorkerId, combinedWorkerThiefId, thiefWorkerGroupSize, victimWorkerId, &victimWorkerData, &stolenTask, &workerJobLength](task_data* task){
384  if (task->task->cl == nullptr) {
385  // Synchronization task, better not
386  return rfs::RemoveResult::KeepContinue;
387  }
388 
389  if (task->best_implementation_by_worker[thiefWorkerId] == -1) {
390  // Can't run this, can't steal
391  return rfs::RemoveResult::KeepContinue;
392  }
393 
394  if (task->task->cl->type == starpu_codelet_type::STARPU_SEQ) {
395  // We want only parallel tasks
396  return rfs::RemoveResult::KeepContinue;
397  }
398 
399  if (task->task->cl->max_parallelism < thiefWorkerGroupSize) {
400  // This thief is too big for this job
401  return rfs::RemoveResult::KeepContinue;
402  }
403 
404  const float pessimismFactor = 1.5f;
405  const auto totalPessimisticThiefLoadAfterSteal = (uint32_t)(task->best_implementation_time_by_worker[thiefWorkerId] * pessimismFactor / (float) thiefWorkerGroupSize);
406  const uint32_t totalRealisticThiefLoadAfterSteal = task->best_implementation_time_by_worker[thiefWorkerId] / thiefWorkerGroupSize;
407  const uint64_t totalVictimLoadAfterSteal = victimWorkerData.total_load - victimWorkerData.current_load - task->best_implementation_time_by_worker[victimWorkerId];
408  if (totalPessimisticThiefLoadAfterSteal >= totalVictimLoadAfterSteal) {
409  // We can steal it, but it won't be helpful
410  return rfs::RemoveResult::KeepContinue;
411  }
412 
413  if (!starpu_combined_worker_can_execute_task(combinedWorkerThiefId, task->task, task->best_implementation_by_worker[thiefWorkerId])) {
414  // But it turns out, that we can't run it
415  return rfs::RemoveResult::KeepContinue;
416  }
417 
418  // It's payday fellas
419  workerJobLength = totalRealisticThiefLoadAfterSteal;
420  starpu_task_set_implementation(task->task, task->best_implementation_by_worker[thiefWorkerId]);
421  victimWorkerData.total_load -= task->best_implementation_time_by_worker[victimWorkerId];
422  stolenTask = task->task;
423  return rfs::RemoveResult::RemoveBreak;
424  });
425 
426  return stolenTask;
427 }

429 static unsigned steal_jobs_until_equilibrium(int thiefWorkerId, int victimWorkerId, worker_data &thiefWorkerData, worker_data &victimWorkerData) {
430  auto thiefIsCPU = starpu_worker_get_type(thiefWorkerId) == starpu_worker_archtype::STARPU_CPU_WORKER;
431  unsigned stolen = 0;
432 
433  rfs::removeMatching(victimWorkerData.queue, [&thiefWorkerData, thiefWorkerId, victimWorkerId, &victimWorkerData, &stolen, thiefIsCPU](task_data* task){
434  if (task->best_implementation_by_worker[thiefWorkerId] == -1) {
435  // Can't steal this
436  return rfs::RemoveResult::KeepContinue;
437  }
438 
439  if (thiefIsCPU && task->task->cl->type != starpu_codelet_type::STARPU_SEQ) {
440  // CPU's should not steal tasks that are parallel on CPU - not yet
441  return rfs::RemoveResult::KeepContinue;
442  }
443 
444  // When stealing, CPU workers ofter underestimate the difficulty of some GPU tasks.
445  // The pessimism factor adjusts for this fact by inflating the difficulty of the task.
446  const float pessimismFactor = 2;
447  const uint64_t totalPessimisticThiefLoadAfterSteal = thiefWorkerData.total_load + (uint64_t)(task->best_implementation_time_by_worker[thiefWorkerId] * pessimismFactor);
448  const uint64_t totalVictimLoadAfterSteal = victimWorkerData.total_load - victimWorkerData.current_load - task->best_implementation_time_by_worker[victimWorkerId];
449  if (totalPessimisticThiefLoadAfterSteal >= totalVictimLoadAfterSteal) {
450  // We can steal it, but it won't be helpful
451  return rfs::RemoveResult::KeepContinue;
452  }
453 
454  // It's payday fellas
455  starpu_task_set_implementation(task->task, task->best_implementation_by_worker[thiefWorkerId]);
456  thiefWorkerData.queue.addLast(task);
457  thiefWorkerData.total_load += task->best_implementation_time_by_worker[thiefWorkerId];
458  victimWorkerData.total_load -= task->best_implementation_time_by_worker[victimWorkerId];
459  stolen++;
460 
461  return rfs::RemoveResult::RemoveContinue;
462  });
463 
464 #if RFS_LOGGING
465  if (stolen > 0)
466  fprintf(stderr, "%d steals from %d: %d tasks\n", thiefWorkerId, victimWorkerId, stolen);
467 #endif
468 
469  return stolen;
470 }

472 static starpu_task * rfs_pop_task(unsigned sched_ctx_id) {
473  RFS_LOG("");
474 
475  auto* data = (rfs_data*)starpu_sched_ctx_get_policy_data(sched_ctx_id);
476  int workerId = starpu_worker_get_id_check();
477  assert(!starpu_worker_is_combined_worker(workerId));
478 
479  auto& ourWorkerData = data->workers[workerId];
480  std::lock_guard<std::mutex> lock(ourWorkerData.workerMutex);
481 
482  ourWorkerData.current_load = 0;
483 
484  // Refill queue only if needed - usually the queue already contains something that can be used right now
485  if (ourWorkerData.queue.empty()) {
486  // Fill up the queue
487 
488  // Tracking which workers were already tried to not bother them again.
489  // Considering all workers that are not present as already checked, to simplify logic down the line.
490  std::bitset<STARPU_NMAXWORKERS> checked = ~data->workers_mask;
491  checked[workerId] = true; // Don't check self
492 
493  // 1. Steal from similar workers
494 
495  int nextSimilarWorkerId = ourWorkerData.nextSimilarWorker;
496  while (nextSimilarWorkerId != workerId /* this is cyclic */) {
497  auto& nextSimilarWorkerData = data->workers[nextSimilarWorkerId];
498 
499  // Only try_lock instead of hard lock to prevent deadlocks
500  if (nextSimilarWorkerData.workerMutex.try_lock()) {
501  checked[nextSimilarWorkerId] = true;
502  steal_jobs_until_equilibrium(workerId, nextSimilarWorkerId, ourWorkerData, nextSimilarWorkerData);
503  nextSimilarWorkerData.workerMutex.unlock();
504  }
505 
506  if (!ourWorkerData.queue.empty()) {
507  goto stealingDone;
508  }
509 
510  nextSimilarWorkerId = nextSimilarWorkerData.nextSimilarWorker;
511  }
512 
513  // 2. Steal from all workers
514 
515  // Using own id as offset to distribute stealing in a round robin fashion
516  for (int stealI = 0; stealI < STARPU_NMAXWORKERS; ++stealI) {
517  const int victimWorker = (workerId + stealI) % STARPU_NMAXWORKERS;
518  if (checked[victimWorker]) {
519  continue;
520  }
521 
522  auto &victimWorkerData = data->workers[victimWorker];
523  if (victimWorkerData.workerMutex.try_lock()) {
524  steal_jobs_until_equilibrium(workerId, victimWorker, ourWorkerData, victimWorkerData);
525  victimWorkerData.workerMutex.unlock();
526 
527  if (!ourWorkerData.queue.empty()) {
528  goto stealingDone;
529  }
530  } // else: Lock failed, worker busy, spare them
531  }
532 
533  // 3. If combined worker captain and no sub-workers are busy, try to steal something combined.
534  if (ourWorkerData.combinedWorkerCaptain >= 0) {
535  // We are a captain! See if companions are free to do some work.
536  int combinedWorkerThiefId = ourWorkerData.combinedWorkerCaptain;
537  int groupSize = 0;
538  int* groupMembers = nullptr;
539  starpu_combined_worker_get_description(combinedWorkerThiefId, &groupSize, &groupMembers);
540 
541  bool allFree = true;
542  for (int i = 0; i < groupSize; ++i) {
543  if (groupMembers[i] != workerId && data->workers[groupMembers[i]].busy) {
544  allFree = false;
545  break;
546  }
547  }
548 
549  if (allFree) {
550  // There is a real chance of a parallel race regarding the busy flag, but it should not crash anything,
551  // so lets not think about that for now.
552 
553  // Pick a task to steal!
554  uint32_t stolenJobPerWorkerLength = 0;
555  starpu_task* stolenJob = nullptr;
556  // Same worker-checking logic as case 2
557  for (int stealI = 0; stealI < STARPU_NMAXWORKERS; ++stealI) {
558  const int victimWorker = (workerId + stealI) % STARPU_NMAXWORKERS;
559  if (checked[victimWorker]) {
560  continue;
561  }
562 
563  auto &victimWorkerData = data->workers[victimWorker];
564  if (victimWorkerData.workerMutex.try_lock()) {
565  stolenJob = steal_single_parallel_job(workerId, combinedWorkerThiefId, groupSize, victimWorker, victimWorkerData,
566  stolenJobPerWorkerLength);
567  victimWorkerData.workerMutex.unlock();
568 
569  break;
570  } // else: Lock failed, worker busy, spare them
571  }
572 
573  if (stolenJob != nullptr) {
574  // Got something, use it!
575  starpu_parallel_task_barrier_init(stolenJob, combinedWorkerThiefId);
576 
577  for (int i = 0; i < groupSize; i++) {
578  const int groupMember = groupMembers[i];
579  data->workers[groupMember].busy = true;
580  // Not setting current_load in case of race condition, in which case it would be wrong.
581  // But not setting it up is probably not a big deal.
582 
583  if (groupMember == workerId) {
584  continue;
585  }
586 
587  struct starpu_task *alias = starpu_task_dup(stolenJob);
