403 // Check whether a CTF is anisotropic
404 bool ProgCtfGroup::isIsotropic(CTFDescription &ctf)
405 {
406  double cosp, sinp, ctfp, diff;
407  Matrix1D<double> freq(2);
408 
409  cosp = COSD(ctf.azimuthal_angle);
410  sinp = SIND(ctf.azimuthal_angle);
411 
412  for (double digres = 0; digres < resol_error; digres+= 0.001)
413  {
414  XX(freq) = cosp * digres;
415  YY(freq) = sinp * digres;
416  digfreq2contfreq(freq, freq, pixel_size);
417  ctf.precomputeValues(XX(freq), YY(freq));
418  ctfp = ctf.getValueAt();
419  ctf.precomputeValues(YY(freq), XX(freq));
420  diff = ABS(ctfp - ctf.getValueAt());
421  if (diff > max_error)
422  {
423  std::cout<<" Anisotropy!"<<digres<<" "<<max_error<<" "<<diff<<" "<<ctfp
424  <<" "<<ctf.getValueAt()<<std::endl;
425  return false;
426  }
427  }
428  return true;
