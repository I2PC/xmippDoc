350 /* Produce all images ------------------------------------------------------ */
351 void Micrograph::produce_all_images(int label, double minCost,
352  const FileName &fn_rootIn, const FileName &fn_image, double ang,
353  double tilt, double psi, bool rmStack, bool fillBorders,
354  bool extractNoise, int Nnoise)
355 {
356  MetaData SF;
357  Image<double> I;
358  Micrograph *M;
359 
360  // Set Source image
361  if (fn_image == "")
362  M = this;
363  else
364  {
365  M = new Micrograph;
366  M->open_micrograph(fn_image/*, swapbyte*/);
367  M->set_window_size(X_window_size, Y_window_size);
368  M->set_transmitance_flag(compute_transmitance);
369  M->set_inverse_flag(compute_inverse);
370  }
371 
372  // Set scale for particles
373  int MXdim, MYdim, thisXdim, thisYdim;
374  M->size(MXdim, MYdim);
375  this->size(thisXdim, thisYdim);
376  double scaleX = (double) MXdim / thisXdim;
377  double scaleY = (double) MYdim / thisYdim;
378 
379  // Compute max and minimum if compute_transmitance
380  // or compute_inverse flags are ON
381  double Dmax=0., Dmin=0.;
382  if (compute_transmitance || compute_inverse)
383  {
384  (*this).computeDoubleMinMax(Dmin, Dmax);
385 
386  if (compute_transmitance)
387  {
388  if (Dmin > 1)
389  Dmin = log10(Dmin);
390  if (Dmax > 1)
391  Dmax = log10(Dmax);
392  }
393  }
394  // Scissor all particles
395  if (ang != 0)
396  std::cout << "Angle from Y axis to tilt axis " << ang << std::endl
397  << " applying appropriate rotation\n";
398  int nmax = ParticleNo();
399  int nparticles = nmax;
400  if (extractNoise && Nnoise>0)
401  nmax=Nnoise;
402  FileName fn_aux;
403  FileName _ext = fn_rootIn.getFileFormat();
404  FileName fn_out;
405  FileName fn_root = fn_rootIn.removeFileFormat().removeLastExtension();
406  if (fn_rootIn.hasStackExtension())
407  fn_out=fn_root.addExtension(_ext);
408  else
409  fn_out=fn_rootIn.addExtension("stk");
410 
411  if (rmStack)
412  fn_out.deleteFile();
413  size_t ii = 0;
414  size_t id;
415 
416  Particle_coords Pnoise;
417  Pnoise.valid = true;
418  Pnoise.label = 0;
419  Pnoise.cost = 1;
420  Pnoise.scoreVar = -1;
421  Pnoise.scoreGini = -1;
422 
423  int minNoiseDistance=Y_window_size/2;
424  std::vector<Particle_coords> noiseCoords;
425 
426  for (int n = 0; n < nmax; n++)
427  {
428  if (coords[n].valid && coords[n].cost > minCost && coords[n].label == label)
429  {
430  fn_aux.compose(++ii, fn_out);
431  id = SF.addObject();
432  // If the ctfRow was set, copy the info to images metadata
433  if (ctfRow.containsLabel(MDL_CTF_DEFOCUSU))
434  SF.setRow(ctfRow, id);
435  SF.setValue(MDL_IMAGE, fn_aux, id);
436  SF.setValue(MDL_MICROGRAPH, M->fn_micrograph, id);
437  SF.setValue(MDL_XCOOR, coords[n].X, id);
438  SF.setValue(MDL_YCOOR, coords[n].Y, id);
439  if (coords[n].scoreVar>0)
440  {
441  SF.setValue(MDL_SCORE_BY_VAR, coords[n].scoreVar, id);
442  SF.setValue(MDL_SCORE_BY_GINI, coords[n].scoreGini, id);
443  }
444  bool t=false;
445  if (extractNoise)
446  {
447  // Look for coordinate that is away from the rest of coordinates
448  bool found=false;
449  while (!found)
450  {
451  Pnoise.X=int(rnd_unif(X_window_size,thisXdim-X_window_size));
452  Pnoise.Y=int(rnd_unif(Y_window_size,thisYdim-Y_window_size));
453  found=true;
454  for (int nn=0; nn<nparticles; nn++)
455  {
456  if (std::abs(Pnoise.X-coords[nn].X)<minNoiseDistance && std::abs(Pnoise.Y-coords[nn].Y)<minNoiseDistance)
457  {
458  found=false;
459  break;
460  }
461  }
462  }
463  noiseCoords.push_back(Pnoise);
464  t = M->scissor(Pnoise, I(), Dmin, Dmax, scaleX, scaleY, false, fillBorders);
465  }
466  else
467  t = M->scissor(coords[n], I(), Dmin, Dmax, scaleX, scaleY, false, fillBorders);
468  if (!t)
469  {
470  std::cout << "Particle " << fn_aux
471  << " is very near the border, "
472  << "corresponding image is set to blank\n";
473  SF.setValue(MDL_ENABLED, -1, id);
474  }
475  else
476  SF.setValue(MDL_ENABLED, 1, id);
477  // if (ang!=0) I().rotate(-ang);
478  I.write(fn_out, ii, true, WRITE_APPEND);
479  }
480  }
481  SF.write(fn_out.withoutExtension() + ".xmd");
482 
483 
484  // Free source image??
485  if (fn_image != "")
486  {
487  M->close_micrograph();
488  delete M;
489  }
490 
491  if (extractNoise)
492  {
493  coords.clear();
494  coords=noiseCoords;
495  }
496 }

776 /* Compute tilting angle --------------------------------------------------- */
777 void TiltPairAligner::computeGamma()
778 {
779 #define TRIANGLE_NO 15000
780 #define MIN_AREA 15
781 #define MAX_AREA 250000
782  gamma = 0;
783  Matrix1D<int> iju(2), iku(2), ijt(2), ikt(2); // From i to j in untilted
784  // From i to k in untilted
785  // From i to j in tilted
786  // From i to k in tilted
787  int triang = 0; // Number of triangles considered
788  int i, j, k, counter1;
789  counter1 = 0;
790  randomize_random_generator();
791  long noCombinations;
792  noCombinations = Nu * (Nu - 1) * (Nu - 2) / 6;
793  while (triang < TRIANGLE_NO && counter1 < noCombinations)
794  {
795  counter1++;
796  i = (int)round(rnd_unif(0, Nu - 1));
797  j = (int)round(rnd_unif(0, Nu - 1));
798  k = (int)round(rnd_unif(0, Nu - 1));
799 
800  // Compute area of triangle in untilted micrograph
801  VECTOR_R2(iju, coordU[j] - coordU[i], coordU[j+1] - coordU[i+1]);
802  VECTOR_R2(iku, coordU[k] - coordU[i], coordU[k+1] - coordU[i+1]);
803  double untilted_area = fabs(dotProduct(iju, iku)/*/2*/);
804  if (untilted_area < MIN_AREA
805  )
806  continue; // For numerical stability
807 
808  // Compute area of the same triangle in the tilted micrograph
809  VECTOR_R2(ijt, coordT[j] - coordT[i], coordT[j+1] - coordT[i+1]);
810  VECTOR_R2(ikt, coordT[k] - coordT[i], coordT[k+1] - coordT[i+1]);
811  double tilted_area = fabs(dotProduct(ijt, ikt)/*/2*/);
812  if (tilted_area < MIN_AREA
813  )
814  continue; // For numerical stability
815  if (tilted_area > MAX_AREA
816  )
817  continue; // micrograph are not perfect
818  // sheets so avoid
819  // very far away particles
820 
821  // Now we know that tilted_area=untilted_area*cos(gamma)
822  if (tilted_area > untilted_area)
823  continue; // There are user errors
824  // In the point selection
825  gamma += acos(tilted_area / untilted_area);
826  triang++;
827  }
828  if (0 == triang) {
829  REPORT_ERROR(ERR_NUMERICAL, "triang is zero (0), which would lead to division by zero");
830  }
831  gamma /= triang;
832  gamma = RAD2DEG(gamma);
833  if (triang < 100)
834  std::cout << "Not many particles, tilt angle may not be accurate"
835  << std::endl;
836 }

839 double matrix_fitness(double *p, void *prm)
840 {
841  TiltPairAligner *aligner = (TiltPairAligner *) prm;
842  Euler_angles2matrix(-p[1], p[3], p[2], aligner->pair_E);
843  double retval = 0;

845  for (int j = 0; j < 2; j++)

848  MAT_ELEM(aligner->pair_E,i, j)
849  - MAT_ELEM(aligner->Put, i, j));

851  }

856 void TiltPairAligner::computeAngles(double &ualpha, double &talpha, double &ogamma)
857 {
858  alpha_u = alpha_t = 0;
859  Matrix1D<double> angles(3);
860  angles.initZeros();
861  double fitness;
862  int iter;
863 
864  // Coarse search
865  double *aux = angles.adaptForNumericalRecipes();
866  double best_alpha_u = 0, best_alpha_t = 0, best_fit = 1e8;
867  aux[3] = gamma;
868  for (aux[1] = 0; aux[1] < 180; aux[1] += 5)
869  for (aux[2] = 0; aux[2] < 180; aux[2] += 5)
870  {
871  double fit = matrix_fitness(aux, this);
872  if (fit < best_fit)
873  {
874  best_fit = fit;
875  best_alpha_u = aux[1];
876  best_alpha_t = aux[2];
877  }
878  }
879  angles.killAdaptationForNumericalRecipes(aux);
880  angles(0) = best_alpha_u;
881  angles(1) = best_alpha_t;
882  angles(2) = gamma;
883 
884  // Fine search
885  Matrix1D<double> steps(3);
886  steps.initConstant(1);
887  powellOptimizer(angles, 1, 3, &matrix_fitness, this, 0.001, fitness, iter,
888  steps, false);
889  alpha_u = angles(0);
890  alpha_t = angles(1);
891  gamma = angles(2);
892 
893  ualpha = alpha_u;
894  talpha = alpha_t;
895  ogamma = gamma;
896 }
