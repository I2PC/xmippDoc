125  mask.resize(Ydim, Xdim);

128  int NInsideMask = (int)(XSIZE(mask) * YSIZE(mask) - mask.sum());
129 
130  FourierFilter Filter;
131  Filter.w1 = -1;
132  if (parent->lpf > 0 && parent->hpf > 0)
133  {
134  Filter.FilterBand = BANDPASS;

145  Filter.FilterBand = HIGHPASS;
146  Filter.w1 = parent->hpf;
147  }
148  Filter.raised_w = Filter.w1 / 3;

150  int ii = 0;
151  bool first = true;
152  Image<double> I;
153  FileName fnImg;

155  FourierTransformer transformer;

162  I.readApplyGeo(SFi, __iter.objId);
163  I().setXmippOrigin();
164  MultidimArray<double> &mI = I();
165 
166  // Bandpass filter images
167  if (Filter.w1 > 0)
168  {
169  if (first)
170  {
171  Filter.generateMask(mI);
172  first = false;
173  }
174  Filter.applyMaskSpace(mI);
175  }
176 
177  // Cut the image outside the largest circle
178  // And compute the DC value inside the mask
179  double meanInside = 0;
180  FOR_ALL_ELEMENTS_IN_ARRAY2D(mask)
181  if (A2D_ELEM(mask,i,j))A2D_ELEM(mI,i,j)=0;
182  else
183  meanInside+=A2D_ELEM(mI,i,j);
184  meanInside /= NInsideMask;
185 
186  // Subtract the mean inside the circle
187  FOR_ALL_ELEMENTS_IN_ARRAY2D(mask)
188  if (!A2D_ELEM(mask,i,j))
189  A2D_ELEM(mI,i,j)-=meanInside;
190 
191  // Compute the Radon transform
192  Radon_Transform(I(), parent->stepAng, RT);
193 
194  // Normalize each line in the Radon Transform so that the
195  // multiplication of any two lines is actually their correlation index
196  RTFourier.resize(YSIZE(RT), XSIZE(mlineiFourier));
197  for (size_t i = 0; i < YSIZE(RT); i++) {
198  memcpy(&(DIRECT_A1D_ELEM(linei,0)),&DIRECT_A2D_ELEM(RT,i,0),XSIZE(linei)*sizeof(double));
199  linei.statisticsAdjust(0.,1.);
200  transformer.FourierTransform();
201  transformer.fReal->initZeros();
202  transformer.inverseFourierTransform();
203  memcpy(&DIRECT_A2D_ELEM(RTFourier,i,0),&(DIRECT_A1D_ELEM(mlineiFourier,0)),XSIZE(mlineiFourier)*sizeof(std::complex<double>));
204 
205  memcpy(&DIRECT_A2D_ELEM(RT,i,0),&(DIRECT_A1D_ELEM(linei,0)),XSIZE(linei)*sizeof(double));
206  }
207 
208  (*(master->blockRTFs))[ii] = RTFourier;
209  (*(master->blockRTs))[ii] = RT;
210  }
211  ii++;
212  }
213  return NULL;

321  int blockIsize = master->RTFsi->size();
322  int blockJsize = master->RTFsj->size();
323  FourierTransformer transformer;

325  linei.resize(parent->Xdim);
326  transformer.setReal(linei);
327  for (int i = 0; i < blockIsize; i++) {
328  long int ii = parent->Nblock * master->i + i;
329  for (int j = 0; j < blockJsize; j++) {

331  long int jj = parent->Nblock * master->j + j;
332  if (ii >= jj)

334  if ((ii * blockJsize + jj + 1) % parent->Nthr != master->myThreadID)
335  continue;
336 
337  // Effectively compare the two images
338  long int idx_ij = ii * parent->Nimg + jj;
339  commonLineTwoImages(
340  *(master->RTFsi), *(master->RTsi), i,
341  *(master->RTFsj), *(master->RTsj), j,

489 {

491  int nmax=Nimg*(Nimg-1)/2;
492  solver.A.resize(nmax,Nimg*2);

494  solver.b.resize(nmax);

501  double sini, cosi, sinj, cosj;
502  sincos(DEG2RAD(cl.angi),&sini,&cosi);
503  sincos(DEG2RAD(cl.angj),&sinj,&cosj);
504  int idxi=2*i;
505  int idxj=2*j;
506  MAT_ELEM(solver.A,n,idxi)=cosi;
507  MAT_ELEM(solver.A,n,idxi+1)=sini;
508  MAT_ELEM(solver.A,n,idxj)=-cosj;
509  MAT_ELEM(solver.A,n,idxj+1)=-sinj;
510  VEC_ELEM(solver.b,n)=cl.jmax;
511  VEC_ELEM(solver.w,n)=cl.percentile;
512  }
513  }
514 
515  const double maxShiftError=1;
516  ransacWeightedLeastSquares(solver,shift,maxShiftError,10000,outlierFraction,Nthr);
