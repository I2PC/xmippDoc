202  header->tags.setValue(MDL_DM3_TAGNAME, stagName, id);
203 
204  /* Reading tags ===================================================================*/
205  if (idTag == 20) // Tag directory
206  {
207  unsigned char dummy;
208  int nTags;
209  xmippFREAD(&dummy,sizeof(unsigned char),1,fimg,false); // 1 = sorted (normally = 1)
210  xmippFREAD(&dummy,sizeof(unsigned char),1,fimg,false); // 0 = closed, 1 = open (normally = 0)
211  xmippFREAD(&nTags,sizeof(int),1,fimg,isLE); // number of tags in tag directory
212 
213

215  header->tags.setValue(MDL_DM3_SIZE, nTags, id);
216 
217  parentId = nodeId;
218  for (int n=1;n<=nTags;n++) // Rest of directories
219  {
220  readTagDM3(fimg, header, parentId, nodeId, isLE, swap);
221  }
222  return 0;
223  }
224  else if (idTag == 21) // Tag
225  {
226  unsigned int nnum;
227  char buf[4]; // to read %%%% symbols
228 
229  xmippFREAD(&buf,1,4,fimg,false); // To read %%%% symbols
230  xmippFREAD(&nnum,sizeof(unsigned int),1,fimg,isLE); // Size of info array
231 
232  int * info;

234  xmippFREAD(info,sizeof(unsigned int),nnum,fimg,isLE); // Reading of Info
235 
236  /* Tag classification =======================================*/
237 
238  if (nnum == 1) // Single entry tag
239  {
240  header->tags.setValue(MDL_DM3_TAGCLASS,(std::string) "Single", id);
241 
242  double tagValue = 0;
243 
244  FREADTagValueDM3(&tagValue,info[0],1,fimg,swap);
245 
246  std::vector<double> vtagValue(1);
247  vtagValue.assign(1,tagValue);
248 
249  header->tags.setValue(MDL_DM3_NUMBER_TYPE, info[0], id);
250  header->tags.setValue(MDL_DM3_VALUE, vtagValue, id);
251  delete []info;
252  return tagValue;
253  }
254  else if(nnum == 3 && info[0]==20) // Tag array
255  {
256  /*nnum = 3
257  info(0) = 20
258  info(1) = number type for all values
259  info(2) = info(nnum) = size of array*/
260 
261  header->tags.setValue(MDL_DM3_TAGCLASS,(std::string) "Array", id);
262 
263  header->tags.setValue(MDL_DM3_NUMBER_TYPE, info[1], id);
264  header->tags.setValue(MDL_DM3_SIZE, info[nnum-1], id);
265  std::vector<double> vtagValue(1);
266  vtagValue.assign(1,(double) ftell(fimg));
267 
268  header->tags.setValue(MDL_DM3_VALUE, vtagValue, id);
269 
270  // Jump the array values
271  int k;
272  if(info[1] == 2 || info[1] == 4)
273  k = 2;
274  else if(info[1] == 3 || info[1] == 5 || info[1] == 6)
275  k = 4;
276  else if(info[1] == 10 )
277  k = 1;
278  else
279  REPORT_ERROR(ERR_ARG_INCORRECT,formatString("unsupported value info[1]=%d, read at %d",info[1],ftell(fimg)));
280 
281  fseek( fimg, ftell(fimg)+(info[nnum-1])*k , SEEK_SET );
282  delete []info;
283  return 0;
284  }
285  else if (info[0]==20 && info[1] == 15) // Tag Group array
286  {
287  /*nnum = size of array
288  info(0) = 20 (array)
289  info(1) = 15 (group)
290  info(2) = 0 (always 0)
291  info(3) = number of values in group
292  info(2*i+3) = number type for value i
293  info(nnum) = size of info array*/
294 
295  header->tags.setValue(MDL_DM3_TAGCLASS, (std::string) "GroupArray", id);
296  header->tags.setValue(MDL_DM3_SIZE, info[3], id);
297  int nBytes=0, k;
298  double fieldValue;
299  for (int n=1;n<=info[3];n++)
300  {
301  fieldValue=0;
302  FREADTagValueDM3(&fieldValue,info[3+2*n],1,fimg,swap);
303 
304  if(info[3+2*n] == 2 || info[3+2*n] == 4)
305  k = 2;
306  else if(info[3+2*n] == 3 || info[3+2*n] == 5)
307  k = 4;
308  else if(info[3+2*n] == 10 )
309  k = 1;
310  nBytes+=k;
311  }
312 
313  // Jump the array values
314  fseek( fimg, ftell(fimg)+(info[nnum-1]-1)*nBytes , SEEK_SET );
315  delete []info;
316  return 0;
317  }
318  else if (info[0] == 15) // Tag Group (struct)
319  {
320  /* info[1] = length group name (always =0)
321  info[2] = number of entries in group */
322  /*nnum = size of info array
323  info(1) = 0fh
324  info(3) = number of values in group
325  info(2*i+3) = number type for value i
326  Other info entries are always zero*/
327 
328  header->tags.setValue(MDL_DM3_TAGCLASS, (std::string) "Group", id);
329  header->tags.setValue(MDL_DM3_SIZE, info[2], id);
330  std::vector<double> vtagValue(info[2]);
331  for (int n=1;n<=info[2];n++)
332  {
333  double fieldValue=0;
334  FREADTagValueDM3(&fieldValue,info[2+2*n],1,fimg,swap);
335  vtagValue.assign(n,fieldValue);
336  }
337  header->tags.setValue(MDL_DM3_VALUE, vtagValue, id);
338  delete []info;
339  return 0;
340  }
341  delete []info;
342  }
343  return 0;
344 }
