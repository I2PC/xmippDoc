31 double orthogonalMatchingPursuit(const Matrix1D<double> &x,
32  const Matrix2D<double> &D, int S, Matrix1D<double> &alpha)
33 {
34  // Compute approximation error
35  double approximationError=0;
36  FOR_ALL_ELEMENTS_IN_MATRIX1D(x)
37  approximationError+=x(i)*x(i);
38  double errorLimit=approximationError*1e-10;
39 
40  // Initialize the number of available atoms
41  int N=D.Ydim(); // Dimension of each atom
42  if (0 == N) REPORT_ERROR(ERR_PARAM_INCORRECT, "Ydim of each atom must not be zero (0)");
43  int K=D.Xdim(); // Number of atoms in the dictionary
44  Matrix1D<int> availableAtoms;
45  availableAtoms.initZeros(K);
46  availableAtoms.initConstant(1);
47 
48  // Compute the projection of x onto each of the atoms and look for the
49  // maximum
50  Matrix1D<double> c, w, e, u;
51  c.initZeros(K);
52  e.initZeros(K); e.initConstant(1);
53  u.initZeros(K); u.initConstant(1);
54 
55  // Compute dot product between x and di
56  int km=0;
57  double ckm=-1;
58  for (int k=0; k<K; k++)
59  {
60  for (int j=0; j<N; j++)
61  c(k)+=D(j,k)*x(j);
62 
63  // Check if this is the maximum
64  if (ABS(c(k))>ckm)
65  {
66  km=k;
67  ckm=ABS(c(k));
68  }
69  }
70 
71  // Delete Js from the available indexes and update the approximation error
72  availableAtoms(km)=0;
73  approximationError-=ckm*ckm;
74 
75  // Annotate Jk
76  std::vector<int> J;
77  int s=0;
78  J.push_back(km);
79 
80  // Build the R matrix
81  Matrix2D<double> R(S,K);
82  R(0,km)=u(km);
83 
84  while (s<S-1 && approximationError>errorLimit)
85  {
86  // Update all the rest of R except Js
87  ckm=0;
88  int nextKm=-1;
89  for (int k=0; k<K; k++)
90  {
91  // If already used, skip it
92  if (!availableAtoms(k)) continue;
93 
94  // Compute the product between the atoms km and k
95  double aux=0;
96  for (int j=0; j<N; j++)
97  aux+=D(j,km)*D(j,k);
98  R(s,k)=aux;
99 
100  // Readjust R
101  for (int n=0; n<s; n++)
102  R(s,k)-=R(n,km)*R(n,k);
103  if (u(km)>XMIPP_EQUAL_ACCURACY)
104  R(s,k)/=u(km);
105 
106  // Update the rest of variables
107  c(k)=c(k)*u(k)-c(km)*R(s,k);
108  e(k)-=R(s,k)*R(s,k);
109  u(k)=sqrt(ABS(e(k)));
110  if (u(k)!=0) c(k)/=u(k);
111 
112  // Check if this is the best c(k)
113  if (ABS(c(k))>ckm)
114  {
115  nextKm=k;
116  ckm=ABS(c(k));
117  }
118  }
119 
120  // Delete km from the available indexes and update the approximation error
121  km=nextKm;
122  J.push_back(km);
123  availableAtoms(km)=0;
124  s++;
125  R(s,km)=u(km);
126  approximationError-=ckm*ckm;
127  }
128 
129  // Perform backsubstitution
130  alpha.initZeros(K);
131  for (int k=s; k>=0; k--)
132  {
133  int Jk=J[k];
134  for (int n=s; n>=k+1; n--)
135  c(Jk)-=R(k,J[n])*c(J[n]);
136  if (R(k,Jk)!=0) c(Jk)/=R(k,Jk);
137  alpha(Jk)=c(Jk);
138  }
139 
140  return sqrt(ABS(approximationError));
141 }

144 /* Lasso (shooting) */
145 /* ------------------------------------------------------------------------- */
146 double lasso(const Matrix1D<double> &x,
147  const Matrix2D<double> &D,
148  const Matrix2D<double> &DtD,
149  const Matrix2D<double> &DtDlambdaInv,
150  double lambda, Matrix1D<double> &alpha,
151  const int maxIter, const double tol)
152 {
153  int K=D.Xdim();
154 
155  // Compute the ridge least squares solution
156  // Compute D^t*x
157  Matrix1D<double> Dtx(K);
158  FOR_ALL_ELEMENTS_IN_MATRIX1D(Dtx)
159  // Compute the dot product of the i-th column of D and x
160  for (size_t k=0; k<D.Ydim(); k++)
161  Dtx(i)+= D(k,i)*x(k);
162 
163  // Now multiply DtdLambdaInv * D^t * x
164  alpha.initZeros(K);
165  FOR_ALL_ELEMENTS_IN_MATRIX2D(DtDlambdaInv)
166  alpha(i)+=DtDlambdaInv(i,j)*Dtx(j);
167 
168  // Iterate over alpha
169  Matrix1D<double> alphaOld;
170  int iter=0;
171  bool finished=false;
172  while (!finished)
173  {
174  // Keep the current alpha
175  alphaOld=alpha;
176 
177  // Update alpha
178  FOR_ALL_ELEMENTS_IN_MATRIX1D(alpha)
179  {
180  double S=-Dtx(i);
181  for (size_t j=0; j<alpha.size(); j++)
182  if (i!=j) S+=DtD(i,j)*alpha(j);
183  if (fabs(S)<lambda)
184  alpha(i)=0;
185  else if (S>lambda)
186  alpha(i)=(lambda-S)/DtD(i,i);
187  else
188  alpha(i)=(-lambda-S)/DtD(i,i);
189  }
190 
191  // Prepare for next iteration
192  iter++;
193  double normDiff=0, normAlpha=0;
194  FOR_ALL_ELEMENTS_IN_MATRIX1D(alpha)
195  {
196  double diff=alpha(i)-alphaOld(i);
197  normDiff+=diff*diff;
198  normAlpha+=alpha(i)*alpha(i);
199  }
200  finished=(iter>=maxIter) || normDiff/normAlpha<tol;
201  }
202 
203  // Compute the approximation error
204  Matrix1D<double> xp(x.size());
205  for (size_t j=0; j<D.Xdim(); j++)
206  if (alpha(j)!=0)
207  for (size_t i=0; i<D.Ydim(); i++)
208  xp(i)+=D(i,j)*alpha(j);
209  double approximationError=0;
210  FOR_ALL_ELEMENTS_IN_MATRIX1D(xp)
211  {
212  double diff=xp(i)-x(i);
213  approximationError+=diff*diff;
214  }
215 
216  return sqrt(ABS(approximationError));
217 }
