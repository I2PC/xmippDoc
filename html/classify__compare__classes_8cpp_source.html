60 void ProgCompareClass::run()

79 
80  // Read the size of the individual classes
81  MetaData MDclass1, MDclass2;
82  FOR_ALL_ELEMENTS_IN_MATRIX1D(MD1classSize)
83  {
84  MDclass1.read(formatString("class%06d_images@%s",ref1[i],fnClass1.c_str()));
85  VEC_ELEM(MD1classSize,i)=MDclass1.size();
86  }
87 
88  FOR_ALL_ELEMENTS_IN_MATRIX1D(MD2classSize)

91  VEC_ELEM(MD2classSize,i)=MDclass2.size();

94  // Now compare the two classifications
95  for (size_t i=0; i<MAT_YSIZE(comparisonMatrix); i++)

97  MDclass1.read(formatString("class%06d_images@%s",ref1[i],fnClass1.c_str()));

106  // Report analysis

123  if (MAT_ELEM(comparisonMatrix,i,j)>0)

125  }

127  for (size_t j=0; j<MAT_XSIZE(comparisonMatrix); j++)
128  {
