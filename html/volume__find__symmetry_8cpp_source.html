326  best_rot=p(0);
327  best_z=p(1);
328  best_corr=-fitness;
329

353  /* Evaluate symmetry ------------------------------------------------------- */
354  double evaluateSymmetry(double *p)
355  {
356  MultidimArray<double> volume_sym, volume_aux;
357  const MultidimArray<double> &mVolume=volume();
358  Matrix2D<double> sym_matrix;
359  if (!helical && !helicalDihedral)
360  {
361  Matrix2D<double> Euler;
362  Matrix1D<double> sym_axis;
363 
364  double rot=p[1];
365  double tilt=p[2];
366 
367  // Compute symmetry axis
368  Euler_angles2matrix(rot, tilt, 0., Euler);
369  Euler.getRow(2, sym_axis);
370  sym_axis.selfTranspose();
371 
372  // Symmetrize along this axis
373  volume_sym = volume();
374  for (int n = 1; n < rot_sym; n++)
375  {
376  rotation3DMatrix(360.0 / rot_sym * n, sym_axis, sym_matrix);
377  if (useSplines)
378  applyGeometry(BSPLINE3, volume_aux, mVolume, sym_matrix,
379  IS_NOT_INV, DONT_WRAP);
380  else
381  applyGeometry(LINEAR, volume_aux, mVolume, sym_matrix,
382  IS_NOT_INV, DONT_WRAP);
383  volume_sym += volume_aux;
384  }
385  return -correlationIndex(mVolume, volume_sym, &mask_prm.get_binary_mask());
386  }
387  else
388  {
389  double rotHelical=p[1];
390  double zHelical=p[2];
391  if (zHelical<0 || zHelical>ZSIZE(mVolume)*0.4)
392  return 1e38;
393  if (zHelical<z0 || zHelical>zF || rotHelical<rot0 || rotHelical>rotF)
394  return 1e38;
395  symmetry_Helical(volume_sym, mVolume, zHelical, DEG2RAD(rotHelical), 0, &mask_prm.get_binary_mask(), helicalDihedral, heightFraction);
396  double corr=correlationIndex(mVolume, volume_sym, &mask_prm.get_binary_mask());
397 //#define DEBUG
398 #ifdef DEBUG
399 
400  Image<double> save;
401  save()=mVolume;
402  save.write("PPPvol.vol");
403  save()=volume_sym;
404  save.write("PPPvolsym.vol");
405  typeCast(mask_prm.get_binary_mask(),save());
406  save.write("PPPmask.vol");
407  std::cout << p[1] << " " << p[2] << " " << p[3] << " -> " << corr << std::endl;
408  char c;
409  std::cin >> c;
410 #endif
411  return -corr;
412  }
