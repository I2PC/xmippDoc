235  void normalize_all_vectors()
236  {
237  FOR_ALL_ELEMENTS_IN_MULTIDIM_VECTORIAL_MATRIX3D(*this)
238  {
239  double mod = sqrt(
240  DIRECT_MULTIDIM_ELEM(__X,n) * DIRECT_MULTIDIM_ELEM(__X,n) +
241  DIRECT_MULTIDIM_ELEM(__Y,n) * DIRECT_MULTIDIM_ELEM(__Y,n) +
242  DIRECT_MULTIDIM_ELEM(__Z,n) * DIRECT_MULTIDIM_ELEM(__Z,n));
243  DIRECT_MULTIDIM_ELEM(__X,n) /= mod;
244  DIRECT_MULTIDIM_ELEM(__Y,n) /= mod;
245  DIRECT_MULTIDIM_ELEM(__Z,n) /= mod;
