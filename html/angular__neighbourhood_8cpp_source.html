77 // Compute Projection Neighbourhood -----------------------------------------
78 void ProgAngularNeighbourhood::run()
79 {
80  show();
81  fn_out.deleteFile();
82  MetaData SF_out;
83  FileName fn2, fn1, fnAux;
84 
85  std::cerr << "Calculating ...\n";
86  FOR_ALL_OBJECTS_IN_METADATA(DF2)
87  {
88  // Read reference projection direction
89  double rot2, tilt2;
90  DF2.getValue(MDL_IMAGE,fn2,__iter.objId);
91  DF2.getValue(MDL_ANGLE_ROT,rot2,__iter.objId);
92  DF2.getValue(MDL_ANGLE_TILT,tilt2,__iter.objId);
93  FOR_ALL_OBJECTS_IN_METADATA(DF1)
94  {
95  // Read assigned angles from document file
96  double rot1, tilt1;
97  DF1.getValue(MDL_ANGLE_ROT,rot1,__iter.objId);
98  DF1.getValue(MDL_ANGLE_TILT,tilt1,__iter.objId);
99  double psi1=0.;
100  double dist = SL.computeDistance(rot2, tilt2, 0., rot1, tilt1, psi1,
101  true, check_mirrors, false);
102  if (dist <= maxdist)
103  {
104  MDRow row;
105  DF1.getRow(row,__iter.objId);
106  SF_out.addRow(row);
107  }
108  }
109 
110  // finished reading all particles for this neighbourhood
111  fnAux=fn2.replaceCharacter('/','_');
112  fnAux=fnAux.replaceCharacter('@','_');
113  fnAux=fnAux.replaceCharacter('.','_');
114  SF_out.write((String)"neighbourhoodOf_"+fnAux.c_str()+"@"+fn_out,MD_APPEND);
115  SF_out.clear();
116  }
117 }
