72 void ProgEliminateEmptyParticles::run()
73 {
74  MetaData SF;

78  CorrelationAux aux;

85  init_progress_bar(SF.size());
86  std::size_t extraPath = fnOut.find_last_of("/");

88  FileName fnDone = fnOut.substr(0, extraPath+1) + "outTemp.xmd";
89  FileName fnElDone = fnOut.substr(0, extraPath+1) + "elimTemp.xmd";
90 
91  std::ifstream ifile1(fnOut.c_str());
92  if (ifile1) MDclass.read(fnOut);
93  std::ifstream ifile2(fnElim.c_str());
94  if (ifile2) MDclassEl.read(fnElim);
95 
96  FOR_ALL_OBJECTS_IN_METADATA(SF)
97  {
98  countItems++;
99  SF.getValue(MDL_IMAGE, fnImg, __iter.objId);
100  SF.getRow(row, countItems);
101 
102  I.read(fnImg);
103  I().setXmippOrigin();
104  centerImageTranslationally(I(), aux);
105 
106  if (useDenoising)
107  realGaussianFilter(I(), denoise);
108  bandpassFilter(I(), 0, 0.1, 0.02);
109 
110  ef.extractVariance(I(), fv);
111 
112  double ratio = fv.back();
113  row.setValue(MDL_SCORE_BY_EMPTINESS, ratio);
114  if (addFeatures)
115  row.setValue(MDL_SCORE_BY_VARIANCE, fv);
116  if (threshold<0 || ratio > threshold)
117  {
118  MDclass.addRow(row);
119  MDclassT.addRow(row);
120  }
121  else
122  {
123  MDclassEl.addRow(row);
124  MDclassElT.addRow(row);
125  }
126 
127  fv.clear();
128  if (countItems%100==0)
129  progress_bar(countItems);
130  }
131  progress_bar(SF.size());
132 
133  if (MDclass.size()>0)
134  MDclass.write(formatString("@%s", fnOut.c_str()), MD_APPEND);
135  if (MDclassEl.size()>0)
136  MDclassEl.write(formatString("@%s", fnElim.c_str()), MD_APPEND);
137  if (MDclassT.size()>0)
138  MDclassT.write(formatString("@%s", fnDone.c_str()), MD_APPEND);
139  if (MDclassElT.size()>0)
140  MDclassElT.write(formatString("@%s", fnElDone.c_str()), MD_APPEND);
141 }
