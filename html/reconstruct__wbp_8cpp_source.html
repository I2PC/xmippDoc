236 void ProgRecWbp::getSampledMatrices(MetaData &SF)
237 {
238  FileName fn_tmp;
239  Matrix2D<double> A(3, 3);
240  Matrix2D<double> L(4, 4), R(4, 4);
241  double newrot, newtilt, newpsi, rot, tilt, dum, weight, totimgs = 0.;
242  bool dumB;
243  std::vector<double> count_imgs;
244 
245  if (verbose > 0)
246  std::cerr << "--> Sampling the filter ..." << std::endl;
247 
248  // Create an (asymmetric part of an) even projection direction distribution
249  std::vector<double> rotList, tiltList;
250  make_even_distribution(rotList, tiltList, sampling, SL, true);
251  size_t NN = rotList.size();
252  count_imgs.resize(NN);
253  // Each experimental image contributes to the nearest of these directions
254  FileName fn_img;
255  FOR_ALL_OBJECTS_IN_METADATA(SF)
256  {
257  SF.getValue(MDL_IMAGE, fn_img, __iter.objId);
258  getAnglesForImage(__iter.objId, rot, tilt, dum, dum, dum, dumB, weight);
259  int idx = find_nearest_direction(rot, tilt, rotList, tiltList, SL, L,
260  R);
261  if (do_weights)
262  count_imgs[idx] += weight;
263  else
264  count_imgs[idx] += 1.;
265  }
266 
267  // Now calculate transformation matrices for all representative directions
268  no_mats = 0;
269  int SL_SymsNo = SL.symsNo();
270  int SL_SymsNo_1 = SL_SymsNo + 1;
271  for (size_t i = 0; i < NN; i++)
272  if (count_imgs[i] > 0.)
273  no_mats += SL_SymsNo_1;
274  mat_g = (WBPInfo*) malloc(no_mats * sizeof(WBPInfo));
275 
276  no_mats = 0;
277  for (size_t i = 0; i < NN; i++)
278  {
279  int count_i = (int)count_imgs[i];
280  if (count_i > 0)
281  {
282  Euler_angles2matrix(rotList[i], -tiltList[i], 0.0, A);
283  mat_g[no_mats].x = MAT_ELEM(A, 2, 0);
284  mat_g[no_mats].y = MAT_ELEM(A, 2, 1);
285  mat_g[no_mats].z = MAT_ELEM(A, 2, 2);
286  mat_g[no_mats].count = count_i;
287  totimgs += count_i;
288  no_mats++;
289 
290  // Expand symmetric directions
291  for (int j = 0; j < SL_SymsNo; j++)
292  {
293  SL.getMatrices(j, L, R, false);
294  Euler_apply_transf(L, R, rot, tilt, 0., newrot, newtilt, newpsi);
295  Euler_angles2matrix(newrot, -newtilt, newpsi, A);
296  mat_g[no_mats].x = MAT_ELEM(A, 2, 0);
297  mat_g[no_mats].y = MAT_ELEM(A, 2, 1);
298  mat_g[no_mats].z = MAT_ELEM(A, 2, 2);
299  mat_g[no_mats].count = count_i;
300  totimgs += count_i;
301  no_mats++;
302  }
303  }
304  }
305 
306  // Adjust relative threshold
307  threshold *= totimgs;
308 }

364 // Simple backprojection of a single image
365 void ProgRecWbp::simpleBackprojection(Projection &img,
366  MultidimArray<double> &vol, int diameter)
367 {
368  //this should be int not size_t ROB
369  int i, j, k, l, m;
370  Matrix2D<double> A(3, 3);
371  double dim2, x, y, z, xp, yp;
372  double value1, value2, scalex, scaley, value;
373  double radius2, x2, y2, z2, z2_plus_y2;
374 
375  // Use minus-tilt, because code copied from OldXmipp
376  Euler_angles2matrix(img.rot(), -img.tilt(), img.psi(), A);
377  A = A.inv();
378 
379  radius2 = diameter / 2.;
380  radius2 = radius2 * radius2;
381  dim2 = dim / 2;
382  double a00 = MAT_ELEM(A,0,0);
383  double a01 = MAT_ELEM(A,0,1);
384  double a10 = MAT_ELEM(A,1,0);
385  double a11 = MAT_ELEM(A,1,1);
386  double a20 = MAT_ELEM(A,2,0);
387  double a21 = MAT_ELEM(A,2,1);
388 
389  double dim1 = dim - 1;
390  const MultidimArray<double> mImg = img();
391  int idim;
392  idim = dim;//cast to int from size_t
393  for (i = 0; i < idim; i++)
394  {
395  z = -i + dim2; /*** Z points upwards ***/
396  z2 = z * z;
397  if (z2 > radius2)
398  continue;
399  double xpz = z * a20 + dim2;
400  double ypz = z * a21 + dim2;
401  for (j = 0; j < idim; j++)
402  {
403  y = j - dim2;
404  y2 = y * y;
405  z2_plus_y2 = z2 + y2;
406  if (z2_plus_y2 > radius2)
407  continue;
408  x = 0 - dim2; /***** X for k == 0 *****/
409  xp = x * a00 + y * a10 + xpz;
410  yp = x * a01 + y * a11 + ypz;
411  if (yp >= dim1 || yp < 0.0)
412  continue;
413  l = (int) yp;
414  scaley = yp - l;
415  double scale1y = 1. - scaley;
416  for (k = 0; k < idim; k++, xp += a00, yp += a01, x++)
417  {
418  x2 = x * x;
419  if (x2 + z2_plus_y2 > radius2)
420  continue;
421  if (xp >= dim1 || xp < 0.0)
422  continue;
423 
424  /**** interpolation ****/
425  m = (int) xp;
426  scalex = xp - m;
427  double scale1x = 1. - scalex;
428  value1 = scalex * dAij(mImg, l, m + 1)
429  + scale1x * dAij(mImg, l, m);
430  value2 = scalex * dAij(mImg, l + 1, m + 1)
431  + scale1x * dAij(mImg, l + 1, m);
432  value = scaley * value2 + scale1y * value1;
433  dAkij(vol, i, j, k) += value;
434  }
435  }
436  }
437 }

439 // Calculate the filter in 2D and apply ======================================
440 void ProgRecWbp::filterOneImage(Projection &proj, Tabsinc &TSINC)
441 {
442  MultidimArray<std::complex<double> > IMG;
443  Matrix2D<double> A(3, 3);
444  double factor, argum, weight, x, y;
445 
446  factor = diameter;
447 
448  //Euler_angles2matrix(proj.rot(), -proj.tilt(), proj.psi(), A);
449  //A = A.inv();
450  Euler_angles2matrix(-proj.rot(), proj.tilt(), -proj.psi(), A);
451  FourierTransform(proj(), IMG);
452  CenterFFT(IMG, true);
453 
454  // loop over all transformation matrices
455  double a00 = MAT_ELEM(A,0,0);
456  double a01 = MAT_ELEM(A,0,1);
457  double a10 = MAT_ELEM(A,1,0);
458  double a11 = MAT_ELEM(A,1,1);
459  double a20 = MAT_ELEM(A,2,0);
460  double a21 = MAT_ELEM(A,2,1);
461  for (int k = 0; k < no_mats; k++)
462  {
463  mat_f[k].x = a00 * mat_g[k].x + a10 * mat_g[k].y + a20 * mat_g[k].z;
464  mat_f[k].y = a01 * mat_g[k].x + a11 * mat_g[k].y + a21 * mat_g[k].z;
465  }
466 
467  double K = ((double) diameter) / dim;
468  FOR_ALL_ELEMENTS_IN_ARRAY2D(IMG)
469  {
470  y = K * i;
471  x = K * j;
472  weight = 0.;
473  for (int k = 0; k < no_mats; k++)
474  {
475  argum = x * mat_f[k].x + y * mat_f[k].y;
476  double daux;
477  TSINCVALUE(TSINC, argum, daux);
478  weight += mat_g[k].count * daux;
479  }
480 
481  if (fabs(weight) < threshold)
482  {
483  count_thr++;
484  A2D_ELEM(IMG, i, j) /= SGN(weight) * (threshold * factor);
485  }
486  else
487  {
488  A2D_ELEM(IMG, i, j) /= (weight * factor);
489  }
490  }
491 
492  // Calculate back-projection with the filtered projection
493  CenterFFT(IMG, false);
494  InverseFourierTransform(IMG, proj());
495 }

515 // Calculate the filter in 2D and apply ======================================
516 void ProgRecWbp::apply2DFilterArbitraryGeometry()
517 {
518  double rot, tilt, psi, xoff, yoff, weight;
519  bool flip;
520  Projection proj;
521  Matrix2D<double> L(4, 4), R(4, 4), A;
522  FileName fn_img;
523 
524  MultidimArray<double> &mReconstructedVolume = reconstructedVolume();
525  mReconstructedVolume.initZeros(dim, dim, dim);
526  mReconstructedVolume.setXmippOrigin();
527  count_thr = 0;
528 
529  // Initialize time bar
530  if (verbose > 0)
531  {
532  std::cerr << "--> Back-projecting ..." << std::endl;
533  init_progress_bar(time_bar_size);
534  }
535 
536  mat_f = (WBPInfo*) malloc(no_mats * sizeof(WBPInfo));
537  Tabsinc TSINC(0.0001, dim);
538 
539  size_t objId, objIndex;
540  while (getImageToProcess(objId, objIndex))
541  {
542  SF.getValue(MDL_IMAGE, fn_img, objId);
543  proj.read(fn_img, false);
544  getAnglesForImage(objId, rot, tilt, psi, xoff, yoff, flip, weight);
545  proj.setRot(rot);
546  proj.setTilt(tilt);
547  proj.setPsi(psi);
548  proj.setShifts(xoff, yoff);
549  proj.setFlip(flip);
550  proj.setWeight(weight);
551  proj.getTransformationMatrix(A, true);
552  if (!A.isIdentity())
553  selfApplyGeometry(BSPLINE3, proj(), A, IS_INV, WRAP);
554  if (do_weights)
555  proj() *= proj.weight();
556  proj().setXmippOrigin();
557  filterOneImage(proj, TSINC);
558  simpleBackprojection(proj, mReconstructedVolume, diameter);
559 
560  showProgress();
561  }
562  if (verbose > 0)
563  progress_bar(time_bar_size);
564 
565  // Symmetrize if necessary
566  if (fn_sym != "")
567  {
568  MultidimArray<double> Vaux;
569  Vaux.resize(mReconstructedVolume);
570  symmetrizeVolume(SL, mReconstructedVolume, Vaux);
571  mReconstructedVolume = Vaux;
572  Mask mask_prm;
573  mask_prm.mode = INNER_MASK;
574  mask_prm.R1 = diameter / 2.;
575  mask_prm.type = BINARY_CIRCULAR_MASK;
576  mask_prm.generate_mask(mReconstructedVolume);
577  mask_prm.apply_mask(mReconstructedVolume, mReconstructedVolume, 0.);
578  }
579 }
