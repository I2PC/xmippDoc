64  def analyzeDefocus(self):
65  micIds=[]
66  particleIds=[]
67  x=[]
68  y=[]
69  defocusU=[]
70  defocusV=[]
71  for particle in self.inputSet.get():
72  micIds.append(particle.getMicId())
73  particleIds.append(particle.getObjId())
74  xi, yi = particle.getCoordinate().getPosition()
75  x.append(xi)
76  y.append(yi)
77  defocusU.append(particle.getCTF().getDefocusU())
78  defocusV.append(particle.getCTF().getDefocusV())
79 
80  uniqueMicIds = list(set(micIds))
81  self.R2={}
82 
83  md = emlib.MetaData()
84 
85  for micId in uniqueMicIds:
86  idx = [i for i,j in enumerate(micIds) if j==micId]
87  defocusUbyId = []
88  defocusVbyId = []
89  meanDefocusbyId = []
90  xbyId = []
91  ybyId = []
92  particleIdsbyMicId = []
93  for idxi in idx:
94  defocusUbyId.append(defocusU[idxi])
95  defocusVbyId.append(defocusV[idxi])
96  meanDefocus = (defocusU[idxi]+defocusV[idxi])/2
97  meanDefocusbyId.append(meanDefocus)
98  xbyId.append(x[idxi])
99  ybyId.append(y[idxi])
100  particleIdsbyMicId.append(particleIds[idxi])
101 
102  #defocus = c*y + b*x + a = A * X; A=[x(i),y(i)]
103  A = np.column_stack([np.ones(len(xbyId)),xbyId,ybyId])
104  polynomial, residuals, _, _ = np.linalg.lstsq(A,meanDefocusbyId,rcond=None)
105  meanDefocusbyIdArray = np.asarray(meanDefocusbyId)
106  coefficients = np.asarray(polynomial)
107  self.R2[micId] = 1 - residuals / sum((meanDefocusbyIdArray - meanDefocusbyIdArray.mean()) ** 2)
108  mdBlock = emlib.MetaData()
109  for xi, yi, deltafi, parti in zip(xbyId,ybyId,meanDefocusbyId,particleIdsbyMicId):
110  objId = mdBlock.addObject()
111  mdBlock.setValue(emlib.MDL_ITEM_ID,long(parti),objId)
112  mdBlock.setValue(emlib.MDL_XCOOR,xi,objId)
113  mdBlock.setValue(emlib.MDL_YCOOR,yi,objId)
114  mdBlock.setValue(emlib.MDL_CTF_DEFOCUSA,deltafi,objId)
115  estimatedVal = coefficients[2]*yi + coefficients[1]*xi + coefficients[0]
116  residuali = deltafi - estimatedVal
117  mdBlock.setValue(emlib.MDL_CTF_DEFOCUS_RESIDUAL,residuali,objId)
118  mdBlock.write("mic_%d@%s"%(micId,self._getExtraPath("micrographDefoci.xmd")),emlib.MD_APPEND)
119  objId = md.addObject()
120  md.setValue(emlib.MDL_CTF_DEFOCUS_COEFS,coefficients.tolist(),objId)
121  md.write(self._getExtraPath("micrographCoef.xmd"),emlib.MD_APPEND)
