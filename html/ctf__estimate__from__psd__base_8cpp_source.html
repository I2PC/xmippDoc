189 /* Produce side information ------------------------------------------------ */
190 void ProgCTFBasicParams::produceSideInfo()
191 {
192  // Resize the frequency
193  x_digfreq.initZeros(YSIZE(*f), XSIZE(*f) / 2);
194  y_digfreq.initZeros(YSIZE(*f), XSIZE(*f) / 2);
195  w_digfreq.initZeros(YSIZE(*f), XSIZE(*f) / 2);
196  w_digfreq_r.initZeros(YSIZE(*f), XSIZE(*f) / 2);
197  x_contfreq.initZeros(YSIZE(*f), XSIZE(*f) / 2);
198  y_contfreq.initZeros(YSIZE(*f), XSIZE(*f) / 2);
199  w_contfreq.initZeros(YSIZE(*f), XSIZE(*f) / 2);
200 
201  Matrix1D<int> idx(2); // Indexes for Fourier plane
202  Matrix1D<double> freq(2); // Frequencies for Fourier plane
203 
204  FOR_ALL_ELEMENTS_IN_ARRAY2D(x_digfreq)
205  {
206  XX(idx) = j;
207  YY(idx) = i;
208 
209  // Digital frequency
210  FFT_idx2digfreq(*f, idx, freq);
211  x_digfreq(i, j) = XX(freq);
212  y_digfreq(i, j) = YY(freq);
213  w_digfreq(i, j) = freq.module();
214  w_digfreq_r(i, j) = (int)(w_digfreq(i,j) * (double)YSIZE(w_digfreq));
215 
216  // Continuous frequency
217  digfreq2contfreq(freq, freq, Tm);
218  x_contfreq(i, j) = XX(freq);
219  y_contfreq(i, j) = YY(freq);
220  w_contfreq(i, j) = freq.module();
221  }
222 
223  // Build frequency mask
224  mask.initZeros(w_digfreq);
225  w_count.initZeros(XSIZE(w_digfreq));
226  FOR_ALL_ELEMENTS_IN_ARRAY2D(w_digfreq)
227  {
228  if (w_digfreq(i, j) >= max_freq
229  || w_digfreq(i, j) <= min_freq)
230  continue;
231  mask(i, j) = 1;
232  w_count(w_digfreq_r(i, j))++;
233  }
234 
235  // Enhance PSD for ctfmodels
236  ProgCTFEnhancePSD prm;
237  prm.filter_w1 = 0.02;
238  prm.filter_w2 = 0.2;
239  prm.decay_width = 0.02;
240  prm.mask_w1 = 0.01;
241  prm.mask_w2 = 0.5;
242  enhanced_ctftomodel() = ctftomodel();
243  prm.applyFilter(enhanced_ctftomodel());
244  if (fn_psd.find('@')==std::string::npos)
245  enhanced_ctftomodel.write(fn_psd.withoutExtension() + "_enhanced_psd.xmp");
246  else
247  enhanced_ctftomodel.write(fn_psd.withoutExtension() + "_enhanced_psd.stk");
248  CenterFFT(enhanced_ctftomodel(), false);
249  enhanced_ctftomodel_fullsize() = enhanced_ctftomodel();
250 
251  // Enhance PSD for optimization
252  prm.filter_w1 = f1;
253  prm.filter_w2 = f2;
254  enhanced_ctftomodel() = ctftomodel();
255  prm.applyFilter(enhanced_ctftomodel());
256  CenterFFT(enhanced_ctftomodel(), false);
257  enhanced_ctftomodel().resize(w_digfreq);
258 
259  // Divide by the number of count at each frequency
260  // and mask between min_freq and max_freq
261  double min_val = enhanced_ctftomodel().computeMin();
262  FOR_ALL_ELEMENTS_IN_ARRAY2D(mask)
263  if (mask(i, j) <= 0)
264  enhanced_ctftomodel(i, j) = min_val;
265 
266  MultidimArray<double> aux;
267  medianFilter3x3(enhanced_ctftomodel(), aux);
268 
269  enhanced_ctftomodel() = aux;
270  enhanced_ctftomodel().rangeAdjust(0, 1);
271 
272  FourierFilter Filter;
273  Filter.FilterShape = RAISED_COSINE;
274  Filter.FilterBand = HIGHPASS;
275  Filter.w1 = 0.01;
276  Filter.raised_w = 0.005;
277  enhanced_ctftomodel().setXmippOrigin();
278  Filter.generateMask(enhanced_ctftomodel());
279  Filter.applyMaskSpace(enhanced_ctftomodel());
280  STARTINGX(enhanced_ctftomodel()) = STARTINGY(enhanced_ctftomodel()) = 0;
281 
282  // Compute now radial average of the enhanced_ctftomodel
283  psd_exp_enhanced_radial_derivative.initZeros(XSIZE(enhanced_ctftomodel()));
284  psd_theo_radial_derivative.initZeros(psd_exp_enhanced_radial_derivative);
285  psd_exp_enhanced_radial.initZeros(psd_exp_enhanced_radial_derivative);
286  psd_exp_radial.initZeros(psd_exp_enhanced_radial_derivative);
287  psd_theo_radial.initZeros(psd_exp_enhanced_radial_derivative);
288  w_digfreq_r_iN.initZeros(psd_exp_enhanced_radial_derivative);
289  FOR_ALL_ELEMENTS_IN_ARRAY2D(enhanced_ctftomodel())
290  {
291  if (w_digfreq(i,j)>min_freq && w_digfreq(i,j)<max_freq)
292  {
293  int r = w_digfreq_r(i, j);
294  w_digfreq_r_iN(r)+=1;
295  psd_exp_enhanced_radial(r) += enhanced_ctftomodel(i, j);
296  psd_exp_radial(r) += ctftomodel(i, j);
297  }
298  }
299  FOR_ALL_ELEMENTS_IN_ARRAY1D(w_digfreq_r_iN)
300  if (w_digfreq_r_iN(i)>0)
301  {
302  w_digfreq_r_iN(i)=1.0/w_digfreq_r_iN(i);
303  psd_exp_enhanced_radial(i)*=w_digfreq_r_iN(i);
304  psd_exp_radial(i)*=w_digfreq_r_iN(i);
305  }
306 
307  // Compute its derivative
308  int state=0;
309  double maxDiff=0;
310  FOR_ALL_ELEMENTS_IN_ARRAY1D(psd_exp_enhanced_radial)
311  {
312  switch (state)
313  {
314  case 0:
315  if (w_digfreq(i,0)>min_freq)
316  state=1;
317  break;
318  case 1:
319  state=2; // Skip first sample
320  break;
321  case 2:
322  if (w_digfreq(i,0)>max_freq)
323  state=3;
324  else
325  {
326  double diff=psd_exp_enhanced_radial(i)-psd_exp_enhanced_radial(i-1);
327  psd_exp_enhanced_radial_derivative(i)=diff;
328  maxDiff=std::max(maxDiff,fabs(diff));
329  }
330  break;
331  }
332  }
333  psd_exp_enhanced_radial_derivative/=maxDiff;
334  if (show_optimization)
335  {
336  psd_exp_enhanced_radial.write("PPPexpEnhanced_fast.txt");
337  psd_exp_radial.write("PPPexp_fast.txt");
338  }
339 
340 }

342 double evaluateIceness(MultidimArray<double> &psd, double Tm)
343 {
344  if (Tm>1.8)
345  return 0.0; // We cannot measure at 3.6A, if the Tm>1.8A
346  double R4_4=floor(XSIZE(psd)*Tm/4.4);
347  double R4_0=floor(XSIZE(psd)*Tm/4.0);
348  double R3_6=floor(XSIZE(psd)*Tm/3.6);
349  R4_4*=R4_4;
350  R4_0*=R4_0;
351  R3_6*=R3_6;
352  double N1=0, N2=0, avg1=0, avg2=0;
353  FOR_ALL_DIRECT_ELEMENTS_IN_ARRAY2D(psd)
354  {
355  double R2=i*i+j*j;
356  if (R2>R4_4)
357  {
358  if (R2<R4_0)
359  {
360  avg1+=A2D_ELEM(psd,i,j);
361  N1+=1;
362  }
363  else if (R2<R3_6)
364  {
365  avg2+=A2D_ELEM(psd,i,j);
366  N2+=1;
367  }
368  }
369  }
370  if (N1>0 && N2>0)
371  {
372  avg1/=N1;
373  avg2/=N2;
374  return avg2/avg1;
375  }
376  return -1;
