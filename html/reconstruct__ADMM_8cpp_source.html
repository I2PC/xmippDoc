304 void ProgReconsADMM::project(const Matrix1D<double> &r1, const Matrix1D<double> &r2, MultidimArray<double> &P, bool adjoint, double weight)
305 {
306  const MultidimArray<double> &mV=CHtb();
307  if (!adjoint)
308  {
309  P.initZeros(std::ceil(XSIZE(mV)/Tp),std::ceil(YSIZE(mV)/Tp));
310  P.setXmippOrigin();
311  }
312 
313  // Tomographic projection
314  for (int k=STARTINGZ(mV); k<=FINISHINGZ(mV); ++k) {
315  // initial computation
316  double rzn = k;
317  double rzn1= rzn*VEC_ELEM(r1,2);
318  double rzn2= rzn*VEC_ELEM(r2,2);
319  for (int i=STARTINGY(mV); i<=FINISHINGY(mV); ++i) {
320  // initial computation
321  double ryn = i;
322  double ryn1= ryn*VEC_ELEM(r1,1);
323  double ryn2= ryn*VEC_ELEM(r2,1);
324  double sx0=rzn1+ryn1;
325  double sy0=rzn2+ryn2;
326 
327  for (int j=STARTINGX(mV); j<=FINISHINGX(mV); ++j) {
328  double rxn = j;
329  double rxn1= rxn*VEC_ELEM(r1,0);
330  double rxn2= rxn*VEC_ELEM(r2,0);
331 
332  double sx = Ti*(sx0+rxn1);
333  double sy = Ti*(sy0+rxn2);
334 
335  int sxmin = std::floor(sx-kernel.supp);
336  int sxmax = std::ceil(sx+kernel.supp);
337  int symin = std::floor(sy-kernel.supp);
338  int symax = std::ceil(sy+kernel.supp);
339  if (sxmin<STARTINGX(P))
340  sxmin = STARTINGX(P);
341  if (sxmax>FINISHINGX(P))
342  sxmax = FINISHINGX(P);
343  if (symin<STARTINGY(P))
344  symin = STARTINGY(P);
345  if (symax>FINISHINGY(P))
346  symax = FINISHINGY(P);
347 
348  if (adjoint)
349  for (int ii=symin; ii<=symax; ii++) {
350  double u=(ii-sy)*Tp;
351  for (int jj=sxmin; jj<=sxmax; jj++) {
352  double v=(jj-sx)*Tp;
353  A3D_ELEM(mV,k,i,j)+=weight*A2D_ELEM(P,ii,jj)*kernel.projectionValueAt(u,v);
354  }
355  }
356  else
357  for (int ii=symin; ii<=symax; ii++) {
358  double u=(ii-sy)*Tp;
359  for (int jj=sxmin; jj<=sxmax; jj++) {
360  double v=(jj-sx)*Tp;
361  A2D_ELEM(P,ii,jj)+=weight*A3D_ELEM(mV,k,i,j)*kernel.projectionValueAt(u,v);
362  }
363  }
364  }
365  }
366  }

369 void ProgReconsADMM::computeHtKH(MultidimArray<double> &kernelV)
370 {
371  kernelV.initZeros(2*ZSIZE(CHtb())-1,2*YSIZE(CHtb())-1,2*XSIZE(CHtb())-1);
372  kernelV.setXmippOrigin();
373 
374  if (rank==0)

381  bool hasWeight=mdIn.containsLabel(MDL_WEIGHT) && useWeights;
382  bool hasCTF=(mdIn.containsLabel(MDL_CTF_MODEL) || mdIn.containsLabel(MDL_CTF_DEFOCUSU)) && useCTF;
383  CTFDescription ctf;
384  MultidimArray<double> kernelAutocorr;
385  if (!hasCTF)
386  kernel.getKernelAutocorrelation(kernelAutocorr);
387  Matrix2D<double> E;
388  Matrix1D<double> r1(3), r2(3);
389  FOR_ALL_OBJECTS_IN_METADATA(mdIn)
390  {
391  if ((i+1)%Nprocs==rank)
392  {

394  mdIn.getValue(MDL_ANGLE_ROT,rot,__iter.objId);
395  mdIn.getValue(MDL_ANGLE_TILT,tilt,__iter.objId);
396  mdIn.getValue(MDL_ANGLE_PSI,psi,__iter.objId);

401  ctf.readFromMetadataRow(mdIn,__iter.objId);

407  // Update kernel
408  Euler_angles2matrix(rot,tilt,psi,E,false);
409  E.getRow(0,r1);
410  E.getRow(1,r2);
411  double iStep=1.0/kernel.autocorrStep;
412  for (int k=((kernelV).zinit); k<=((kernelV).zinit + (int)(kernelV).zdim - 1); ++k)
413  {
414  double r1_z=k*ZZ(r1);
415  double r2_z=k*ZZ(r2);
416  for (int i=((kernelV).yinit); i<=((kernelV).yinit + (int)(kernelV).ydim - 1); ++i)
417  {
418  double r1_yz=i*YY(r1)+r1_z;
419  double r2_yz=i*YY(r2)+r2_z;
420  for (int j=((kernelV).xinit); j<=((kernelV).xinit + (int)(kernelV).xdim - 1); ++j)
421  {
422  double r1_xyz=j*XX(r1)+r1_yz;
423  double r2_xyz=j*XX(r2)+r2_yz;
424  A3D_ELEM(kernelV,k,i,j)+=weight*kernelAutocorr.interpolatedElement2D(r1_xyz*iStep,r2_xyz*iStep);
425  }
426  }
427  }
428  }
429 
430  i++;
431  if (i%100==0 && rank==0)
432  progress_bar(i);
433  }
434  if (rank==0)
435  progress_bar(mdIn.size());
436 
437 #ifndef SYMMETRIZE_PROJECTIONS
438  MultidimArray<double> kernelVsym;
439  symmetrizeVolume(SL, kernelV, kernelVsym, false, false, true);
440  kernelV=kernelVsym;
441 #endif
442 
443  shareVolume(kernelV);
444 }

446 void addGradientTerm(double mu, AdmmKernel &kernel, MultidimArray<double> &L, FourierTransformer &transformer,
447  MultidimArray<std::complex<double> >&fourierKernelV, char direction)
448 {
449  kernel.computeGradient(L,direction);
450 
451  transformer.FourierTransform();
452  double K=mu*MULTIDIM_SIZE(L);
453  double xdim_2=(double)(XSIZE(L)/2);
454  double Kargument=2*PI*xdim_2/XSIZE(L);
455  FOR_ALL_ELEMENTS_IN_ARRAY3D(fourierKernelV)
456  {
457  // Calculation of L^t*L
458  // -L^2(r)*exp(-i*2*pi*(N/2)/N*(k+i+j)): the last term is a phase shift due to FFT
459  double argument=Kargument*(k+i+j);
460  double s, c;
461  sincos(argument,&s,&c);
462  A3D_ELEM(fourierKernelV,k,i,j)+=-A3D_ELEM(transformer.fFourier,k,i,j)*
463  A3D_ELEM(transformer.fFourier,k,i,j)*std::complex<double>(K*c,K*s);
464  }
465 }

543 void ProgReconsADMM::applyConjugateGradient()
544 {
545  // Conjugate gradient is solving Ax=b, when A is symmetric (=> positive semidefinite) and we apply it
546  // to the problem
547  // (H^T K H+mu L^TL + lambda I) x = H^Tb + mu L^T(u-d)
548  // Being H the projection operator
549  // K the CTF operator
550  // L the gradient operator
551 
552  // Compute H^tb+mu*L^t(u-d)
553  MultidimArray<double> r;
554  applyLtFilter(fourierLx,ux,dx);
555  r=ud;
556  applyLtFilter(fourierLy,uy,dy);
557  r+=ud;
558  applyLtFilter(fourierLz,uz,dz);
559  r+=ud;
560  r*=mu;
561 
562  r+=CHtb();
563 
564  // Apply A^tA to the current estimate of the reconstruction
565  MultidimArray<double> AtAVk;
566  applyKernel3D(Ck(),AtAVk);
567 
568  // Compute first residual. This is the negative gradient of ||Ax-b||^2
569  r-=AtAVk;
570  double d=r.sum2();
571 
572  // Search direction
573  MultidimArray<double> p, AtAp;
574  p=r;
575 
576  // Perform CG iterations
577  MultidimArray<double> &mCk=Ck();
578 
579  for (int iter=0; iter<Ncgiter; ++iter)
580  {
581  applyKernel3D(p,AtAp);
582  double alpha=d/p.dotProduct(AtAp);
583

808 void AdmmKernel::computeGradient(MultidimArray<double> &gradient, char direction, bool adjoint)

814  {

819  for (int j=-supp; j<=supp; ++j)
