40 /* Split histograms -------------------------------------------------------- */
41 // Split several histograms within the indexes l0 and lF so that
42 // the entropy after division is maximized
43 int splitHistogramsUsingEntropy(const std::vector<Histogram1D> &hist,
44  size_t l0, size_t lF)
45 {
46  // Number of classes
47  int K = hist.size();
48 
49  // Set everything outside l0 and lF to zero, and make it a PDF
50  std::vector<Histogram1D> histNorm;
51  for (int k = 0; k < K; k++)
52  {
53  Histogram1D histaux = hist[k];
54  for (size_t l = 0; l < XSIZE(histaux); l++)
55  if (l < l0 || l > lF)
56  DIRECT_A1D_ELEM(histaux,l) = 0;
57  histaux *= 1.0/histaux.sum();
58  histNorm.push_back(histaux);
59  }
60 
61  // Compute for each class the probability of being l<=l0 and l>l0
62  MultidimArray<double> p(K, 2);
63  for (int k = 0; k < K; k++)
64  {
65  const Histogram1D& histogram=histNorm[k];
66  DIRECT_A2D_ELEM(p,k, 0) = DIRECT_A1D_ELEM(histogram,l0);
67  DIRECT_A2D_ELEM(p,k, 1) = 0;
68  for (size_t l = l0 + 1; l <= lF; l++)
69  DIRECT_A2D_ELEM(p,k, 1) += DIRECT_A1D_ELEM(histogram,l);
70  }
71 
72  // Compute the splitting l giving maximum entropy
73  double maxEntropy = 0;
74  int lmaxEntropy = -1;
75  size_t l = l0;
76  while (l < lF)
77  {
78  // Compute the entropy of the classes if we split by l
79  double entropy = 0;
80  FOR_ALL_DIRECT_ELEMENTS_IN_MULTIDIMARRAY(p)
81  {
82  double aux=DIRECT_MULTIDIM_ELEM(p,n);
83  if (aux != 0)
84  entropy -= aux * log10(aux);
85  }
86 
87 #ifdef DEBUG_SPLITTING_USING_ENTROPY
88  std::cout << "Splitting at " << l << " entropy=" << entropy
89  << std::endl;
90 #endif
91 
92  // Check if this is the maximum
93  if (entropy > maxEntropy)
94  {
95  maxEntropy = entropy;
96  lmaxEntropy = l;
97  }
98 
99  // Move to next split point
100  ++l;
101 
102  // Update probabilities of being l<=l0 and l>l0
103  for (int k = 0; k < K; k++)
104  {
105  const Histogram1D& histogram=histNorm[k];
106  double aux=DIRECT_A1D_ELEM(histogram,l);
107  DIRECT_A2D_ELEM(p,k, 0) += aux;
108  DIRECT_A2D_ELEM(p,k, 1) -= aux;
109  }
110  }
111 
112 #ifdef DEBUG_SPLITTING_USING_ENTROPY
113  std::cout << "Finally in l=[" << l0 << "," << lF
114  << " Max Entropy:" << maxEntropy
115  << " lmax=" << lmaxEntropy << std::endl;
116 #endif
117 
118  // If the point giving the maximum entropy is too much on the extreme,
119  // substitute it by the middle point
120  if (lmaxEntropy<=2 || lmaxEntropy>=(int)lF-2)
121  lmaxEntropy = (int)ceil((lF + l0)/2.0);
122 
123  return lmaxEntropy;
124 }

127 LeafNode::LeafNode(const std::vector < MultidimArray<double> > &leafFeatures,
128  int discrete_levels)
129 {
130  __discreteLevels = discrete_levels;
131  K = leafFeatures.size();
132  if (__discreteLevels==0)
133  {
134  // This is a dummy node for features that cannot classify
135  MultidimArray<int> newBins(1);
136  A1D_ELEM(newBins,0)=0;
137  Histogram1D hist;
138  hist.resize(1);
139  A1D_ELEM(hist,0)=1;
140  IrregularHistogram1D irregHist;
141  for (int k=0; k<K; k++)
142  {
143  irregHist.init(hist, newBins);
144  irregHist.selfNormalize();
145  __leafPDF.push_back(irregHist);
146  }
147  }
148  else
149  {
150  // Compute the minimum and maximum of each class
151  double minval=0., maxval=0.;
152  for(int k=0; k<K; k++)
153  {
154  double minvalk=0., maxvalk=0.;
155  leafFeatures[k].computeDoubleMinMax(minvalk, maxvalk);
156  if (k==0)
157  {
158  minval=minvalk;
159  maxval=maxvalk;
160  }
161  else
162  {
163  minval=std::min(minval,minvalk);
164  maxval=std::max(maxval,maxvalk);
165  }
166  }
167  if (minval==maxval)
168  {
169  __discreteLevels=0;
170  return;
171  }
172 
173  // Compute the PDF of each class
174  std::vector<Histogram1D> hist(K);
175  for (int k=0; k<K; k++)
176  {
177  // There is variation of this feature for this class
178  compute_hist(leafFeatures[k], hist[k], minval, maxval, 100);
179  hist[k] += 1; // Apply Laplace correction
180  }
181 
182  // Split the histograms into discrete_level (power of 2) bins
183  std::queue< Matrix1D<int> > intervals, splittedIntervals;
184  Matrix1D<int> limits(2);
185  VECTOR_R2(limits,0,99);
186  intervals.push(limits);
187  int imax=ROUND(log2(__discreteLevels));
188  for (int i=0; i<imax; i++)
189  {
190  // Split all the intervals in the queue
191  while (!intervals.empty())
192  {
193  Matrix1D<int> currentInterval = intervals.front();
194  intervals.pop();
195  int lsplit = splitHistogramsUsingEntropy(hist,
196  currentInterval(0), currentInterval(1));
197  VECTOR_R2(limits,currentInterval(0),lsplit);
198  splittedIntervals.push(limits);
199  VECTOR_R2(limits,lsplit+1, currentInterval(1));
200  splittedIntervals.push(limits);
201  }
202 
203  // Copy the splitted intervals to the interval list
204  while (!splittedIntervals.empty())
205  {
206  intervals.push(splittedIntervals.front());
207  splittedIntervals.pop();
208  }
209  }
210 
211  // Compute the bins of the split
212  MultidimArray<int> newBins(__discreteLevels);
213  imax=intervals.size();
214  for (int i=0; i<imax; i++)
215  {
216  A1D_ELEM(newBins,i) = intervals.front()(1);
217  intervals.pop();
218  }
219 
220  // Compute now the irregular histograms
221  IrregularHistogram1D irregHist;
222  for (int k=0; k<K; k++)
223  {
224  irregHist.init(hist[k], newBins);
225  irregHist.selfNormalize();
226  __leafPDF.push_back(irregHist);
227  }
228  }
229 }

352 int NaiveBayes::doInference(const MultidimArray<double> &newFeatures, double &cost,
353  Matrix1D<double> &classesProbs, Matrix1D<double> &allCosts)
354 {
355  classesProbs=__priorProbsLog10;
356  for(int f=0; f<Nfeatures; f++)
357  {
358  const LeafNode &leaf_f=*(__leafs[f]);
359  double newFeatures_f=DIRECT_A1D_ELEM(newFeatures,f);
360  for (int k=0; k<K; k++)
361  {
362  double p = leaf_f.assignProbability(newFeatures_f, k);
363 
364  if (fabs(p) < 1e-2)
365  VEC_ELEM(classesProbs,k) += -2*DIRECT_A1D_ELEM(__weights,f);
366  else
367  VEC_ELEM(classesProbs,k) += DIRECT_A1D_ELEM(__weights,f)*std::log10(p);
368 
369 #ifdef DEBUG_FINE_CLASSIFICATION
370 
371  if(debugging == true)
372  {
373  std::cout << "Feature " << f
374  << " Probability for class " << k << " = "
375  << classesProbs(k) << " increase= " << p
376  << std::endl;
377  char c;
378  // COSS std::cin >> c;
379  // if (c=='q') debugging = false;
380  }
381 #endif
382 
383  }
384  }
385 
386  classesProbs-=classesProbs.computeMax();
387  // std::cout << "classesProbs " << classesProbs.transpose() << std::endl;
388 
389  for (int k=0; k<K; k++)
390  VEC_ELEM(classesProbs,k)=pow(10.0,VEC_ELEM(classesProbs,k));
391  classesProbs*=1.0/classesProbs.sum();
392  // std::cout << "classesProbs norm " << classesProbs.transpose() << std::endl;
393 
394  allCosts=__cost*classesProbs;
395  // std::cout << "allCosts " << allCosts.transpose() << std::endl;
396 
397  int bestk=0;
398  cost=VEC_ELEM(allCosts,0)=std::log10(VEC_ELEM(allCosts,0));
399  for (int k=1; k<K; k++)
400  {
401  VEC_ELEM(allCosts,k)=std::log10(VEC_ELEM(allCosts,k));
402  if (VEC_ELEM(allCosts,k)<cost)
403  {
404  cost=VEC_ELEM(allCosts,k);
405  bestk=k;
406  }
407  }
408 
409 #ifdef DEBUG_CLASSIFICATION
410  if(debugging == true)
411  {
412  for (int k=0; k<K; k++)
413  classesProbs(k)=log10(classesProbs(k));
414  std::cout << "Class probababilities=" << classesProbs.transpose()
415  << "\n costs=" << allCosts.transpose()
416  << " best class=" << bestk << " cost=" << cost << std::endl;
417  char c;
418  // COSS std::cin >> c;
419  // if (c=='q') debugging = false;
420  }
421 #endif
422  return bestk;
423 }

442  double samplingFeatures, double samplingIndividuals,
443  const std::string &newJudgeCombination)

447  K=features.size();
448  judgeCombination=newJudgeCombination;
449 
450 #ifdef WEIGHTED_SAMPLING
451  // Measure the classification power of each variable
452  NaiveBayes *nb_weights=new NaiveBayes(features, priorProbs, discreteLevels);
453  MultidimArray<double> weights=nb_weights->__weights;
454  delete nb_weights;
455  double sumWeights=weights.sum();
456 #endif
457 
458  for (int n=0; n<numberOfClassifiers; n++)
459  {
460  // Produce the set of features for this subclassifier
461  MultidimArray<int> subFeatures(NsubFeatures);
462  FOR_ALL_ELEMENTS_IN_ARRAY1D(subFeatures)
463  {
464 #ifdef WEIGHTED_SAMPLING

467  do
468  {
469  double wj=DIRECT_A1D_ELEM(weights,j);
470  if (wj<random_sum_weight)
471  {
472  random_sum_weight-=wj;
473  j++;
474  if (j==NFeatures)

476  j=NFeatures-1;
477  break;
478  }
479  }
480  else

482  }

484  DIRECT_A1D_ELEM(subFeatures,i)=j;
485 #else
486 
487  DIRECT_A1D_ELEM(subFeatures,i)=round(rnd_unif(0,NFeatures-1));
488 #endif
489 
490  }
491

493  std::vector< MultidimArray<double> > newFeatures;
494 
495  // Produce the data set for each class
496  for (int k=0; k<K; k++)
497  {
498  int NIndividuals=YSIZE(features[k]);

500  MultidimArray<int> subIndividuals(NsubIndividuals);

511

513  }
514

564  maxCost(k)=costn;

568  bestClass=k;
569  }
570  }

572  cost=minCost(bestClass);
573  else if (judgeCombination[bestClass]=='M')
