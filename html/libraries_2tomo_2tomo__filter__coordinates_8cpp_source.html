95 void ProgTomoFilterCoordinates::calculateCoordinateStatistics(MultidimArray<double> &tom)
96 {
97  MetaDataVec scoredMd;
98  MDRowVec row;
99 
100  for (size_t i = 0; i < inputCoords.size(); i++)
101  {
102  Point3D coor = inputCoords[i];
103 
104  if (((coor.z - radius) < 0) || ((coor.z + radius) > (zDim-1)) || ((coor.y - radius) < 0) || ((coor.y + radius) > (yDim-1)) || ((coor.x - radius) < 0) || ((coor.x + radius) > (xDim-1)))
105  {
106  std::cout << "WARNNING: Coordinate at (x=" << coor.x<< ", y=" << coor.y << ", z=" << coor.z << ") masked out." << std::endl;
107  continue;
108  }
109 
110  double meanCoor = 0;
111  double meanCoor2 = 0;
112  double stdCoor = 0;
113  // double medianCoor = 0;
114  // double madCoor = 0;
115  double value = 0;
116  size_t Nelems = 0;
117 
118 
119  for (int i = -radius; i < radius; i++)
120  {
121  size_t i2 = i*i;
122  for (int j = -radius; j < radius; j++)
123  {
124  size_t j2i2 = i2 + j*j;
125  for (int k = -radius; k < radius; k++)
126  {
127  size_t r2 = j2i2 + k*k;
128 
129  if (r2 <= radius)
130  {
131  auto value = DIRECT_A3D_ELEM(tom, coor.z + k, coor.y + i, coor.x + j);
132  meanCoor += value;
133  Nelems++;
134  meanCoor2 += value*value;
135  }
136  }
137  }
138  }
139 
140  meanCoor /= Nelems;
141  stdCoor = sqrt(meanCoor2/Nelems - meanCoor*meanCoor);
142 
143  row.setValue(MDL_XCOOR, coor.x);
144  row.setValue(MDL_YCOOR, coor.y);
145  row.setValue(MDL_ZCOOR, coor.z);
146  row.setValue(MDL_AVG, meanCoor);
147  row.setValue(MDL_STDDEV, stdCoor);
148  // row.setValue(MDL_VOLUME_SCORE1, medianCoor);
149  // row.setValue(MDL_VOLUME_SCORE2, madCoor);
150  scoredMd.addRow(row);
151  }
152  scoredMd.write(fnOut);
153 }
