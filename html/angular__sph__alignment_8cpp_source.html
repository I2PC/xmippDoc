253 double continuousSphCost(double *x, void *_prm)
254 {
255  ProgAngularSphAlignment *prm=(ProgAngularSphAlignment *)_prm;
256  double deltax=x[prm->pos+1];
257  double deltay=x[prm->pos+2];
258  double deltaRot=x[prm->pos+3];
259  double deltaTilt=x[prm->pos+4];
260  double deltaPsi=x[prm->pos+5];
261  if (prm->maxShift>0 && deltax*deltax+deltay*deltay>prm->maxShift*prm->maxShift)
262  return 1e38;
263  if (prm->maxAngularChange>0 && (fabs(deltaRot)>prm->maxAngularChange || fabs(deltaTilt)>prm->maxAngularChange || fabs(deltaPsi)>prm->maxAngularChange))
264  return 1e38;
265 
266  MAT_ELEM(prm->A,0,2)=prm->old_shiftX+deltax;
267  MAT_ELEM(prm->A,1,2)=prm->old_shiftY+deltay;
268  MAT_ELEM(prm->A,0,0)=1;
269  MAT_ELEM(prm->A,0,1)=0;
270  MAT_ELEM(prm->A,1,0)=0;
271  MAT_ELEM(prm->A,1,1)=1;
272 
273  return prm->tranformImageSph(x,prm->old_rot+deltaRot, prm->old_tilt+deltaTilt, prm->old_psi+deltaPsi,
274  prm->A);
275 }

533 void ProgAngularSphAlignment::deformVol(MultidimArray<double> &mVD, const MultidimArray<double> &mV, double &def)
534 {
535  int l,n,m;
536  size_t idxY0=(VEC_XSIZE(clnm)-5)/4;
537  double Ncount=0.0;
538  double totalVal=0.0;
539  double voxModg=0.0;
540  double diff2=0.0;
541 
542  def=0.0;
543  size_t idxZ0=2*idxY0;
544  size_t idxR=3*idxY0;
545  sumVd=0.0;
546  for (int k=STARTINGZ(mV); k<=FINISHINGZ(mV); k++)
547  {
548  for (int i=STARTINGY(mV); i<=FINISHINGY(mV); i++)
549  {
550  for (int j=STARTINGX(mV); j<=FINISHINGX(mV); j++)
551  {
552 
553  double gx=0.0, gy=0.0, gz=0.0;
554  for (size_t idx=0; idx<idxY0; idx++)
555  {
556  double RmaxF=VEC_ELEM(clnm,idx+idxR);
557  double RmaxF2=RmaxF*RmaxF;
558  double iRmaxF=1.0/RmaxF;
559  double k2=k*k;
560  double kr=k*iRmaxF;
561  double k2i2=k2+i*i;
562  double ir=i*iRmaxF;
563  double r2=k2i2+j*j;
564  double jr=j*iRmaxF;
565  double zsph=0.0;
566  double rr=sqrt(r2)*iRmaxF;
567  if (r2<RmaxF2)
568  {
569  spherical_index2lnm(idx+1,l,n,m);
570  zsph=ZernikeSphericalHarmonics(l,n,m,jr,ir,kr,rr);
571  }
572  if (rr>0 || l==0)
573  {
574  gx += VEC_ELEM(clnm,idx) *(zsph);
575  gy += VEC_ELEM(clnm,idx+idxY0) *(zsph);
576  gz += VEC_ELEM(clnm,idx+idxZ0) *(zsph);
577  }
578  }
579  mVD(k,i,j) = mV.interpolatedElement3D(j+gx,i+gy,k+gz);
580  double voxelR=A3D_ELEM(mV,k,i,j);
581  double absVoxelR=std::abs(voxelR);
582  voxModg += absVoxelR*(gx*gx+gy*gy+gz*gz);
583  totalVal += absVoxelR;
584 
585  double voxelI=A3D_ELEM(mVD,k,i,j);
586  double diff=voxelR-voxelI;
587  diff2+=absVoxelR*diff*diff;
588  sumVd+=voxelI;
589  }
590  }
591  }
592 
593  // COSS def=5e-3*sqrt(diff2/totalVal)+sqrt(voxModg/totalVal);
594  def=sqrt(voxModg/totalVal);
595  totalDeformation = sqrt(voxModg/totalVal);
596 }
