87  void run()
88  {
89  // Get angles ==============================================================
90  MetaData angles;
91  angles.read(fnIn);
92  size_t AngleNo = angles.size();
93  if (AngleNo == 0 || !angles.containsLabel(MDL_ANGLE_ROT))
94  REPORT_ERROR(ERR_MD_BADLABEL, "Input file doesn't contain angular information");
95 
96  double maxWeight = -99.e99;
97  MultidimArray<double> weight;
98  weight.initZeros(AngleNo);
99  if (angles.containsLabel(MDL_WEIGHT))
100  {
101  // Find maximum weight
102  int i=0;
103  FOR_ALL_OBJECTS_IN_METADATA(angles)
104  {
105  double w;
106  angles.getValue(MDL_WEIGHT,w,__iter.objId);
107  DIRECT_A1D_ELEM(weight,i++)=w;
108  maxWeight=XMIPP_MAX(w,maxWeight);
109  }
110  }
111  double imaxWeight=1.0/maxWeight;
112 
113  // Build vector tables ======================================================
114  std::vector< Matrix1D<double> > v, v_ang;
115  v.reserve(AngleNo);
116  v_ang.reserve(AngleNo);
117  Matrix1D<double> aux(3);
118  Matrix1D<double> aux_ang(3);
119  FOR_ALL_OBJECTS_IN_METADATA(angles)
120  {
121  double rot, tilt, psi;
122  angles.getValue(MDL_ANGLE_ROT,rot,__iter.objId);
123  angles.getValue(MDL_ANGLE_TILT,tilt,__iter.objId);
124  angles.getValue(MDL_ANGLE_PSI,psi,__iter.objId);
125  if (up_down_correction && fabs(tilt)>90)
126  \
127  Euler_up_down(rot,tilt,psi,rot,tilt,psi);
128  Euler_direction(rot, tilt, psi, aux);
129  v.push_back(aux);
130  VECTOR_R3(aux_ang, rot, tilt, psi);
131  v_ang.push_back(aux_ang);
132  }
133 
134  if (type=="histogram")
135  {
136  // Compute minimum distance table
137  MultidimArray<double> dist;
138  dist.initZeros(AngleNo);
139  for (size_t i = 0; i < AngleNo; i++)
140  {
141  const Matrix1D<double> &vi=v[i];
142  for (size_t j = i + 1; j < AngleNo; j++)
143  {
144  const Matrix1D<double> &vj=v[j];
145  // Since the two vectors are in the unit sphere, the spherical distance
146  // is simply the arc cosine
147  double d = acos(XX(vi)*XX(vj) + YY(vi)*YY(vj) + ZZ(vi)*ZZ(vj));
148  d=RAD2DEG(d);
149  if (DIRECT_A1D_ELEM(dist,i) == 0 || d < DIRECT_A1D_ELEM(dist,i))
150  DIRECT_A1D_ELEM(dist,i) = d;
151  if (DIRECT_A1D_ELEM(dist,j) == 0 || d < DIRECT_A1D_ELEM(dist,j))
152  DIRECT_A1D_ELEM(dist,j) = d;
153  }
154  }
155 
156  Histogram1D dist_hist;
157  compute_hist(dist, dist_hist, steps);
158  dist_hist.write(fnOut);
159  }
160  else if (type=="chimera")
161  {
162  std::ofstream fh_bild;
163  fh_bild.open(fnOut.c_str(), std::ios::out);
164  if (!fh_bild)
165  REPORT_ERROR(ERR_IO_NOWRITE, fnOut);
166  fh_bild << ".color 1 0 0" << std::endl;
167 
168  int imax=v.size();

170  {
171  double r=rmax;
172  if (maxWeight>0)
173  r *= DIRECT_A1D_ELEM(weight,i)*imaxWeight;
174  fh_bild
175  << ".sphere "
176  << R*XX(v[i]) + shift_center << " "
177  << R*YY(v[i]) + shift_center << " "
178  << R*ZZ(v[i]) + shift_center << " "

196 #define TO_PS(x,y) \
197  tmp=y; \
198  y=400.0f-x*250.0f/60; \
199  x=300.0f+tmp*250.0f/60;

201  Matrix1D<double> p0(4), p1(4), p2(4), p3(4), view_direction, pp(3);
202  Matrix2D<double> A, euler_view;
203  Euler_angles2matrix(rot_view, tilt_view, 0., euler_view);
204  euler_view.getRow(2, view_direction);
205  double tmp;
206  int imax=v.size();
207  for (int i=0; i<imax; i++)
208  {

212 
213  // Initially the triangle is on the floor of the projection plane
214  VECTOR_R3(p0, 0 , 0 , 0);
215  VECTOR_R3(p1, 0 , r*2 / 3*SIND(60), 0);
216  VECTOR_R3(p2, r / 2*0.6, -r*1 / 3*SIND(60), 0);
217  VECTOR_R3(p3, -r / 2*0.6, -r*1 / 3*SIND(60), 0);
218 
219  // Convert to homogeneous coordinates
220  p0(3) = 1;
221  p1(3) = 1;
222  p2(3) = 1;
223  p3(3) = 1;
224 
225  // Compute Transformation matrix
226  const Matrix1D<double> &v_angi=v_ang[i];
227  const Matrix1D<double> &vi=v[i];
228  Euler_angles2matrix(VEC_ELEM(v_angi,1), VEC_ELEM(v_angi,2),
229  VEC_ELEM(v_angi,3), A, true);
230  A = A.transpose(); // We go from the projection plane to the universal coordinates
231 
232  // Apply a translation to the sphere of radius R
233  MAT_ELEM(A, 0, 3) = R * XX(vi);
234  MAT_ELEM(A, 1, 3) = R * YY(vi);
235  MAT_ELEM(A, 2, 3) = R * ZZ(vi);
236 
237  // Convert triangle coordinates to universal ones
238  p0 = A * p0;
239  p1 = A * p1;
240  p2 = A * p2;
241  p3 = A * p3;
242

244  if (solid_sphere)
245  {
246  // Point-plane distance
247  double d=XX(p0)*XX(view_direction)+
248  YY(p0)*YY(view_direction)+
249  ZZ(p0)*ZZ(view_direction);
250  if (d < 0)
251  continue;
252  }
253 
254  // Project this triangle onto the view plane and write in PS
255  Uproject_to_plane(p1, euler_view, pp);
256  TO_PS(XX(pp), YY(pp));
257  fh_ps << "newpath\n";
258  fh_ps << XX(pp) << " " << YY(pp) << " moveto\n";
259 
260  Uproject_to_plane(p2, euler_view, pp);
261  TO_PS(XX(pp), YY(pp));
262  fh_ps << XX(pp) << " " << YY(pp) << " lineto\n";
