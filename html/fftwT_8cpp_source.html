37 template<typename T>
38 void FFTwT<T>::init(const HW &cpu, const FFTSettingsNew<T> &settings, bool reuse) {
39  bool canReuse = m_isInit
40  && reuse
41  // we can reuse if the helper arrays are bigger, or non-existent
42  && (m_settings->sBytesBatch() >= settings.sBytesBatch())
43  && (m_settings->fBytesBatch() >= settings.fBytesBatch())
44  && (needsAuxArray(*m_settings) == needsAuxArray(settings));
45  bool mustAllocate = !canReuse;
46  if (mustAllocate) {
47  release();
48  }
49  // previous plan has to be released, otherwise we will get CPU memory leak
50  release(m_plan);
51  delete m_settings;
52 
53  m_settings = new FFTSettingsNew<T>(settings);
54  try {
55  m_cpu = &dynamic_cast<const CPU&>(cpu);
56  } catch (std::bad_cast&) {
57  REPORT_ERROR(ERR_ARG_INCORRECT, "Instance of CPU expected");
58  }
59 
60  check();
61 
62  m_plan = createPlan(*m_cpu, *m_settings, true);
63  if (mustAllocate) {
64  allocate();
65  }
66 
67  m_isInit = true;
68 }

147 void FFTwT<T>::check() {
148  if (m_settings->sDim().x() < 1) {
149  REPORT_ERROR(ERR_LOGIC_ERROR, "X dim must be at least 1 (one)");
150  }
151  if ((m_settings->sDim().y() > 1)
152  && (m_settings->sDim().x() < 2)) {
153  REPORT_ERROR(ERR_LOGIC_ERROR, "X dim must be at least 2 (two) for 2D/3D transformations");
154  }
155  if ((m_settings->sDim().z() > 1)
156  && (m_settings->sDim().y() < 2)) {
157  REPORT_ERROR(ERR_LOGIC_ERROR, "Y dim must be at least 2 (two) for 3D transformations");
158  }
159  if (m_settings->batch() > m_settings->sDim().n()) {
160  REPORT_ERROR(ERR_LOGIC_ERROR, "Batch size must be smaller or equal to number of signals");
161  }

202 template<typename T>
203 std::complex<T>* FFTwT<T>::fft(const T *in,
204  std::complex<T> *out) {
205  auto isReady = m_isInit && m_settings->isForward();
206  if ( ! isReady) {
207  REPORT_ERROR(ERR_LOGIC_ERROR, "Not ready to perform Fourier Transform. "
208  "Call init() function first");
209  }
210 
211  // process signals in batches
212  for (size_t offset = 0; offset < m_settings->sDim().n(); offset += m_settings->batch()) {
213  size_t signalsToProcess = std::min(m_settings->batch(), m_settings->sDim().n() - offset);
214  size_t signalsToSkip = m_settings->batch() - signalsToProcess;
215  bool useAux = (m_settings->batch() != signalsToProcess) && needsAuxArray(*m_settings);
216 
217  // set in / out pointers for normal case
218  const T *batchSrc = in
219  + (offset - signalsToSkip) * m_settings->sDim().xyzPadded();
220  std::complex<T> *batchDest = out
221  + (offset - signalsToSkip) * m_settings->fDim().xyzPadded();
222 
223  if (useAux) {
224  // copy data to aux array
225  memcpy(m_SD,
226  in + offset * m_settings->sDim().xyzPadded(),
227  signalsToProcess * m_settings->sBytesSingle());
228  // set pointers (we should be here only for in-place transforms)
229  batchSrc = (const T*)m_SD;
230  batchDest = (std::complex<T>*)m_SD;
231  }
232 
233  fft(cast(m_plan), batchSrc, batchDest);
234 
235  if (useAux) {
236  // copy data from aux array (skip already processed data)
237  memcpy(out + offset * m_settings->fDim().xyzPadded(),
238  batchDest,
239  signalsToProcess * m_settings->fBytesSingle());
240  }
241  }
242  return out;
243 }

250 template<typename T>
251 T* FFTwT<T>::ifft(const std::complex<T> *in,
252  T *out) {
253  auto isReady = m_isInit && ( ! m_settings->isForward());
254  if ( ! isReady) {
255  REPORT_ERROR(ERR_LOGIC_ERROR, "Not ready to perform Inverse Fourier Transform. "
256  "Call init() function first");
257  }
258 
259  // process signals in batches
260  for (size_t offset = 0; offset < m_settings->fDim().n(); offset += m_settings->batch()) {
261  size_t signalsToProcess = std::min(m_settings->batch(), m_settings->fDim().n() - offset);
262  size_t signalsToSkip = m_settings->batch() - signalsToProcess;
263  bool useAux = (m_settings->batch() != signalsToProcess) || needsAuxArray(*m_settings);
264 
265  // copy data
266  memcpy(m_FD,
267  in + offset * m_settings->fDim().xyzPadded(),
268  signalsToProcess * m_settings->fBytesSingle());
269 
270  // set the destination
271  T *batchDest = useAux
272  ? m_SD
273  : out + (offset - signalsToSkip) * m_settings->sDim().xyzPadded();
274 
275  ifft(cast(m_plan), m_FD, batchDest);
276 
277  if (useAux) {
278  // copy data from aux array
279  memcpy(out + offset * m_settings->sDim().xyzPadded(),
280  batchDest,
281  signalsToProcess * m_settings->sBytesSingle());
282  }
283  }
284  return out;

346 template<typename T>
347 template<typename U, typename F>
348 U FFTwT<T>::planHelper(const FFTSettingsNew<T> &settings, F function,
349  int threads,
350  bool isDataAligned) {
351  auto n = std::array<int, 3>{(int)settings.sDim().z(), (int)settings.sDim().y(), (int)settings.sDim().x()};
352  int rank = 3;
353  if (settings.sDim().z() == 1) rank--;
354  if ((2 == rank) && (settings.sDim().y() == 1)) rank--;
355  int offset = 3 - rank;
356 
357  void *in = nullptr;
358  void *out = settings.isInPlace() ? in : &m_mockOut;
359 
360  // no input-preserving algorithms are implemented for multi-dimensional c2r transforms
361  // see http://www.fftw.org/fftw3_doc/Planner-Flags.html#Planner-Flags
362  auto flags = FFTW_ESTIMATE
363  | (settings.isForward() ? FFTW_PRESERVE_INPUT : FFTW_DESTROY_INPUT);
364  if ( ! isDataAligned) {
365  flags = flags | FFTW_UNALIGNED;
366  }
367 
368  int idist;
369  int odist;
370  if (settings.isForward()) {
371  idist = settings.sDim().xyzPadded();
372  odist = settings.fDim().xyzPadded();
373  } else {
374  idist = settings.fDim().xyzPadded();
375  odist = settings.sDim().xyzPadded();
376  }
377  // set threads
378  fftw_plan_with_nthreads(threads);
379  fftwf_plan_with_nthreads(threads);
380 
381  auto tmp = function(rank, &n[offset], settings.batch(),
382  in, nullptr,
383  1, idist,
384  out, nullptr,
385  1, odist,
386  flags);
387  return tmp;
388 }
